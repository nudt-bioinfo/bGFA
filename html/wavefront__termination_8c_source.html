34/*
35 * Detect alignment termination (end of alignment)
36 */
37FORCE_NO_INLINE bool wavefront_termination_end2end(
38 wavefront_aligner_t* const wf_aligner,
39 wavefront_t* const mwavefront,
40 const int score,
41 const int score_mod) {
42 // Parameters
43 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
44 const int pattern_length = sequences->pattern_length;
45 const int text_length = sequences->text_length;
46 const affine2p_matrix_type component_end = wf_aligner->component_end;
47 const int alignment_k = DPMATRIX_DIAGONAL(text_length,pattern_length);
48 const wf_offset_t alignment_offset = DPMATRIX_OFFSET(text_length,pattern_length);
49 // Select end component
50 switch (component_end) {
51 case affine2p_matrix_M: {
52 // Check diagonal/offset
53 if (mwavefront->lo > alignment_k || alignment_k > mwavefront->hi) return false; // Not done
54 const wf_offset_t moffset = mwavefront->offsets[alignment_k];
55 if (moffset < alignment_offset) return false; // Not done
56 // We are done
57 wf_aligner->alignment_end_pos.score = score;
58 wf_aligner->alignment_end_pos.k = alignment_k;
59 wf_aligner->alignment_end_pos.offset = alignment_offset;
60 return true;
61 }
62 case affine2p_matrix_I1: {
63 // Fetch I1-wavefront & check diagonal/offset
64 wavefront_t* const i1wavefront = wf_aligner->wf_components.i1wavefronts[score_mod];
65 if (i1wavefront == NULL || i1wavefront->lo > alignment_k || alignment_k > i1wavefront->hi) return false; // Not done
66 const wf_offset_t i1offset = i1wavefront->offsets[alignment_k];
67 if (i1offset < alignment_offset) return false; // Not done
68 // We are done
69 wf_aligner->alignment_end_pos.score = score;
70 wf_aligner->alignment_end_pos.k = alignment_k;
71 wf_aligner->alignment_end_pos.offset = alignment_offset;
72 return true;
73 }
74 case affine2p_matrix_I2: {
75 // Fetch I2-wavefront & check diagonal/offset
76 wavefront_t* const i2wavefront = wf_aligner->wf_components.i2wavefronts[score_mod];
77 if (i2wavefront == NULL || i2wavefront->lo > alignment_k || alignment_k > i2wavefront->hi) return false; // Not done
78 const wf_offset_t i2offset = i2wavefront->offsets[alignment_k];
79 if (i2offset < alignment_offset) return false; // Not done
80 // We are done
81 wf_aligner->alignment_end_pos.score = score;
82 wf_aligner->alignment_end_pos.k = alignment_k;
83 wf_aligner->alignment_end_pos.offset = alignment_offset;
84 return true;
85 }
86 case affine2p_matrix_D1: {
87 // Fetch D1-wavefront & check diagonal/offset
88 wavefront_t* const d1wavefront = wf_aligner->wf_components.d1wavefronts[score_mod];
89 if (d1wavefront == NULL || d1wavefront->lo > alignment_k || alignment_k > d1wavefront->hi) return false; // Not done
90 const wf_offset_t d1offset = d1wavefront->offsets[alignment_k];
91 if (d1offset < alignment_offset) return false; // Not done
92 // We are done
93 wf_aligner->alignment_end_pos.score = score;
94 wf_aligner->alignment_end_pos.k = alignment_k;
95 wf_aligner->alignment_end_pos.offset = alignment_offset;
96 return true;
97 }
98 case affine2p_matrix_D2: {
99 // Fetch D2-wavefront & check diagonal/offset
100 wavefront_t* const d2wavefront = wf_aligner->wf_components.d2wavefronts[score_mod];
101 if (d2wavefront == NULL || d2wavefront->lo > alignment_k || alignment_k > d2wavefront->hi) return false; // Not done
102 const wf_offset_t d2offset = d2wavefront->offsets[alignment_k];
103 if (d2offset < alignment_offset) return false; // Not done
104 // We are done
105 wf_aligner->alignment_end_pos.score = score;
106 wf_aligner->alignment_end_pos.k = alignment_k;
107 wf_aligner->alignment_end_pos.offset = alignment_offset;
108 return true;
109 }
110 default:
111 break;
112 }
113 return false;
114}
115FORCE_NO_INLINE bool wavefront_termination_endsfree(
116 wavefront_aligner_t* const wf_aligner,
117 wavefront_t* const mwavefront,
118 const int score,
119 const int k,
120 const wf_offset_t offset) {
121 // Parameters
122 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
123 const int pattern_length = sequences->pattern_length;
124 const int text_length = sequences->text_length;
125 // Check ends-free reaching boundaries
126 const int h_pos = WAVEFRONT_H(k,offset);
127 const int v_pos = WAVEFRONT_V(k,offset);
128 if (h_pos >= text_length) { // Text is aligned
129 // Is Pattern end-free?
130 const int pattern_left = pattern_length - v_pos;
131 const int pattern_end_free = wf_aligner->alignment_form.pattern_end_free;
132 if (pattern_left <= pattern_end_free) {
133 #ifdef WFA_PARALLEL
134 #pragma omp critical
135 #endif
136 {
137 wf_aligner->alignment_end_pos.score = score;
138 wf_aligner->alignment_end_pos.k = k;
139 wf_aligner->alignment_end_pos.offset = offset;
140 }
141 return true; // Quit (we are done)
142 }
143 }
144 if (v_pos >= pattern_length) { // Pattern is aligned
145 // Is text end-free?
146 const int text_left = text_length - h_pos;
147 const int text_end_free = wf_aligner->alignment_form.text_end_free;
148 if (text_left <= text_end_free) {
149 #ifdef WFA_PARALLEL
150 #pragma omp critical
151 #endif
152 {
153 wf_aligner->alignment_end_pos.score = score;
154 wf_aligner->alignment_end_pos.k = k;
155 wf_aligner->alignment_end_pos.offset = offset;
156 }
157 return true; // Quit (we are done)
158 }
159 }
160 // Not done
161 return false;
162}
