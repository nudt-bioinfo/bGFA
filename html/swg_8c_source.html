37/*
38 * SWG traceback
39 */
40void swg_traceback(
41 affine_matrix_t* const affine_matrix,
42 affine_penalties_t* const penalties,
43 const int pattern_length,
44 const int text_length,
45 const int target_v,
46 const int target_h,
47 cigar_t* const cigar) {
48 // Parameters
49 affine_cell_t** const dp = affine_matrix->columns;
50 char* const operations = cigar->operations;
51 cigar->end_offset = cigar->max_operations;
52 int op_sentinel = cigar->end_offset - 1;
53 // Add final insertions/deletions
54 int i;
55 for (i=target_v;i<pattern_length;++i) { operations[op_sentinel--] = 'D'; }
56 for (i=target_h;i<text_length;++i) { operations[op_sentinel--] = 'I'; }
57 // Compute traceback
58 affine_matrix_type matrix_type = affine_matrix_M;
59 int h = target_h;
60 int v = target_v;
61 while (h>0 && v>0) {
62 switch (matrix_type) {
63 case affine_matrix_D:
64 // Traceback D-matrix
65 operations[op_sentinel--] = 'D';
66 if (dp[h][v].D != dp[h][v-1].D + penalties->gap_extension) {
67 matrix_type = affine_matrix_M;
68 }
69 --v;
70 break;
71 case affine_matrix_I:
72 // Traceback I-matrix
73 operations[op_sentinel--] = 'I';
74 if (dp[h][v].I != dp[h-1][v].I + penalties->gap_extension) {
75 matrix_type = affine_matrix_M;
76 }
77 --h;
78 break;
79 case affine_matrix_M:
80 // Traceback M-matrix
81 if (dp[h][v].M == dp[h-1][v-1].M + penalties->mismatch) {
82 operations[op_sentinel--] = 'X';
83 --h; --v;
84 } else if (dp[h][v].M == dp[h][v].D) {
85 matrix_type = affine_matrix_D;
86 } else if (dp[h][v].M == dp[h][v].I) {
87 matrix_type = affine_matrix_I;
88 } else if (dp[h][v].M == dp[h-1][v-1].M + penalties->match) {
89 operations[op_sentinel--] = 'M';
90 --h; --v;
91 } else {
92 fprintf(stderr,"SWG backtrace. No backtrace operation found\n");
93 exit(1);
94 }
95 break;
96 }
97 }
98 // Add initial deletions/insertions
99 while (v>0) { operations[op_sentinel--] = 'D'; --v; }
100 while (h>0) { operations[op_sentinel--] = 'I'; --h; }
101 cigar->begin_offset = op_sentinel+1;
102 cigar->score = dp[target_h][target_v].M;
103}
104/*
105 * SWG alignment
106 */
107void swg_align(
108 affine_matrix_t* const affine_matrix,
109 affine_penalties_t* const penalties,
110 const char* const pattern,
111 const int pattern_length,
112 const char* const text,
113 const int text_length,
114 cigar_t* const cigar) {
115 // Parameters
116 affine_cell_t** const dp = affine_matrix->columns;
117 int h, v;
118 // Init DP
119 dp[0][0].D = AFFINE_SCORE_MAX;
120 dp[0][0].I = AFFINE_SCORE_MAX;
121 dp[0][0].M = 0;
122 for (v=1;v<=pattern_length;++v) { // Init first column
123 dp[0][v].D = penalties->gap_opening + v*penalties->gap_extension;
124 dp[0][v].I = AFFINE_SCORE_MAX;
125 dp[0][v].M = dp[0][v].D;
126 }
127 for (h=1;h<=text_length;++h) { // Init first row
128 dp[h][0].D = AFFINE_SCORE_MAX;
129 dp[h][0].I = penalties->gap_opening + h*penalties->gap_extension;
130 dp[h][0].M = dp[h][0].I;
131 }
132 // Compute DP
133 for (h=1;h<=text_length;++h) {
134 for (v=1;v<=pattern_length;++v) {
135 // Update DP.D
136 const int del_new = dp[h][v-1].M + penalties->gap_opening + penalties->gap_extension;
137 const int del_ext = dp[h][v-1].D + penalties->gap_extension;
138 const int del = MIN(del_new,del_ext);
139 dp[h][v].D = del;
140 // Update DP.I
141 const int ins_new = dp[h-1][v].M + penalties->gap_opening + penalties->gap_extension;
142 const int ins_ext = dp[h-1][v].I + penalties->gap_extension;
143 const int ins = MIN(ins_new,ins_ext);
144 dp[h][v].I = ins;
145 // Update DP.M
146 const int m_match = dp[h-1][v-1].M + ((pattern[v-1]==text[h-1]) ? penalties->match : penalties->mismatch);
147 dp[h][v].M = MIN(m_match,MIN(ins,del));
148 }
149 }
150 // Compute traceback
151 swg_traceback(
152 affine_matrix,penalties,
153 pattern_length,text_length,
154 pattern_length,text_length,cigar);
155 // DEBUG
156 //affine_matrix_print(stderr,affine_matrix,pattern,text);
157}
158/*
159 * SWG alignment (ends-free)
160 */
161void swg_align_endsfree(
162 affine_matrix_t* const affine_matrix,
163 affine_penalties_t* const penalties,
164 const char* const pattern,
165 const int pattern_length,
166 const char* const text,
167 const int text_length,
168 const int pattern_begin_free,
169 const int pattern_end_free,
170 const int text_begin_free,
171 const int text_end_free,
172 cigar_t* const cigar) {
173 // Parameters
174 affine_cell_t** const dp = affine_matrix->columns;
175 const int pattern_min_v = pattern_length - pattern_end_free;
176 const int text_min_h = text_length - text_end_free;
177 int h, v;
178 // Init DP
179 dp[0][0].D = AFFINE_SCORE_MAX;
180 dp[0][0].I = AFFINE_SCORE_MAX;
181 dp[0][0].M = 0;
182 for (v=1;v<=pattern_length;++v) { // Init first column
183 dp[0][v].D = (v > pattern_begin_free) ? penalties->gap_opening + (v+1-pattern_begin_free)*penalties->gap_extension : 0;
184 dp[0][v].I = AFFINE_SCORE_MAX;
185 dp[0][v].M = dp[0][v].D;
186 }
187 for (h=1;h<=text_length;++h) { // Init first row
188 dp[h][0].D = AFFINE_SCORE_MAX;
189 dp[h][0].I = (h > text_begin_free) ? penalties->gap_opening + (h+1-text_begin_free)*penalties->gap_extension : 0;
190 dp[h][0].M = dp[h][0].I;
191 }
192 // Keep minimum score
193 int min_v = 0, min_h = 0, min_score = AFFINE_SCORE_MAX;
194 if (text_min_h==0 && dp[0][pattern_length].M<min_score) {
195 min_score = dp[0][pattern_length].M;
196 min_v = pattern_length;
197 min_h = 0;
198 }
199 // Compute DP
200 for (h=1;h<=text_length;++h) {
201 for (v=1;v<=pattern_length;++v) {
202 // Update DP.D
203 const int del_new = dp[h][v-1].M + penalties->gap_opening + penalties->gap_extension;
204 const int del_ext = dp[h][v-1].D + penalties->gap_extension;
205 const int del = MIN(del_new,del_ext);
206 dp[h][v].D = del;
207 // Update DP.I
208 const int ins_new = dp[h-1][v].M + penalties->gap_opening + penalties->gap_extension;
209 const int ins_ext = dp[h-1][v].I + penalties->gap_extension;
210 const int ins = MIN(ins_new,ins_ext);
211 dp[h][v].I = ins;
212 // Update DP.M
213 const int m_match = dp[h-1][v-1].M + ((pattern[v-1]==text[h-1]) ? penalties->match : penalties->mismatch);
214 dp[h][v].M = MIN(m_match,MIN(ins,del));
215 }
216 // Pattern aligned. Keep minimum score
217 if (h>=text_min_h && dp[h][pattern_length].M<min_score) {
218 min_score = dp[h][pattern_length].M;
219 min_v = pattern_length;
220 min_h = h;
221 }
222 }
223 // Text aligned. Keep minimum score
224 for (v=pattern_length;v>=0;--v) {
225 if (v>=pattern_min_v && dp[text_length][v].M<min_score) {
226 min_score = dp[text_length][v].M;
227 min_v = v;
228 min_h = text_length;
229 }
230 }
231 // Compute traceback
232 swg_traceback(
233 affine_matrix,penalties,
234 pattern_length,text_length,
235 min_v,min_h,cigar);
236 // DEBUG
237 //affine_matrix_print(stderr,affine_matrix,pattern,text);
238}
239/*
240 * SWG alignment (banded)
241 */
242void swg_align_banded(
243 affine_matrix_t* const affine_matrix,
244 affine_penalties_t* const penalties,
245 const char* const pattern,
246 const int pattern_length,
247 const char* const text,
248 const int text_length,
249 const int bandwidth,
250 cigar_t* const cigar) {
251 // Parameters
252 const int k_end = ABS(text_length-pattern_length)+1;
253 int effective_bandwidth = bandwidth;
254 if (effective_bandwidth < k_end) effective_bandwidth = k_end;
255 if (effective_bandwidth > pattern_length) effective_bandwidth = pattern_length;
256 if (effective_bandwidth > text_length) effective_bandwidth = text_length;
257 affine_cell_t** const dp = affine_matrix->columns;
258 int h, v;
259 // Initialize
260 dp[0][0].D = AFFINE_SCORE_MAX;
261 dp[0][0].I = AFFINE_SCORE_MAX;
262 dp[0][0].M = 0;
263 for (v=1;v<=effective_bandwidth;++v) {
264 dp[0][v].D = penalties->gap_opening + v*penalties->gap_extension;
265 dp[0][v].I = AFFINE_SCORE_MAX;
266 dp[0][v].M = dp[0][v].D;
267 }
268 // Compute DP
269 for (h=1;h<=text_length;++h) {
270 // Compute lo limit
271 int lo;
272 if (h <= effective_bandwidth) {
273 lo = 1;
274 dp[h][lo-1].D = AFFINE_SCORE_MAX;
275 dp[h][lo-1].I = penalties->gap_opening + h*penalties->gap_extension;
276 dp[h][lo-1].M = dp[h][lo-1].I;
277 } else {
278 lo = h - effective_bandwidth;
279 dp[h][lo-1].D = AFFINE_SCORE_MAX;
280 dp[h][lo-1].I = AFFINE_SCORE_MAX;
281 dp[h][lo-1].M = AFFINE_SCORE_MAX;
282 }
283 // Compute hi limit
284 int hi = h + effective_bandwidth - 1;
285 if (hi > pattern_length) {
286 hi = pattern_length;
287 } else if (h > 1) {
288 dp[h-1][hi].D = AFFINE_SCORE_MAX;
289 dp[h-1][hi].I = AFFINE_SCORE_MAX;
290 dp[h-1][hi].M = AFFINE_SCORE_MAX;
291 }
292 // Compute column
293 for (v=lo;v<=hi;++v) {
294 // Update DP.D
295 const int del_new = dp[h][v-1].M + penalties->gap_opening + penalties->gap_extension;
296 const int del_ext = dp[h][v-1].D + penalties->gap_extension;
297 const int del = MIN(del_new,del_ext);
298 dp[h][v].D = del;
299 // Update DP.I
300 const int ins_new = dp[h-1][v].M + penalties->gap_opening + penalties->gap_extension;
301 const int ins_ext = dp[h-1][v].I + penalties->gap_extension;
302 const int ins = MIN(ins_new,ins_ext);
303 dp[h][v].I = ins;
304 // Update DP.M
305 const int m_match = dp[h-1][v-1].M + ((pattern[v-1]==text[h-1]) ? penalties->match : penalties->mismatch);
306 dp[h][v].M = MIN(m_match,MIN(ins,del));
307 }
308 }
309 // Compute traceback
310 swg_traceback(
311 affine_matrix,penalties,
312 pattern_length,text_length,
313 pattern_length,text_length,cigar);
314}
