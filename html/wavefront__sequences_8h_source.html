37/*
38 * Custom extend-match function, e.g.:
39 *
40 * typedef struct {
41 * char* pattern;
42 * int pattern_length;
43 * char* text;
44 * int text_length;
45 * } match_function_params_t;
46 *
47 * int match_function(int v,int h,void* arguments) {
48 * // Extract parameters
49 * match_function_params_t* match_arguments = (match_function_params_t*)arguments;
50 * // Check match
51 * if (v > match_arguments->pattern_length || h > match_arguments->text_length) return 0;
52 * return (match_arguments->pattern[v] == match_arguments->text[h]);
53 * }
54 */
55typedef int (*alignment_match_funct_t)(int,int,void*);

60typedef enum {
61 wf_sequences_ascii = 0,
62 wf_sequences_lambda = 1,
63 wf_sequences_packed2bits = 2,
64} wf_sequences_mode_t;
65typedef struct {
66 // Mode
67 wf_sequences_mode_t mode; // Sequences mode
68 bool reverse; // Reverse sequences
69 // Current sequences & bounds
70 char* pattern; // Pointer to current pattern sequence (padded)
71 char* text; // Pointer to current text sequence (padded)
72 int pattern_begin; // Pattern begin offset
73 int pattern_length; // Pattern length
74 int text_begin; // Text begin offset
75 int text_length; // Text length
76 // Lambda Sequence
77 alignment_match_funct_t match_funct; // Custom matching function (match(v,h,args))
78 void* match_funct_arguments; // Generic arguments passed to matching function (args)
79 // Internal buffers (ASCII encoded)
80 char* seq_buffer; // Internal buffer
81 int seq_buffer_allocated; // Internal buffer allocated
82 char* pattern_buffer; // Source pattern sequence
83 char* text_buffer; // Source text sequence
84 int pattern_buffer_length; // Source pattern length
85 int text_buffer_length; // Source text length
86 char pattern_eos; // Source pattern char at EOS
87 char text_eos; // Source pattern char at EOS
88} wavefront_sequences_t;
