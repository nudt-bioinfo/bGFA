39void affine_matrix_allocate(
40 affine_matrix_t* const matrix,
41 const int num_rows,
42 const int num_columns,
43 mm_allocator_t* const mm_allocator) {
44 // Allocate Matrix
45 int h;
46 matrix->num_rows = num_rows;
47 matrix->num_columns = num_columns;
48 matrix->columns = mm_allocator_malloc(mm_allocator,num_columns*sizeof(affine_cell_t*)); // Columns
49 for (h=0;h<num_columns;++h) {
50 matrix->columns[h] = mm_allocator_calloc(mm_allocator,num_rows,affine_cell_t,false); // Rows
51 }
52}
53void affine_matrix_free(
54 affine_matrix_t* const matrix,
55 mm_allocator_t* const mm_allocator) {
56 const int num_columns = matrix->num_columns;
57 int h;
58 for (h=0;h<num_columns;++h) {
59 mm_allocator_free(mm_allocator,matrix->columns[h]);
60 }
61 mm_allocator_free(mm_allocator,matrix->columns);
62}

72void affine_matrix_print(
73 FILE* const stream,
74 const affine_matrix_t* const matrix,
75 const char* const pattern,
76 const char* const text) {
77 // Parameters
78 affine_cell_t** const dp = matrix->columns;
79 int i, j;
80 // Print Header
81 fprintf(stream," ");

83 fprintf(stream," %c ",text[i]);

85 fprintf(stream,"\n ");
86 for (i=0;i<matrix->num_columns;++i) {
87 fprintf(stream," ");
88 AFFINE_MATRIX_PRINT_VALUE(dp[i][0].M);
89 }
90 fprintf(stream,"\n");
91 // Print Rows
92 for (i=1;i<matrix->num_rows;++i) {
93 fprintf(stream,"%c",pattern[i-1]);
94 for (j=0;j<matrix->num_columns;++j) {
95 fprintf(stream," ");
96 AFFINE_MATRIX_PRINT_VALUE(dp[j][i].M);
97 }
98 fprintf(stream,"\n");
