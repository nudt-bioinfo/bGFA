37/*
38 * Compute limits
39 */
40void wavefront_compute_limits_input(
41 wavefront_aligner_t* const wf_aligner,
42 const wavefront_set_t* const wavefront_set,
43 int* const lo,
44 int* const hi) {
45 // Parameters
46 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
47 const wavefront_t* const m_misms = wavefront_set->in_mwavefront_misms;
48 const wavefront_t* const m_open1 = wavefront_set->in_mwavefront_open1;
49 // Init
50 int min_lo = m_misms->lo;
51 int max_hi = m_misms->hi;
52 // Gap-linear
53 if (min_lo > m_open1->lo-1) min_lo = m_open1->lo-1;
54 if (max_hi < m_open1->hi+1) max_hi = m_open1->hi+1;
55 if (distance_metric == gap_linear) {
56 *lo = min_lo;
57 *hi = max_hi;
58 return;
59 }
60 // Parameters
61 const wavefront_t* const i1_ext = wavefront_set->in_i1wavefront_ext;
62 const wavefront_t* const d1_ext = wavefront_set->in_d1wavefront_ext;
63 // Gap-affine
64 if (min_lo > i1_ext->lo+1) min_lo = i1_ext->lo+1;
65 if (max_hi < i1_ext->hi+1) max_hi = i1_ext->hi+1;
66 if (min_lo > d1_ext->lo-1) min_lo = d1_ext->lo-1;
67 if (max_hi < d1_ext->hi-1) max_hi = d1_ext->hi-1;
68 if (distance_metric == gap_affine) {
69 *lo = min_lo;
70 *hi = max_hi;
71 return;
72 }
73 // Parameters
74 const wavefront_t* const m_open2 = wavefront_set->in_mwavefront_open2;
75 const wavefront_t* const i2_ext = wavefront_set->in_i2wavefront_ext;
76 const wavefront_t* const d2_ext = wavefront_set->in_d2wavefront_ext;
77 // Gap-affine-2p
78 if (min_lo > m_open2->lo-1) min_lo = m_open2->lo-1;
79 if (max_hi < m_open2->hi+1) max_hi = m_open2->hi+1;
80 if (min_lo > i2_ext->lo+1) min_lo = i2_ext->lo+1;
81 if (max_hi < i2_ext->hi+1) max_hi = i2_ext->hi+1;
82 if (min_lo > d2_ext->lo-1) min_lo = d2_ext->lo-1;
83 if (max_hi < d2_ext->hi-1) max_hi = d2_ext->hi-1;
84 *lo = min_lo;
85 *hi = max_hi;
86}
87void wavefront_compute_limits_output(
88 wavefront_aligner_t* const wf_aligner,
89 const int lo,
90 const int hi,
91 int* const effective_lo,
92 int* const effective_hi) {
93 // Parameters
94 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
95 const int max_score_scope = wf_components->max_score_scope;
96 // Add padding to avoid compute-kernel peeling
97 const int eff_lo = lo - (max_score_scope + 1);
98 const int eff_hi = hi + (max_score_scope + 1);
99 // Consider historic (to avoid errors using heuristics)
100 *effective_lo = MIN(eff_lo,wf_components->historic_min_lo);
101 *effective_hi = MAX(eff_hi,wf_components->historic_max_hi);
102 wf_components->historic_min_lo = *effective_lo;
103 wf_components->historic_max_hi = *effective_hi;
104}

121/*
122 * Compute ends-free init conditions
123 */
124bool wavefront_compute_endsfree_required(
125 wavefront_aligner_t* const wf_aligner,
126 const int score) {
127 // Parameters
128 alignment_form_t* const alg_form = &wf_aligner->alignment_form;
129 wavefront_penalties_t* const penalties = &wf_aligner->penalties;
130 // Return if ends-free initialization is required
131 if (penalties->match == 0) return false;
132 if (alg_form->span != alignment_endsfree) return false;
133 if (alg_form->text_begin_free == 0 &&
134 alg_form->pattern_begin_free == 0) return false;
135 if (score % (-penalties->match) != 0) return false;
136 // Check boundary conditions for ends-free
137 const int endsfree_k = score/(-penalties->match); // (h/v)-coordinate for boundary conditions
138 const bool text_begin_free = (alg_form->text_begin_free >= endsfree_k);
139 const bool pattern_begin_free = (alg_form->pattern_begin_free >= endsfree_k);
140 if (!text_begin_free && !pattern_begin_free) return false;
141 // Ok
142 return true;
143}
144void wavefront_compute_endsfree_limits(
145 wavefront_aligner_t* const wf_aligner,
146 const int score,
147 int* const lo,
148 int* const hi) {
149 // Parameters
150 alignment_form_t* const alg_form = &wf_aligner->alignment_form;
151 wavefront_penalties_t* const penalties = &wf_aligner->penalties;
152 // Consider ends-free conditions
153 const int endsfree_k = score/(-penalties->match);
154 *hi = (alg_form->text_begin_free >= endsfree_k) ? endsfree_k : INT_MIN;
155 *lo = (alg_form->pattern_begin_free >= endsfree_k) ? -endsfree_k : INT_MAX;
156}
157void wavefront_compute_endsfree_init_offset(
158 wavefront_aligner_t* const wf_aligner,
159 wavefront_t* const wavefront,
160 const int k,
161 const int v,
162 const int h) {
163 // Parameters
164 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
165 wf_offset_t* const offsets = wavefront->offsets;
166 // Set offset
167 offsets[k] = DPMATRIX_OFFSET(h,v);
168 if (wf_components->bt_piggyback) {
169 wavefront->bt_pcigar[k] = 0;
170 wavefront->bt_prev[k] =
171 wf_backtrace_buffer_init_block(wf_components->bt_buffer,v,h);
172 }
173}
174void wavefront_compute_endsfree_init(
175 wavefront_aligner_t* const wf_aligner,
176 wavefront_t* const wavefront,
177 const int score) {
178 // Parameters
179 alignment_form_t* const alg_form = &wf_aligner->alignment_form;
180 wavefront_penalties_t* const penalties = &wf_aligner->penalties;
181 const int lo = wavefront->lo;
182 const int hi = wavefront->hi;
183 // Consider ends-free conditions
184 int endsfree_k = score/(-penalties->match);
185 wf_offset_t* const offsets = wavefront->offsets;
186 // Consider text begin-free
187 int k;
188 if (alg_form->text_begin_free >= endsfree_k) {
189 if (hi >= endsfree_k) {
190 if (offsets[endsfree_k] <= DPMATRIX_OFFSET(endsfree_k,0)) {
191 wavefront_compute_endsfree_init_offset(wf_aligner,wavefront,endsfree_k,0,endsfree_k);
192 }
193 } else {
194 for (k=hi+1;k<endsfree_k;++k) {
195 offsets[k] = WAVEFRONT_OFFSET_NULL;
196 }
197 wavefront_compute_endsfree_init_offset(wf_aligner,wavefront,endsfree_k,0,endsfree_k);
198 wavefront->hi = endsfree_k;
199 }
200 }
201 // Consider pattern begin-free
202 if (alg_form->pattern_begin_free >= endsfree_k) {
203 endsfree_k = -endsfree_k;
204 if (lo <= endsfree_k) {
205 if (offsets[endsfree_k] <= DPMATRIX_OFFSET(0,endsfree_k)) {
206 wavefront_compute_endsfree_init_offset(wf_aligner,wavefront,endsfree_k,-endsfree_k,0);
207 }
208 } else {
209 wavefront_compute_endsfree_init_offset(wf_aligner,wavefront,endsfree_k,-endsfree_k,0);
210 for (k=endsfree_k+1;k<lo;k++) {
211 offsets[k] = WAVEFRONT_OFFSET_NULL;
212 }
213 wavefront->lo = endsfree_k;
214 }
215 }

217wavefront_t* wavefront_compute_endsfree_allocate_null(
218 wavefront_aligner_t* const wf_aligner,
219 const int score) {
220 // Parameters
221 wavefront_slab_t* const wavefront_slab = wf_aligner->wavefront_slab;
222 alignment_form_t* const alg_form = &wf_aligner->alignment_form;
223 wavefront_penalties_t* const penalties = &wf_aligner->penalties;
224 // Consider ends-free conditions
225 const int endsfree_k = score/(-penalties->match);
226 const bool text_begin_free = (alg_form->text_begin_free >= endsfree_k);
227 const bool pattern_begin_free = (alg_form->pattern_begin_free >= endsfree_k);
228 int lo = 0, hi = 0;
229 if (text_begin_free && pattern_begin_free) {
230 lo = -endsfree_k;
231 hi = endsfree_k;
232 } else if (text_begin_free) {
233 lo = endsfree_k;
234 hi = endsfree_k;
235 } else if (pattern_begin_free) {
236 lo = -endsfree_k;
237 hi = -endsfree_k;
238 }
239 // Compute effective hi/lo dimensions
240 int effective_lo, effective_hi;
241 wavefront_compute_limits_output(wf_aligner,lo,hi,&effective_lo,&effective_hi);
242 // Allocate & initialize
243 wavefront_t* const wavefront = wavefront_slab_allocate(wavefront_slab,effective_lo,effective_hi);
244 wf_offset_t* const offsets = wavefront->offsets;
245 int k;
246 for (k=lo;k<=hi;k++) {
247 offsets[k] = WAVEFRONT_OFFSET_NULL;
248 }
249 if (text_begin_free) {
250 wavefront_compute_endsfree_init_offset(wf_aligner,wavefront,endsfree_k,0,endsfree_k);
251 }
252 if (pattern_begin_free) {
253 wavefront_compute_endsfree_init_offset(wf_aligner,wavefront,-endsfree_k,endsfree_k,0);
254 }
255 wavefront->lo = lo;
256 wavefront->hi = hi;
257 // Set max/min init elements
258 wavefront->wf_elements_init_min = lo;
259 wavefront->wf_elements_init_max = hi;
260 // Return
261 return wavefront;
262}

306void wavefront_compute_fetch_input(
307 wavefront_aligner_t* const wf_aligner,
308 wavefront_set_t* const wavefront_set,
309 const int score) {
310 // Parameters
311 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
312 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
313 // Compute scores
314 const wavefront_penalties_t* const penalties = &(wf_aligner->penalties);
315 if (distance_metric == gap_linear) {
316 int mismatch = score - penalties->mismatch;
317 int gap_open1 = score - penalties->gap_opening1;
318 // Modular wavefront
319 if (wf_components->memory_modular) {
320 const int max_score_scope = wf_components->max_score_scope;
321 if (mismatch > 0) mismatch = mismatch % max_score_scope;
322 if (gap_open1 > 0) gap_open1 = gap_open1 % max_score_scope;
323 }
324 // Fetch wavefronts
325 wavefront_set->in_mwavefront_misms = wavefront_compute_get_mwavefront(wf_components,mismatch);
326 wavefront_set->in_mwavefront_open1 = wavefront_compute_get_mwavefront(wf_components,gap_open1);
327 } else { // distance_metric == gap_affine || distance_metric == gap_affine_2p
328 int mismatch = score - penalties->mismatch;
329 int gap_open1 = score - penalties->gap_opening1 - penalties->gap_extension1;
330 int gap_extend1 = score - penalties->gap_extension1;
331 int gap_open2 = score - penalties->gap_opening2 - penalties->gap_extension2;
332 int gap_extend2 = score - penalties->gap_extension2;
333 // Modular wavefront
334 if (wf_components->memory_modular) {
335 const int max_score_scope = wf_components->max_score_scope;
336 if (mismatch > 0) mismatch = mismatch % max_score_scope;
337 if (gap_open1 > 0) gap_open1 = gap_open1 % max_score_scope;
338 if (gap_extend1 > 0) gap_extend1 = gap_extend1 % max_score_scope;
339 if (gap_open2 > 0) gap_open2 = gap_open2 % max_score_scope;
340 if (gap_extend2 > 0) gap_extend2 = gap_extend2 % max_score_scope;
341 }
342 // Fetch wavefronts
343 wavefront_set->in_mwavefront_misms = wavefront_compute_get_mwavefront(wf_components,mismatch);
344 wavefront_set->in_mwavefront_open1 = wavefront_compute_get_mwavefront(wf_components,gap_open1);
345 wavefront_set->in_i1wavefront_ext = wavefront_compute_get_i1wavefront(wf_components,gap_extend1);
346 wavefront_set->in_d1wavefront_ext = wavefront_compute_get_d1wavefront(wf_components,gap_extend1);
347 if (distance_metric == gap_affine) return;
348 wavefront_set->in_mwavefront_open2 = wavefront_compute_get_mwavefront(wf_components,gap_open2);
349 wavefront_set->in_i2wavefront_ext = wavefront_compute_get_i2wavefront(wf_components,gap_extend2);
350 wavefront_set->in_d2wavefront_ext = wavefront_compute_get_d2wavefront(wf_components,gap_extend2);
351 }
352}
353/*
354 * Output wavefronts (allocate)
355 */
356void wavefront_compute_free_output(
357 wavefront_aligner_t* const wf_aligner,
358 const int score_mod) {
359 // Parameters
360 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
361 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
362 wavefront_slab_t* const wavefront_slab = wf_aligner->wavefront_slab;
363 // Free
364 if (wf_components->mwavefronts[score_mod]) {
365 wavefront_slab_free(wavefront_slab,wf_components->mwavefronts[score_mod]);
366 }
367 if (distance_metric == gap_linear) return;
368 if (wf_components->i1wavefronts[score_mod]) {
369 wavefront_slab_free(wavefront_slab,wf_components->i1wavefronts[score_mod]);
370 }
371 if (wf_components->d1wavefronts[score_mod]) {
372 wavefront_slab_free(wavefront_slab,wf_components->d1wavefronts[score_mod]);
373 }
374 if (distance_metric == gap_affine) return;
375 if (wf_components->i2wavefronts[score_mod]) {
376 wavefront_slab_free(wavefront_slab,wf_components->i2wavefronts[score_mod]);
377 }
378 if (wf_components->d2wavefronts[score_mod]) {
379 wavefront_slab_free(wavefront_slab,wf_components->d2wavefronts[score_mod]);
380 }
381}
382void wavefront_compute_allocate_output_null(
383 wavefront_aligner_t* const wf_aligner,
384 const int score) {
385 // Parameters
386 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
387 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
388 // Modular wavefront
389 int score_mod = score;
390 if (wf_components->memory_modular) {
391 score_mod = score % wf_components->max_score_scope;
392 wavefront_compute_free_output(wf_aligner,score_mod);
393 }
394 // Consider ends-free (M!=0)
395 if (wavefront_compute_endsfree_required(wf_aligner,score)) {
396 wf_components->mwavefronts[score_mod] =
397 wavefront_compute_endsfree_allocate_null(wf_aligner,score);
398 } else {
399 wf_components->mwavefronts[score_mod] = NULL;
400 }
401 // Nullify Wavefronts
402 if (distance_metric == gap_linear) return;
403 wf_components->i1wavefronts[score_mod] = NULL;
404 wf_components->d1wavefronts[score_mod] = NULL;
405 if (distance_metric == gap_affine) return;
406 wf_components->i2wavefronts[score_mod] = NULL;
407 wf_components->d2wavefronts[score_mod] = NULL;
408}
409void wavefront_compute_allocate_output(
410 wavefront_aligner_t* const wf_aligner,
411 wavefront_set_t* const wavefront_set,
412 const int score,
413 const int lo,
414 const int hi) {
415 // Parameters
416 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
417 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
418 wavefront_slab_t* const wavefront_slab = wf_aligner->wavefront_slab;
419 // Consider ends-free (M!=0)
420 int effective_lo, effective_hi;
421 if (wavefront_compute_endsfree_required(wf_aligner,score)) {
422 int endsfree_lo, endsfree_hi;
423 wavefront_compute_endsfree_limits(wf_aligner,score,&endsfree_lo,&endsfree_hi);
424 effective_lo = MIN(lo,endsfree_lo);
425 effective_hi = MAX(hi,endsfree_hi);
426 } else {
427 effective_lo = lo;
428 effective_hi = hi;
429 }
430 // Compute effective hi/lo dimensions
431 wavefront_compute_limits_output(
432 wf_aligner,effective_lo,effective_hi,
433 &effective_lo,&effective_hi);
434 // Resize null/victim wavefronts
435 wavefront_components_resize_null__victim(wf_components,effective_lo,effective_hi);
436 // Modular wavefront
437 int score_mod = score;
438 if (wf_components->memory_modular) {
439 score_mod = score % wf_components->max_score_scope;
440 wavefront_compute_free_output(wf_aligner,score_mod);
441 }
442 // Check
443 if (score_mod >= wf_components->num_wavefronts) {
444 fprintf(stderr,"[WFA::Compute] Maximum allocated wavefronts reached\n");
445 exit(1);
446 }
447 // Allocate M-Wavefront
448 wavefront_set->out_mwavefront = wavefront_slab_allocate(wavefront_slab,effective_lo,effective_hi);
449 wf_components->mwavefronts[score_mod] = wavefront_set->out_mwavefront;
450 wf_components->mwavefronts[score_mod]->lo = lo;
451 wf_components->mwavefronts[score_mod]->hi = hi;
452 if (distance_metric == gap_linear) return;
453 // Allocate I1-Wavefront
454 if (!wavefront_set->in_mwavefront_open1->null || !wavefront_set->in_i1wavefront_ext->null) {
455 wavefront_set->out_i1wavefront = wavefront_slab_allocate(wavefront_slab,effective_lo,effective_hi);
456 wf_components->i1wavefronts[score_mod] = wavefront_set->out_i1wavefront;
457 wf_components->i1wavefronts[score_mod]->lo = lo;
458 wf_components->i1wavefronts[score_mod]->hi = hi;
459 } else {
460 wavefront_set->out_i1wavefront = wf_components->wavefront_victim;
461 wf_components->i1wavefronts[score_mod] = NULL;
462 }
463 // Allocate D1-Wavefront
464 if (!wavefront_set->in_mwavefront_open1->null || !wavefront_set->in_d1wavefront_ext->null) {
465 wavefront_set->out_d1wavefront = wavefront_slab_allocate(wavefront_slab,effective_lo,effective_hi);
466 wf_components->d1wavefronts[score_mod] = wavefront_set->out_d1wavefront;
467 wf_components->d1wavefronts[score_mod]->lo = lo;
468 wf_components->d1wavefronts[score_mod]->hi = hi;
469 } else {
470 wavefront_set->out_d1wavefront = wf_components->wavefront_victim;
471 wf_components->d1wavefronts[score_mod] = NULL;
472 }
473 if (distance_metric == gap_affine) return;
474 // Allocate I2-Wavefront
475 if (!wavefront_set->in_mwavefront_open2->null || !wavefront_set->in_i2wavefront_ext->null) {
476 wavefront_set->out_i2wavefront = wavefront_slab_allocate(wavefront_slab,effective_lo,effective_hi);
477 wf_components->i2wavefronts[score_mod] = wavefront_set->out_i2wavefront;
478 wf_components->i2wavefronts[score_mod]->lo = lo;
479 wf_components->i2wavefronts[score_mod]->hi = hi;
480 } else {
481 wavefront_set->out_i2wavefront = wf_components->wavefront_victim;
482 wf_components->i2wavefronts[score_mod] = NULL;
483 }
484 // Allocate D2-Wavefront
485 if (!wavefront_set->in_mwavefront_open2->null || !wavefront_set->in_d2wavefront_ext->null) {
486 wavefront_set->out_d2wavefront = wavefront_slab_allocate(wavefront_slab,effective_lo,effective_hi);
487 wf_components->d2wavefronts[score_mod] = wavefront_set->out_d2wavefront;
488 wf_components->d2wavefronts[score_mod]->lo = lo;
489 wf_components->d2wavefronts[score_mod]->hi = hi;
490 } else {
491 wavefront_set->out_d2wavefront = wf_components->wavefront_victim;
492 wf_components->d2wavefronts[score_mod] = NULL;
493 }
494}
495/*
496 * Initialize wavefronts ends
497 */
498void wavefront_compute_init_ends_wf_lower(
499 wavefront_t* const wavefront,
500 const int min_lo) {
501 // Check initialization (lowest element)
502 if (wavefront->wf_elements_init_min <= min_lo) return;
503 // Initialize lower elements
504 wf_offset_t* const offsets = wavefront->offsets;
505 const int min_init = MIN(wavefront->wf_elements_init_min,wavefront->lo);
506 int k;
507 for (k=min_lo;k<min_init;++k) {
508 offsets[k] = WAVEFRONT_OFFSET_NULL;

576/*
577 * Trim wavefronts ends
578 */
579void wavefront_compute_trim_ends(
580 wavefront_aligner_t* const wf_aligner,
581 wavefront_t* const wavefront) {
582 // Parameters
583 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
584 const int pattern_length = sequences->pattern_length;
585 const int text_length = sequences->text_length;
586 wf_offset_t* const offsets = wavefront->offsets;
587 // Trim from hi
588 int k;
589 const int lo = wavefront->lo;
590 for (k=wavefront->hi;k>=lo;--k) {
591 // Fetch offset
592 const wf_offset_t offset = offsets[k];
593 // Check boundaries
594 const uint32_t h = WAVEFRONT_H(k,offset); // Make unsigned to avoid checking negative
595 const uint32_t v = WAVEFRONT_V(k,offset); // Make unsigned to avoid checking negative
596 if (h <= text_length && v <= pattern_length) break;
597 }
598 wavefront->hi = k; // Set new hi
599 wavefront->wf_elements_init_max = k;
600 // Trim from lo
601 const int hi = wavefront->hi;
602 for (k=wavefront->lo;k<=hi;++k) {
603 // Fetch offset
604 const wf_offset_t offset = offsets[k];
605 // Check boundaries
606 const uint32_t h = WAVEFRONT_H(k,offset); // Make unsigned to avoid checking negative
607 const uint32_t v = WAVEFRONT_V(k,offset); // Make unsigned to avoid checking negative
608 if (h <= text_length && v <= pattern_length) break;
609 }
610 wavefront->lo = k; // Set new lo
611 wavefront->wf_elements_init_min = k;
612 wavefront->null = (wavefront->lo > wavefront->hi);
613}
614void wavefront_compute_process_ends(
615 wavefront_aligner_t* const wf_aligner,
616 wavefront_set_t* const wavefront_set,
617 const int score) {
618 // Parameters
619 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
620 // Consider ends-free (M!=0)
621 if (wavefront_compute_endsfree_required(wf_aligner,score)) {
622 wavefront_compute_endsfree_init(wf_aligner,wavefront_set->out_mwavefront,score);
623 }
624 // Trim ends from non-null WFs
625 if (wavefront_set->out_mwavefront) wavefront_compute_trim_ends(wf_aligner,wavefront_set->out_mwavefront);
626 if (distance_metric == gap_linear) return;
627 if (wavefront_set->out_i1wavefront) wavefront_compute_trim_ends(wf_aligner,wavefront_set->out_i1wavefront);
628 if (wavefront_set->out_d1wavefront) wavefront_compute_trim_ends(wf_aligner,wavefront_set->out_d1wavefront);
629 if (distance_metric == gap_affine) return;
630 if (wavefront_set->out_i2wavefront) wavefront_compute_trim_ends(wf_aligner,wavefront_set->out_i2wavefront);
631 if (wavefront_set->out_d2wavefront) wavefront_compute_trim_ends(wf_aligner,wavefront_set->out_d2wavefront);
632}
