48/*
49 * Alignment scope
50 */
51typedef enum {
52 compute_score, // Only distance/score
53 compute_alignment, // Full alignment CIGAR
54} alignment_scope_t;
55typedef enum {
56 alignment_end2end, // End-to-end alignment (aka global)
57 alignment_endsfree, // Ends-free alignment (semiglobal, glocal, etc)
58} alignment_span_t;
59typedef struct {
60 // Mode
61 alignment_span_t span; // Alignment form (End-to-end/Ends-free)
62 // Extension
63 bool extension; // Activate extension-like alignment
64 // Ends-free
65 int pattern_begin_free; // Allow free-gap at the beginning of the pattern
66 int pattern_end_free; // Allow free-gap at the end of the pattern
67 int text_begin_free; // Allow free-gap at the beginning of the text
68 int text_end_free; // Allow free-gap at the end of the text
69} alignment_form_t;
70
71/*
72 * Alignment system configuration
73 */
74typedef struct {
75 // Limits
76 int max_alignment_steps; // Maximum WFA-steps allowed before quit
77 // Probing intervals
78 int probe_interval_global; // Score-ticks interval to check any limits
79 int probe_interval_compact; // Score-ticks interval to check BT-buffer compacting
80 // Memory
81 uint64_t max_partial_compacts; // Maximum partial-compacts before attempting full-compact
82 uint64_t max_memory_compact; // Maximum BT-buffer memory allowed before trigger compact
83 uint64_t max_memory_resident; // Maximum memory allowed to be buffered before reap
84 uint64_t max_memory_abort; // Maximum memory allowed to be used before aborting alignment
85 // Verbose
86 // 0 - Quiet
87 // 1 - Report each sequence aligned (brief)
88 // 2 - Report each sequence/subsequence aligned (brief)
89 // 3 - Report WFA progress (heavy tasks) (verbose)
90 // 4 - Full report of each sequence/subsequence aligned (very verbose)
91 int verbose; // Verbose (regulates messages during alignment)
92 // Debug
93 bool check_alignment_correct; // Verify that the alignment CIGAR output is correct
94 // Profile
95 profiler_timer_t timer; // Time alignment
96 // OS
97 int max_num_threads; // Maximum number of threads to use to compute/extend WFs
98 int min_offsets_per_thread; // Minimum amount of offsets to spawn a thread
99} alignment_system_t;
100
101/*
102 * Low-memory modes
103 */
104typedef enum {
105 wavefront_memory_high = 0, // High-memore mode (fastest, stores all WFs explicitly)
106 wavefront_memory_med = 1, // Succing-memory mode piggyback-based (medium, offloads half-full BT-blocks)
107 wavefront_memory_low = 2, // Succing-memory mode piggyback-based (slow, offloads only full BT-blocks)
108 wavefront_memory_ultralow = 3, // Bidirectional WFA
109} wavefront_memory_t;
110
111/*
112 * Wavefront Aligner Attributes
113 */
114typedef struct {
115 // Distance model
116 distance_metric_t distance_metric; // Alignment metric/distance used
117 alignment_scope_t alignment_scope; // Alignment scope (score only or full-CIGAR)
118 alignment_form_t alignment_form; // Alignment mode (end-to-end/ends-free)
119 // Penalties
120 linear_penalties_t linear_penalties; // Gap-linear penalties (placeholder)
121 affine_penalties_t affine_penalties; // Gap-affine penalties (placeholder)
122 affine2p_penalties_t affine2p_penalties; // Gap-affine-2p penalties (placeholder)
123 // Heuristic strategy
124 wavefront_heuristic_t heuristic; // Wavefront heuristic
125 // Memory model
126 wavefront_memory_t memory_mode; // Wavefront memory strategy (modular wavefronts and piggyback)
127 // External MM (instead of allocating one inside)
128 mm_allocator_t* mm_allocator; // MM-Allocator
129 // Display
130 wavefront_plot_attr_t plot; // Plot wavefront
131 // System
132 alignment_system_t system; // System related parameters
133} wavefront_aligner_attr_t;
