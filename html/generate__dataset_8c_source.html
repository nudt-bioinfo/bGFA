77/*
78 * Random number generator [min,max)
79 */
80uint64_t rand_iid(
81 const uint64_t min,
82 const uint64_t max) {
83 const int n_rand = rand(); // [0, RAND_MAX]
84 const uint64_t range = max - min;
85 const uint64_t rem = RAND_MAX % range;
86 const uint64_t sample = RAND_MAX / range;
87 // Consider the small interval within remainder of RAND_MAX
88 if (n_rand < RAND_MAX - rem) {
89 return min + n_rand/sample;
90 } else {
91 return rand_iid(min,max);
92 }
93}

97void sequence_generate_random(
98 char* const sequence,
99 const int length) {
100 // Generate random characters
101 int i;
102 for (i=0;i<length;++i) {
103 sequence[i] = alphabet[rand_iid(0,ALPHABET_SIZE)];
104 }
105 sequence[length] = '\0';
106}
107/*
108 * Extract sequence
109 */
110int sequence_extract(
111 char* const seqlong,
112 const int seqlong_length,
113 char* const seqshort,
114 const int seqshort_length) {
115 const int length_diff = seqlong_length - seqshort_length;
116 const int offset = rand_iid(0,length_diff+1);
117 strncpy(seqshort,seqlong+offset,seqshort_length);
118 // Return offset
119 return offset;
120}

141/*
142 * Sequence errors generation
143 */
144void sequence_generate_mismatch(
145 char* const sequence,
146 const int sequence_length,
147 seq_error_t* const errors_log) {
148 // Generate random mismatch
149 char character;
150 int position;
151 do {
152 position = rand_iid(0,sequence_length);
153 character = alphabet[rand_iid(0,ALPHABET_SIZE)];
154 } while (sequence[position] == character);
155 // Log
156 errors_log->operation = 'M';
157 errors_log->position = position;
158 errors_log->base_char = sequence[position];
159 errors_log->replacement_char = character;
160 // Set mismatch
161 sequence[position] = character;
162}
163void sequence_generate_deletion(
164 char* const sequence,
165 int* const sequence_length,
166 seq_error_t* const errors_log) {
167 // Generate random deletion
168 const int position = rand_iid(0,*sequence_length);
169 const int new_sequence_length = *sequence_length - 1;
170 int i;
171 for (i=position;i<new_sequence_length;++i) {
172 sequence[i] = sequence[i+1];
173 }
174 *sequence_length = new_sequence_length;
175 // Log
176 errors_log->operation = 'D';
177 errors_log->position = position;
178}
179void sequence_generate_insertion(
180 char* const sequence,
181 int* const sequence_length,
182 seq_error_t* const errors_log) {
183 // Generate random insertion
184 const int position = rand_iid(0,*sequence_length);
185 const int new_sequence_length = *sequence_length + 1;
186 int i;
187 for (i=new_sequence_length-1;i>position;--i) {
188 sequence[i] = sequence[i-1];
189 }
190 *sequence_length = new_sequence_length;
191 // Insert random character
192 sequence[position] = alphabet[rand_iid(0,ALPHABET_SIZE)];
193 // Log
194 errors_log->operation = 'I';
195 errors_log->position = position;
196}
197int sequence_generate_errors(
198 char* const sequence,
199 const int sequence_length,
200 const int num_errors,
201 seq_error_t* const errors_log) {
202 int length = sequence_length;
203 // Generate random errors
204 int i;
205 for (i=0;i<num_errors;++i) {
206 int error_type = rand_iid(0,3);
207 switch (error_type) {
208 case 0: sequence_generate_mismatch(sequence,length,errors_log+i); break;
209 case 1: sequence_generate_deletion(sequence,&length,errors_log+i); break;
210 default: sequence_generate_insertion(sequence,&length,errors_log+i); break;
211 }
212 }
213 // Close sequence and return length
214 sequence[length] = '\0';
215 errors_log[num_errors].operation = '\0';
216 return length;
217}
218/*
219 * Large-Indel generation
220 */
221void sequence_generate_indel(
222 char* const sequence,
223 int* const sequence_length,
224 const int deletion_length,
225 seq_error_t* const errors_log) {
226 // Generate random deletion
227 const int position = rand_iid(0,*sequence_length);
228 if (deletion_length >= *sequence_length) return;
229 const int new_sequence_length = *sequence_length - deletion_length;
230 int i;
231 for (i=position;i<new_sequence_length;++i) {
232 sequence[i] = sequence[i+deletion_length];
233 }
234 *sequence_length = new_sequence_length;
235 // Log
236 errors_log->operation = 'N';
237 errors_log->position = position;
238}
239int sequence_generate_indels(
240 char* const sequence,
241 const int sequence_length,
242 const int max_indels,
243 const int deletion_length,
244 seq_error_t* const errors_log) {
245 int length = sequence_length;
246 // Generate random indels
247 int i;
248 const int indels = rand_iid(0,max_indels+1);
249 for (i=0;i<indels;++i) {
250 sequence_generate_indel(sequence,&length,deletion_length,errors_log+i);
251 }
252 // Close sequence and return length
253 sequence[length] = '\0';
254 errors_log[indels].operation = '\0';
255 return length;
256}

297void parse_arguments(int argc,char** argv) {
298 struct option long_options[] = {
299 { "num-patterns", required_argument, 0, 'n' },
300 { "output", required_argument, 0, 'o' },
301 { "length", required_argument, 0, 'l' },
302 { "length-diff", required_argument, 0, 1000 },
303 { "error", required_argument, 0, 'e' },
304 { "indels", required_argument, 0, 1001 },
305 { "debug", no_argument, 0, 'g' },
306 { "help", no_argument, 0, 'h' },
307 { 0, 0, 0, 0 } };
308 int c,option_index;
309 if (argc <= 1) {
310 usage();
311 exit(0);
312 }
313 while (1) {
314 c=getopt_long(argc,argv,"n:o:l:e:gh",long_options,&option_index);
315 if (c==-1) break;
316 switch (c) {
317 case 'n':
318 parameters.num_reads = atoi(optarg);
319 break;
320 case 'o':
321 parameters.output = optarg;
322 break;
323 case 'l':
324 parameters.length = atoi(optarg);
325 break;
326 case 1000: // --length-diff
327 parameters.length_diff = atof(optarg);
328 break;
329 case 'e':
330 parameters.error_degree = atof(optarg);
331 break;
332 case 1001: { // --indels
333 char* sentinel = strtok(optarg,",");
334 parameters.indels_num = atof(sentinel);
335 sentinel = strtok(NULL,",");
336 parameters.indels_length = atof(sentinel);
337 break;
338 }
339 case 'g':
340 parameters.debug = 1;
341 break;
342 case 'h':
343 usage();
344 exit(1);
345 case '?': default:
346 fprintf(stderr, "Option not recognized \n"); exit(1);
347 }
348 }

350int main(int argc,char* argv[]) {
351 // Parsing command-line options
352 parse_arguments(argc,argv);
353 // Open file
354 FILE* const output_file = (parameters.output==NULL) ? stdout : fopen(parameters.output,"w");
355 // Allocate sequences
356 const int pattern_length = parameters.length;
357 const int text_length = ceil((float)pattern_length * parameters.length_diff);
358 const int seqlong_length = MAX(pattern_length,text_length); // Long sequence
359 const int seqshort_length = MIN(pattern_length,text_length); // Short sequence
360 const int num_errors = (parameters.error_degree >= 1.0) ?
361 (int) parameters.error_degree : ceil((float)pattern_length * parameters.error_degree);
362 char* const seqlong = malloc(seqlong_length+1);
363 char* const seqshort = malloc(seqshort_length+1+num_errors+parameters.indels_num*parameters.indels_length);
364 seq_error_t* const errors_log = malloc((num_errors+parameters.indels_num+1)*sizeof(seq_error_t));
365 // Read-align loop
366 srand(time(0));
367 int i;
368 for (i=0;i<parameters.num_reads;++i) {
369 // Generate random sequence
370 sequence_generate_random(seqlong,seqlong_length);
371 // Extract short sequence from long
372 const int offset = sequence_extract(seqlong,seqlong_length,seqshort,seqshort_length);
373 // Generate errors
374 if (num_errors > 0) sequence_generate_errors(seqshort,seqshort_length,num_errors,errors_log);
375 // Generate indels
376 if (parameters.indels_num > 0) {
377 sequence_generate_indels(seqshort,seqshort_length,
378 parameters.indels_num,parameters.indels_length,errors_log+num_errors);
379 }
380 // DEBUG
381 if (parameters.debug) {
382 fprintf(output_file,"#DEBUG offset=%d errors=",offset);
383 sequence_errors_print(output_file,errors_log);
384 fprintf(output_file,"\n");
385 }
386 // Print
387 if (pattern_length <= text_length) {
388 fprintf(output_file,">%s\n",seqshort);
389 fprintf(output_file,"<%s\n",seqlong);
390 } else {
391 fprintf(output_file,"<%s\n",seqlong);
392 fprintf(output_file,">%s\n",seqshort);
393 }
394 }
395 // Close files & free
396 if (parameters.output!=NULL) fclose(output_file);
397 free(seqlong);
398 free(seqshort);
399 free(errors_log);
400}
