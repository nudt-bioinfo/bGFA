44void updateLink(GFA &merged_graph,
45 const std::vector<GFA> &graphs,
46 const NewIdMap &new_id_list)
47{
48 // 0 is the reference graph
49 uword_t graph_id = 1;
50
51 for (const auto &gfa : graphs)
52 {
53 const auto &all_links = gfa.getLinksSet().getAllLinks();
54 for (uword_t from_id = 0; from_id < all_links.size(); from_id++)
55 {
56 const auto &data = all_links[from_id];
57 for (const auto &to_value : data)
58 {
59
60 uword_t graph_from_id =
61 ((graph_id << (WORD_BIT - MERGE_PLACEHOLDER)) | from_id);
62
63 uword_t graph_to_id =
64 ((graph_id << (WORD_BIT - MERGE_PLACEHOLDER)) |
65 (to_value >> LINK_DIR_BIT));
66
67 auto from_it = new_id_list.find(graph_from_id);
68 auto to_it = new_id_list.find(graph_to_id);
69
70 if (from_it != new_id_list.end() && to_it != new_id_list.end())
71 {
72 // From last id -> To first id
73 merged_graph.addLink(
74 from_it->second.second,
75 (to_it->second.first << LINK_DIR_BIT | (to_value & 0x03)));
76 }
77 }
78 }
79
80 graph_id++;
81 }
82 return;
83}

119uword_t getSimilarScore(MinimizerSet &seg_ms, MinimizerSet &ref_ms)
120{
121 MinimizerSetVect seg_ms_vect = getMinimizerVect(seg_ms);
122 MinimizerSetVect ref_ms_vect = getMinimizerVect(ref_ms);
123
124 size_t m = seg_ms_vect.size();
125 size_t n = ref_ms_vect.size();
126 if (m == 0 || n == 0)
127 return 0;
128
129 // TODO: 硬编码，后续换成参数传入
130 const int MATCH = 1; // hash match +1
131 const int ORDER_BONUS = 1; // additional +1 if previous match preserves order
132 const int MISMATCH = -1; // mismatch -1
133 const int GAP = -1; // gap penalty -1
134
135 // Use rolling 1D DP (two rows) for Smith-Waterman local alignment to save memory.
136 std::vector<int> prev(n + 1, 0);
137 std::vector<int> curr(n + 1, 0);
138 int max_score = 0;
139
140 for (size_t i = 1; i <= m; ++i)
141 {
142 curr[0] = 0;
143 for (size_t j = 1; j <= n; ++j)
144 {
145 int diag_score;
146 // compare hashes (second element of pair)
147 if (seg_ms_vect[i - 1].second == ref_ms_vect[j - 1].second)
148 {
149 diag_score = prev[j - 1] + MATCH;
150 // if previous items exist and offsets are increasing in both sequences,
151 // treat this as preserving order and give an extra bonus
152 if (i > 1 && j > 1 &&
153 seg_ms_vect[i - 2].first < seg_ms_vect[i - 1].first &&
154 ref_ms_vect[j - 2].first < ref_ms_vect[j - 1].first)
155 {
156 diag_score += ORDER_BONUS;
157 }
158 }
159 else
160 {
161 diag_score = prev[j - 1] + MISMATCH;
162 }
163
164 int up_score = prev[j] + GAP;
165 int left_score = curr[j - 1] + GAP;
166
167 int cell = std::max(0, std::max(diag_score, std::max(up_score, left_score)));
168 curr[j] = cell;
169 if (cell > max_score)
170 max_score = cell;
171 }
172
173 // move current row to previous, and reset current
174 std::swap(prev, curr);
175 std::fill(curr.begin(), curr.end(), 0);
176 }
177
178 return (uword_t)max_score;
179}

181uword_t getMergeTargetbyMinimizers(GFA &merged_graph,
182 MinimizerSet &ref_ms, MinimizerSet &seg_ms)
183{
184 uword_t target_seg_id = uword_t(-1);
185 // id -> (current_score, last_seg_offset)
186 std::unordered_map<uword_t, uword_t> seg_minimizer_hit;
187
188 // hit count for each minimizer in the segment
189 for (const auto &seg_minimizer : seg_ms.getMinimizerTable())
190 {
191 uint64_t minimizer_hash = seg_minimizer.first;
192
193 // Check if this minimizer exists in the reference minimizer set
194 auto ref_it = ref_ms.getMinimizerTable().find(minimizer_hash);
195
196 // Calculate hits
197 if (ref_it != ref_ms.getMinimizerTable().end())
198 {
199 for (const auto &pos : ref_it->second.getPositions())
200 {
201 uword_t ref_seg_id = getPosID(pos);
202 uword_t ref_seg_offset = getPosOffset(pos);
203
204 auto score_it = seg_minimizer_hit.find(ref_seg_id);
205 if (score_it != seg_minimizer_hit.end())
206 {
207 seg_minimizer_hit[ref_seg_id] += 1;
208 }
209 else
210 {
211 seg_minimizer_hit[ref_seg_id] = 1;
212 }
213 }
214 }
215 }
216
217 if (seg_minimizer_hit.empty())
218 {
219 return target_seg_id;
220 }
221
222 // hit order
223 // TODO: 阈值不应该是硬编码，后续需要调整
224 uword_t min_hit_count = seg_ms.getMinimizerTable().size() * 0.7;
225 uword_t max_score = 0;
226 for (const auto &hit_pair : seg_minimizer_hit)
227 {
228 if (hit_pair.second < min_hit_count)
229 {
230 continue;
231 }
232
233 target_seg_id = hit_pair.first;
234 Segment target_seg = merged_graph.getSegmentByIndex(hit_pair.first);
235 MinimizerSet target_seg_ms = MinimizerSet(
236 target_seg,
237 ref_ms.getLength(),
238 ref_ms.getWindowSize(),
239 ref_ms.getMode());
240 if (max_score < getSimilarScore(seg_ms, target_seg_ms))
241 {
242 max_score = getSimilarScore(seg_ms, target_seg_ms);
243 target_seg_id = hit_pair.first;
244 }
245 }
246 return target_seg_id;
247}

262void mergeExactCompare(GFA &merged_graph, NewIdMap &new_id_list, uword_t g_id,
263 const Segment &segment, const std::string &target_seq,
264 const uword_t target_id, bool &is_merged,
265 uword_t merge_limited, uint8_t alignment_kernel)
266{
267 const std::string segment_seq = segment.getSequenceAsString();
268 const uword_t segment_id = segment.getId();
269
270 // Skip if length difference exceeds threshold
271 double length_diff = std::abs((int)segment_seq.size() - (int)target_seq.size());
272 length_diff /= std::max(segment_seq.size(), target_seq.size());
273 if (length_diff > (100 - merge_limited) / 100.0)
274 return;
275
276#ifdef INFO_DEBUG
277 std::cerr << "[mergeExactCompare] segment_id=" << segment_id
278 << " target_id=" << target_id
279 << " seg_len=" << segment_seq.size()
280 << " target_len=" << target_seq.size()
281 << " seg_seq=" << segment_seq
282 << " target_seq=" << target_seq
283 << " length_diff=" << length_diff << "\n";
284#endif
285
286 std::string cigar_str = alignmentSequence(segment_seq, target_seq);
287 if (cigar_str.empty() || cigar_str == "0M")
288 return;
289
290 uint64_t m_number = 0;
291 uint64_t cigar_number = 0;
292 uint64_t total_ops = 0;
293 {
294 size_t p = 0, n = cigar_str.size();
295 const char *s1 = segment_seq.data();
296 const char *s2 = target_seq.data();
297 size_t pos1_tmp = 0, pos2_tmp = 0;
298 while (p < n)
299 {
300 uint64_t num = 0;
301 while (p < n && std::isdigit(static_cast<unsigned char>(cigar_str[p])))
302 {
303 num = num * 10 + (cigar_str[p] - '0');
304 ++p;
305 }
306 if (p >= n)
307 break; // malformed
308 char op = cigar_str[p++];
309 ++total_ops;
310 cigar_number += num;
311
312 if (op == 'M')
313 {
314
315 m_number += num;
316 pos1_tmp += num;
317 pos2_tmp += num;
318 }
319 else if (op == 'I')
320 {
321 pos2_tmp += num;
322 }
323 else if (op == 'D')
324 {
325 pos1_tmp += num;
326 }
327 else if (op == 'X')
328 {
329 pos1_tmp += num;
330 pos2_tmp += num;
331 }
332 }
333 }
334
335 if (cigar_number == 0)
336 return;
337
338#ifdef INFO_DEBUG
339 std::cerr << "\t\t cigar=" << cigar_str
340 << " cigar_len=" << cigar_number
341 << " m_number=" << m_number
342 << " total_ops=" << total_ops
343 << " merge_limited=" << merge_limited << "\n";
344#endif
345
346 if (m_number * 100 < merge_limited * cigar_number)
347 return;
348 if (total_ops > (static_cast<uint64_t>(1) << EXACT_MERGE_PLACEHOLDER) - 1)
349 return;
350
351 // Default: not merged
352 is_merged = false;
353
354 uword_t first_new_target = -1, last_new_target = -1;
355 uword_t first_new_segment = -1, last_new_segment = -1;
356
357 size_t pos1 = 0, pos2 = 0;
358 bool is_first_seg1 = true, is_first_seg2 = true;
359 uword_t seq1_cur_id = uword_t(-1), seq2_cur_id = uword_t(-1);
360
361 {
362 size_t p = 0, n = cigar_str.size();
363 uword_t op_idx = 0;
364 const char *s1 = segment_seq.data();
365 const char *s2 = target_seq.data();
366
367 while (p < n)
368 {
369 uint64_t num = 0;
370 while (p < n && std::isdigit(static_cast<unsigned char>(cigar_str[p])))
371 {
372 num = num * 10 + (cigar_str[p] - '0');
373 ++p;
374 }
375 if (p >= n)
376 break;
377 char op = cigar_str[p++];
378
379 uword_t id_tag = static_cast<uword_t>(op_idx) << (WORD_BIT - EXACT_MERGE_PLACEHOLDER);
380 ++op_idx;
381
382 if (op == 'M')
383 {
384 pos1 += num;
385 pos2 += num;
386 const size_t start1 = pos1 - num;
387 const size_t start2 = pos2 - num;
388
389 // create (or reuse) merged segments for the matched region
390 if (num > 0)
391 {
392 std::string piece1 = segment_seq.substr(start1, num);
393 std::string piece2 = target_seq.substr(start2, num);
394
395 if (piece1 == piece2)
396 {
397 Segment tmp(0, piece1);
398 uword_t new_idx = merged_graph.addSegment(tmp);
399 if (is_first_seg1)
400 {
401 first_new_segment = new_idx;
402 is_first_seg1 = false;
403 }
404 last_new_segment = new_idx;
405
406 if (is_first_seg2)
407 {
408 first_new_target = new_idx;
409 is_first_seg2 = false;
410 }
411 last_new_target = new_idx;
412
413 if (seq1_cur_id != uword_t(-1))
414 {
415 merged_graph.addLink(seq1_cur_id, (new_idx << LINK_DIR_BIT) | 0x03);
416 }
417 seq1_cur_id = new_idx;
418
419 if (seq2_cur_id != uword_t(-1))
420 {
421 merged_graph.addLink(seq2_cur_id, (new_idx << LINK_DIR_BIT) | 0x03);
422 }
423 seq2_cur_id = new_idx;
424 }
425 else
426 {
427 // different sequences in M -> create two nodes
428 Segment tmp1(0, piece1);
429 Segment tmp2(0, piece2);
430 uword_t new_idx1 = merged_graph.addSegment(tmp1);
431 uword_t new_idx2 = merged_graph.addSegment(tmp2);
432
433 if (is_first_seg1)
434 {
435 first_new_segment = new_idx1;
436 is_first_seg1 = false;
437 }
438 last_new_segment = new_idx1;
439
440 if (is_first_seg2)
441 {
442 first_new_target = new_idx2;
443 is_first_seg2 = false;
444 }
445 last_new_target = new_idx2;
446
447 if (seq1_cur_id != uword_t(-1))
448 {
449 merged_graph.addLink(seq1_cur_id, (new_idx1 << LINK_DIR_BIT) | 0x03);
450 }
451 seq1_cur_id = new_idx1;
452
453 if (seq2_cur_id != uword_t(-1))
454 {
455 merged_graph.addLink(seq2_cur_id, (new_idx2 << LINK_DIR_BIT) | 0x03);
456 }
457 seq2_cur_id = new_idx2;
458 }
459 }
460 }
461 else if (op == 'X')
462 {
463 pos1 += num;
464 pos2 += num;
465 const size_t start1 = pos1 - num;
466 const size_t start2 = pos2 - num;
467
468 // Create separate nodes for mismatched region on both sides
469 if (num > 0)
470 {
471 std::string piece1 = segment_seq.substr(start1, num);
472 std::string piece2 = target_seq.substr(start2, num);
473
474 Segment tmp1(0, piece1);
475 Segment tmp2(0, piece2);
476 uword_t new_idx1 = merged_graph.addSegment(tmp1);
477 uword_t new_idx2 = merged_graph.addSegment(tmp2);
478
479 if (is_first_seg1)
480 {
481 first_new_segment = new_idx1;
482 is_first_seg1 = false;
483 }
484 last_new_segment = new_idx1;
485 if (seq1_cur_id != uword_t(-1))
486 {
487 merged_graph.addLink(seq1_cur_id, (new_idx1 << LINK_DIR_BIT) | 0x03);
488 }
489 seq1_cur_id = new_idx1;
490
491 if (is_first_seg2)
492 {
493 first_new_target = new_idx2;
494 is_first_seg2 = false;
495 }
496 last_new_target = new_idx2;
497 if (seq2_cur_id != uword_t(-1))
498 {
499 merged_graph.addLink(seq2_cur_id, (new_idx2 << LINK_DIR_BIT) | 0x03);
500 }
501 seq2_cur_id = new_idx2;
502 }
503 }
504 else if (op == 'I')
505 {
506 pos2 += num;
507 const size_t start2 = pos2 - num;
508 if (num > 0)
509 {
510 std::string piece2 = target_seq.substr(start2, num);
511 Segment tmp2(0, piece2);
512 uword_t new_idx2 = merged_graph.addSegment(tmp2);
513 if (is_first_seg2)
514 {
515 first_new_target = new_idx2;
516 is_first_seg2 = false;
517 }
518 last_new_target = new_idx2;
519 if (seq2_cur_id != uword_t(-1))
520 {
521 merged_graph.addLink(seq2_cur_id, (new_idx2 << LINK_DIR_BIT) | 0x03);
522 }
523 seq2_cur_id = new_idx2;
524 }
525 }
526 else if (op == 'D')
527 {
528 pos1 += num;
529 const size_t start1 = pos1 - num;
530 if (num > 0)
531 {
532 std::string piece1 = segment_seq.substr(start1, num);
533 Segment tmp1(0, piece1);
534 uword_t new_idx1 = merged_graph.addSegment(tmp1);
535 if (is_first_seg1)
536 {
537 first_new_segment = new_idx1;
538 is_first_seg1 = false;
539 }
540 last_new_segment = new_idx1;
541 if (seq1_cur_id != uword_t(-1))
542 {
543 merged_graph.addLink(seq1_cur_id, (new_idx1 << LINK_DIR_BIT) | 0x03);
544 }
545 seq1_cur_id = new_idx1;
546 }
547 }
548 else
549 {
550 continue;
551 }
552 }
553 }
554
555 // Only consider "merged" if new nodes are actually created on the source side, and write to the mapping
556 if (first_new_segment != uword_t(-1) && last_new_segment != uword_t(-1))
557 {
558 is_merged = true;
559
560 // Redirect links for the Target segment（仅当目标侧也产生了首尾时）
561 if (first_new_target != uword_t(-1) && last_new_target != uword_t(-1))
562 {
563 // Redirect all incoming edges X -> target to X -> first_new_target
564 merged_graph.redirectLinksTo(target_id, first_new_target);
565
566 // Move all outgoing edges target -> Y to last_new_target -> Y
567 merged_graph.redirectLinksFrom(target_id, last_new_target);
568
569 // Remove any remaining edges of target (if any) and mark segment deleted
570 merged_graph.removeLinksOf(target_id);
571 merged_graph.markSegmentDeleted(target_id);
572 }
573
574 // Set Information in new_id_list
575 new_id_list[(g_id << (WORD_BIT - MERGE_PLACEHOLDER)) | segment_id] =
576 std::make_pair(first_new_segment, last_new_segment);
577 }
578}

590void mergeGFAStrict(GFA &merged_graph, const std::vector<GFA> &graphs)
591{
592 uword_t graph_id = 0;
593 NodeMap segment_set;
594 NewIdMap new_id_list;
595
596 updateSegmentSet(segment_set, merged_graph, 0);
597 graph_id++;
598
599 for (const auto &gfa : graphs)
600 {
601 for (const auto &segment : gfa.getAllSegments())
602 {
603 std::string segment_seq = segment.getSequenceAsString();
604
605 if (segment_set.find(segment_seq) == segment_set.end())
606 {
607 segment_set[segment_seq] = merged_graph.addSegment(segment);
608 }
609 else
610 {
611 new_id_list[(graph_id << (WORD_BIT - MERGE_PLACEHOLDER)) |
612 segment.getId()] =
613 std::make_pair(segment_set[segment_seq],
614 segment_set[segment_seq]);
615 }
616 }
617 graph_id++;
618 }
619
620 updateLink(merged_graph, graphs, new_id_list);
621}

680#ifdef INFO_DEBUG

682 << "[mergeGFAExact] New segment_id="
683 << segment_id << " seq=" << segment_seq << "\n";
684#endif
685 }
686 }
687 else
688 {
689 new_id_list[(graph_id << (WORD_BIT - MERGE_PLACEHOLDER)) |
690 segment.getId()] =
691 std::make_pair(segment_set[segment_seq],

694 }

696 }

698 updateLink(merged_graph, graphs, new_id_list);
699}
