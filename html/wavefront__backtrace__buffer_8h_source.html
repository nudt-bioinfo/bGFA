63/*
64 * Backtrace Buffer
65 */
66typedef struct {
67 // Locator
68 int segment_idx; // Current segment idx
69 int segment_offset; // Current free position within segment
70 bt_block_t* block_next; // Next BT-block free
71 // Buffers
72 vector_t* segments; // Memory segments (bt_block_t*)
73 vector_t* alignment_init_pos; // Buffer to store alignment's initial coordinates (h,v) (wf_backtrace_init_pos_t)
74 bt_block_idx_t num_compacted_blocks; // Total compacted blocks in BT-buffer compacted (dense from 0..num_compacted_blocks-1)
75 int num_compactions; // Total compactions performed
76 // Internal buffers
77 vector_t* alignment_packed; // Temporal buffer to store final alignment (pcigar_t)
78 vector_t* prefetch_blocks_idxs; // Temporal buffer to store blocks_idxs (bt_block_idx_t)
79 // MM
80 mm_allocator_t* mm_allocator;
81} wf_backtrace_buffer_t;
