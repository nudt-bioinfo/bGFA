44heatmap_t* heatmap_new(
45 const heatmap_type type,
46 const int min_v,
47 const int max_v,
48 const int min_h,
49 const int max_h,
50 const int resolution_points) {
51 // Alloc
52 heatmap_t* const heatmap = (heatmap_t*)malloc(sizeof(heatmap_t));
53 // Configure
54 heatmap->type = type;
55 heatmap->min_v = min_v;
56 heatmap->max_v = max_v;
57 heatmap->min_h = min_h;
58 heatmap->max_h = max_h;
59 // Binning (mantain aspect ratio)
60 const int v_range = (max_v-min_v+1);
61 const int h_range = (max_h-min_h+1);
62 const int max_range = MAX(v_range,h_range);
63 if (max_range <= resolution_points) {
64 heatmap->binning_factor = 1.0f;
65 heatmap->num_rows = v_range;
66 heatmap->num_columns = h_range;
67 } else {
68 heatmap->binning_factor = (float)max_range/(float)resolution_points;
69 heatmap->num_rows = (float)v_range/heatmap->binning_factor;
70 heatmap->num_columns = (float)h_range/heatmap->binning_factor;
71 }
72 // Allocate matrix
73 heatmap->values = (int**)malloc(heatmap->num_rows*sizeof(int*));
74 int i;
75 for (i=0;i<heatmap->num_rows;++i) {
76 heatmap->values[i] = (int*)malloc(heatmap->num_columns*sizeof(int)); // Allocate row
77 }
78 // Clear
79 heatmap_clear(heatmap);
80 // Return
81 return heatmap;
82}
83void heatmap_clear(
84 heatmap_t* const heatmap) {
85 // Parameters
86 const heatmap_type type = heatmap->type;
87 const int num_rows = heatmap->num_rows;
88 const int num_columns = heatmap->num_columns;
89 // Clear values
90 int i, j;
91 for (i=0;i<num_rows;++i) {
92 switch (type) {
93 case heatmap_min: {
94 for (j=0;j<num_columns;++j) heatmap->values[i][j] = HEATMAP_INT_MAX;
95 break;
96 }
97 case heatmap_max:
98 case heatmap_value:
99 default: {
100 for (j=0;j<num_columns;++j) heatmap->values[i][j] = HEATMAP_INT_MIN;
101 break;
102 }
103 }
104 }
105}

118void heatmap_set(
119 heatmap_t* const heatmap,
120 const int v,
121 const int h,
122 const int value) {
123 // Paramters
124 const int num_rows = heatmap->num_rows;
125 const int num_columns = heatmap->num_columns;
126 // Check position
127 if (v<heatmap->min_v || v>heatmap->max_v) return;
128 if (h<heatmap->min_h || h>heatmap->max_h) return;
129 // Adjust position into binning
130 int v_adjusted = (float)(v - heatmap->min_v) / heatmap->binning_factor;
131 int h_adjusted = (float)(h - heatmap->min_h) / heatmap->binning_factor;
132 if (v_adjusted >= num_rows) v_adjusted = num_rows-1;
133 if (h_adjusted >= num_columns) h_adjusted = num_columns-1;
134 // Set value
135 switch (heatmap->type) {
136 case heatmap_min:
137 heatmap->values[v_adjusted][h_adjusted] = MIN(heatmap->values[v_adjusted][h_adjusted],value);
138 break;
139 case heatmap_max:
140 heatmap->values[v_adjusted][h_adjusted] = MAX(heatmap->values[v_adjusted][h_adjusted],value);
141 break;
142 case heatmap_value:
143 default:
144 heatmap->values[v_adjusted][h_adjusted] = value;
145 break;
146 }
147}

151void heatmap_print(
152 FILE* const stream,
153 heatmap_t* const heatmap) {
154 // Parameters
155 const int num_rows = heatmap->num_rows;
156 const int num_columns = heatmap->num_columns;
157 // Print values
158 int v, h;
159 for (v=0;v<num_rows;++v) {
160 for (h=0;h<num_columns;++h) {
161 if (h>0) fprintf(stream,"%c",HEATMAP_SEPARATOR);
162 const int value = heatmap->values[v][h];
163 if (value == HEATMAP_INT_MIN || value == HEATMAP_INT_MAX) {
164 fprintf(stream,"-1");
165 } else {
166 fprintf(stream,"%d",value);
167 }
168 }
169 fprintf(stream,"\n");
170 }
