24def random_dna(length: int, rng: random.Random) -> str:
25 """Generate a random DNA string of 90~110% length using provided RNG."""
26 # Choose an actual length uniformly in [0.9*length, 1.1*length]
27 min_len = max(1, int(length * 0.9))
28 max_len = max(min_len, int(length * 1.1))
29 real_len = rng.randint(min_len, max_len)
30 return "".join(rng.choice(DNA_ALPHABET) for _ in range(real_len))

33def make_edges(
34 num_nodes: int,
35 complexity: int,
36 loop_prob: float,
37 rng: random.Random,
38) -> List[Tuple[int, int, str, str]]:
39 """Construct edges for a test graph.
40
41 Returns a list of edges in the form:
42 (from_id, to_id, from_orient, to_orient), with orientations in '+' or '-'.
43 """
44
45 edges: List[Tuple[int, int, str, str]] = []
46
47 # Base linear backbone
48 for node_id in range(1, num_nodes):
49 edges.append((node_id, node_id + 1, "+", "+"))
50
51 # Add branches and loops per complexity
52 extra_edges = 0
53 if complexity <= 0:
54 return edges
55 elif complexity == 1:
56 extra_edges = max(1, num_nodes // 4)
57 elif complexity == 2:
58 extra_edges = max(2, num_nodes // 3)
59 else:
60 extra_edges = max(3, num_nodes // 2)
61
62 # Helper: check if adding edge (a->b) will create a cycle (loop)
63 def would_create_cycle(src: int, dst: int, current_edges: List[Tuple[int, int, str, str]]) -> bool:
64 """Return True if adding edge src->dst forms a cycle.
65
66 A cycle exists when src == dst or there is already a path dst->...->src.
67 """
68 if src == dst:
69 return True
70 # BFS/DFS to see if there is a path from b to a in current graph
71 adj = defaultdict(list)
72 for u, v, _oa, _ob in current_edges:
73 adj[u].append(v)
74 seen = set()
75 dq = deque([dst])
76 while dq:
77 v = dq.popleft()
78 if v == src:
79 return True
80 if v in seen:
81 continue
82 seen.add(v)
83 for w in adj.get(v, []):
84 if w not in seen:
85 dq.append(w)
86 return False
87
88 for _ in range(extra_edges):
89 a = rng.randint(1, num_nodes)
90 b = rng.randint(1, num_nodes)
91 # Decide if we add a cycle-forming edge
92 if would_create_cycle(a, b, edges):
93 if rng.random() < loop_prob:
94 edges.append(
95 (a, b, rng.choice(["+", "-"]), rng.choice(["+", "-"]))
96 )
97 # else: skip cycle edge this time
98 else:
99 # Non-cycle edge: add directly (no extra branch_prob gating)
100 edges.append(
101 (a, b, rng.choice(["+", "-"]), rng.choice(["+", "-"])))
102
103 return edges

106def _is_strict_linear(num_nodes: int, edges: List[Tuple[int, int, str, str]]) -> bool:
107 """Check if the whole graph forms a strict linear chain."""
108 if num_nodes == 0:
109 return False
110 # Must have exactly n-1 edges
111 if len(edges) != max(0, num_nodes - 1):
112 return False
113 indeg = defaultdict(int)
114 outdeg = defaultdict(int)
115 nodes = set(range(1, num_nodes + 1))
116 for a, b, _oa, _ob in edges:
117 outdeg[a] += 1
118 indeg[b] += 1
119 # Degree profile checks
120 starts = [v for v in nodes if indeg[v] == 0 and outdeg[v] == 1]
121 ends = [v for v in nodes if indeg[v] == 1 and outdeg[v] == 0]
122 mids_ok = all((indeg[v] == 1 and outdeg[v] == 1) or (
123 indeg[v] + outdeg[v] == 0) for v in nodes if v not in set(starts + ends))
124 if len(starts) != 1 or len(ends) != 1 or not mids_ok:
125 return False
126 # Connectivity check
127 s = starts[0]
128 seen = set([s])
129 cur = s
130 for _ in range(num_nodes - 1):
131 nxt = None
132 for a, b, _oa, _ob in edges:
133 if a == cur:
134 nxt = b
135 break
136 if nxt is None or nxt in seen:
137 return False
138 seen.add(nxt)
139 cur = nxt
140 return len(seen) == num_nodes

143def write_gfa(output_path: str, sequences: List[str], edges: List[Tuple[int, int, str, str]]) -> None:
144 """Write GFA v1: S/L records; add one P record if strictly linear.
145
146 If `output_path` is '-', emit to stdout instead of a file.
147 """
148 if output_path == "-":
149 f = sys.stdout
150 # Avoid mixing binary and text modes; stdout is fine
151 f.write("H\tVN:Z:1\n")
152 for i, seq in enumerate(sequences, start=1):
153 f.write(f"S\t{i}\t{seq}\n")
154 for (a, b, oa, ob) in edges:
155 f.write(f"L\t{a}\t{oa}\t{b}\t{ob}\t0M\n")
156 if len(sequences) > 0 and _is_strict_linear(len(sequences), edges):
157 path_nodes = ",".join(str(i) for i in range(1, len(sequences) + 1))
158 f.write(f"P\tlinear\t{path_nodes}\t*\n")
159 return
160
161 os.makedirs(os.path.dirname(output_path) or ".", exist_ok=True)
162 with open(output_path, "w", encoding="utf-8") as f:
163 f.write("H\tVN:Z:1\n")
164 for i, seq in enumerate(sequences, start=1):
165 f.write(f"S\t{i}\t{seq}\n")
166 for (a, b, oa, ob) in edges:
167 f.write(f"L\t{a}\t{oa}\t{b}\t{ob}\t0M\n")

169 path_nodes = ",".join(str(i) for i in range(1, len(sequences) + 1))

173def compress_linear(
174 sequences: List[str],
175 edges: List[Tuple[int, int, str, str]],
176) -> Tuple[List[str], List[Tuple[int, int, str, str]], Dict[int, int]]:
177 """
178 Merge maximal strictly linear chains where internal nodes have in-degree==1 and out-degree==1.
179 Returns:
180 - New sequence list (reindexed 1..M)
181 - New edge list (orientations normalized to '+')
182 - Mapping from old id to new id
183 Note: For simplicity and robustness, orientations are normalized to '+', and overlap stays as '0M'.
184 """
185 num_nodes = len(sequences)
186 id_to_seq: Dict[int, str] = {i + 1: s for i, s in enumerate(sequences)}
187 outgoing: Dict[int, List[Tuple[int, int, str, str]]] = defaultdict(list)
188 incoming: Dict[int, List[Tuple[int, int, str, str]]] = defaultdict(list)
189 for e in edges:
190 a, b, oa, ob = e
191 outgoing[a].append(e)
192 incoming[b].append(e)
193
194 indeg = {i: len(incoming.get(i, [])) for i in range(1, num_nodes + 1)}
195 outdeg = {i: len(outgoing.get(i, [])) for i in range(1, num_nodes + 1)}
196
197 visited = set()
198 chains: List[List[int]] = []
199
200 # Find chain starts: out==1 and in!=1, or sources with in==0
201 for v in range(1, num_nodes + 1):
202 if v in visited and indeg.get(v, 0) == 0 and outdeg.get(v, 0) == 0:
203 continue
204 if outdeg.get(v, 0) == 1 and indeg.get(v, 0) != 1:
205 cur = v
206 chain = [cur]
207 visited.add(cur)
208 while outdeg.get(cur, 0) == 1:
209 e = outgoing[cur][0]
210 nxt = e[1]
211 if indeg.get(nxt, 0) != 1 or nxt in visited:
212 break
213 chain.append(nxt)
214 visited.add(nxt)
215 cur = nxt
216 chains.append(chain)
217
218 # Handle unvisited segments with out==1 and in==1 (find leftmost and expand)
219 for v in range(1, num_nodes + 1):
220 if v in visited:
221 continue
222 if outdeg.get(v, 0) == 1 and indeg.get(v, 0) == 1:
223 # Backtrack to the leftmost
224 left = v
225 while indeg.get(left, 0) == 1:
226 prev_e = incoming[left][0]
227 prev_v = prev_e[0]
228 if outdeg.get(prev_v, 0) != 1 or prev_v in visited:
229 break
230 left = prev_v
231 # Expand to the right from left
232 cur = left
233 chain = [cur]
234 visited.add(cur)
235 while outdeg.get(cur, 0) == 1:
236 e = outgoing[cur][0]
237 nxt = e[1]
238 if indeg.get(nxt, 0) != 1 or nxt in visited:
239 break
240 chain.append(nxt)
241 visited.add(nxt)
242 cur = nxt
243 chains.append(chain)
244
245 # Remaining isolated nodes (out==0 and in==0) are also chains
246 for v in range(1, num_nodes + 1):
247 if v not in visited:
248 chains.append([v])
249 visited.add(v)
250
251 # Build new graph
252 new_id = 0
253 old_to_new: Dict[int, int] = {}
254 new_seq_by_id: Dict[int, str] = {}
255 for chain in chains:
256 new_id += 1
257 merged_seq = "".join(id_to_seq[x] for x in chain)
258 new_seq_by_id[new_id] = merged_seq
259 for x in chain:
260 old_to_new[x] = new_id
261
262 # Rebuild edges: keep inter-chain edges; remove internal; normalize orient to '+'
263 new_edges_set = set()
264 for a, b, _oa, _ob in edges:
265 na, nb = old_to_new[a], old_to_new[b]
266 if na == nb:
267 continue
268 new_edges_set.add((na, nb, "+", "+"))
269 new_edges = list(new_edges_set)
270
271 # Sort by new id and emit lists
272 sequences_out = [new_seq_by_id[i] for i in range(1, new_id + 1)]
273 return sequences_out, new_edges, old_to_new

296def main(argv: List[str]) -> int:
297 """Entry point: build sequences and edges, compress, and write GFA."""
298 args = parse_args(argv)
299 if args.nodes <= 0 or args.length <= 0:
300 print(
301 "[gen_graph] Node count and length must be positive integers", file=sys.stderr)
302 return 2
303
304 rng = random.Random(args.seed)
305 sequences = [random_dna(args.length, rng) for _ in range(args.nodes)]
306 edges = make_edges(
307 num_nodes=args.nodes,
308 complexity=args.complexity,
309 loop_prob=args.loop_prob,
310 rng=rng,
311 )
312 # Stats before compression
313 n0, m0 = len(sequences), len(edges)
314 # Perform linear-chain compression
315 sequences2, edges2, mapping = compress_linear(sequences, edges)
316 n1, m1 = len(sequences2), len(edges2)
317 output_path = "-" if args.emit == "stdout" or args.output == "-" else args.output
318 write_gfa(output_path, sequences2, edges2)
319 # Summary report
320 merged_nodes = n0 - n1
321 print(
322 f"[gen_graph] Original: nodes={n0}, edges={m0}; Compressed: nodes={n1}, edges={m1}; merged_nodes={merged_nodes}")
323 print(
324 f"[gen_graph] Written: {'stdout' if output_path == '-' else output_path} (len={args.length}, complexity={args.complexity})")
325 return 0
