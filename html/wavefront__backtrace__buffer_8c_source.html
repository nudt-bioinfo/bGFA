50void wf_backtrace_buffer_segment_add(
51 wf_backtrace_buffer_t* const bt_buffer) {
52 bt_block_t* const bt_segment = mm_allocator_calloc(
53 bt_buffer->mm_allocator,BT_BUFFER_SEGMENT_LENGTH,bt_block_t,false);
54 vector_insert(bt_buffer->segments,bt_segment,bt_block_t*);
55}
56void wf_backtrace_buffer_segment_reserve(
57 wf_backtrace_buffer_t* const bt_buffer) {
58 // Reset position
59 bt_buffer->segment_offset = 0;
60 ++(bt_buffer->segment_idx);
61 // Check segments
62 if (bt_buffer->segment_idx >= vector_get_used(bt_buffer->segments)) {
63 // Check segment position
64 const uint64_t block_idx = ((uint64_t)bt_buffer->segment_idx+1) * BT_BUFFER_SEGMENT_LENGTH;
65 if (block_idx >= BT_BLOCK_IDX_MAX) {
66 fprintf(stderr,"[WFA::BacktraceBuffer] Reached maximum addressable index"); exit(-1);
67 }
68 // Add segment
69 wf_backtrace_buffer_segment_add(bt_buffer);
70 }
71 // Set pointer to next block free
72 bt_block_t** const segments = vector_get_mem(bt_buffer->segments,bt_block_t*);
73 bt_buffer->block_next = segments[bt_buffer->segment_idx];
74}

78wf_backtrace_buffer_t* wf_backtrace_buffer_new(
79 mm_allocator_t* const mm_allocator) {
80 // Alloc
81 wf_backtrace_buffer_t* const bt_buffer =
82 mm_allocator_alloc(mm_allocator,wf_backtrace_buffer_t);
83 bt_buffer->mm_allocator = mm_allocator;
84 // Initialize
85 bt_buffer->segment_idx = 0;
86 bt_buffer->segment_offset = 0;
87 bt_buffer->segments = vector_new(10,bt_block_t*);
88 wf_backtrace_buffer_segment_add(bt_buffer); // Add initial segment
89 bt_buffer->block_next = vector_get_mem(bt_buffer->segments,bt_block_t*)[0];
90 bt_buffer->num_compacted_blocks = 0;
91 bt_buffer->num_compactions = 0;
92 bt_buffer->alignment_init_pos = vector_new(100,wf_backtrace_init_pos_t);
93 bt_buffer->alignment_packed = vector_new(100,pcigar_t);
94 bt_buffer->prefetch_blocks_idxs = vector_new(500,bt_block_idx_t);
95 // Return
96 return bt_buffer;
97}

107void wf_backtrace_buffer_reap(
108 wf_backtrace_buffer_t* const bt_buffer) {
109 // Reap segments beyond the first
110 const int num_segments = vector_get_used(bt_buffer->segments);
111 bt_block_t** const segments = vector_get_mem(bt_buffer->segments,bt_block_t*);
112 int i;
113 for (i=1;i<num_segments;++i) {
114 mm_allocator_free(bt_buffer->mm_allocator,segments[i]);
115 }
116 vector_set_used(bt_buffer->segments,1);
117 // Clear
118 bt_buffer->segment_idx = 0;
119 bt_buffer->segment_offset = 0;
120 bt_buffer->block_next = vector_get_mem(bt_buffer->segments,bt_block_t*)[0];
121 bt_buffer->num_compacted_blocks = 0;
122 bt_buffer->num_compactions = 0;
123}

250/*
251 * Unpack CIGAR
252 */
253bt_block_t* wf_backtrace_buffer_traceback_pcigar(
254 wf_backtrace_buffer_t* const bt_buffer,
255 bt_block_t* bt_block) {
256 // Clear temporal buffer
257 vector_t* const alignment_packed = bt_buffer->alignment_packed;
258 vector_clear(alignment_packed);
259 // Traverse-back the BT-blocks and store all the pcigars
260 while (bt_block->prev_idx != BT_BLOCK_IDX_NULL) {
261 vector_insert(alignment_packed,bt_block->pcigar,pcigar_t);
262 const bt_block_idx_t prev_idx = bt_block->prev_idx;
263 bt_block = wf_backtrace_buffer_get_block(bt_buffer,prev_idx);
264 }
265 // Return initial block (start coordinate)
266 return bt_block;
267}
268void wf_backtrace_buffer_unpack_cigar_linear(
269 wf_backtrace_buffer_t* const bt_buffer,
270 wavefront_sequences_t* const sequences,
271 const int begin_v,
272 const int begin_h,
273 const int end_v,
274 const int end_h,
275 cigar_t* const cigar) {
276 // Parameters
277 const int pattern_length = sequences->pattern_length;
278 const int text_length = sequences->text_length;
279 // Add init insertions/deletions
280 cigar_clear(cigar);
281 char* cigar_buffer = cigar->operations;
282 int i;
283 int v = begin_v;
284 int h = begin_h;
285 for (i=0;i<h;++i) {*cigar_buffer = 'I'; ++cigar_buffer;};
286 for (i=0;i<v;++i) {*cigar_buffer = 'D'; ++cigar_buffer;};
287 // Traverse-forward the pcigars and unpack the cigar
288 const int num_palignment_blocks = vector_get_used(bt_buffer->alignment_packed);
289 pcigar_t* const palignment_blocks = vector_get_mem(bt_buffer->alignment_packed,pcigar_t);
290 for (i=num_palignment_blocks-1;i>=0;--i) {
291 // Unpack block
292 int cigar_block_length = 0;
293 pcigar_unpack_linear(
294 palignment_blocks[i],sequences,&v,&h,
295 cigar_buffer,&cigar_block_length);
296 // Update CIGAR
297 cigar_buffer += cigar_block_length;
298 }
299 // Account for last stroke of matches
300 const int num_matches = MIN(end_v-v,end_h-h);
301 for (i=0;i<num_matches;++i) {*cigar_buffer = 'M'; ++cigar_buffer;};
302 v += num_matches;
303 h += num_matches;
304 // Account for last stroke of insertion/deletion
305 while (h < text_length) {*cigar_buffer = 'I'; ++cigar_buffer; ++h;};
306 while (v < pattern_length) {*cigar_buffer = 'D'; ++cigar_buffer; ++v;};
307 // Close CIGAR
308 *cigar_buffer = '\0';
309 cigar->end_offset = cigar_buffer - cigar->operations;
310}

358void wf_backtrace_buffer_mark_backtrace(
359 wf_backtrace_buffer_t* const bt_buffer,
360 const bt_block_idx_t bt_block_idx,
361 bitmap_t* const bitmap) {
362 // Parameters
363 const bt_block_idx_t num_compacted_blocks = bt_buffer->num_compacted_blocks;
364 // Traverse-back the BT-blocks while not marked
365 bt_block_t bt_block_last = { .prev_idx = bt_block_idx };
366 bt_block_t* bt_block = &bt_block_last;
367 // Check marked and fetch previous (until already marked or NULL is found)
368 while (bt_block->prev_idx != BT_BLOCK_IDX_NULL &&
369 bt_block->prev_idx >= num_compacted_blocks &&
370 !bitmap_check__set(bitmap,bt_block->prev_idx)) {
371 // Fetch previous BT-block
372 const bt_block_idx_t prev_idx = bt_block->prev_idx;
373 bt_block = wf_backtrace_buffer_get_block(bt_buffer,prev_idx);
374 }

376void wf_backtrace_buffer_mark_backtrace_batch(
377 wf_backtrace_buffer_t* const bt_buffer,
378 wf_offset_t* const offsets,
379 bt_block_idx_t* const bt_block_idxs,
380 const int num_block_idxs,
381 bitmap_t* const bitmap) {
382 // Parameters
383 const bt_block_idx_t num_compacted_blocks = bt_buffer->num_compacted_blocks;
384 // Reserve prefetch-buffer
385 const int max_batch_size = 100;
386 vector_reserve(bt_buffer->prefetch_blocks_idxs,max_batch_size,false);
387 bt_block_idx_t* const pf_block_idx = vector_get_mem(bt_buffer->prefetch_blocks_idxs,bt_block_idx_t);
388 // Fill-in loop (+ initial prefetch)
389 int active_blocks = 0, next_idx = 0;
390 while (active_blocks < max_batch_size && next_idx < num_block_idxs) {
391 // Check NULL
392 const bt_block_idx_t block_idx = bt_block_idxs[next_idx];
393 if (offsets[next_idx] >= 0 &&
394 block_idx >= num_compacted_blocks) { // NOTE block_idx != BT_BLOCK_IDX_NULL
395 // Prefetch (bt-block and bt_block)
396 BITMAP_PREFETCH_BLOCK(bitmap,block_idx);
397 wf_backtrace_buffer_prefetch_block(bt_buffer,block_idx);
398 // Store
399 pf_block_idx[active_blocks] = block_idx;
400 ++active_blocks;
401 }
402 ++next_idx; // Next
403 }
404 // Batch process+prefetch loop
405 int i = 0;
406 while (active_blocks > 0) {
407 // Fetch BT-block & BM-block
408 const bt_block_idx_t block_idx = pf_block_idx[i];
409 BITMAP_GET_BLOCK(bitmap,block_idx,block_bm_ptr);
410 // Check marked
411 if (!BM_BLOCK_IS_SET(*block_bm_ptr,block_idx)) {
412 BM_BLOCK_SET(*block_bm_ptr,block_idx);
413 // Fetch next block
414 bt_block_t* const bt_block = wf_backtrace_buffer_get_block(bt_buffer,block_idx);
415 const bt_block_idx_t prev_block_idx = bt_block->prev_idx;
416 // Check NULL
417 if (prev_block_idx != BT_BLOCK_IDX_NULL &&
418 prev_block_idx >= num_compacted_blocks) {
419 // Update next bt-block index and prefetch
420 pf_block_idx[i] = prev_block_idx;
421 BITMAP_PREFETCH_BLOCK(bitmap,prev_block_idx);
422 wf_backtrace_buffer_prefetch_block(bt_buffer,prev_block_idx);
423 // Next in batch
424 i = (i+1) % active_blocks;
425 continue;
426 }
427 }
428 // Refill
429 while (true /* !refilled */) {
430 if (next_idx < num_block_idxs) {
431 // Check NULL
432 if (offsets[next_idx] < 0 || bt_block_idxs[next_idx] < num_compacted_blocks) {
433 ++next_idx; // Next
434 continue;
435 }
436 // Prefetch (bt-block and bt_block)
437 const bt_block_idx_t block_idx = bt_block_idxs[next_idx];
438 BITMAP_PREFETCH_BLOCK(bitmap,block_idx);
439 wf_backtrace_buffer_prefetch_block(bt_buffer,block_idx);
440 // Store
441 pf_block_idx[i] = block_idx;
442 ++next_idx;
443 // Next in batch
444 i = (i+1) % active_blocks;
445 break;
446 } else {
447 // Take the last in the batch
448 --active_blocks;
449 pf_block_idx[i] = pf_block_idx[active_blocks];
450 // Next in batch
451 if (active_blocks) i = (i+1) % active_blocks;
452 break;
453 }
454 }
455 }

457bt_block_idx_t wf_backtrace_buffer_compact_marked(
458 wf_backtrace_buffer_t* const bt_buffer,
459 bitmap_t* const bitmap,
460 const int verbose) {
461 // Parameters
462 const int num_segments = vector_get_used(bt_buffer->segments);
463 bt_block_t** const segments = vector_get_mem(bt_buffer->segments,bt_block_t*);
464 const bt_block_idx_t num_compacted_blocks = bt_buffer->num_compacted_blocks;
465 // Sentinels
466 bt_block_idx_t read_global_pos = num_compacted_blocks;
467 bt_block_idx_t write_global_pos = num_compacted_blocks;
468 bt_block_idx_t read_segidx = BT_BUFFER_SEGMENT_IDX(read_global_pos);
469 bt_block_idx_t read_offset = BT_BUFFER_SEGMENT_OFFSET(read_global_pos);
470 bt_block_idx_t write_segidx = BT_BUFFER_SEGMENT_IDX(write_global_pos);
471 bt_block_idx_t write_offset = BT_BUFFER_SEGMENT_OFFSET(write_global_pos);
472 bt_block_t* read_block = segments[read_segidx] + read_offset;
473 bt_block_t* write_block = segments[write_segidx] + write_offset;
474 // Traverse all BT-blocks from the beginning (stored marked)
475 const bt_block_idx_t max_block_idx = BT_BUFFER_IDX(bt_buffer->segment_idx,bt_buffer->segment_offset);
476 while (read_global_pos < max_block_idx) {
477 // Check marked block
478 BITMAP_GET_BLOCK(bitmap,read_global_pos,block_bitmap_ptr);
479 if (BM_BLOCK_IS_SET(*block_bitmap_ptr,read_global_pos)) {
480 // Store pcigar in compacted BT-buffer
481 write_block->pcigar = read_block->pcigar;
482 // Translate and store index in compacted BT-buffer
483 if (read_block->prev_idx == BT_BLOCK_IDX_NULL ||
484 read_block->prev_idx < num_compacted_blocks) {
485 write_block->prev_idx = read_block->prev_idx;
486 } else {
487 write_block->prev_idx = num_compacted_blocks + bitmap_erank(bitmap,read_block->prev_idx);
488 }
489 // Next write
490 ++write_offset; ++write_block; ++write_global_pos;
491 if (write_offset >= BT_BUFFER_SEGMENT_LENGTH) {
492 // Next segment
493 write_block = segments[++write_segidx];
494 write_offset = 0;
495 }
496 }
497 // Next read
498 ++read_offset; ++read_block; ++read_global_pos;
499 if (read_offset >= BT_BUFFER_SEGMENT_LENGTH) {
500 // Next segment
501 if (++read_segidx >= num_segments) break;
502 read_block = segments[read_segidx];
503 read_offset = 0;
504 }
505 }
506 // Update next BT-buffer index
507 bt_buffer->segment_offset = write_offset;
508 bt_buffer->segment_idx = write_segidx;
509 bt_buffer->block_next = write_block;
510 bt_buffer->num_compactions++;
511 // DEBUG
512 if (verbose >= 3) {
513 fprintf(stderr,"[WFA::BacktraceBuffer] Compacted from %lu MB to %lu MB (%2.2f%%)",
514 CONVERT_B_TO_MB(read_global_pos*sizeof(bt_block_t)),
515 CONVERT_B_TO_MB(write_global_pos*sizeof(bt_block_t)),
516 100.0f*(float)write_global_pos/(float)read_global_pos);
517 }
518 // Return last index
519 return write_global_pos - 1;
520}
