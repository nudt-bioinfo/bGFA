26std::string convertKswCigar2MX(const std::string &cigar_str_in,
27 const std::string &tseq, const std::string &qseq)
28{
29 if (cigar_str_in.empty())
30 return cigar_str_in;
31
32 std::string new_cigar;
33 new_cigar.reserve(cigar_str_in.size());
34
35 size_t p = 0, n = cigar_str_in.size();
36 size_t pos_t = 0, pos_q = 0; // positions in tseq / qseq
37
38 auto append_op = [&](uint64_t len, char op)
39 {
40 if (len == 0)
41 return;
42 if (!new_cigar.empty())
43 {
44 // if previous op char equals current op, merge lengths
45 char last_op = new_cigar.back();
46 if (last_op == op)
47 {
48 // find start index of last token's number (the digits before last_op)
49 size_t pos_op = new_cigar.size() - 1;
50 size_t num_start = pos_op;
51 while (num_start > 0 && std::isdigit(static_cast<unsigned char>(new_cigar[num_start - 1])))
52 --num_start;
53 size_t num_len = pos_op - num_start;
54 uint64_t last_len = 0;
55 if (num_len > 0)
56 {
57 last_len = std::stoull(new_cigar.substr(num_start, num_len));
58 }
59 uint64_t merged = last_len + len;
60 new_cigar.erase(num_start);
61 new_cigar += std::to_string(merged);
62 new_cigar.push_back(op);
63 return;
64 }
65 }
66 new_cigar += std::to_string(len);
67 new_cigar.push_back(op);
68 };
69
70 while (p < n)
71 {
72 uint64_t num = 0;
73 while (p < n && std::isdigit(static_cast<unsigned char>(cigar_str_in[p])))
74 {
75 num = num * 10 + (cigar_str_in[p] - '0');
76 ++p;
77 }
78 if (p >= n)
79 break; // malformed
80 char op = cigar_str_in[p++];
81
82 if (op == 'M')
83 {
84 // split M into runs of matches (M) and mismatches (X)
85 uint64_t run_len = 0;
86 bool run_match = false;
87 for (uint64_t i = 0; i < num; ++i)
88 {
89 char a = (pos_t + i < tseq.size()) ? tseq[pos_t + i] : '\0';
90 char b = (pos_q + i < qseq.size()) ? qseq[pos_q + i] : '\0';
91 bool equal = (std::toupper(static_cast<unsigned char>(a)) ==
92 std::toupper(static_cast<unsigned char>(b)));
93 if (i == 0)
94 {
95 run_match = equal;
96 run_len = 1;
97 }
98 else if (equal == run_match)
99 {
100 ++run_len;
101 }
102 else
103 {
104 append_op(run_len, run_match ? 'M' : 'X');
105 run_match = equal;
106 run_len = 1;
107 }
108 }
109 if (run_len)
110 append_op(run_len, run_match ? 'M' : 'X');
111
112 pos_t += num;
113 pos_q += num;
114 }
115 else if (op == 'I')
116 {
117 append_op(num, 'I');
118 pos_q += num;
119 }
120 else if (op == 'D')
121 {
122 append_op(num, 'D');
123 pos_t += num;
124 }
125 else
126 {
127 // keep other ops as-is
128 append_op(num, op);
129 }
130 }
131
132 return new_cigar;
133}

141std::string alignmentSequenceKSW(const std::string &tseq,
142 const std::string &qseq)
143{
144
145 std::string cigar_str;
146 if (tseq.empty() || qseq.empty())
147 {
148 return "0M";
149 }
150
151 int i, a = 2, b = 4; // a>0 and b<0
152 int8_t mat[25] = {a, b, b, b, 0,
153 b, a, b, b, 0,
154 b, b, a, b, 0,
155 b, b, b, a, 0,
156 0, 0, 0, 0, 0};
157 int tl = tseq.size(), ql = qseq.size();
158 uint8_t *ts, *qs, c[256];
159 ksw_extz_t ez;
160
161 memset(&ez, 0, sizeof(ksw_extz_t));
162 memset(c, 4, 256);
163 c['A'] = c['a'] = 0;
164 c['C'] = c['c'] = 1;
165 c['G'] = c['g'] = 2;
166 c['T'] = c['t'] = 3; // build the encoding table
167 ts = (uint8_t *)malloc(tl);
168 qs = (uint8_t *)malloc(ql);
169 for (i = 0; i < tl; ++i)
170 ts[i] = c[(uint8_t)tseq[i]]; // encode to 0/1/2/3
171 for (i = 0; i < ql; ++i)
172 qs[i] = c[(uint8_t)qseq[i]];
173
174 ksw_extz2_sse(0, ql, qs, tl, ts, 5, mat, 4, 2, -1, -1, 0, 0, &ez);
175
176 if (ez.n_cigar == 0)
177 {
178 free(ts);
179 free(qs);
180 return "0M";
181 }
182 cigar_str.reserve(ez.n_cigar * 4); // Reserve space for CIGAR string
183 for (i = 0; i < ez.n_cigar; ++i) // convert CIGAR to string
184 {
185 int len = ez.cigar[i] >> 4;
186 char op = GlobalVariant::bin2cigarop[ez.cigar[i] & 0xf];
187 cigar_str += std::to_string(len) + op;
188 }
189 free(ez.cigar);
190 free(ts);
191 free(qs);
192
193 // Convert KSW CIGAR (M include mismatches) into M/X-aware CIGAR
194 cigar_str = convertKswCigar2MX(cigar_str, tseq, qseq);
195
196#ifdef INFO_DEBUG
197 std::cout << "[KSW Alignment] CIGAR: " << cigar_str
198 << "\n\t\tTarget Seq: " << tseq
199 << "\n\t\tQuery Seq: " << qseq << std::endl;
200#endif
201
202 return cigar_str;
203}

211std::string alignmentSequenceWFA(const std::string &tseq,
212 const std::string &qseq)
213{
214
215 std::string cigar_str;
216 if (tseq.empty() || qseq.empty())
217 {
218 return "0M"; // Return empty CIGAR string if either sequence is empty
219 }
220
221 WFAlignerGapAffine aligner(4, 6, 2, WFAligner::Alignment,
222 WFAligner::MemoryHigh);
223
224 // Normalize sequences to uppercase to ensure consistent behavior
225 // across aligner backends (WFA expects canonical bases).
226 std::string t_up = tseq;
227 std::string q_up = qseq;
228 for (char &c : t_up)
229 c = static_cast<char>(std::toupper(static_cast<unsigned char>(c)));
230 for (char &c : q_up)
231 c = static_cast<char>(std::toupper(static_cast<unsigned char>(c)));
232
233 aligner.alignEnd2End(t_up, q_up);
234
235 cigar_str = aligner.getCIGAR(true);
236 std::replace(cigar_str.begin(), cigar_str.end(), '=', 'M');
237
238#ifdef INFO_DEBUG
239 std::cout << "[WFA Alignment] CIGAR: " << cigar_str
240 << "\n\t\tTarget Seq: " << tseq
241 << "\n\t\tQuery Seq: " << qseq << std::endl;
242#endif
243
244 return cigar_str;
245}

247std::string alignmentSequence(const std::string &tseq,
248 const std::string &qseq,
249 const uint8_t kernel = 1)
250{
251 if (kernel == 0) // Use KSW alignment kernel
252 {
253 return alignmentSequenceKSW(tseq, qseq);

255 else if (kernel == 1) // Use WFA alignment kernel

261 return "0M";
262 }
263}
