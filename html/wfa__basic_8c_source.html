35int main(int argc,char* argv[]) {
36 // Patter & Text
37 char* pattern = "TCTTTACTCGCGCGTTGGAGAAATACAATAGT";
38 char* text = "TCTATACTGCGCGTTTGGAGAAATAAAATAGT";
39 // Configure alignment attributes
40 wavefront_aligner_attr_t attributes = wavefront_aligner_attr_default;
41 attributes.distance_metric = gap_affine;
42 attributes.affine_penalties.match = 0;
43 attributes.affine_penalties.mismatch = 4;
44 attributes.affine_penalties.gap_opening = 6;
45 attributes.affine_penalties.gap_extension = 2;
46 // Initialize Wavefront Aligner
47 wavefront_aligner_t* const wf_aligner = wavefront_aligner_new(&attributes);
48 // Align
49 wavefront_align(wf_aligner,pattern,strlen(pattern),text,strlen(text));
50 fprintf(stderr,"WFA-Alignment returns score %d\n",wf_aligner->cigar->score);
51 // Display alignment
52 fprintf(stderr," PATTERN %s\n",pattern);
53 fprintf(stderr," TEXT %s\n",text);
54 fprintf(stderr," SCORE (RE)COMPUTED %d\n",
55 cigar_score_gap_affine(wf_aligner->cigar,&attributes.affine_penalties));
56 cigar_print_pretty(stderr,wf_aligner->cigar,
57 pattern,strlen(pattern),text,strlen(text));
58 // Free
59 wavefront_aligner_delete(wf_aligner);
60}
