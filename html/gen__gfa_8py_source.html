1"""
2Generate GFA graphs with two generation modes.
3
4Mode A (basic):
5- Given node count, average length (±20%), and edge multiplier (1.2/1.5/2.0),
6 build sequences and a non-linear connection scheme.
7- Ensures exactly one source-only node (indeg==0) and one sink-only node (outdeg==0).
8 All other nodes have at least one predecessor and one successor.
9- The graph is not strictly linear (adds cross edges beyond a simple chain).
10
11Mode B (distribution):
12- Given total length and a node-length distribution across 6 bins
13 (<10, 10-100, 100-1000, 1k-10k, 10k-100k, >100k),
14 either via explicit percentages or presets, estimate node count and sample lengths.
15- Then generate connections with the same degree constraints and edge multiplier.
16
17Author: grampart
18Version: 0.1
19License: MIT License
20"""

79 # Always at least a chain; then add up to round(mult*n)
80 return max(n - 1, int(round(mult * n)))

83def _generate_edges(num_nodes: int, edge_mult: float, rng: random.Random) -> List[Tuple[int, int, str, str]]:
84 if num_nodes < 2:
85 raise ValueError(
86 "num_nodes must be >= 2 to ensure unique source and sink")
87
88 # Start with a simple chain to guarantee connectivity and degree constraints for inner nodes
89 edges: List[Tuple[int, int, str, str]] = [
90 (i, i + 1, "+", "+") for i in range(1, num_nodes)]
91 edge_set = set(edges) # for duplicate checks (includes orientation)
92
93 target = _target_edge_count(num_nodes, edge_mult)
94
95 # Add cross edges forward in index to avoid cycles and preserve source/sink uniqueness
96 # Source is 1 (indeg=0), sink is num_nodes (outdeg=0). We avoid edges into 1 or out of N implicitly by a<b.
97 while len(edges) < target:
98 a = rng.randint(1, num_nodes - 1)
99 b = rng.randint(a + 1, num_nodes)
100 oa = rng.choice(["+", "-"])
101 ob = rng.choice(["+", "-"])
102 candidate = (a, b, oa, ob)
103 if candidate in edge_set:
104 continue
105 edge_set.add(candidate)
106 edges.append(candidate)
107
108 return edges

111def _sample_lengths_mode_a(n: int, avg_len: int, rng: random.Random) -> List[str]:
112 if avg_len <= 0:

121 avgs: List[float] = []

128def _normalize_dist(dist: List[float]) -> List[float]:
129 s = sum(dist)
130 if s <= 0:
131 raise ValueError("Distribution sum must be > 0")
132 return [x / s for x in dist]

135def _perturb_percentages(dist_pct: List[float], jitter: float, rng: random.Random) -> List[float]:
136 """Apply ±jitter relative fluctuation to each percentage and rescale to sum 100.
137
138 - jitter=0.1 means each value multiplied by U[0.9, 1.1].
139 - Ensures non-negative and final sum exactly 100.0 (within float error).
140 """
141 jitter = max(0.0, jitter)
142 if jitter == 0.0:
143 return list(dist_pct)
144 perturbed = []
145 low = 1.0 - jitter
146 high = 1.0 + jitter
147 for w in dist_pct:
148 factor = rng.uniform(low, high)
149 val = max(0.0, w * factor)
150 perturbed.append(val)
151 total = sum(perturbed)
152 if total <= 0:
153 return list(dist_pct)
154 scale = 100.0 / total
155 return [v * scale for v in perturbed]

158def _round_counts(total: int, weights: List[float]) -> List[int]:
159 # Largest remainder method
160 raw = [total * w for w in weights]
161 floors = [int(x) for x in raw]
162 rem = total - sum(floors)
163 fracs = [(i, raw[i] - floors[i]) for i in range(len(weights))]
164 fracs.sort(key=lambda t: t[1], reverse=True)
165 for i in range(rem):
166 floors[fracs[i][0]] += 1
167 return floors

170def _sample_lengths_mode_b(total_len: int, dist_pct: List[float], max_len_last_bin: int,
171 rng: random.Random, tol: float = 0.1) -> List[str]:
172 if total_len <= 0:
173 raise ValueError("Total length must be positive")
174 if max_len_last_bin < 100001:
175 raise ValueError("max length for last bin must be >= 100001")
176
177 weights = _normalize_dist(dist_pct)
178 avgs = _bin_avgs(max_len_last_bin)
179 exp_avg = sum(w * a for w, a in zip(weights, avgs))
180 n_est = max(2, int(round(total_len / max(1.0, exp_avg))))
181
182 counts = _round_counts(n_est, weights)
183
184 # Build ranges with applied max for last bin
185 ranges: List[Tuple[int, int]] = []
186 for low, high in BINS_SPECS:
187 h = max_len_last_bin if high is None else high
188 ranges.append((low, h))
189
190 # First pass: sample lengths within their bins; keep track of assigned bin ranges
191 seqs: List[str] = []
192 lengths: List[int] = []
193 assigned_ranges: List[Tuple[int, int]] = []
194 for (low, high), cnt in zip(ranges, counts):
195 for _ in range(cnt):
196 length = rng.randint(low, high)
197 lengths.append(length)
198 assigned_ranges.append((low, high))
199 seqs.append(random_dna(length, rng))
200
201 # In rare cases, if counts rounded down to <2 due to extremely small total_len, pad to 2 nodes
202 while len(seqs) < 2:
203 seqs.append(random_dna(10, rng))
204
205 # Top-up to reach at least total_len by extending nodes within bin caps, then add nodes if needed
206 current_total = sum(lengths)
207 # desired bounds for tolerance
208 tol = max(0.0, min(1.0, tol))
209 lower_bound = int(math.floor(total_len * (1.0 - tol)))
210 upper_bound = int(math.ceil(total_len * (1.0 + tol)))
211
212 if current_total < total_len:
213 gap = total_len - current_total
214 # Extend existing nodes up to their bin upper bounds
215 idx = 0
216 n_nodes = len(lengths)
217 while gap > 0 and idx < n_nodes:
218 low, high = assigned_ranges[idx]
219 can_inc = max(0, high - lengths[idx])
220 if can_inc > 0:
221 inc = min(can_inc, gap)
222 lengths[idx] += inc
223 # append random bases deterministically per node
224 seqs[idx] += random_dna(inc, rng)
225 gap -= inc
226 idx += 1
227
228 # If still not enough, append new nodes sampled by weights
229 if gap > 0:
230 # Build cumulative weights for sampling bin index
231 cum: List[float] = []
232 s = 0.0
233 for w in weights:
234 s += w
235 cum.append(s)
236
237 def pick_bin() -> int:
238 r = rng.random()
239 for i, c in enumerate(cum):
240 if r <= c:
241 return i
242 return len(cum) - 1
243
244 while gap > 0:
245 bi = pick_bin()
246 low, high = ranges[bi]
247 # choose a length trying not to overshoot too much, but stay within bin
248 # if gap < low, we will overshoot; that's acceptable to satisfy minimum total length
249 desired = max(low, min(high, gap))
250 seq = random_dna(desired, rng)
251 seqs.append(seq)
252 lengths.append(desired)
253 assigned_ranges.append((low, high))
254 # update gap (allow negative to indicate overshoot handled)
255 if desired >= gap:
256 gap = 0
257 else:
258 gap -= desired
259
260 current_total = sum(lengths)
261
262 # If we exceed the upper bound, try to shrink towards target without going below bin lows
263 if current_total > upper_bound:
264 # Aim for target total_len if achievable; otherwise get as low as allowed by lows
265 target_total = max(lower_bound, min(total_len, current_total))
266 shrink = current_total - target_total
267 if shrink > 0:
268 idx = 0
269 n_nodes = len(lengths)
270 while shrink > 0 and idx < n_nodes:
271 low, _high = assigned_ranges[idx]
272 dec_cap = max(0, lengths[idx] - low)
273 if dec_cap > 0:
274 dec = min(dec_cap, shrink)
275 lengths[idx] -= dec
276 # trim sequence tail by dec bases
277 if dec > 0:
278 seqs[idx] = seqs[idx][:-
279 dec] if dec <= len(seqs[idx]) else ""
280 shrink -= dec
281 idx += 1
282
283 # Recompute; if still above upper bound, we've hit lower limits; accept best effort
284 # If we undershot below lower bound due to rounding, we can extend 1 base back to fix, but unlikely.
285
286 return seqs
