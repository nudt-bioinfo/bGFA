44void wavefront_compute_indel_idm(
45 wavefront_aligner_t* const wf_aligner,
46 wavefront_t* const wf_prev,
47 wavefront_t* const wf_curr,
48 const int lo,
49 const int hi) {
50 // Parameters
51 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
52 const int pattern_length = sequences->pattern_length;
53 const int text_length = sequences->text_length;
54 const wf_offset_t* const prev_offsets = wf_prev->offsets;
55 wf_offset_t* const curr_offsets = wf_curr->offsets;
56 // Compute-Next kernel loop
57 int k;
58 PRAGMA_LOOP_VECTORIZE
59 for (k=lo;k<=hi;++k) {
60 // Compute maximum offset
61 const wf_offset_t ins = prev_offsets[k-1] + 1;
62 const wf_offset_t del = prev_offsets[k+1];
63 wf_offset_t max = MAX(del,ins);
64 // Adjust offset out of boundaries !(h>tlen,v>plen) (here to allow vectorization)
65 const wf_unsigned_offset_t h = WAVEFRONT_H(k,max); // Make unsigned to avoid checking negative
66 const wf_unsigned_offset_t v = WAVEFRONT_V(k,max); // Make unsigned to avoid checking negative
67 if (h > text_length) max = WAVEFRONT_OFFSET_NULL;
68 if (v > pattern_length) max = WAVEFRONT_OFFSET_NULL;
69 curr_offsets[k] = max;
70 }
71}

104void wavefront_compute_indel_idm_piggyback(
105 wavefront_aligner_t* const wf_aligner,
106 wavefront_t* const wf_prev,
107 wavefront_t* const wf_curr,
108 const int lo,
109 const int hi,
110 const int score) {
111 // Parameters
112 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
113 const int pattern_length = sequences->pattern_length;
114 const int text_length = sequences->text_length;
115 // Previous WF
116 const wf_offset_t* const prev_offsets = wf_prev->offsets;
117 const pcigar_t* const prev_pcigar = wf_prev->bt_pcigar;
118 const bt_block_idx_t* const prev_bt_idx = wf_prev->bt_prev;
119 // Current WF
120 wf_offset_t* const curr_offsets = wf_curr->offsets;
121 pcigar_t* const curr_pcigar = wf_curr->bt_pcigar;
122 bt_block_idx_t* const curr_bt_idx = wf_curr->bt_prev;
123 // Compute-Next kernel loop
124 int k;
125 PRAGMA_LOOP_VECTORIZE // Ifs predicated by the compiler
126 for (k=lo;k<=hi;++k) {
127 // Compute maximum offset
128 const wf_offset_t ins = prev_offsets[k-1] + 1;
129 const wf_offset_t del = prev_offsets[k+1];
130 wf_offset_t max = MAX(del,ins);
131 // Update pcigar & bt-block
132 if (max == del) {
133 curr_pcigar[k] = PCIGAR_PUSH_BACK_DEL(prev_pcigar[k+1]);
134 curr_bt_idx[k] = prev_bt_idx[k+1];
135 } else { // max == ins
136 curr_pcigar[k] = PCIGAR_PUSH_BACK_INS(prev_pcigar[k-1]);
137 curr_bt_idx[k] = prev_bt_idx[k-1];
138 }
139 // Adjust offset out of boundaries !(h>tlen,v>plen) (here to allow vectorization)
140 const wf_unsigned_offset_t h = WAVEFRONT_H(k,max); // Make unsigned to avoid checking negative
141 const wf_unsigned_offset_t v = WAVEFRONT_V(k,max); // Make unsigned to avoid checking negative
142 if (h > text_length) max = WAVEFRONT_OFFSET_NULL;
143 if (v > pattern_length) max = WAVEFRONT_OFFSET_NULL;
144 curr_offsets[k] = max;
145 }

196/*
197 * Exact pruning paths
198 */
199int wf_compute_edit_best_score(
200 const int pattern_length,
201 const int text_length,
202 const int k,
203 const wf_offset_t offset) {
204 // Compute best-alignment case
205 const int left_v = pattern_length - WAVEFRONT_V(k,offset);
206 const int left_h = text_length - WAVEFRONT_H(k,offset);
207 return (left_v >= left_h) ? left_v - left_h : left_h - left_v;
208}
209int wf_compute_edit_worst_score(
210 const int pattern_length,
211 const int text_length,
212 const int k,
213 const wf_offset_t offset) {
214 // Compute worst-alignment case
215 const int left_v = pattern_length - WAVEFRONT_V(k,offset);
216 const int left_h = text_length - WAVEFRONT_H(k,offset);
217 return MAX(left_v,left_h);
218}
219void wavefront_compute_edit_exact_prune(
220 wavefront_aligner_t* const wf_aligner,
221 wavefront_t* const wavefront) {
222 // Parameters
223 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
224 const int plen = sequences->pattern_length;
225 const int tlen = sequences->text_length;
226 wf_offset_t* const offsets = wavefront->offsets;
227 const int lo = wavefront->lo;
228 const int hi = wavefront->hi;
229 // Speculative compute if needed
230 if (WAVEFRONT_LENGTH(lo,hi) < 1000) return;
231 const int sample_k = lo + (hi-lo)/2;
232 const wf_offset_t sample_offset = offsets[sample_k];
233 if (sample_offset < 0) return; // Unlucky null in the middle
234 const int smax_sample = wf_compute_edit_worst_score(plen,tlen,sample_k,offsets[sample_k]);
235 const int smin_lo = wf_compute_edit_best_score(plen,tlen,lo,offsets[lo]);
236 const int smin_hi = wf_compute_edit_best_score(plen,tlen,hi,offsets[hi]);
237 if (smin_lo <= smax_sample && smin_hi <= smax_sample) return;
238 /*
239 * Suggested by Heng Li as an effective exact-prunning technique
240 * for sequences of very different length where some diagonals
241 * can be proven impossible to yield better alignments.
242 */
243 // Compute the best worst-case-alignment
244 int score_min_worst = INT_MAX;
245 int k;
246 for (k=lo;k<=hi;++k) {
247 const wf_offset_t offset = offsets[k];
248 if (offset < 0) continue; // Skip nulls
249 // Compute worst-alignment case
250 const int score_worst = wf_compute_edit_worst_score(plen,tlen,k,offset);
251 if (score_worst < score_min_worst) score_min_worst = score_worst;
252 }
253 // Compare against the best-case-alignment (Prune from bottom)
254 int lo_reduced = lo;
255 for (k=lo;k<=hi;++k) {
256 // Compute best-alignment case
257 const wf_offset_t offset = offsets[k];
258 const int score_best = wf_compute_edit_best_score(plen,tlen,k,offset);
259 // Compare best and worst
260 if (score_best <= score_min_worst) break;
261 ++lo_reduced;
262 }
263 wavefront->lo = lo_reduced;
264 // Compare against the best-case-alignment (Prune from top)
265 int hi_reduced = hi;
266 for (k=hi;k>lo_reduced;--k) {
267 // Compute best-alignment case
268 const wf_offset_t offset = offsets[k];
269 const int score_best = wf_compute_edit_best_score(plen,tlen,k,offset);
270 // Compare best and worst
271 if (score_best <= score_min_worst) break;
272 --hi_reduced;
273 }
274 wavefront->hi = hi_reduced;
275}

279void wavefront_compute_edit_dispatcher(
280 wavefront_aligner_t* const wf_aligner,
281 const int score,
282 wavefront_t* const wf_prev,
283 wavefront_t* const wf_curr,
284 const int lo,
285 const int hi) {
286 if (wf_aligner->wf_components.bt_piggyback) {
287 if (wf_aligner->penalties.distance_metric == indel) {
288 wavefront_compute_indel_idm_piggyback(wf_aligner,wf_prev,wf_curr,lo,hi,score);
289 } else {
290 wavefront_compute_edit_idm_piggyback(wf_aligner,wf_prev,wf_curr,lo,hi,score);
291 }
292 } else {
293 if (wf_aligner->penalties.distance_metric == indel) {
294 wavefront_compute_indel_idm(wf_aligner,wf_prev,wf_curr,lo,hi);
295 } else {
296 wavefront_compute_edit_idm(wf_aligner,wf_prev,wf_curr,lo,hi);
297 }
298 }

300void wavefront_compute_edit_dispatcher_omp(
301 wavefront_aligner_t* const wf_aligner,
302 wavefront_t* const wf_prev,
303 wavefront_t* const wf_curr,
304 const int lo,
305 const int hi,
306 const int score) {
307 // Parameters
308 const int num_threads = wavefront_compute_num_threads(wf_aligner,lo,hi);
309 // Multithreading dispatcher
310 if (num_threads == 1) {
311 // Compute next wavefront
312 wavefront_compute_edit_dispatcher(
313 wf_aligner,score,wf_prev,wf_curr,lo,hi);
314 } else {
315#ifdef WFA_PARALLEL
316 // Compute next wavefront in parallel
317 #pragma omp parallel num_threads(num_threads)
318 {
319 int t_lo, t_hi;
320 const int thread_id = omp_get_thread_num();
321 const int thread_num = omp_get_num_threads();
322 wavefront_compute_thread_limits(thread_id,thread_num,lo,hi,&t_lo,&t_hi);
323 wavefront_compute_edit_dispatcher(
324 wf_aligner,score,wf_prev,wf_curr,t_lo,t_hi);
325 }
326#endif
327 }

329void wavefront_compute_edit(
330 wavefront_aligner_t* const wf_aligner,
331 const int score) {
332 // Parameters
333 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
334 // Compute scores
335 int score_prev = score - 1;
336 int score_curr = score;
337 if (wf_components->memory_modular) { // Modular wavefront
338 score_prev = score_prev % wf_components->max_score_scope;
339 score_curr = score_curr % wf_components->max_score_scope;
340 if (wf_components->mwavefronts[score_curr]) { // Free
341 wavefront_slab_free(wf_aligner->wavefront_slab,wf_components->mwavefronts[score_curr]);
342 }
343 }
344 // Fetch previous wavefront, compute limits & initialize
345 wavefront_t* const wf_prev = wf_components->mwavefronts[score_prev];
346 const int lo = wf_prev->lo - 1;
347 const int hi = wf_prev->hi + 1;
348 // wf_components->historic_min_lo = min_lo;
349 // wf_components->historic_max_hi = max_hi;
350 wf_prev->offsets[lo-1] = WAVEFRONT_OFFSET_NULL;
351 wf_prev->offsets[lo] = WAVEFRONT_OFFSET_NULL;
352 wf_prev->offsets[hi] = WAVEFRONT_OFFSET_NULL;
353 wf_prev->offsets[hi+1] = WAVEFRONT_OFFSET_NULL;
354 // Allocate output wavefront
355 wavefront_t* const wf_curr = wavefront_slab_allocate(wf_aligner->wavefront_slab,lo-2,hi+2);
356 wf_components->mwavefronts[score_curr] = wf_curr;
357 wf_components->mwavefronts[score_curr]->lo = lo;
358 wf_components->mwavefronts[score_curr]->hi = hi;
359 // Compute Wavefront
360 wavefront_compute_edit_dispatcher_omp(wf_aligner,wf_prev,wf_curr,lo,hi,score);
361 // Offload backtrace (if necessary)
362 if (wf_components->bt_piggyback && score % PCIGAR_MAX_LENGTH == 0) {
363 wavefront_backtrace_offload_blocks_linear(
364 wf_aligner,wf_curr->offsets,wf_curr->bt_pcigar,wf_curr->bt_prev,lo,hi);
365 }
366 // Trim wavefront ends
367 wavefront_compute_trim_ends(wf_aligner,wf_curr);
368 if (wf_curr->null) wf_aligner->align_status.num_null_steps = INT_MAX;
369 // Exact pruning paths
370 if (wf_aligner->alignment_form.span == alignment_end2end &&
371 wf_aligner->penalties.distance_metric == edit) {
372 wavefront_compute_edit_exact_prune(wf_aligner,wf_curr);
373 }
