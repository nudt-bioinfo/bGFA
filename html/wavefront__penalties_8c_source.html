63void wavefront_penalties_set_linear(
64 wavefront_penalties_t* const wf_penalties,
65 linear_penalties_t* const linear_penalties) {
66 // Set distance model
67 wf_penalties->distance_metric = gap_linear;
68 // Check base penalties
69 if (linear_penalties->match > 0) {
70 fprintf(stderr,"[WFA::Penalties] Match score must be negative or zero (M=%d)\n",linear_penalties->match);
71 exit(1);
72 } else if (linear_penalties->mismatch <= 0 || linear_penalties->indel <= 0) {
73 fprintf(stderr,"[WFA::Penalties] Penalties (X=%d,D=%d,I=%d) must be (X>0,D>0,I>0)\n",
74 linear_penalties->mismatch,linear_penalties->indel,linear_penalties->indel);
75 exit(1);
76 }
77 // Set penalties (if needed, adjust using Eizenga's formula)
78 if (linear_penalties->match < 0) {
79 wf_penalties->match = linear_penalties->match;
80 wf_penalties->mismatch = 2*linear_penalties->mismatch - 2*linear_penalties->match;
81 wf_penalties->gap_opening1 = 2*linear_penalties->indel - linear_penalties->match;
82 } else {
83 wf_penalties->match = 0;
84 wf_penalties->mismatch = linear_penalties->mismatch;
85 wf_penalties->gap_opening1 = linear_penalties->indel;
86 }
87 // Set unused
88 wf_penalties->gap_extension1 = -1;
89 wf_penalties->gap_opening2 = -1;
90 wf_penalties->gap_extension2 = -1;
91 // Internals
92 wf_penalties->linear_penalties = *linear_penalties;
93 wf_penalties->internal_gap_e = linear_penalties->indel;
94}
95void wavefront_penalties_set_affine(
96 wavefront_penalties_t* const wf_penalties,
97 affine_penalties_t* const affine_penalties) {
98 // Set distance model
99 wf_penalties->distance_metric = gap_affine;
100 // Check base penalties
101 if (affine_penalties->match > 0) {
102 fprintf(stderr,"[WFA::Penalties] Match score must be negative or zero (M=%d)\n",affine_penalties->match);
103 exit(1);
104 } else if (affine_penalties->mismatch <= 0 ||
105 affine_penalties->gap_opening < 0 ||
106 affine_penalties->gap_extension <= 0) {
107 fprintf(stderr,"[WFA::Penalties] Penalties (X=%d,O=%d,E=%d) must be (X>0,O>=0,E>0)\n",
108 affine_penalties->mismatch,
109 affine_penalties->gap_opening,
110 affine_penalties->gap_extension);
111 exit(1);
112 }
113 // Set penalties (if needed, adjust using Eizenga's formula)
114 if (affine_penalties->match < 0) {
115 wf_penalties->match = affine_penalties->match;
116 wf_penalties->mismatch = 2*affine_penalties->mismatch - 2*affine_penalties->match;
117 wf_penalties->gap_opening1 = 2*affine_penalties->gap_opening;
118 wf_penalties->gap_extension1 = 2*affine_penalties->gap_extension - affine_penalties->match;
119 } else {
120 wf_penalties->match = 0;
121 wf_penalties->mismatch = affine_penalties->mismatch;
122 wf_penalties->gap_opening1 = affine_penalties->gap_opening;
123 wf_penalties->gap_extension1 = affine_penalties->gap_extension;
124 }
125 // Set unused
126 wf_penalties->gap_opening2 = -1;
127 wf_penalties->gap_extension2 = -1;
128 // Internals
129 wf_penalties->affine_penalties = *affine_penalties;
130 wf_penalties->internal_gap_e = affine_penalties->gap_extension;
131}

147 affine2p_penalties->mismatch,
148 affine2p_penalties->gap_opening1,
149 affine2p_penalties->gap_extension1,
150 affine2p_penalties->gap_opening2,
151 affine2p_penalties->gap_extension2);
152 exit(1);
153 }
154 // Set penalties (if needed, adjust using Eizenga's formula)
155 if (affine2p_penalties->match < 0) {
156 wf_penalties->match = affine2p_penalties->match;
157 wf_penalties->mismatch = 2*affine2p_penalties->mismatch - 2*wf_penalties->match;
158 wf_penalties->gap_opening1 = 2*affine2p_penalties->gap_opening1;
159 wf_penalties->gap_extension1 = 2*affine2p_penalties->gap_extension1 - affine2p_penalties->match;
160 wf_penalties->gap_opening2 = 2*affine2p_penalties->gap_opening2;
161 wf_penalties->gap_extension2 = 2*affine2p_penalties->gap_extension2 - affine2p_penalties->match;
162 } else {
163 wf_penalties->match = 0;
164 wf_penalties->mismatch = affine2p_penalties->mismatch;
165 wf_penalties->gap_opening1 = affine2p_penalties->gap_opening1;
166 wf_penalties->gap_extension1 = affine2p_penalties->gap_extension1;
167 wf_penalties->gap_opening2 = affine2p_penalties->gap_opening2;
168 wf_penalties->gap_extension2 = affine2p_penalties->gap_extension2;
169 }
170 // Internals
171 wf_penalties->affine2p_penalties = *affine2p_penalties;
172 wf_penalties->internal_gap_e = affine2p_penalties->gap_extension1;
173}

177void wavefront_penalties_print(
178 FILE* const stream,
179 wavefront_penalties_t* const wf_penalties) {
180 // Select penalties mode
181 switch (wf_penalties->distance_metric) {
182 case indel:
183 fprintf(stream,"(Indel,0,inf,1)");
184 break;
185 case edit:
186 fprintf(stream,"(Edit,0,1,1)");
187 break;
188 case gap_linear:
189 fprintf(stream,"(GapLinear,%d,%d,%d)",
190 wf_penalties->linear_penalties.match,
191 wf_penalties->linear_penalties.mismatch,
192 wf_penalties->linear_penalties.indel);
193 break;
194 case gap_affine:
195 fprintf(stream,"(GapAffine,%d,%d,%d,%d)",
196 wf_penalties->affine_penalties.match,
197 wf_penalties->affine_penalties.mismatch,
198 wf_penalties->affine_penalties.gap_opening,
199 wf_penalties->affine_penalties.gap_extension);
200 break;
201 case gap_affine_2p:
202 fprintf(stream,"(GapAffine2p,%d,%d,%d,%d,%d,%d)",
203 wf_penalties->affine2p_penalties.match,
204 wf_penalties->affine2p_penalties.mismatch,
205 wf_penalties->affine2p_penalties.gap_opening1,
206 wf_penalties->affine2p_penalties.gap_extension1,
207 wf_penalties->affine2p_penalties.gap_opening2,
208 wf_penalties->affine2p_penalties.gap_extension2);
209 break;
210 default:
211 break;
212 }
