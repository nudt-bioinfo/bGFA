35/*
36 * Gap-affine 2-piece backtrace using dynamic-programming matrix
37 */
38void affine2p_dp_traceback(
39 affine2p_matrix_t* const matrix,
40 affine2p_penalties_t* const penalties,
41 const int pattern_length,
42 const int text_length,
43 cigar_t* const cigar) {
44 // Parameters
45 affine2p_cell_t** const dp = matrix->columns;
46 char* const operations = cigar->operations;
47 cigar->end_offset = cigar->max_operations;
48 int op_sentinel = cigar->end_offset - 1;
49 // Compute traceback
50 int h = text_length;
51 int v = pattern_length;
52 affine2p_matrix_type matrix_type = affine2p_matrix_M;
53 while (h>0 && v>0) {
54 switch (matrix_type) {
55 case affine2p_matrix_D1:
56 // Traceback D1-matrix
57 operations[op_sentinel--] = 'D';
58 if (dp[h][v].D1 != dp[h][v-1].D1 + penalties->gap_extension1) {
59 matrix_type = affine2p_matrix_M;
60 }
61 --v;
62 break;
63 case affine2p_matrix_D2:
64 // Traceback D2-matrix
65 operations[op_sentinel--] = 'D';
66 if (dp[h][v].D2 != dp[h][v-1].D2 + penalties->gap_extension2) {
67 matrix_type = affine2p_matrix_M;
68 }
69 --v;
70 break;
71 case affine2p_matrix_I1:
72 // Traceback I1-matrix
73 operations[op_sentinel--] = 'I';
74 if (dp[h][v].I1 != dp[h-1][v].I1 + penalties->gap_extension1) {
75 matrix_type = affine2p_matrix_M;
76 }
77 --h;
78 break;
79 case affine2p_matrix_I2:
80 // Traceback I2-matrix
81 operations[op_sentinel--] = 'I';
82 if (dp[h][v].I2 != dp[h-1][v].I2 + penalties->gap_extension2) {
83 matrix_type = affine2p_matrix_M;
84 }
85 --h;
86 break;
87 case affine2p_matrix_M:
88 // Traceback M-matrix
89 if (dp[h][v].M == dp[h-1][v-1].M + penalties->mismatch) {
90 operations[op_sentinel--] = 'X';
91 --h; --v;
92 } else if (dp[h][v].M == dp[h][v].D2) {
93 matrix_type = affine2p_matrix_D2;
94 } else if (dp[h][v].M == dp[h][v].D1) {
95 matrix_type = affine2p_matrix_D1;
96 } else if (dp[h][v].M == dp[h][v].I2) {
97 matrix_type = affine2p_matrix_I2;
98 } else if (dp[h][v].M == dp[h][v].I1) {
99 matrix_type = affine2p_matrix_I1;
100 } else if (dp[h][v].M == dp[h-1][v-1].M + penalties->match) {
101 operations[op_sentinel--] = 'M';
102 --h; --v;
103 } else {
104 fprintf(stderr,"Affine-2P Backtrace. No backtrace operation found\n");
105 exit(1);
106 }
107 break;
108 }
109 }
110 while (h>0) {operations[op_sentinel--] = 'I'; --h;}
111 while (v>0) {operations[op_sentinel--] = 'D'; --v;}
112 cigar->begin_offset = op_sentinel+1;
113}
114/*
115 * Gap-affine 2-piece alignment computation using dynamic-programming matrix
116 */
117void affine2p_dp_align(
118 affine2p_matrix_t* const matrix,
119 affine2p_penalties_t* const penalties,
120 const char* const pattern,
121 const int pattern_length,
122 const char* const text,
123 const int text_length,
124 cigar_t* const cigar) {
125 // Parameters
126 affine2p_cell_t** const dp = matrix->columns;
127 int h, v;
128 // Initial conditions
129 dp[0][0].I1 = AFFINE2P_SCORE_MAX;
130 dp[0][0].I2 = AFFINE2P_SCORE_MAX;
131 dp[0][0].D1 = AFFINE2P_SCORE_MAX;
132 dp[0][0].D2 = AFFINE2P_SCORE_MAX;
133 dp[0][0].M = 0;
134 for (v=1;v<=pattern_length;++v) { // Initialize first column
135 dp[0][v].I1 = AFFINE2P_SCORE_MAX;
136 dp[0][v].I2 = AFFINE2P_SCORE_MAX;
137 dp[0][v].D1 = penalties->gap_opening1 + v*penalties->gap_extension1;
138 dp[0][v].D2 = penalties->gap_opening2 + v*penalties->gap_extension2;
139 dp[0][v].M = MIN(dp[0][v].D1,dp[0][v].D2);
140 }
141 for (h=1;h<=text_length;++h) { // Initialize first row
142 dp[h][0].I1 = penalties->gap_opening1 + h*penalties->gap_extension1;
143 dp[h][0].I2 = penalties->gap_opening2 + h*penalties->gap_extension2;
144 dp[h][0].D1 = AFFINE2P_SCORE_MAX;
145 dp[h][0].D2 = AFFINE2P_SCORE_MAX;
146 dp[h][0].M = MIN(dp[h][0].I1,dp[h][0].I2);
147 }
148 // Compute DP
149 for (h=1;h<=text_length;++h) {
150 for (v=1;v<=pattern_length;++v) {
151 // Update DP.I
152 const int ins1_new = dp[h-1][v].M + penalties->gap_opening1 + penalties->gap_extension1;
153 const int ins1_ext = dp[h-1][v].I1 + penalties->gap_extension1;
154 const int ins1 = MIN(ins1_new,ins1_ext);
155 dp[h][v].I1 = ins1;
156 const int ins2_new = dp[h-1][v].M + penalties->gap_opening2 + penalties->gap_extension2;
157 const int ins2_ext = dp[h-1][v].I2 + penalties->gap_extension2;
158 const int ins2 = MIN(ins2_new,ins2_ext);
159 dp[h][v].I2 = ins2;
160 // Update DP.D
161 const int del1_new = dp[h][v-1].M + penalties->gap_opening1 + penalties->gap_extension1;
162 const int del1_ext = dp[h][v-1].D1 + penalties->gap_extension1;
163 const int del1 = MIN(del1_new,del1_ext);
164 dp[h][v].D1 = del1;
165 const int del2_new = dp[h][v-1].M + penalties->gap_opening2 + penalties->gap_extension2;
166 const int del2_ext = dp[h][v-1].D2 + penalties->gap_extension2;
167 const int del2 = MIN(del2_new,del2_ext);
168 dp[h][v].D2 = del2;
169 // Update DP.M
170 const int m_match = dp[h-1][v-1].M + ((pattern[v-1]==text[h-1]) ? penalties->match : penalties->mismatch);
171 const int min_dels = MIN(del1,del2);
172 const int min_inss = MIN(ins1,ins2);
173 const int min_gaps = MIN(min_dels,min_inss);
174 dp[h][v].M = MIN(m_match,min_gaps);
175 }
176 }
177 // Compute traceback
178 affine2p_dp_traceback(matrix,penalties,pattern_length,text_length,cigar);
179// // DEBUG
180// affine2p_matrix_print(stderr,matrix,pattern,text);
181}
