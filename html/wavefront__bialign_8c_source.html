55void wavefront_bialign_debug(
56 wf_bialign_breakpoint_t* const breakpoint,
57 const int align_level) {
58 // Parameters
59 const int breakpoint_h = WAVEFRONT_H(breakpoint->k_forward,breakpoint->offset_forward);
60 const int breakpoint_v = WAVEFRONT_V(breakpoint->k_forward,breakpoint->offset_forward);
61 // Prinf debug info
62 fprintf(stderr,"[WFA::BiAlign][Recursion=%d] ",align_level);
63 int i; for (i=0;i<align_level;++i) fprintf(stderr," ");
64 fprintf(stderr,"Breakpoint at (h,v,score,comp) = (%d,%d,%d,",
65 breakpoint_h,breakpoint_v,breakpoint->score);
66 switch (breakpoint->component) {
67 case affine2p_matrix_M: fprintf(stderr,"M"); break;
68 case affine2p_matrix_I1: fprintf(stderr,"I1"); break;
69 case affine2p_matrix_I2: fprintf(stderr,"I2"); break;
70 case affine2p_matrix_D1: fprintf(stderr,"D1"); break;
71 case affine2p_matrix_D2: fprintf(stderr,"D2"); break;
72 default: fprintf(stderr,"?"); break;
73 }
74 fprintf(stderr,")\n");
75}
76/*
77 * Init
78 */
79void wavefront_bialign_init(
80 wavefront_bialigner_t* const bialigner,
81 const distance_metric_t distance_metric,
82 alignment_form_t* const form,
83 const affine2p_matrix_type component_begin,
84 const affine2p_matrix_type component_end,
85 const int align_level,
86 const int verbose) {
87 // Parameters
88 wavefront_aligner_t* const wf_forward = bialigner->wf_forward;
89 wavefront_aligner_t* const wf_reverse = bialigner->wf_reverse;
90 // Configure WF-compute function
91 switch (distance_metric) {
92 case indel:
93 case edit:
94 bialigner->wf_align_compute = &wavefront_compute_edit;
95 break;
96 case gap_linear:
97 bialigner->wf_align_compute = &wavefront_compute_linear;
98 break;
99 case gap_affine:
100 bialigner->wf_align_compute = &wavefront_compute_affine;
101 break;
102 case gap_affine_2p:
103 bialigner->wf_align_compute = &wavefront_compute_affine2p;
104 break;
105 default:
106 fprintf(stderr,"[WFA] Distance function not implemented\n");
107 exit(1);
108 break;
109 }
110 // Initialize wavefront-aligner (forward)
111 alignment_span_t span_forward =
112 (form->pattern_begin_free > 0 || form->text_begin_free > 0) ?
113 alignment_endsfree : alignment_end2end;
114 alignment_form_t form_forward = {
115 .span = span_forward,
116 .pattern_begin_free = form->pattern_begin_free,
117 .pattern_end_free = 0,
118 .text_begin_free = form->text_begin_free,
119 .text_end_free = 0,
120 };
121 wf_forward->alignment_form = form_forward;
122 wf_forward->component_begin = component_begin;
123 wf_forward->component_end = component_end;
124 wavefront_aligner_init(wf_forward,align_level);
125 // Initialize wavefront-aligner (reverse)
126 alignment_span_t span_reverse =
127 (form->pattern_end_free > 0 || form->text_end_free > 0) ?
128 alignment_endsfree : alignment_end2end;
129 alignment_form_t form_reverse = {
130 .span = span_reverse,
131 .pattern_begin_free = form->pattern_end_free,
132 .pattern_end_free = 0,
133 .text_begin_free = form->text_end_free,
134 .text_end_free = 0,
135 };
136 wf_reverse->alignment_form = form_reverse;
137 wf_reverse->component_begin = component_end;
138 wf_reverse->component_end = component_begin;
139 wavefront_aligner_init(wf_reverse,align_level);
140 // Plot
141 const bool plot_enabled = (wf_forward->plot != NULL);
142 if (plot_enabled) {
143 wavefront_plot(wf_forward,0,align_level);
144 wavefront_plot(wf_reverse,0,align_level);
145 }
146 // DEBUG
147 if (verbose >= 2) {
148 wavefront_debug_begin(wf_forward);
149 wavefront_debug_begin(wf_reverse);
150 }
151}
152/*
153 * Bidirectional Alignment (base cases)
154 */
155int wavefront_bialign_base(
156 wavefront_aligner_t* const wf_aligner,
157 alignment_form_t* const form,
158 const affine2p_matrix_type component_begin,
159 const affine2p_matrix_type component_end,
160 const int align_level) {
161 // Parameters
162 wavefront_aligner_t* const wf_base = wf_aligner->bialigner->wf_base;
163 const int verbose = wf_base->system.verbose;
164 // Configure
165 wf_base->alignment_form = *form;
166 wavefront_unialign_init(wf_base,component_begin,component_end);
167 // DEBUG
168 if (verbose >= 2) wavefront_debug_begin(wf_base);
169 // Wavefront align sequences
170 wavefront_unialign(wf_base);
171 // DEBUG
172 if (verbose >= 2) {
173 wavefront_debug_end(wf_base);
174 wavefront_debug_check_correct(wf_base);
175 }
176 // Append CIGAR
177 cigar_append_forward(wf_aligner->cigar,wf_base->cigar);
178 // Set status and return
179 const int align_status = wf_base->align_status.status;
180 if (align_status == WF_STATUS_ALG_COMPLETED) {
181 return WF_STATUS_OK;
182 } else {
183 return WF_STATUS_UNATTAINABLE;
184 }
185}
186/*
187 * Bidirectional check breakpoints
188 */
189void wavefront_bialign_breakpoint_indel2indel(
190 wavefront_aligner_t* const wf_aligner,
191 const bool breakpoint_forward,
192 const int score_0,
193 const int score_1,
194 wavefront_t* const dwf_0,
195 wavefront_t* const dwf_1,
196 const affine2p_matrix_type component,
197 wf_bialign_breakpoint_t* const breakpoint) {
198 // Parameters
199 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
200 const int text_length = sequences->text_length;
201 const int pattern_length = sequences->pattern_length;
202 const int gap_open =
203 (component==affine2p_matrix_I1 || component==affine2p_matrix_D1) ?
204 wf_aligner->penalties.gap_opening1 : wf_aligner->penalties.gap_opening2;
205 // Check wavefronts overlapping
206 const int lo_0 = dwf_0->lo;
207 const int hi_0 = dwf_0->hi;
208 const int lo_1 = WAVEFRONT_K_INVERSE(dwf_1->hi,pattern_length,text_length);
209 const int hi_1 = WAVEFRONT_K_INVERSE(dwf_1->lo,pattern_length,text_length);
210 if (hi_1 < lo_0 || hi_0 < lo_1) return;
211 // Compute overlapping interval
212 const int min_hi = MIN(hi_0,hi_1);
213 const int max_lo = MAX(lo_0,lo_1);
214 int k_0;
215 for (k_0=max_lo;k_0<=min_hi;k_0++) {
216 const int k_1 = WAVEFRONT_K_INVERSE(k_0,pattern_length,text_length);
217 // Fetch offsets
218 const wf_offset_t doffset_0 = dwf_0->offsets[k_0];
219 const wf_offset_t doffset_1 = dwf_1->offsets[k_1];
220 const int dh_0 = WAVEFRONT_H(k_0,doffset_0);
221 const int dh_1 = WAVEFRONT_H(k_1,doffset_1);
222 // Check breakpoint d2d
223 if (dh_0 + dh_1 >= text_length && score_0 + score_1 - gap_open < breakpoint->score) {
224 if (breakpoint_forward) {
225 // Check out-of-bounds coordinates
226 const int v = WAVEFRONT_V(k_0,dh_0);
227 const int h = WAVEFRONT_H(k_0,dh_0);
228 if (v > pattern_length || h > text_length) continue;
229 // Set breakpoint
230 breakpoint->score_forward = score_0;
231 breakpoint->score_reverse = score_1;
232 breakpoint->k_forward = k_0;
233 breakpoint->k_reverse = k_1;
234 breakpoint->offset_forward = dh_0;
235 breakpoint->offset_reverse = dh_1;
236 } else {
237 // Check out-of-bounds coordinates
238 const int v = WAVEFRONT_V(k_1,dh_1);
239 const int h = WAVEFRONT_H(k_1,dh_1);
240 if (v > pattern_length || h > text_length) continue;
241 // Set breakpoint
242 breakpoint->score_forward = score_1;
243 breakpoint->score_reverse = score_0;
244 breakpoint->k_forward = k_1;
245 breakpoint->k_reverse = k_0;
246 breakpoint->offset_forward = dh_1;
247 breakpoint->offset_reverse = dh_0;
248 }
249 breakpoint->score = score_0 + score_1 - gap_open;
250 breakpoint->component = component;
251 // wavefront_bialign_debug(breakpoint,-1); // DEBUG
252 // No need to keep searching
253 return;
254 }
255 }

257void wavefront_bialign_breakpoint_m2m(
258 wavefront_aligner_t* const wf_aligner,
259 const bool breakpoint_forward,
260 const int score_0,
261 const int score_1,
262 wavefront_t* const mwf_0,
263 wavefront_t* const mwf_1,
264 wf_bialign_breakpoint_t* const breakpoint) {
265 // Parameters
266 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
267 const int text_length = sequences->text_length;
268 const int pattern_length = sequences->pattern_length;
269 // Check wavefronts overlapping
270 const int lo_0 = mwf_0->lo;
271 const int hi_0 = mwf_0->hi;
272 const int lo_1 = WAVEFRONT_K_INVERSE(mwf_1->hi,pattern_length,text_length);
273 const int hi_1 = WAVEFRONT_K_INVERSE(mwf_1->lo,pattern_length,text_length);
274 if (hi_1 < lo_0 || hi_0 < lo_1) return;
275 // Compute overlapping interval
276 const int min_hi = MIN(hi_0,hi_1);
277 const int max_lo = MAX(lo_0,lo_1);
278 int k_0;
279 for (k_0=max_lo;k_0<=min_hi;k_0++) {
280 const int k_1 = WAVEFRONT_K_INVERSE(k_0,pattern_length,text_length);
281 // Fetch offsets
282 const wf_offset_t moffset_0 = mwf_0->offsets[k_0];
283 const wf_offset_t moffset_1 = mwf_1->offsets[k_1];
284 const int mh_0 = WAVEFRONT_H(k_0,moffset_0);
285 const int mh_1 = WAVEFRONT_H(k_1,moffset_1);
286 // Check breakpoint m2m
287 if (mh_0 + mh_1 >= text_length && score_0 + score_1 < breakpoint->score) {
288 if (breakpoint_forward) {
289 breakpoint->score_forward = score_0;
290 breakpoint->score_reverse = score_1;
291 breakpoint->k_forward = k_0;
292 breakpoint->k_reverse = k_1;
293 breakpoint->offset_forward = moffset_0;
294 breakpoint->offset_reverse = moffset_1;
295 } else {
296 breakpoint->score_forward = score_1;
297 breakpoint->score_reverse = score_0;
298 breakpoint->k_forward = k_1;
299 breakpoint->k_reverse = k_0;
300 breakpoint->offset_forward = moffset_1;
301 breakpoint->offset_reverse = moffset_0;
302 }
303 breakpoint->score = score_0 + score_1;
304 breakpoint->component = affine2p_matrix_M;
305 // wavefront_bialign_debug(breakpoint,-1); // DEBUG
306 // No need to keep searching
307 return;
308 }
309 }

311/*
312 * Bidirectional find overlaps
313 */
314void wavefront_bialign_overlap(
315 wavefront_aligner_t* const wf_aligner_0,
316 wavefront_aligner_t* const wf_aligner_1,
317 const int score_0,
318 const int score_1,
319 const bool breakpoint_forward,
320 wf_bialign_breakpoint_t* const breakpoint) {
321 // Parameters
322 const int max_score_scope = wf_aligner_0->wf_components.max_score_scope;
323 const distance_metric_t distance_metric = wf_aligner_0->penalties.distance_metric;
324 const int gap_opening1 = wf_aligner_0->penalties.gap_opening1;
325 const int gap_opening2 = wf_aligner_0->penalties.gap_opening2;
326 // Fetch wavefronts-0
327 const int score_mod_0 = score_0 % max_score_scope;
328 wavefront_t* const mwf_0 = wf_aligner_0->wf_components.mwavefronts[score_mod_0];
329 if (mwf_0 == NULL) return;
330 wavefront_t* d1wf_0 = NULL, *i1wf_0 = NULL;
331 if (distance_metric >= gap_affine) {
332 d1wf_0 = wf_aligner_0->wf_components.d1wavefronts[score_mod_0];
333 i1wf_0 = wf_aligner_0->wf_components.i1wavefronts[score_mod_0];
334 }
335 wavefront_t* d2wf_0 = NULL, *i2wf_0 = NULL;
336 if (distance_metric == gap_affine_2p) {
337 d2wf_0 = wf_aligner_0->wf_components.d2wavefronts[score_mod_0];
338 i2wf_0 = wf_aligner_0->wf_components.i2wavefronts[score_mod_0];
339 }
340 // Traverse all scores-1
341 int i;
342 for (i=0;i<max_score_scope;++i) {
343 // Compute score
344 const int score_i = score_1 - i;
345 if (score_i < 0) break;
346 const int score_mod_i = score_i % max_score_scope;
347 // Check I2/D2-breakpoints (gap_affine_2p)
348 if (distance_metric == gap_affine_2p) {
349 if (score_0 + score_i - gap_opening2 >= breakpoint->score) continue;
350 // Check breakpoint d2d
351 wavefront_t* const d2wf_1 = wf_aligner_1->wf_components.d2wavefronts[score_mod_i];
352 if (d2wf_0 != NULL && d2wf_1 != NULL) {
353 wavefront_bialign_breakpoint_indel2indel(
354 wf_aligner_0,breakpoint_forward,score_0,score_i,
355 d2wf_0,d2wf_1,affine2p_matrix_D2,breakpoint);
356 }
357 // Check breakpoint i2i
358 wavefront_t* const i2wf_1 = wf_aligner_1->wf_components.i2wavefronts[score_mod_i];
359 if (i2wf_0 != NULL && i2wf_1 != NULL) {
360 wavefront_bialign_breakpoint_indel2indel(
361 wf_aligner_0,breakpoint_forward,score_0,score_i,
362 i2wf_0,i2wf_1,affine2p_matrix_I2,breakpoint);
363 }
364 }
365 // Check I1/D1-breakpoints (gap_affine)
366 if (distance_metric >= gap_affine) {
367 if (score_0 + score_i - gap_opening1 >= breakpoint->score) continue;
368 // Check breakpoint d2d
369 wavefront_t* const d1wf_1 = wf_aligner_1->wf_components.d1wavefronts[score_mod_i];
370 if (d1wf_0 != NULL && d1wf_1 != NULL) {
371 wavefront_bialign_breakpoint_indel2indel(
372 wf_aligner_0,breakpoint_forward,score_0,score_i,
373 d1wf_0,d1wf_1,affine2p_matrix_D1,breakpoint);
374 }
375 // Check breakpoint i2i
376 wavefront_t* const i1wf_1 = wf_aligner_1->wf_components.i1wavefronts[score_mod_i];
377 if (i1wf_0 != NULL && i1wf_1 != NULL) {
378 wavefront_bialign_breakpoint_indel2indel(
379 wf_aligner_0,breakpoint_forward,score_0,score_i,
380 i1wf_0,i1wf_1,affine2p_matrix_I1,breakpoint);
381 }
382 }
383 // Check M-breakpoints (indel, edit, gap-linear)
384 if (score_0 + score_i >= breakpoint->score) continue;
385 wavefront_t* const mwf_1 = wf_aligner_1->wf_components.mwavefronts[score_mod_i];
386 if (mwf_1 != NULL) {
387 wavefront_bialign_breakpoint_m2m(
388 wf_aligner_0,breakpoint_forward,
389 score_0,score_i,mwf_0,mwf_1,breakpoint);
390 }
391 }

393/*
394 * Bidirectional breakpoint detection
395 */
396int wavefront_bialign_overlap_gopen_adjust(
397 wavefront_aligner_t* const wf_aligner,
398 const distance_metric_t distance_metric) {
399 switch (distance_metric) {
400 case gap_affine:
401 return wf_aligner->penalties.gap_opening1;
402 case gap_affine_2p:
403 return MAX(wf_aligner->penalties.gap_opening1,wf_aligner->penalties.gap_opening2);
404 case indel:
405 case edit:
406 case gap_linear:
407 default:
408 return 0;
409 }
410}
411int wavefront_bialign_find_breakpoint(
412 wavefront_bialigner_t* const bialigner,
413 const distance_metric_t distance_metric,
414 alignment_form_t* const form,
415 const affine2p_matrix_type component_begin,
416 const affine2p_matrix_type component_end,
417 wf_bialign_breakpoint_t* const breakpoint,
418 const int align_level) {
419 // Parameters
420 wavefront_aligner_t* const wf_forward = bialigner->wf_forward;
421 wavefront_aligner_t* const wf_reverse = bialigner->wf_reverse;
422 alignment_system_t* const system = &wf_forward->system;
423 const bool plot_enabled = (wf_forward->plot != NULL);
424 const int verbose = system->verbose;
425 // Init bialignment
426 wavefront_bialign_init(bialigner,distance_metric,form,component_begin,component_end,align_level,verbose);
427 // Sequences
428 wavefront_sequences_t* const sequences = &wf_forward->sequences;
429 const int text_length = sequences->text_length;
430 const int pattern_length = sequences->pattern_length;
431 // Operators
432 void (*wf_align_compute)(wavefront_aligner_t* const,const int) = bialigner->wf_align_compute;
433 // Parameters
434 const int max_alignment_steps = wf_forward->system.max_alignment_steps;
435 const int max_antidiagonal = DPMATRIX_ANTIDIAGONAL(pattern_length,text_length) - 1; // Note: Even removing -1
436 int score_forward = 0, score_reverse = 0, forward_max_ak = 0, reverse_max_ak = 0;
437 bool reachability_quit;
438 // Prepare and perform first bialignment step
439 breakpoint->score = INT_MAX;
440 reachability_quit = wavefront_extend_end2end_max(wf_forward,score_forward,&forward_max_ak);
441 if (reachability_quit) return wf_forward->align_status.status;
442 reachability_quit = wavefront_extend_end2end_max(wf_reverse,score_reverse,&reverse_max_ak);
443 if (reachability_quit) return wf_reverse->align_status.status;
444 // Compute wavefronts of increasing score until both wavefronts overlap
445 int max_ak = 0;
446 bool last_wf_forward = false;
447 while (true) {
448 // Check close-to-collision
449 if (forward_max_ak + reverse_max_ak >= max_antidiagonal) break;
450 /*
451 * Compute next wavefront (Forward)
452 */
453 ++score_forward;
454 (*wf_align_compute)(wf_forward,score_forward);
455 if (plot_enabled) wavefront_plot(wf_forward,score_forward,align_level); // Plot
456 // Extend
457 reachability_quit = wavefront_extend_end2end_max(wf_forward,score_forward,&max_ak);
458 if (forward_max_ak < max_ak) forward_max_ak = max_ak;
459 last_wf_forward = true;
460 // Check end-reached and close-to-collision
461 if (reachability_quit) return wf_forward->align_status.status;
462 if (forward_max_ak + reverse_max_ak >= max_antidiagonal) break;
463 /*
464 * Compute next wavefront (Reverse)
465 */
466 ++score_reverse;
467 (*wf_align_compute)(wf_reverse,score_reverse);
468 if (plot_enabled) wavefront_plot(wf_reverse,score_reverse,align_level); // Plot
469 // Extend
470 reachability_quit = wavefront_extend_end2end_max(wf_reverse,score_reverse,&max_ak);
471 if (reverse_max_ak < max_ak) reverse_max_ak = max_ak;
472 last_wf_forward = false;
473 // Check end-reached and max-steps-reached
474 if (reachability_quit) return wf_reverse->align_status.status;
475 if (score_reverse + score_forward >= max_alignment_steps) return WF_STATUS_MAX_STEPS_REACHED;
476 // DEBUG
477 if (verbose >= 3 && score_forward % system->probe_interval_global == 0) {
478 wavefront_unialign_print_status(stderr,wf_forward,score_forward);
479 }
480 }
481 // Advance until overlap is found
482 const int max_score_scope = wf_forward->wf_components.max_score_scope;
483 const int gap_opening = wavefront_bialign_overlap_gopen_adjust(wf_forward,distance_metric);
484 while (true) {
485 if (last_wf_forward) {
486 // Check overlapping wavefronts
487 const int min_score_reverse = (score_reverse > max_score_scope-1) ? score_reverse - (max_score_scope-1) : 0;
488 if (score_forward + min_score_reverse - gap_opening >= breakpoint->score) break; // Done!
489 wavefront_bialign_overlap(wf_forward,wf_reverse,score_forward,score_reverse,true,breakpoint);
490 /*
491 * Compute next wavefront (Reverse)
492 */
493 ++score_reverse;
494 (*wf_align_compute)(wf_reverse,score_reverse);
495 if (plot_enabled) wavefront_plot(wf_reverse,score_reverse,align_level); // Plot
496 // Extend & check end-reached
497 reachability_quit = wavefront_extend_end2end(wf_reverse,score_reverse);
498 if (reachability_quit) return wf_reverse->align_status.status;
499 }
500 // Check overlapping wavefronts
501 const int min_score_forward = (score_forward > max_score_scope-1) ? score_forward - (max_score_scope-1) : 0;
502 if (min_score_forward + score_reverse - gap_opening >= breakpoint->score) break; // Done!
503 wavefront_bialign_overlap(wf_reverse,wf_forward,score_reverse,score_forward,false,breakpoint);
504 /*
505 * Compute next wavefront (Forward)
506 */
507 ++score_forward;
508 (*wf_align_compute)(wf_forward,score_forward);
509 if (plot_enabled) wavefront_plot(wf_forward,score_forward,align_level); // Plot
510 // Extend & check end-reached/max-steps-reached
511 reachability_quit = wavefront_extend_end2end(wf_forward,score_forward);
512 if (reachability_quit) return wf_forward->align_status.status;
513 if (score_reverse + score_forward >= max_alignment_steps) return WF_STATUS_MAX_STEPS_REACHED;
514 // Enable always
515 last_wf_forward = true;
516 }
517 // Return OK
518 return WF_STATUS_OK;
519}
520int wavefront_bialign_find_breakpoint_exception(
521 wavefront_aligner_t* const wf_aligner,
522 alignment_form_t* const form,
523 const affine2p_matrix_type component_begin,
524 const affine2p_matrix_type component_end,
525 const int align_level,
526 const int align_status) {
527 // Breakpoint was not found, check end reached
528 if (align_status == WF_STATUS_END_REACHED) {
529 wavefront_aligner_t* const wf_forward = wf_aligner->bialigner->wf_forward;
530 wavefront_aligner_t* const wf_reverse = wf_aligner->bialigner->wf_reverse;
531 // Retrieve score when end was reached
532 int score_reached;
533 if (wf_forward->align_status.status == WF_STATUS_END_REACHED) {
534 score_reached = wf_forward->align_status.score;
535 } else {
536 score_reached = wf_reverse->align_status.score;
537 }
538 // Fallback if possible

542 return WF_STATUS_END_UNREACHABLE; // To no avail
543 }
544 } else { // Other unrecoverable conditions

581int wavefront_bialign_alignment(
582 wavefront_aligner_t* const wf_aligner,
583 alignment_form_t* const form,
584 const affine2p_matrix_type component_begin,
585 const affine2p_matrix_type component_end,
586 const int score_remaining,
587 const int align_level) {
588 // Parameters
589 wavefront_sequences_t* const sequences = &wf_aligner->bialigner->wf_forward->sequences;
590 const int pattern_begin = sequences->pattern_begin;
591 const int pattern_end = sequences->pattern_begin + sequences->pattern_length;
592 const int text_begin = sequences->text_begin;
593 const int text_end = sequences->text_begin + sequences->text_length;
594 const int pattern_length = pattern_end - pattern_begin;
595 const int text_length = text_end - text_begin;
596 // Trivial cases
597 if (text_length == 0) {
598 cigar_append_deletion(wf_aligner->cigar,pattern_length);
599 return WF_STATUS_OK;
600 } else if (pattern_length == 0) {
601 cigar_append_insertion(wf_aligner->cigar,text_length);
602 return WF_STATUS_OK;
603 } else if (score_remaining <= WF_BIALIGN_FALLBACK_MIN_SCORE) {
604 // Fall back to regular WFA
605 return wavefront_bialign_base(wf_aligner,form,
606 component_begin,component_end,align_level);
607 }
608 // Find breakpoint in the alignment
609 wf_bialign_breakpoint_t breakpoint;
610 int align_status = wavefront_bialign_find_breakpoint(
611 wf_aligner->bialigner,wf_aligner->penalties.distance_metric,
612 form,component_begin,component_end,&breakpoint,align_level);
613 // DEBUG
614 if (wf_aligner->system.verbose >= 2) {
615 wf_aligner->bialigner->wf_forward->align_status.status = align_status;
616 wf_aligner->bialigner->wf_reverse->align_status.status = align_status;
617 wavefront_debug_end(wf_aligner->bialigner->wf_forward);
618 wavefront_debug_end(wf_aligner->bialigner->wf_reverse);
619 }
620 // Check status
621 if (align_status != WF_STATUS_OK) {
622 return wavefront_bialign_find_breakpoint_exception(
623 wf_aligner,form,component_begin,component_end,align_level,align_status);
624 }
625 // Breakpoint found
626 const int breakpoint_h = WAVEFRONT_H(breakpoint.k_forward,breakpoint.offset_forward);
627 const int breakpoint_v = WAVEFRONT_V(breakpoint.k_forward,breakpoint.offset_forward);
628 // DEBUG
629 if (wf_aligner->system.verbose >= 3) wavefront_bialign_debug(&breakpoint,align_level);
630 // Align half_0
631 alignment_form_t form_0;
632 wavefront_bialigner_set_sequences_bounds(wf_aligner->bialigner,
633 pattern_begin,pattern_begin+breakpoint_v,
634 text_begin,text_begin+breakpoint_h);
635 wavefront_bialign_init_half_0(form,&form_0);
636 align_status = wavefront_bialign_alignment(wf_aligner,
637 &form_0,component_begin,breakpoint.component,
638 breakpoint.score_forward,align_level+1);
639 if (align_status != WF_STATUS_OK) return align_status;
640 // Align half_1
641 alignment_form_t form_1;
642 wavefront_bialigner_set_sequences_bounds(wf_aligner->bialigner,
643 pattern_begin+breakpoint_v,pattern_end,
644 text_begin+breakpoint_h,text_end);
645 wavefront_bialign_init_half_1(form,&form_1);
646 align_status = wavefront_bialign_alignment(wf_aligner,
647 &form_1,breakpoint.component,component_end,
648 breakpoint.score_reverse,align_level+1);
649 if (align_status != WF_STATUS_OK) return align_status;
650 // Set score (Strictly speaking, only needed at level-0)
651 if (align_level == 0) {
652 cigar_t* const cigar = wf_aligner->cigar;
653 cigar->score = wavefront_compute_classic_score(wf_aligner,pattern_length,text_length,breakpoint.score);
654 cigar->end_v = pattern_length;
655 cigar->end_h = text_length;
656 }
657 return WF_STATUS_OK; // All good
658}
659/*
660 * Bidirectional Score-only
661 */
662int wavefront_bialign_compute_score(
663 wavefront_aligner_t* const wf_aligner) {
664 // Parameters
665 wavefront_aligner_t* const wf_forward = wf_aligner->bialigner->wf_forward;
666 wavefront_aligner_t* const wf_reverse = wf_aligner->bialigner->wf_reverse;
667 wavefront_sequences_t* const sequences = &wf_forward->sequences;
668 const int text_length = sequences->text_length;
669 const int pattern_length = sequences->pattern_length;
670 // Clear cigar
671 cigar_clear(wf_aligner->cigar);
672 // Find breakpoint in the alignment
673 wf_bialign_breakpoint_t breakpoint;
674 const int align_status = wavefront_bialign_find_breakpoint(wf_aligner->bialigner,
675 wf_aligner->penalties.distance_metric,&wf_aligner->alignment_form,
676 affine_matrix_M,affine_matrix_M,&breakpoint,0);
677 // DEBUG
678 if (wf_aligner->system.verbose >= 2) {
679 wavefront_debug_end(wf_forward);
680 wavefront_debug_end(wf_reverse);
681 }
682 // Check status
683 cigar_t* const cigar = wf_aligner->cigar;
684 if (align_status == WF_STATUS_OK || align_status == WF_STATUS_END_REACHED) {
685 if (align_status == WF_STATUS_END_REACHED) {
686 breakpoint.score = (wf_forward->align_status.status == WF_STATUS_END_REACHED) ?
687 wf_forward->align_status.score : wf_reverse->align_status.score;
688 }
689 // Set status & score
690 cigar->score = wavefront_compute_classic_score(wf_aligner,pattern_length,text_length,breakpoint.score);
691 cigar->end_v = pattern_length;
692 cigar->end_h = text_length;
693 // Return OK
694 return WF_STATUS_OK;
695 } else {
696 // Other cases
697 return align_status;
698 }
699}
