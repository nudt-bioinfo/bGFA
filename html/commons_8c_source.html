35 * Random number generator [min,max)
36 */
37uint64_t rand_iid(
38 const uint64_t min,
39 const uint64_t max) {
40 const int n_rand = rand(); // [0, RAND_MAX]
41 const uint64_t range = max - min;
42 const uint64_t rem = RAND_MAX % range;
43 const uint64_t sample = RAND_MAX / range;
44 // Consider the small interval within remainder of RAND_MAX
45 if (n_rand < RAND_MAX - rem) {
46 return min + n_rand/sample;
47 } else {
48 return rand_iid(min,max);
49 }
50}
