184 int argc,
185 char** argv) {
186 struct option long_options[] = {
187 /* Input */
188 { "algorithm", required_argument, 0, 'a' },
189 { "input", required_argument, 0, 'i' },
190 { "output", required_argument, 0, 'o' },
191 { "output-full", required_argument, 0, 800 },
192 /* Penalties */
193 { "linear-penalties", required_argument, 0, 'p' },
194 { "affine-penalties", required_argument, 0, 'g' },
195 { "affine2p-penalties", required_argument, 0, 900 },
196 /* Wavefront parameters */
197 { "wfa-score-only", no_argument, 0, 1000 },
198 { "wfa-span", required_argument, 0, 1001 },
199 { "wfa-memory", required_argument, 0, 1002 },
200 { "wfa-heuristic", required_argument, 0, 1003 },
201 { "wfa-heuristic-parameters", required_argument, 0, 1004 },
202 { "wfa-max-memory", required_argument, 0, 1005 },
203 { "wfa-max-steps", required_argument, 0, 1006 },
204 { "wfa-max-threads", required_argument, 0, 1007 },
205 { "wfa-lambda", no_argument, 0, 1008 },
206 /* Other alignment parameters */
207 { "bandwidth", required_argument, 0, 2000 },
208 /* Misc */
209 { "check", required_argument, 0, 'c' },
210 { "check-distance", required_argument, 0, 3001 },
211 { "check-bandwidth", required_argument, 0, 3002 },
212 { "plot", optional_argument, 0, 3003 },
213 /* System */
214 { "num-threads", required_argument, 0, 't' },
215 { "batch-size", required_argument, 0, 4000 },
216 { "progress", required_argument, 0, 'P' },
217 { "verbose", required_argument, 0, 4001 },
218 { "verbose1", no_argument, 0, 'v' },
219 { "quiet", no_argument, 0, 'q' },
220 { "help", no_argument, 0, 'h' },
221 { 0, 0, 0, 0 } };
222 int c,option_index;
223 if (argc <= 1) {
224 usage();
225 exit(0);
226 }
227 while (1) {
228 c=getopt_long(argc,argv,"a:i:o:p:g:P:c:vqt:h",long_options,&option_index);
229 if (c==-1) break;
230 switch (c) {
231 /*
232 * Algorithm
233 */
234 case 'a': {
235 /* Test bench */
236 if (strcmp(optarg,"test")==0) {
237 parameters.algorithm = alignment_test;
238 // Indel
239 } else if (strcmp(optarg,"indel-wfa")==0) {
240 parameters.algorithm = alignment_indel_wavefront;
241 // Edit
242 } else if (strcmp(optarg,"edit-bpm")==0) {
243 parameters.algorithm = alignment_edit_bpm;
244 } else if (strcmp(optarg,"edit-dp")==0) {
245 parameters.algorithm = alignment_edit_dp;
246 } else if (strcmp(optarg,"edit-dp-banded")==0) {
247 parameters.algorithm = alignment_edit_dp_banded;
248 } else if (strcmp(optarg,"edit-wfa")==0) {
249 parameters.algorithm = alignment_edit_wavefront;
250 // Gap-Linear
251 } else if (strcmp(optarg,"gap-linear-nw")==0 ||
252 strcmp(optarg,"gap-linear-dp")==0) {
253 parameters.algorithm = alignment_gap_linear_nw;
254 } else if (strcmp(optarg,"gap-linear-wfa")==0) {
255 parameters.algorithm = alignment_gap_linear_wavefront;
256 // Gap-Affine
257 } else if (strcmp(optarg,"gap-affine-swg")==0 ||
258 strcmp(optarg,"gap-affine-dp")==0) {
259 parameters.algorithm = alignment_gap_affine_swg;
260 } else if (strcmp(optarg,"gap-affine-swg-banded")==0 ||
261 strcmp(optarg,"gap-affine-dp-banded")==0) {
262 parameters.algorithm = alignment_gap_affine_swg_banded;
263 } else if (strcmp(optarg,"gap-affine-wfa")==0) {
264 parameters.algorithm = alignment_gap_affine_wavefront;
265 // Gap-Affine 2-Pieces
266 } else if (strcmp(optarg,"gap-affine2p-dp")==0) {
267 parameters.algorithm = alignment_gap_affine2p_dp;
268 } else if (strcmp(optarg,"gap-affine2p-wfa")==0) {
269 parameters.algorithm = alignment_gap_affine2p_wavefront;
270 } else {
271 fprintf(stderr,"Algorithm '%s' not recognized\n",optarg);
272 exit(1);
273 }
274 break;
275 }
276 /*
277 * Input/Output
278 */
279 case 'i':
280 parameters.input_filename = optarg;
281 break;
282 case 'o':
283 parameters.output_filename = optarg;
284 break;
285 case 800: // --output-full
286 parameters.output_filename = optarg;
287 parameters.output_full = true;
288 break;
289 /*
290 * Penalties
291 */
292 case 'p': { // --linear-penalties M,X,I
293 char* sentinel = strtok(optarg,",");
294 parameters.linear_penalties.match = atoi(sentinel);
295 sentinel = strtok(NULL,",");
296 parameters.linear_penalties.mismatch = atoi(sentinel);
297 sentinel = strtok(NULL,",");
298 parameters.linear_penalties.indel = atoi(sentinel);
299 break;
300 }
301 case 'g': { // --affine-penalties M,X,O,E
302 char* sentinel = strtok(optarg,",");
303 parameters.affine_penalties.match = atoi(sentinel);
304 sentinel = strtok(NULL,",");
305 parameters.affine_penalties.mismatch = atoi(sentinel);
306 sentinel = strtok(NULL,",");
307 parameters.affine_penalties.gap_opening = atoi(sentinel);
308 sentinel = strtok(NULL,",");
309 parameters.affine_penalties.gap_extension = atoi(sentinel);
310 break;
311 }
312 case 900: { // --affine2p-penalties M,X,O1,E1,O2,E2
313 char* sentinel = strtok(optarg,",");
314 parameters.affine2p_penalties.match = atoi(sentinel);
315 sentinel = strtok(NULL,",");
316 parameters.affine2p_penalties.mismatch = atoi(sentinel);
317 sentinel = strtok(NULL,",");
318 parameters.affine2p_penalties.gap_opening1 = atoi(sentinel);
319 sentinel = strtok(NULL,",");
320 parameters.affine2p_penalties.gap_extension1 = atoi(sentinel);
321 sentinel = strtok(NULL,",");
322 parameters.affine2p_penalties.gap_opening2 = atoi(sentinel);
323 sentinel = strtok(NULL,",");
324 parameters.affine2p_penalties.gap_extension2 = atoi(sentinel);
325 break;
326 }
327 /*
328 * Wavefront parameters
329 */
330 case 1000: // --wfa-score-only
331 parameters.wfa_score_only = true;
332 break;
333 case 1001: { // --wfa-span P0,Pf,T0,Tf
334 if (strcmp(optarg,"global")==0) {
335 parameters.align_span_global = true;
336 parameters.align_span_extension = false;
337 parameters.align_span_endsfree = false;
338 } else if (strcmp(optarg,"extension")==0 ||
339 strcmp(optarg,"extend")==0) {
340 parameters.align_span_global = false;
341 parameters.align_span_extension = true;
342 parameters.align_span_endsfree = false;
343 } else if (strncmp(optarg,"ends-free",9)==0 ||
344 strncmp(optarg,"endsfree",8)==0) {
345 parameters.align_span_global = false;
346 parameters.align_span_extension = false;
347 parameters.align_span_endsfree = true;
348 // Parse arguments
349 char* sentinel = strtok(optarg,",");
350 sentinel = strtok(NULL,","); // Skip span-keyword
351 parameters.pattern_begin_free = atof(sentinel);
352 sentinel = strtok(NULL,",");
353 parameters.pattern_end_free = atof(sentinel);
354 sentinel = strtok(NULL,",");
355 parameters.text_begin_free = atof(sentinel);
356 sentinel = strtok(NULL,",");
357 parameters.text_end_free = atof(sentinel);
358 } else {
359 fprintf(stderr,"Option '--wfa-span' must be in {'global','extension','ends-free'}\n");
360 exit(1);
361 }
362 break;
363 }
364 case 1002: // --wfa-memory in {'high','med','low'}
365 if (strcmp(optarg,"high")==0) {
366 parameters.wfa_memory_mode = wavefront_memory_high;
367 } else if (strcmp(optarg,"med")==0) {
368 parameters.wfa_memory_mode = wavefront_memory_med;
369 } else if (strcmp(optarg,"low")==0) {
370 parameters.wfa_memory_mode = wavefront_memory_low;
371 } else if (strcmp(optarg,"ultralow")==0) {
372 parameters.wfa_memory_mode = wavefront_memory_ultralow;
373 } else {
374 fprintf(stderr,"Option '--wfa-memory' must be in {'high','med','low','ultralow'}\n");
375 exit(1);
376 }
377 break;
378 case 1003: // --wfa-heuristic in {'none'|'banded-static'|'banded-adaptive'|'wfa-adaptive'|'xdrop'|'zdrop'}
379 if (strcmp(optarg,"none")==0) {
380 parameters.wfa_heuristic = wf_heuristic_none;
381 } else if (strcmp(optarg,"banded-static")==0 || strcmp(optarg,"banded")==0) {
382 parameters.wfa_heuristic = wf_heuristic_banded_static;
383 } else if (strcmp(optarg,"banded-adaptive")==0) {
384 parameters.wfa_heuristic = wf_heuristic_banded_adaptive;
385 } else if (strcmp(optarg,"wfa-adaptive")==0) {
386 parameters.wfa_heuristic = wf_heuristic_wfadaptive;
387 } else if (strcmp(optarg,"xdrop")==0) {
388 parameters.wfa_heuristic = wf_heuristic_xdrop;
389 } else if (strcmp(optarg,"zdrop")==0) {
390 parameters.wfa_heuristic = wf_heuristic_zdrop;
391 } else {
392 fprintf(stderr,"Option '--wf-heuristic' must be in {'none'|'banded-static'|'banded-adaptive'|'wfa-adaptive'|'xdrop'|'zdrop'}\n");
393 exit(1);
394 }
395 break;
396 case 1004: { // --wfa-heuristic-parameters <P1>,<P2>[,<P3>]
397 char* sentinel = strtok(optarg,",");
398 const int p1 = atoi(sentinel);
399 parameters.wfa_heuristic_p1 = p1;
400 sentinel = strtok(NULL,",");
401 const int p2 = atoi(sentinel);
402 parameters.wfa_heuristic_p2 = p2;
403 sentinel = strtok(NULL,",");
404 if (sentinel != NULL) {
405 const int p3 = atoi(sentinel);
406 parameters.wfa_heuristic_p3 = p3;
407 }
408 break;
409 }
410 case 1005: // --wfa-max-memory
411 parameters.wfa_max_memory = atol(optarg);
412 break;
413 case 1006: // --wfa-max-steps
414 parameters.wfa_max_steps = atoi(optarg);
415 break;
416 case 1007: // --wfa-max-threads
417 parameters.wfa_max_threads = atoi(optarg);
418 break;
419 case 1008: // --wfa-lambda
420 parameters.wfa_lambda = true;
421 break;
422 /*
423 * Other alignment parameters
424 */
425 case 2000: // --bandwidth
426 parameters.bandwidth = atoi(optarg);
427 break;
428 /*
429 * Misc
430 */
431 case 'c':
432 if (optarg == NULL) { // default = correct
433 parameters.check_correct = true;
434 parameters.check_score = false;
435 parameters.check_alignments = false;
436 } else if (strcasecmp(optarg,"display")==0) {
437 parameters.check_display = true;
438 } else if (strcasecmp(optarg,"correct")==0) {
439 parameters.check_correct = true;
440 parameters.check_score = false;
441 parameters.check_alignments = false;
442 } else if (strcasecmp(optarg,"score")==0) {
443 parameters.check_correct = true;
444 parameters.check_score = true;
445 parameters.check_alignments = false;
446 } else if (strcasecmp(optarg,"alignment")==0) {
447 parameters.check_correct = true;
448 parameters.check_score = true;
449 parameters.check_alignments = true;
450 } else {
451 fprintf(stderr,"Option '--check' must be in {'correct','score','alignment'}\n");
452 exit(1);
453 }
454 break;
455 case 3001: // --check-distance in {'indel','edit','linear','affine','affine2p'}
456 if (strcasecmp(optarg,"indel")==0) {
457 parameters.check_metric = ALIGN_DEBUG_CHECK_DISTANCE_METRIC_INDEL;
458 } else if (strcasecmp(optarg,"edit")==0) {
459 parameters.check_metric = ALIGN_DEBUG_CHECK_DISTANCE_METRIC_EDIT;
460 } else if (strcasecmp(optarg,"linear")==0) {
461 parameters.check_metric = ALIGN_DEBUG_CHECK_DISTANCE_METRIC_GAP_LINEAR;
462 } else if (strcasecmp(optarg,"affine")==0) {
463 parameters.check_metric = ALIGN_DEBUG_CHECK_DISTANCE_METRIC_GAP_AFFINE;
464 } else if (strcasecmp(optarg,"affine2p")==0) {
465 parameters.check_metric = ALIGN_DEBUG_CHECK_DISTANCE_METRIC_GAP_AFFINE2P;
466 } else {
467 fprintf(stderr,"Option '--check-distance' must be in {'indel','edit','linear','affine','affine2p'}\n");
468 exit(1);
469 }
470 break;
471 case 3002: // --check-bandwidth
472 parameters.check_bandwidth = atoi(optarg);
473 break;
474 case 3003: // --plot
475 parameters.plot = (optarg==NULL) ? 1 : atoi(optarg);
476 break;
477 /*
478 * System
479 */
480 case 't': // --num-threads
481 parameters.num_threads = atoi(optarg);
482 break;
483 case 4000: // --batch-size
484 parameters.batch_size = atoi(optarg);
485 break;
486 case 'P':
487 parameters.progress = atoi(optarg);
488 break;
489 case 'v':
490 parameters.verbose = 1;
491 break;
492 case 4001: // --verbose (long option)
493 parameters.verbose = atoi(optarg);
494 if (parameters.verbose < 0 || parameters.verbose > 4) {
495 fprintf(stderr,"Option '--verbose' must be in {0,1,2,3,4}\n");
496 exit(1);
497 }
498 break;
499 case 'q':
500 parameters.verbose = -1;
501 break;
502 case 'h':
503 usage();
504 exit(1);
505 // Other
506 case '?': default:
507 fprintf(stderr,"Option not recognized \n");
508 exit(1);
509 }
510 }
511 // Checks general
512 if (parameters.algorithm!=alignment_test && parameters.input_filename==NULL) {
513 fprintf(stderr,"Option --input is required \n");
514 exit(1);
515 }
516 // Check 'ends-free' parameter
517 if (parameters.align_span_endsfree || parameters.align_span_extension) {
518 switch (parameters.algorithm) {
519 case alignment_indel_wavefront:
520 case alignment_edit_wavefront:
521 case alignment_gap_linear_wavefront:
522 case alignment_gap_affine_wavefront:
523 case alignment_gap_affine2p_wavefront:
524 break;
525 default:
526 fprintf(stderr,"Ends-free/Extension variant not implemented for the selected algorithm\n");
527 exit(1);
528 break;
529 }
530 }
531 // Check 'bandwidth' parameter
532 switch (parameters.algorithm) {
533 case alignment_edit_dp_banded:
534 case alignment_gap_affine_swg_banded:
535 if (parameters.bandwidth == -1) {
536 fprintf(stderr,"Parameter 'bandwidth' has to be provided for banded algorithms\n");
537 exit(1);
538 }
539 break;
540 default:
541 if (parameters.bandwidth != -1) {
542 fprintf(stderr,"Parameter 'bandwidth' has no effect with the selected algorithm\n");
543 exit(1);
544 }
545 break;
546 }
547 // Check 'wfa-heuristic'
548 switch (parameters.wfa_heuristic) {
549 case wf_heuristic_banded_static:
550 case wf_heuristic_xdrop:
551 case wf_heuristic_zdrop:
552 if (parameters.wfa_heuristic_p1 == -1 ||
553 parameters.wfa_heuristic_p2 == -1) {
554 fprintf(stderr,"Heuristic requires parameters '--wfa-heuristic-parameters' <P1>,<P2>\n");
555 exit(1);
556 }
557 break;
558 case wf_heuristic_banded_adaptive:
559 case wf_heuristic_wfadaptive:
560 if (parameters.wfa_heuristic_p1 == -1 ||
561 parameters.wfa_heuristic_p2 == -1 ||
562 parameters.wfa_heuristic_p3 == -1) {
563 fprintf(stderr,"Heuristic requires parameters '--wfa-heuristic-parameters' <P1>,<P2>,<P3>\n");
564 exit(1);
565 }
566 break;
567 default:
568 break;
569 }
570 // Checks parallel
571 if (parameters.num_threads > 1) {
572 if (parameters.plot > 0) {
573 fprintf(stderr,"Parameter 'plot' disabled for parallel executions\n");
574 parameters.plot = 0;
575 }
576 }
