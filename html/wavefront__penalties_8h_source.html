39/*
40 * Distance metrics
41 */
42typedef enum {
43 indel = 0, // Longest Common Subsequence - LCS
44 edit = 1, // Levenshtein
45 gap_linear = 2, // Needleman-Wunsch
46 gap_affine = 3, // Smith-Waterman-Gotoh
47 gap_affine_2p = 4 // Gap-Affine 2-pieces
48} distance_metric_t;
49
50/*
51 * Wavefront Penalties
52 */
53typedef struct {
54 // Alignment metric/distance used
55 distance_metric_t distance_metric;
56 // Penalty values
57 int match; // (M <= 0)
58 int mismatch; // (X > 0)
59 int gap_opening1; // (O1 >= 0)
60 int gap_extension1; // (E1 > 0)
61 int gap_opening2; // (O2 >= 0)
62 int gap_extension2; // (E2 > 0)
63 // Internals
64 linear_penalties_t linear_penalties; // Original gap-linear penalties
65 affine_penalties_t affine_penalties; // Original gap-affine penalties
66 affine2p_penalties_t affine2p_penalties; // Original gap-affine2p penalties
67 int internal_gap_e; // Original gap-extension value (used for z-drop)
68} wavefront_penalties_t;
