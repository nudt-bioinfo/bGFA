45wavefront_slab_t* wavefront_slab_new(
46 const int init_wf_length,
47 const bool allocate_backtrace,
48 const wf_slab_mode_t slab_mode,
49 mm_allocator_t* const mm_allocator) {
50 // Allocate
51 wavefront_slab_t* const wavefront_slab =
52 mm_allocator_alloc(mm_allocator,wavefront_slab_t);
53 // Attributes
54 wavefront_slab->allocate_backtrace = allocate_backtrace;
55 wavefront_slab->slab_mode = slab_mode;
56 // Wavefront Slabs
57 wavefront_slab->init_wf_length = init_wf_length;
58 wavefront_slab->current_wf_length = init_wf_length;
59 wavefront_slab->wavefronts = vector_new(WF_SLAB_QUEUES_LENGTH_INIT,wavefront_t*);
60 wavefront_slab->wavefronts_free = vector_new(WF_SLAB_QUEUES_LENGTH_INIT,wavefront_t*);
61 // Stats
62 wavefront_slab->memory_used = 0;
63 // MM
64 wavefront_slab->mm_allocator = mm_allocator;
65 // Return
66 return wavefront_slab;
67}
68void wavefront_slab_reap_free(
69 wavefront_slab_t* const wavefront_slab) {
70 // Parameters
71 wavefront_t** const wavefronts = vector_get_mem(wavefront_slab->wavefronts,wavefront_t*);
72 const int num_wavefronts = vector_get_used(wavefront_slab->wavefronts);
73 mm_allocator_t* const mm_allocator = wavefront_slab->mm_allocator;
74 // Remove "deallocated" and free wavefronts
75 int i, valid_idx = 0;
76 for (i=0;i<num_wavefronts;++i) {
77 switch (wavefronts[i]->status) {
78 case wavefront_status_deallocated:
79 mm_allocator_free(mm_allocator,wavefronts[i]); // Delete handler
80 break;
81 case wavefront_status_busy:
82 wavefronts[valid_idx++] = wavefronts[i]; // Valid wavefront
83 break;
84 case wavefront_status_free:
85 wavefront_free(wavefronts[i],mm_allocator); // Free wavefront
86 wavefront_slab->memory_used -= wavefront_get_size(wavefronts[i]);
87 mm_allocator_free(mm_allocator,wavefronts[i]); // Delete handler
88 break;
89 }
90 }
91 vector_set_used(wavefront_slab->wavefronts,valid_idx);
92 vector_clear(wavefront_slab->wavefronts_free);
93}
94void wavefront_slab_reap_repurpose(
95 wavefront_slab_t* const wavefront_slab) {
96 // Parameters
97 const int current_wf_length = wavefront_slab->current_wf_length;
98 wavefront_t** const wavefronts = vector_get_mem(wavefront_slab->wavefronts,wavefront_t*);
99 const int num_wavefronts = vector_get_used(wavefront_slab->wavefronts);
100 mm_allocator_t* const mm_allocator = wavefront_slab->mm_allocator;
101 // Clear free
102 vector_reserve(wavefront_slab->wavefronts_free,num_wavefronts,false);
103 wavefront_t** const wavefronts_free = vector_get_mem(wavefront_slab->wavefronts_free,wavefront_t*);
104 // Remove "deallocated" and repurpose all we can of current wf-length
105 int i, valid_idx = 0;
106 for (i=0;i<num_wavefronts;++i) {
107 switch (wavefronts[i]->status) {
108 case wavefront_status_deallocated:
109 mm_allocator_free(mm_allocator,wavefronts[i]); // Delete handler
110 break;
111 case wavefront_status_busy:
112 case wavefront_status_free:
113 if (wavefronts[i]->wf_elements_allocated == current_wf_length) {
114 wavefronts[i]->status = wavefront_status_free; // Set free
115 wavefronts[valid_idx] = wavefronts[i]; // Valid wavefront
116 wavefronts_free[valid_idx] = wavefronts[i]; // Free wavefront
117 valid_idx++;
118 } else {
119 wavefront_free(wavefronts[i],mm_allocator); // Free wavefront
120 wavefront_slab->memory_used -= wavefront_get_size(wavefronts[i]);
121 mm_allocator_free(mm_allocator,wavefronts[i]); // Delete handler
122 }
123 break;
124 }
125 }
126 vector_set_used(wavefront_slab->wavefronts,valid_idx);
127 vector_set_used(wavefront_slab->wavefronts_free,valid_idx);
128}
129void wavefront_slab_reap(
130 wavefront_slab_t* const wavefront_slab) {
131 // Back to initial size
132 wavefront_slab->current_wf_length = wavefront_slab->init_wf_length;
133 wavefront_slab_reap_repurpose(wavefront_slab); // Repurpose all wavefronts
134}
135void wavefront_slab_clear(
136 wavefront_slab_t* const wavefront_slab) {
137 // Select slab mode
138 switch (wavefront_slab->slab_mode) {
139 case wf_slab_reuse:
140 wavefront_slab_reap_repurpose(wavefront_slab);
141 break;
142 case wf_slab_tight:
143 // Back to initial size
144 wavefront_slab->current_wf_length = wavefront_slab->init_wf_length;
145 wavefront_slab_reap_repurpose(wavefront_slab);
146 break;
147 }
148}
149void wavefront_slab_delete(
150 wavefront_slab_t* const wavefront_slab) {
151 // Parameters
152 mm_allocator_t* const mm_allocator = wavefront_slab->mm_allocator;
153 // Delete free vector
154 vector_delete(wavefront_slab->wavefronts_free);
155 // Free wavefronts
156 wavefront_t** const wavefronts =
157 vector_get_mem(wavefront_slab->wavefronts,wavefront_t*);
158 const int num_wavefronts = vector_get_used(wavefront_slab->wavefronts);
159 int i;
160 for (i=0;i<num_wavefronts;++i) {
161 if (wavefronts[i]->status == wavefront_status_deallocated) {
162 mm_allocator_free(mm_allocator,wavefronts[i]); // Delete handler
163 } else {
164 wavefront_free(wavefronts[i],mm_allocator); // Free wavefront
165 mm_allocator_free(mm_allocator,wavefronts[i]); // Delete handler
166 }
167 }
168 vector_delete(wavefront_slab->wavefronts);
169 // Handler
170 mm_allocator_free(wavefront_slab->mm_allocator,wavefront_slab);
171}

190wavefront_t* wavefront_slab_allocate_new(
191 wavefront_slab_t* const wavefront_slab,
192 const int wf_length_requested,
193 const int min_lo,
194 const int max_hi) {
195 // Allocate a new wavefront
196 mm_allocator_t* const mm_allocator = wavefront_slab->mm_allocator;
197 wavefront_t* const wavefront = mm_allocator_alloc(mm_allocator,wavefront_t);
198 wavefront_allocate(wavefront,wf_length_requested,wavefront_slab->allocate_backtrace,mm_allocator);
199 vector_insert(wavefront_slab->wavefronts,wavefront,wavefront_t*);
200 wavefront_slab->memory_used += wavefront_get_size(wavefront);
201 // Init wavefront
202 wavefront->status = wavefront_status_busy;
203 wavefront_init(wavefront,min_lo,max_hi);
204 // Return
205 return wavefront;
206}
207wavefront_t* wavefront_slab_allocate_free(
208 wavefront_slab_t* const wavefront_slab,
209 const int min_lo,
210 const int max_hi) {
211 // Parameters
212 vector_t* const wavefronts_free = wavefront_slab->wavefronts_free;
213 // Reuse wavefront
214 wavefront_t* const wavefront = *(vector_get_last_elm(wavefronts_free,wavefront_t*));
215 vector_dec_used(wavefronts_free);
216 // Init wavefront
217 wavefront->status = wavefront_status_busy;
218 wavefront_init(wavefront,min_lo,max_hi);
219 // Return
220 return wavefront;
221}
222wavefront_t* wavefront_slab_allocate(
223 wavefront_slab_t* const wavefront_slab,
224 const int min_lo,
225 const int max_hi) {
226 // Parameters
227 vector_t* const wavefronts_free = wavefront_slab->wavefronts_free;
228 const int wf_length = WAVEFRONT_LENGTH(min_lo,max_hi);
229 // Check slab-mode
230 if (wavefront_slab->slab_mode == wf_slab_reuse) {
231 // Check max-length of pre-allocated wavefronts
232 if (wf_length > wavefront_slab->current_wf_length) {
233 const int proposed_wf_length = (float)wf_length * WF_SLAB_EXPAND_FACTOR;
234 wavefront_slab->current_wf_length = proposed_wf_length; // New slab size
235 wavefront_slab_reap_free(wavefront_slab); // Reap free wavefronts
236 }
237 // Check for a free wavefront (pre-allocated in the slab)
238 if (vector_get_used(wavefronts_free) > 0) {
239 return wavefront_slab_allocate_free(wavefront_slab,min_lo,max_hi);
240 } else {
241 // Allocate a new wavefront
242 return wavefront_slab_allocate_new(wavefront_slab,
243 wavefront_slab->current_wf_length,min_lo,max_hi);
244 }
245 } else { // wf_slab_tight
246 if (wf_length <= wavefront_slab->init_wf_length) {
247 // Check for a free wavefront (pre-allocated in the slab)
248 if (vector_get_used(wavefronts_free) > 0) {
249 return wavefront_slab_allocate_free(wavefront_slab,min_lo,max_hi);
250 } else {
251 return wavefront_slab_allocate_new(wavefront_slab,
252 wavefront_slab->init_wf_length,min_lo,max_hi); // Allocate new
253 }
254 } else {
255 return wavefront_slab_allocate_new(wavefront_slab,
256 wf_length,min_lo,max_hi); // Allocate new
257 }
258 }

260void wavefront_slab_free(
261 wavefront_slab_t* const wavefront_slab,
262 wavefront_t* const wavefront) {
263 // Check reasons to repurpose wavefront (NOTE: Tight-mode never slab_frees())
264 // (A) Reuse-mode and wavefront has current wf-length
265 // (B) Tight-mode and wavefront has init wf-length
266 const int wf_length = wavefront->wf_elements_allocated;
267 const bool repurpose_reuse =
268 (wavefront_slab->slab_mode == wf_slab_reuse) &&
269 (wf_length == wavefront_slab->current_wf_length);
270 const bool repurpose_tight =
271 (wavefront_slab->slab_mode == wf_slab_tight) &&
272 (wf_length == wavefront_slab->init_wf_length);
273 if (repurpose_reuse || repurpose_tight) {
274 // Return wavefront to slab as free (Good job recycling)
275 wavefront->status = wavefront_status_free;
276 vector_insert(wavefront_slab->wavefronts_free,wavefront,wavefront_t*);
277 } else {
278 // Delete wavefront
279 wavefront->status = wavefront_status_deallocated;
280 wavefront_slab->memory_used -= wavefront_get_size(wavefront);
281 wavefront_free(wavefront,wavefront_slab->mm_allocator);
282 }
283}
