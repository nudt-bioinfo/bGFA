42 * Checker (given the correct CIGAR+SCORE)
43 */
44void benchmark_check_alignment_using_solution(
45 align_input_t* const align_input,
46 cigar_t* const cigar_computed,
47 const int score_computed,
48 cigar_t* const cigar_correct,
49 const int score_correct) {
50 counter_add(&(align_input->align),1);
51 counter_add(&(align_input->align_score_total),ABS(score_computed));
52 // Debug
53 if (align_input->debug_flags) {
54 // Display info
55 if (align_input->debug_flags & ALIGN_DEBUG_DISPLAY_INFO) {
56 benchmark_print_alignment(stderr,align_input,score_computed,cigar_computed,-1,NULL);
57 }
58 // Check correct
59 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_CORRECT) {
60 bool correct = cigar_check_alignment(stderr,
61 align_input->pattern,align_input->pattern_length,
62 align_input->text,align_input->text_length,
63 cigar_computed,align_input->verbose);
64 if (!correct) {
65 // Print
66 if (align_input->verbose) {
67 fprintf(stderr,"INCORRECT ALIGNMENT\n");
68 benchmark_print_alignment(stderr,align_input,-1,cigar_computed,-1,NULL);
69 }
70 // Quit
71 return;
72 } else {
73 counter_add(&(align_input->align_correct),1);
74 }
75 // CIGAR Stats
76 int i;
77 counter_add(&(align_input->align_bases),align_input->pattern_length);
78 for (i=cigar_computed->begin_offset;i<cigar_computed->end_offset;++i) {
79 switch (cigar_computed->operations[i]) {
80 case 'M': counter_add(&(align_input->align_matches),1); break;
81 case 'X': counter_add(&(align_input->align_mismatches),1); break;
82 case 'I': counter_add(&(align_input->align_ins),1); break;
83 case 'D': default: counter_add(&(align_input->align_del),1); break;
84 }
85 }
86 }
87 // Check score
88 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_SCORE) {
89 if (score_computed != score_correct) {
90 // Print
91 if (align_input->verbose) {
92 benchmark_print_alignment(
93 stderr,align_input,
94 score_computed,cigar_computed,
95 score_correct,cigar_correct);
96 fprintf(stderr,"(#%d)\t INACCURATE SCORE computed=%d\tcorrect=%d\n",
97 align_input->sequence_id,score_computed,score_correct);
98 }
99 counter_add(&(align_input->align_score_diff),ABS(score_computed-score_correct));
100 // Quit
101 return;
102 } else {
103 counter_add(&(align_input->align_score),1);
104 }
105 }
106 // Check alignment
107 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_ALIGNMENT) {
108 if (cigar_cmp(cigar_computed,cigar_correct) != 0) {
109 // Print
110 if (align_input->verbose) {
111 fprintf(stderr,"INACCURATE ALIGNMENT\n");
112 benchmark_print_alignment(
113 stderr,align_input,
114 -1,cigar_computed,
115 -1,cigar_correct);
116 }
117 // Quit
118 return;
119 } else {
120 counter_add(&(align_input->align_cigar),1);
121 }
122 }
123 }
124}

128void benchmark_check_alignment_indel(
129 align_input_t* const align_input,
130 cigar_t* const cigar_computed) {
131 score_matrix_t score_matrix;
132 score_matrix_allocate(
133 &score_matrix,align_input->pattern_length+1,
134 align_input->text_length+1,align_input->mm_allocator);
135 cigar_t* const cigar = cigar_new(
136 align_input->pattern_length+align_input->text_length);
137 indel_dp_compute(&score_matrix,
138 align_input->pattern,align_input->pattern_length,
139 align_input->text,align_input->text_length,cigar);
140 const int score_correct = cigar_score_edit(cigar);
141 const int score_computed = cigar_score_edit(cigar_computed);
142 // Check alignment
143 benchmark_check_alignment_using_solution(
144 align_input,cigar_computed,score_computed,
145 cigar,score_correct);
146 // Free
147 score_matrix_free(&score_matrix);
148 cigar_free(cigar);
149}

268void benchmark_check_alignment(
269 align_input_t* const align_input,
270 cigar_t* const cigar_computed) {
271 // Compute correct CIGAR
272 if ((align_input->debug_flags & ALIGN_DEBUG_CHECK_SCORE) ||
273 (align_input->debug_flags & ALIGN_DEBUG_CHECK_ALIGNMENT)) {
274 // Check gap-affine-2p
275 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_DISTANCE_METRIC_GAP_AFFINE2P) {
276 benchmark_check_alignment_gap_affine2p(align_input,cigar_computed);
277 }
278 // Check gap-affine
279 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_DISTANCE_METRIC_GAP_AFFINE) {
280 benchmark_check_alignment_gap_affine(align_input,cigar_computed);
281 }
282 // Check gap-linear
283 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_DISTANCE_METRIC_GAP_LINEAR) {
284 benchmark_check_alignment_gap_linear(align_input,cigar_computed);
285 }
286 // Check edit
287 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_DISTANCE_METRIC_EDIT) {
288 benchmark_check_alignment_edit(align_input,cigar_computed);
289 }
290 // Check indel
291 if (align_input->debug_flags & ALIGN_DEBUG_CHECK_DISTANCE_METRIC_INDEL) {
292 benchmark_check_alignment_indel(align_input,cigar_computed);
293 }
294 } else if ((align_input->debug_flags & ALIGN_DEBUG_CHECK_CORRECT) ||
295 (align_input->debug_flags & ALIGN_DEBUG_DISPLAY_INFO)) { // Check CORRECT or just DISPLAY
296 // Delegate check alignment
297 benchmark_check_alignment_using_solution(
298 align_input,cigar_computed,-1,NULL,-1);
299 }
300}
