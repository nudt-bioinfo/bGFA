38void score_matrix_allocate(
39 score_matrix_t* const score_matrix,
40 const int num_rows,
41 const int num_columns,
42 mm_allocator_t* const mm_allocator) {
43 // Allocate DP matrix
44 int h;
45 score_matrix->num_rows = num_rows;
46 score_matrix->num_columns = num_columns;
47 score_matrix->columns = mm_allocator_malloc(mm_allocator,num_columns*sizeof(int*)); // Columns
48 for (h=0;h<num_columns;++h) {
49 score_matrix->columns[h] = mm_allocator_calloc(mm_allocator,num_rows,int,false); // Rows
50 }
51 // MM
52 score_matrix->mm_allocator = mm_allocator;
53}
54void score_matrix_free(
55 score_matrix_t* const score_matrix) {
56 // Parameters
57 mm_allocator_t* const mm_allocator = score_matrix->mm_allocator;
58 // DP matrix
59 const int num_columns = score_matrix->num_columns;
60 int h;
61 for (h=0;h<num_columns;++h) {
62 mm_allocator_free(mm_allocator,score_matrix->columns[h]);
63 }
64 mm_allocator_free(mm_allocator,score_matrix->columns);
65}

83void score_matrix_print(
84 FILE* const stream,
85 const score_matrix_t* const score_matrix,
86 const char* const pattern,
87 const char* const text) {
88 // Parameters
89 int** const matrix = score_matrix->columns;
90 const int num_columns = score_matrix->num_columns;
91 const int num_rows = score_matrix->num_rows;
92 int h;
93 // Print Header
94 fprintf(stream," ");
95 for (h=0;h<num_columns-1;++h) {
96 score_matrix_print_char(stream,text[h]);
97 }
98 fprintf(stream,"\n ");
99 for (h=0;h<num_columns;++h) {
100 score_matrix_print_score(stream,h);
101 }
102 fprintf(stream,"\n ");
103 for (h=0;h<num_columns;++h) {
104 score_matrix_print_score(stream,matrix[h][0]);
105 }
106 fprintf(stream,"\n");
107 // Print Rows
108 int v;
109 for (v=1;v<num_rows;++v) {
110 fprintf(stream,"%c",pattern[v-1]);
111 for (h=0;h<num_columns;++h) {
112 score_matrix_print_score(stream,matrix[h][v]);
113 }
114 fprintf(stream,"\n");
115 }
116 fprintf(stream,"\n");
117}
