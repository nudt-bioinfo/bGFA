38sequence_buffer_t* sequence_buffer_new(
39 const uint64_t num_sequences_hint,
40 const uint64_t sequence_length_hint) {
41 // Alloc
42 sequence_buffer_t* const sequence_buffer = malloc(sizeof(sequence_buffer_t));
43 // ID
44 sequence_buffer->sequence_id = 1;
45 // Initialize sequences
46 sequence_buffer->offsets = malloc(num_sequences_hint*sizeof(sequence_offset_t));
47 sequence_buffer->offsets_used = 0;
48 sequence_buffer->offsets_allocated = num_sequences_hint;
49 // Initialize buffer
50 const uint64_t buffer_size = num_sequences_hint*sequence_length_hint;
51 sequence_buffer->buffer = malloc(buffer_size);
52 sequence_buffer->buffer_used = 0;
53 sequence_buffer->buffer_allocated = buffer_size;
54 // Stats
55 sequence_buffer->max_pattern_length = 0;
56 sequence_buffer->max_text_length = 0;
57 // Return
58 return sequence_buffer;
59}

77void sequence_buffer_add_offsets(
78 sequence_buffer_t* const sequence_buffer,
79 const uint64_t pattern_offset,
80 const uint64_t pattern_length,
81 const uint64_t text_offset,
82 const uint64_t text_length) {
83 // Check allocated memory
84 if (sequence_buffer->offsets_used == sequence_buffer->offsets_allocated) {
85 const uint64_t num_offsets = (float)(sequence_buffer->offsets_used+1) * (3.0/2.0);
86 sequence_buffer->offsets = realloc(sequence_buffer->offsets,num_offsets*sizeof(sequence_offset_t));
87 sequence_buffer->offsets_allocated = num_offsets;
88 }
89 // Add sequence pair
90 sequence_offset_t* const current_offsets = sequence_buffer->offsets + sequence_buffer->offsets_used;
91 current_offsets->pattern_offset = pattern_offset;
92 current_offsets->pattern_length = pattern_length;
93 current_offsets->text_offset = text_offset;
94 current_offsets->text_length = text_length;
95 sequence_buffer->offsets_used += 1;
96}
97void sequence_buffer_add_pair(
98 sequence_buffer_t* const sequence_buffer,
99 char* const pattern,
100 const uint64_t pattern_length,
101 char* const text,
102 const uint64_t text_length) {
103 // Allocate memory
104 const uint64_t bytes_required = pattern_length + text_length + 4; // Padding
105 if (sequence_buffer->buffer_used+bytes_required > sequence_buffer->buffer_allocated) {
106 const uint64_t proposed_buffer_size = (3*(sequence_buffer->buffer_used+bytes_required))/2;
107 sequence_buffer->buffer_allocated = proposed_buffer_size;
108 sequence_buffer->buffer = realloc(sequence_buffer->buffer,proposed_buffer_size);
109 }
110 // Copy pattern sequence
111 char* mem_ptr = sequence_buffer->buffer + sequence_buffer->buffer_used;
112 memcpy(mem_ptr,pattern,pattern_length);
113 mem_ptr += pattern_length;
114 mem_ptr[0] = '\0';
115 mem_ptr[1] = '!';
116 mem_ptr += 2; // Padding sentinels
117 // Copy text sequence
118 memcpy(mem_ptr,text,text_length);
119 mem_ptr += text_length;
120 mem_ptr[0] = '\0';
121 mem_ptr[1] = '?';
122 mem_ptr += 2; // Padding sentinels
123 // Add sequence pair
124 const uint64_t pattern_offset = sequence_buffer->buffer_used;
125 const uint64_t text_offset = pattern_offset + pattern_length + 2;
126 sequence_buffer_add_offsets(
127 sequence_buffer,pattern_offset,
128 pattern_length,text_offset,text_length);
129 // Update used
130 sequence_buffer->buffer_used += bytes_required;
131 // Update stats
132 sequence_buffer->max_pattern_length = MAX(sequence_buffer->max_pattern_length,pattern_length);
133 sequence_buffer->max_text_length = MAX(sequence_buffer->max_text_length,text_length);
134}
