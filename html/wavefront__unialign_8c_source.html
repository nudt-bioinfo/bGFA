51/*
52 * Initialize alignment
53 */
54void wavefront_unialign_init(
55 wavefront_aligner_t* const wf_aligner,
56 const affine2p_matrix_type component_begin,
57 const affine2p_matrix_type component_end) {
58 // Parameters
59 wavefront_align_status_t* const align_status = &wf_aligner->align_status;
60 alignment_form_t* const alignment_form = &wf_aligner->alignment_form;
61 const bool end2end = (alignment_form->span == alignment_end2end);
62 // Configure WF-compute function
63 switch (wf_aligner->penalties.distance_metric) {
64 case indel:
65 case edit:
66 align_status->wf_align_compute = &wavefront_compute_edit;
67 break;
68 case gap_linear:
69 align_status->wf_align_compute = &wavefront_compute_linear;
70 break;
71 case gap_affine:
72 align_status->wf_align_compute = &wavefront_compute_affine;
73 break;
74 case gap_affine_2p:
75 align_status->wf_align_compute = &wavefront_compute_affine2p;
76 break;
77 default:
78 fprintf(stderr,"[WFA] Distance function not implemented\n");
79 exit(1);
80 break;
81 }
82 // Configure WF-extend function
83 if (end2end) {
84 align_status->wf_align_extend = &wavefront_extend_end2end;
85 } else {
86 align_status->wf_align_extend = &wavefront_extend_endsfree;
87 }
88 // Initialize wavefront-aligner (to perform a new alignment)
89 wf_aligner->component_begin = component_begin;
90 wf_aligner->component_end = component_end;
91 wavefront_aligner_init(wf_aligner,0);
92 // Clear cigar
93 cigar_clear(wf_aligner->cigar);
94}
95/*
96 * Limits
97 */
98bool wavefront_unialign_reached_limits(
99 wavefront_aligner_t* const wf_aligner,
100 const int score) {
101 // Check alignment-score limit
102 if (score >= wf_aligner->system.max_alignment_steps) {
103 wf_aligner->cigar->score = -wf_aligner->system.max_alignment_steps;
104 wf_aligner->align_status.status = WF_STATUS_MAX_STEPS_REACHED;
105 wf_aligner->align_status.score = score;
106 return true; // Stop
107 }
108 // Global probing interval
109 alignment_system_t* const system = &wf_aligner->system;
110 if (score % system->probe_interval_global != 0) return false; // Continue
111 if (system->verbose >= 3) {
112 wavefront_unialign_print_status(stderr,wf_aligner,score); // DEBUG
113 }
114 // BT-Buffer
115 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
116 if (wf_components->bt_buffer!=NULL && (score%system->probe_interval_compact)==0) {
117 uint64_t bt_memory = wf_backtrace_buffer_get_size_used(wf_components->bt_buffer);
118 // Check BT-buffer memory
119 if (bt_memory > system->max_memory_compact) {
120 // Compact BT-buffer
121 wavefront_components_compact_bt_buffer(wf_components,score,wf_aligner->system.verbose);
122 // Set new buffer limit
123 bt_memory = wf_backtrace_buffer_get_size_used(wf_components->bt_buffer);
124 uint64_t proposed_mem = (double)bt_memory * TELESCOPIC_FACTOR;
125 if (system->max_memory_compact < proposed_mem && proposed_mem < system->max_memory_abort) {
126 proposed_mem = system->max_memory_compact;
127 }
128 // Reset (if maximum compacts has been performed)
129 if (wf_components->bt_buffer->num_compactions >= system->max_partial_compacts) {
130 wf_backtrace_buffer_reset_compaction(wf_components->bt_buffer);
131 }
132 }
133 }
134 // Check overall memory used
135 const uint64_t wf_memory_used = wavefront_aligner_get_size(wf_aligner);
136 if (wf_memory_used > system->max_memory_abort) {
137 wf_aligner->align_status.status = WF_STATUS_OOM;
138 wf_aligner->align_status.score = score;
139 return true; // Stop
140 }
141 // Otherwise continue
142 return false;
143}
144/*
145 * Terminate alignment (backtrace)
146 */
147void wavefront_unialign_terminate(
148 wavefront_aligner_t* const wf_aligner,
149 const int score) {
150 // Parameters
151 wavefront_align_status_t* const align_status = &wf_aligner->align_status;
152 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
153 const int pattern_length = sequences->pattern_length;
154 const int text_length = sequences->text_length;
155 cigar_t* const cigar = wf_aligner->cigar;
156 // Select alignment scope
157 align_status->score = score;
158 if (wf_aligner->alignment_scope == compute_score) {
159 // Set end-alignment position & score
160 if (align_status->status == WF_STATUS_END_REACHED) {
161 cigar->end_v = pattern_length;
162 cigar->end_h = text_length;
163 cigar->score = wavefront_compute_classic_score(wf_aligner,pattern_length,text_length,score);
164 align_status->status = WF_STATUS_ALG_COMPLETED;
165 } else {
166 const int k = wf_aligner->alignment_end_pos.k;
167 const int offset = wf_aligner->alignment_end_pos.offset;
168 cigar->end_v = WAVEFRONT_V(k,offset);
169 cigar->end_h = WAVEFRONT_H(k,offset);
170 cigar->score = wavefront_compute_classic_score(wf_aligner,cigar->end_v,cigar->end_h,score);
171 align_status->dropped = true;
172 align_status->status = WF_STATUS_ALG_PARTIAL;
173 }
174 } else {
175 // Parameters
176 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
177 const int alignment_end_k = wf_aligner->alignment_end_pos.k;
178 const wf_offset_t alignment_end_offset = wf_aligner->alignment_end_pos.offset;
179 if (alignment_end_offset != WAVEFRONT_OFFSET_NULL) {
180 if (wf_components->bt_piggyback) {
181 // Fetch wavefront
182 const bool memory_modular = wf_aligner->wf_components.memory_modular;
183 const int max_score_scope = wf_aligner->wf_components.max_score_scope;
184 const int score_mod = (memory_modular) ? score % max_score_scope : score;
185 wavefront_t* const mwavefront = wf_components->mwavefronts[score_mod];
186 // Backtrace alignment from buffer (unpacking pcigar)
187 wavefront_backtrace_pcigar(
188 wf_aligner,alignment_end_k,alignment_end_offset,
189 mwavefront->bt_pcigar[alignment_end_k],
190 mwavefront->bt_prev[alignment_end_k]);
191 } else {
192 // Backtrace alignment
193 if (wf_aligner->penalties.distance_metric <= gap_linear) {
194 wavefront_backtrace_linear(wf_aligner,
195 score,alignment_end_k,alignment_end_offset);
196 } else {
197 wavefront_backtrace_affine(wf_aligner,
198 wf_aligner->component_begin,wf_aligner->component_end,
199 score,alignment_end_k,alignment_end_offset);
200 }
201 }
202 }
203 /*
204 * Post-processing (Extension-Trim, Score, and Ends)
205 *
206 * | Alignment-Regular | Alignment-Extension |
207 * |------------------------------------------------------------------------------|
208 * | END_REACHABLE | NoTrim + ALG_COMPLETED | Trim + ALG_PARTIAL|ALG_COMPLETED |
209 * |END_UNREACHABLE | Trim + ALG_PARTIAL | Trim + ALG_PARTIAL |
210 */
211 const bool do_extension = wf_aligner->alignment_form.extension;
212 const bool unreachable = (align_status->status == WF_STATUS_END_UNREACHABLE);
213 align_status->dropped = unreachable;
214 if (do_extension || unreachable) {
215 // Alignment extension (maximal score)
216 const bool cigar_trimmed = wavefront_aligner_maxtrim_cigar(wf_aligner);
217 if (cigar_trimmed) {
218 align_status->status = WF_STATUS_ALG_PARTIAL;
219 } else {
220 align_status->status = (align_status->status == WF_STATUS_END_UNREACHABLE) ?
221 WF_STATUS_ALG_PARTIAL : WF_STATUS_ALG_COMPLETED;
222 }
223 } else {
224 const int k = wf_aligner->alignment_end_pos.k;
225 const int offset = wf_aligner->alignment_end_pos.offset;
226 cigar->end_v = WAVEFRONT_V(k,offset);
227 cigar->end_h = WAVEFRONT_H(k,offset);
228 cigar->score = wavefront_compute_classic_score(wf_aligner,cigar->end_v,cigar->end_h,score);
229 // Set status
230 if (unreachable) {
231 align_status->status = WF_STATUS_ALG_PARTIAL;
232 } else {
233 align_status->status = WF_STATUS_ALG_COMPLETED;
234 }
235 }
236 }
237}
238/*
239 * Classic WF-Alignment (Unidirectional)
240 */
241#define WFA_UNIALIGN_DEBUG() wavefront_aligner_print(stderr,wf_aligner,0,score,7,0)
242int wavefront_unialign(
243 wavefront_aligner_t* const wf_aligner) {
244 // Parameters
245 wavefront_align_status_t* const align_status = &wf_aligner->align_status;
246 void (*wf_align_compute)(wavefront_aligner_t* const,const int) = align_status->wf_align_compute;
247 int (*wf_align_extend)(wavefront_aligner_t* const,const int) = align_status->wf_align_extend;
248 // Compute wavefronts of increasing score
249 int score = align_status->score;
250 // WFA_UNIALIGN_DEBUG(); // DEBUG
251 while (true) {
252 // Exact extend s-wavefront
253 const int finished = (*wf_align_extend)(wf_aligner,score);
254 if (finished) {
255 // WFA_UNIALIGN_DEBUG(); // DEBUG
256 if (align_status->status == WF_STATUS_END_REACHED ||
257 align_status->status == WF_STATUS_END_UNREACHABLE) {
258 wavefront_unialign_terminate(wf_aligner,score);
259 }
260 return align_status->status;
261 }
262 // Compute (s+1)-wavefront
263 ++score;
264 (*wf_align_compute)(wf_aligner,score);
265 // Probe limits
266 if (wavefront_unialign_reached_limits(wf_aligner,score)) return align_status->status;
267 // Plot
268 if (wf_aligner->plot != NULL) wavefront_plot(wf_aligner,score,0);
269 // WFA_UNIALIGN_DEBUG(); // DEBUG
270 }
271 // Unreachable code
272 return WF_STATUS_OK;
273}
274/*
275 * Display
276 */
277void wavefront_unialign_print_status(
278 FILE* const stream,
279 wavefront_aligner_t* const wf_aligner,
280 const int score) {
281 // Parameters
282 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
283 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
284 const int pattern_length = sequences->pattern_length;
285 const int text_length = sequences->text_length;
286 // Approximate progress
287 const int dist_total = MAX(text_length,pattern_length);
288 int s = (wf_components->memory_modular) ? score%wf_components->max_score_scope : score;
289 wavefront_t* wavefront = wf_components->mwavefronts[s];
290 if (wavefront==NULL && s>0) {
291 s = (wf_components->memory_modular) ? (score-1)%wf_components->max_score_scope : (score-1);
292 wavefront = wf_components->mwavefronts[s];
293 }
294 int dist_max = -1, wf_len = -1, k;
295 if (wavefront!=NULL) {
296 wf_offset_t* const offsets = wavefront->offsets;
297 for (k=wavefront->lo;k<=wavefront->hi;++k) {
298 const int dist = MAX(WAVEFRONT_V(k,offsets[k]),WAVEFRONT_H(k,offsets[k]));
299 dist_max = MAX(dist_max,dist);
300 }
301 wf_len = wavefront->hi-wavefront->lo+1;
302 }
303 // Memory used
304 const uint64_t slab_size = wavefront_slab_get_size(wf_aligner->wavefront_slab);
305 const uint64_t bt_buffer_used = (wf_components->bt_buffer) ?
306 wf_backtrace_buffer_get_size_used(wf_components->bt_buffer) : 0;
307 // Progress
308 const float aligned_progress = (dist_max>=0) ? (100.0f*(float)dist_max/(float)dist_total) : -1.0f;
309 const float million_offsets = (wf_len>=0) ? (float)wf_len/1000000.0f : -1.0f;
310 // Print one-line status
311 fprintf(stream,"[");
312 wavefront_aligner_print_mode(stream,wf_aligner);
313 fprintf(stream,
314 "] SequenceLength=(%d,%d) Score %d (~ %2.3f%% aligned). "
315 "MemoryUsed(WF-Slab,BT-buffer)=(%lu MB,%lu MB). "
316 "Wavefronts ~ %2.3f Moffsets\n",
317 pattern_length,
318 text_length,
319 score,
320 aligned_progress,
321 CONVERT_B_TO_MB(slab_size),
322 CONVERT_B_TO_MB(bt_buffer_used),
323 million_offsets);
324}
