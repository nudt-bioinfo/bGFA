97static std::string normalize(const std::string &s)
98{
99 std::string out;
100 out.reserve(s.size());
101 bool inSpace = false;
102 for (char c : s)
103 {
104 if (c == '\t' || c == ' ')
105 {
106 if (!inSpace)
107 {
108 out.push_back(' ');
109 inSpace = true;
110 }
111 }
112 else
113 {
114 out.push_back(c);
115 inSpace = false;
116 }
117 }
118 if (!out.empty() && out.front() == ' ')
119 out.erase(out.begin());
120 if (!out.empty() && out.back() == ' ')
121 out.pop_back();
122 return out;
123}
