39bitmap_t* bitmap_new(
40 const uint64_t length,
41 mm_allocator_t* const mm_allocator) {
42 // Allocate
43 bitmap_t* const bitmap =
44 mm_allocator_alloc(mm_allocator,bitmap_t);
45 bitmap->mm_allocator = mm_allocator;
46 // Allocate bitmap-blocks
47 const uint64_t num_blocks = DIV_CEIL(length,BITMAP_BLOCK_ELEMENTS);
48 bitmap->num_blocks = num_blocks;
49 bitmap->bitmap_blocks = mm_allocator_calloc(mm_allocator,num_blocks,bitmap_block_t,true);
50 // Return
51 return bitmap;
52}

101void bitmap_update_counters(
102 bitmap_t* const bitmap) {
103 // Parameters
104 const uint64_t num_blocks = bitmap->num_blocks;
105 bitmap_block_t* bitmap_block = bitmap->bitmap_blocks;
106 // Update all counters
107 uint64_t acc_count = 0;
108 uint64_t i;
109 for (i=0;i<num_blocks;++i,++bitmap_block) {
110 bitmap_block->counter = acc_count;
111 acc_count += POPCOUNT_64(bitmap_block->bitmap);
112 }
113}
114uint64_t bitmap_erank(
115 bitmap_t* const bitmap,
116 const uint64_t position) {
117 // Locate block
118 const uint64_t block_num = position / BITMAP_BLOCK_ELEMENTS;
119 const uint64_t block_pos = position % BITMAP_BLOCK_ELEMENTS;
120 // Compute e(xclusive)rank (number of bits set to one before the given position, not included)
121 bitmap_block_t* const bitmap_block = bitmap->bitmap_blocks + block_num;
122 const uint64_t bitmap_masked = (block_pos!=0) ? bitmap_block->bitmap << (BITMAP_BLOCK_ELEMENTS - block_pos) : 0;
123 const uint64_t bitmap_count = POPCOUNT_64(bitmap_masked);
124 return bitmap_block->counter + bitmap_count;
125}
