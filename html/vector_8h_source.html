108#define vector_alloc_new(vector,type,return_element_pointer) { \
109 vector_reserve_additional(vector,1); \
110 return_element_pointer = vector_get_free_elm(vector,type); \
111 vector_inc_used(vector); \
112}
113#define vector_insert(vector,element,type) { \
114 vector_reserve_additional(vector,1); \
115 *(vector_get_free_elm(vector,type)) = element; \
116 vector_inc_used(vector); \
117}
