35int main(int argc,char* argv[]) {
36 // Patter & Text
37 char* pattern = "TCTTTACTCGCGCGTTGGAGAAATACAATAGT";
38 char* text = "TCTATACTGCGCGTTTGGAGAAATAAAATAGT";
39 // Configure alignment attributes
40 wavefront_aligner_attr_t attributes = wavefront_aligner_attr_default;
41 attributes.distance_metric = gap_affine;
42 attributes.affine_penalties.match = 0;
43 attributes.affine_penalties.mismatch = 4;
44 attributes.affine_penalties.gap_opening = 6;
45 attributes.affine_penalties.gap_extension = 2;
46 // Set heuristic wf-adaptive
47 attributes.heuristic.strategy = wf_heuristic_wfadaptive;
48 attributes.heuristic.min_wavefront_length = 10;
49 attributes.heuristic.max_distance_threshold = 50;
50 attributes.heuristic.steps_between_cutoffs = 1;
51 // Initialize Wavefront Aligner
52 wavefront_aligner_t* const wf_aligner = wavefront_aligner_new(&attributes);
53 // Align
54 wavefront_align(wf_aligner,pattern,strlen(pattern),text,strlen(text));
55 fprintf(stderr,"WFA-Alignment returns score %d\n",wf_aligner->cigar->score);
56 // Count mismatches, deletions, and insertions
57 int i, misms=0, ins=0, del=0;
58 cigar_t* const cigar = wf_aligner->cigar;
59 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
60 switch (cigar->operations[i]) {
61 case 'M': break;
62 case 'X': ++misms; break;
63 case 'D': ++del; break;
64 case 'I': ++ins; break;
65 }
66 }
67 fprintf(stderr,"Alignment contains %d mismatches, "
68 "%d insertions, and %d deletions\n",misms,ins,del);
69 // Free
70 wavefront_aligner_delete(wf_aligner);
71}
