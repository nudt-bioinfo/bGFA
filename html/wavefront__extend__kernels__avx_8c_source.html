41/*
42 * Wavefront-Extend Inner Kernel (Scalar)
43 */
44FORCE_INLINE wf_offset_t wavefront_extend_matches_packed_kernel(
45 wavefront_aligner_t* const wf_aligner,
46 const int k,
47 wf_offset_t offset) {
48 // Fetch pattern/text blocks
49 uint64_t* pattern_blocks = (uint64_t*)(wf_aligner->sequences.pattern+WAVEFRONT_V(k,offset));
50 uint64_t* text_blocks = (uint64_t*)(wf_aligner->sequences.text+WAVEFRONT_H(k,offset));
51 // Compare 64-bits blocks
52 uint64_t cmp = *pattern_blocks ^ *text_blocks;
53 while (__builtin_expect(cmp==0,0)) {
54 // Increment offset (full block)
55 offset += 8;
56 // Next blocks
57 ++pattern_blocks;
58 ++text_blocks;
59 // Compare
60 cmp = *pattern_blocks ^ *text_blocks;
61 }
62 // Count equal characters
63 const int equal_right_bits = __builtin_ctzl(cmp);
64 const int equal_chars = DIV_FLOOR(equal_right_bits,8);
65 offset += equal_chars;
66 // Return extended offset
67 return offset;
68}
69/*
70 * SIMD clz, use a native instruction when available (AVX512 CD or VL
71 * extensions), or emulate the clz behavior.
72 */
73FORCE_INLINE __m256i avx2_lzcnt_epi32(__m256i v) {
74#if __AVX512CD__ && __AVX512VL__
75 return _mm256_lzcnt_epi32(v);
76#else
77 // Emulate clz for AVX2: https://stackoverflow.com/a/58827596
78 v = _mm256_andnot_si256(_mm256_srli_epi32(v,8),v); // keep 8 MSB
79 v = _mm256_castps_si256(_mm256_cvtepi32_ps(v)); // convert an integer to float
80 v = _mm256_srli_epi32(v,23); // shift down the exponent
81 v = _mm256_subs_epu16(_mm256_set1_epi32(158),v); // undo bias
82 v = _mm256_min_epi16(v,_mm256_set1_epi32(32)); // clamp at 32
83 return v;
84#endif
85}
86/*
87 * Wavefront-Extend Inner Kernel (SIMD AVX2/AVX512)
88 */
89FORCE_NO_INLINE void wavefront_extend_matches_packed_end2end_avx2(
90 wavefront_aligner_t* const wf_aligner,
91 wavefront_t* const mwavefront,
92 const int lo,
93 const int hi) {
94 // Parameters
95 wf_offset_t* const offsets = mwavefront->offsets;
96 int k_min = lo;
97 int k_max = hi;
98 const char* pattern = wf_aligner->sequences.pattern;
99 const char* text = wf_aligner->sequences.text;
100 const __m256i vector_null = _mm256_set1_epi32(-1);
101 const __m256i fours = _mm256_set1_epi32(4);
102 const __m256i eights = _mm256_set1_epi32(8);
103 const __m256i vecShuffle = _mm256_set_epi8(28,29,30,31,24,25,26,27,
104 20,21,22,23,16,17,18,19,
105 12,13,14,15, 8, 9,10,11,
106 4 , 5, 6, 7, 0, 1, 2 ,3);
107 const int elems_per_register = 8;
108 int num_of_diagonals = k_max - k_min + 1;
109 int loop_peeling_iters = num_of_diagonals % elems_per_register;
110 int k;
111 for (k=k_min;k<k_min+loop_peeling_iters;k++) {
112 const wf_offset_t offset = offsets[k];
113 if (offset < 0) continue;
114 // Extend offset
115 offsets[k] = wavefront_extend_matches_packed_kernel(wf_aligner,k,offset);
116 }
117 if (num_of_diagonals < elems_per_register) return;
118 k_min += loop_peeling_iters;
119 __m256i ks = _mm256_set_epi32 (
120 k_min+7,k_min+6,k_min+5,k_min+4,
121 k_min+3,k_min+2,k_min+1,k_min);
122 // Main SIMD extension loop
123 for (k=k_min;k<=k_max;k+=elems_per_register) {
124 __m256i offsets_vector = _mm256_lddqu_si256 ((__m256i*)&offsets[k]);
125 __m256i h_vector = offsets_vector;
126 __m256i v_vector = _mm256_sub_epi32(offsets_vector,ks);
127 ks =_mm256_add_epi32 (ks, eights);
128 // NULL offsets will read at index 0 (avoid segfaults)
129 __m256i null_mask = _mm256_cmpgt_epi32(offsets_vector,vector_null);
130 v_vector = _mm256_and_si256(null_mask,v_vector);
131 h_vector = _mm256_and_si256(null_mask,h_vector);
132 __m256i pattern_vector = _mm256_i32gather_epi32((int const*)&pattern[0],v_vector,1);
133 __m256i text_vector = _mm256_i32gather_epi32((int const*)&text[0],h_vector,1);
134 // Change endianess to make the xor + clz character comparison
135 pattern_vector = _mm256_shuffle_epi8(pattern_vector,vecShuffle);
136 text_vector = _mm256_shuffle_epi8(text_vector,vecShuffle);
137 __m256i xor_result_vector = _mm256_xor_si256(pattern_vector,text_vector);
138 __m256i clz_vector = avx2_lzcnt_epi32(xor_result_vector);
139 // Divide clz by 8 to get the number of equal characters
140 // Assume there are sentinels on sequences so we won't count characters
141 // outside the sequences
142 __m256i equal_chars = _mm256_srli_epi32(clz_vector,3);
143 offsets_vector = _mm256_add_epi32 (offsets_vector,equal_chars);
144 v_vector = _mm256_add_epi32 (v_vector,fours);
145 h_vector = _mm256_add_epi32 (h_vector,fours);
146 // Lanes to continue == 0xffffffff, other lanes = 0
147 __m256i vector_mask = _mm256_cmpeq_epi32(equal_chars,fours);
148 _mm256_storeu_si256((__m256i*)&offsets[k],offsets_vector);
149 int mask = _mm256_movemask_epi8(vector_mask);
150 if(mask == 0) continue;
151 // ctz(0) is undefined
152 while (mask != 0) {
153 int tz = __builtin_ctz(mask);
154 int curr_k = k + (tz/4);
155 const wf_offset_t offset = offsets[curr_k];
156 // Extend offset
157 if (offset >= 0) {
158 offsets[curr_k] = wavefront_extend_matches_packed_kernel(wf_aligner,curr_k,offset);
159 } else {
160 offsets[curr_k] = WAVEFRONT_OFFSET_NULL;
161 }
162 mask &= (0xfffffff0 << tz);
163 }
164 }
165}
