37void indel_dp_traceback(
38 score_matrix_t* const score_matrix,
39 cigar_t* const cigar) {
40 // Parameters
41 int** const dp = score_matrix->columns;
42 char* const operations = cigar->operations;
43 cigar->end_offset = cigar->max_operations;
44 int op_sentinel = cigar->end_offset - 1;
45 int h, v;
46 // Compute traceback
47 h = score_matrix->num_columns-1;
48 v = score_matrix->num_rows-1;
49 while (h>0 && v>0) {
50 if (dp[h][v]==dp[h][v-1]+1) {
51 operations[op_sentinel--] = 'D';
52 --v;
53 } else if (dp[h][v]==dp[h-1][v]+1) {
54 operations[op_sentinel--] = 'I';
55 --h;
56 } else if (dp[h][v]==dp[h-1][v-1]) {
57 operations[op_sentinel--] = 'M';
58 --h;
59 --v;
60 } else {
61 fprintf(stderr,"Indel backtrace. No backtrace operation found");
62 exit(1);
63 }
64 }
65 while (h>0) {operations[op_sentinel--] = 'I'; --h;}
66 while (v>0) {operations[op_sentinel--] = 'D'; --v;}
67 cigar->begin_offset = op_sentinel + 1;
68}
69void indel_dp_compute(
70 score_matrix_t* const score_matrix,
71 const char* const pattern,
72 const int pattern_length,
73 const char* const text,
74 const int text_length,
75 cigar_t* const cigar) {
76 // Parameters
77 int** dp = score_matrix->columns;
78 int h, v;
79 // Init DP
80 for (v=0;v<=pattern_length;++v) dp[0][v] = v; // No ends-free
81 for (h=0;h<=text_length;++h) dp[h][0] = h; // No ends-free
82 // Compute DP
83 for (h=1;h<=text_length;++h) {
84 for (v=1;v<=pattern_length;++v) {
85 int min = MIN(dp[h-1][v]+1,dp[h][v-1]+1); // Indel
86 if (text[h-1] == pattern[v-1]) {
87 min = MIN(min,dp[h-1][v-1]);
88 }
89 dp[h][v] = min;
90 }
91 }
92 // Compute traceback
93 indel_dp_traceback(score_matrix,cigar);
94 // DEBUG
95 // score_matrix_print(stderr,score_matrix,pattern,text);
96}
