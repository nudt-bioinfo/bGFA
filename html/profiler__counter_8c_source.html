43void counter_add(
44 profiler_counter_t* const counter,
45 const uint64_t amount) {
46 // Add to total & increment number of samples
47 counter->total += amount;
48 ++(counter->samples);
49 // From http://www.johndcook.com/standard_deviation.html
50 // See Knuth TAOCP vol 2, 3rd edition, page 232
51 if (counter->samples == 1) {
52 counter->min = amount;
53 counter->max = amount;
54 counter->m_oldM = amount;
55 counter->m_newM = amount;
56 counter->m_oldS = 0.0;
57 } else {
58 counter->min = MIN(counter->min,amount);
59 counter->max = MAX(counter->max,amount);
60 counter->m_newM = counter->m_oldM + ((double)amount-counter->m_oldM)/(double)counter->samples;
61 counter->m_newS = counter->m_oldS + ((double)amount-counter->m_oldM)*((double)amount-counter->m_newM);
62 counter->m_oldM = counter->m_newM;
63 counter->m_oldS = counter->m_newS;
64 }
65}

78double counter_get_mean(const profiler_counter_t* const counter) {
79 return (double)counter->total/(double)counter->samples;
80}
81double counter_get_variance(const profiler_counter_t* const counter) {
82 return ((counter->samples > 1) ? counter->m_newS/(double)(counter->samples - 1) : 0.0);
83}
84double counter_get_stddev(const profiler_counter_t* const counter) {
85 return sqrt(counter_get_variance(counter));
86}
87void counter_combine_sum(
88 profiler_counter_t* const counter_dst,
89 profiler_counter_t* const counter_src) {
90 counter_dst->total += counter_src->total;
91 counter_dst->samples += counter_src->samples;
92 counter_dst->min = MIN(counter_dst->min,counter_src->min);
93 counter_dst->max = MAX(counter_dst->max,counter_src->max);
94 if (counter_src->m_newS!=0.0) counter_dst->m_newS = counter_src->m_newS;
95 if (counter_src->m_newM!=0.0) counter_dst->m_newM = counter_src->m_newM;
96 if (counter_src->m_oldS!=0.0) counter_dst->m_oldS = counter_src->m_oldS;
97 if (counter_src->m_oldM!=0.0) counter_dst->m_oldM = counter_src->m_oldM;
98}
99void counter_print_stats(
100 FILE* const stream,
101 const profiler_counter_t* const counter,
102 const profiler_counter_t* const ref_counter,
103 const char* const units) {
104 // Print Samples
105 const uint64_t num_samples = counter_get_num_samples(counter);
106 if (num_samples >= METRIC_FACTOR_1G) {
107 fprintf(stream," (samples=%" PRIu64 "G",num_samples/METRIC_FACTOR_1G);
108 } else if (num_samples >= METRIC_FACTOR_1M) {
109 fprintf(stream," (samples=%" PRIu64 "M",num_samples/METRIC_FACTOR_1M);
110 } else if (num_samples >= METRIC_FACTOR_1K) {
111 fprintf(stream," (samples=%" PRIu64 "K",num_samples/METRIC_FACTOR_1K);
112 } else {
113 fprintf(stream," (samples=%" PRIu64 "",num_samples);
114 if (num_samples==0) {
115 fprintf(stream,",--n/a--)}\n");
116 return;
117 }
118 }
119 // Print Mean
120 const double mean = counter_get_mean(counter);
121 if (mean >= METRIC_FACTOR_1G) {
122 fprintf(stream,"{mean%.2fG",mean/METRIC_FACTOR_1G);
123 } else if (mean >= METRIC_FACTOR_1M) {
124 fprintf(stream,"{mean%.2fM",mean/METRIC_FACTOR_1M);
125 } else if (mean >= METRIC_FACTOR_1K) {
126 fprintf(stream,"{mean%.2fK",mean/METRIC_FACTOR_1K);
127 } else {
128 fprintf(stream,"{mean%.2f",mean);
129 }
130 // Print Min
131 const uint64_t min = counter_get_min(counter);
132 if (min >= METRIC_FACTOR_1G) {
133 fprintf(stream,",min%.2fG",(double)min/METRIC_FACTOR_1G);
134 } else if (min >= METRIC_FACTOR_1M) {
135 fprintf(stream,",min%.2fM",(double)min/METRIC_FACTOR_1M);
136 } else if (min >= METRIC_FACTOR_1K) {
137 fprintf(stream,",min%.2fK",(double)min/METRIC_FACTOR_1K);
138 } else {
139 fprintf(stream,",min%.2f",(double)min);
140 }
141 // Print Max
142 const uint64_t max = counter_get_max(counter);
143 if (max >= METRIC_FACTOR_1G) {
144 fprintf(stream,",Max%.2fG",(double)max/METRIC_FACTOR_1G);
145 } else if (max >= METRIC_FACTOR_1M) {
146 fprintf(stream,",Max%.2fM",(double)max/METRIC_FACTOR_1M);
147 } else if (max >= METRIC_FACTOR_1K) {
148 fprintf(stream,",Max%.2fK",(double)max/METRIC_FACTOR_1K);
149 } else {
150 fprintf(stream,",Max%.2f",(double)max);
151 }
152 // Print Variance
153 const uint64_t var = counter_get_variance(counter);
154 if (var >= METRIC_FACTOR_1G) {
155 fprintf(stream,",Var%.2fG",(double)var/METRIC_FACTOR_1G);
156 } else if (var >= METRIC_FACTOR_1M) {
157 fprintf(stream,",Var%.2fM",(double)var/METRIC_FACTOR_1M);
158 } else if (var >= METRIC_FACTOR_1K) {
159 fprintf(stream,",Var%.2fK",(double)var/METRIC_FACTOR_1K);
160 } else {
161 fprintf(stream,",Var%.2f",(double)var);
162 }
163 // Print Standard Deviation
164 const uint64_t stdDev = counter_get_stddev(counter);
165 if (stdDev >= METRIC_FACTOR_1G) {
166 fprintf(stream,",StdDev%.2fG)}\n",(double)stdDev/METRIC_FACTOR_1G);
167 } else if (stdDev >= METRIC_FACTOR_1M) {
168 fprintf(stream,",StdDev%.2fM)}\n",(double)stdDev/METRIC_FACTOR_1M);
169 } else if (stdDev >= METRIC_FACTOR_1K) {
170 fprintf(stream,",StdDev%.2fK)}\n",(double)stdDev/METRIC_FACTOR_1K);
171 } else {
172 fprintf(stream,",StdDev%.2f)}\n",(double)stdDev);
173 }

175void counter_print(
176 FILE* const stream,
177 const profiler_counter_t* const counter,
178 const profiler_counter_t* const ref_counter,
179 const char* const units,
180 const bool full_report) {
181 const uint64_t total = counter_get_total(counter);
182 // Print Total
183 if (total >= METRIC_FACTOR_1G) {
184 fprintf(stream,"%7.2f G%s",(double)total/METRIC_FACTOR_1G,units);
185 } else if (total >= METRIC_FACTOR_1M) {
186 fprintf(stream,"%7.2f M%s",(double)total/METRIC_FACTOR_1M,units);
187 } else if (total >= METRIC_FACTOR_1K) {
188 fprintf(stream,"%7.2f K%s",(double)total/METRIC_FACTOR_1K,units);
189 } else {
190 fprintf(stream,"%7.2f %s ",(double)total,units);
191 }
192 // Print percentage wrt reference
193 if (ref_counter!=NULL) {
194 if (total==0) {
195 fprintf(stream," ( 0.00 %%)");
196 } else {
197 const uint64_t total_ref = counter_get_total(ref_counter);
198 if (total_ref==0) {
199 fprintf(stream," ( n/a %%)");
200 } else {
201 const double percentage = (double)(total*100)/(double)total_ref;
202 fprintf(stream," (%6.02f %%)",percentage);
203 }
204 }
205 } else {
206 fprintf(stream," ");
207 }
208 // Full report
209 if (!full_report) {
210 fprintf(stream,"\n");
211 return;
212 } else {
213 counter_print_stats(stream,counter,ref_counter,units);
214 }
215}
216void percentage_print(
217 FILE* const stream,
218 const profiler_counter_t* const counter,
219 const char* const units) {
220 // Print Mean
221 const double mean = counter_get_mean(counter);
222 fprintf(stream,"%7.2f %%%s\t\t",mean,units);
223 // Print Samples
224 const uint64_t num_samples = counter_get_num_samples(counter);
225 if (num_samples >= METRIC_FACTOR_1G) {
226 fprintf(stream," (samples=%" PRIu64 "G",num_samples/METRIC_FACTOR_1G);
227 } else if (num_samples >= METRIC_FACTOR_1M) {
228 fprintf(stream," (samples=%" PRIu64 "M",num_samples/METRIC_FACTOR_1M);
229 } else if (num_samples >= METRIC_FACTOR_1K) {
230 fprintf(stream," (samples=%" PRIu64 "K",num_samples/METRIC_FACTOR_1K);
231 } else {
232 fprintf(stream," (samples=%" PRIu64 "",num_samples);
233 }
234 if (num_samples == 0) {
235 fprintf(stream,")\n");
236 return;
237 }
238 // Print Min/Max
239 fprintf(stream,",min%.2f%%,Max%.2f%%",
240 (double)counter_get_min(counter),(double)counter_get_max(counter));
241 // Print Variance/StandardDeviation
242 fprintf(stream,",Var%.2f,StdDev%.2f)\n",
243 counter_get_variance(counter),counter_get_stddev(counter));
244}
