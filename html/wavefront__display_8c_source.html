46int wavefront_display_compute_row_width(
47 const distance_metric_t distance_metric,
48 const int bt_length) {
49 // Compute number of components
50 int num_components = 1;
51 switch (distance_metric) {
52 case indel:
53 case edit:
54 case gap_linear: num_components = 1; break;
55 case gap_affine: num_components = 3; break;
56 case gap_affine_2p: num_components = 5; break;
57 default: break;
58 }
59 // Return
60 return ((num_components)*(4+(bt_length)));
61}
62void wavefront_display_compute_limits(
63 wavefront_aligner_t* const wf_aligner,
64 const int score_begin,
65 const int score_end,
66 int* const out_max_k,
67 int* const out_min_k) {
68 // Parameters
69 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
70 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
71 // Compute min/max k
72 int i, max_k=INT_MIN, min_k=INT_MAX;
73 for (i=score_begin;i<=score_end;++i) {
74 const int s = (wf_components->memory_modular) ? i%wf_components->max_score_scope : i;
75 wavefront_t* const mwavefront = wf_components->mwavefronts[s];
76 if (mwavefront != NULL) {
77 max_k = MAX(max_k,mwavefront->hi);
78 min_k = MIN(min_k,mwavefront->lo);
79 }
80 if (distance_metric <= gap_linear) continue;
81 wavefront_t* const i1wavefront = wf_components->i1wavefronts[s];
82 if (i1wavefront != NULL) {
83 max_k = MAX(max_k,i1wavefront->hi);
84 min_k = MIN(min_k,i1wavefront->lo);
85 }
86 wavefront_t* const d1wavefront = wf_components->d1wavefronts[s];
87 if (d1wavefront != NULL) {
88 max_k = MAX(max_k,d1wavefront->hi);
89 min_k = MIN(min_k,d1wavefront->lo);
90 }
91 if (distance_metric == gap_affine) continue;
92 wavefront_t* const i2wavefront = wf_components->i2wavefronts[s];
93 if (i2wavefront != NULL) {
94 max_k = MAX(max_k,i2wavefront->hi);
95 min_k = MIN(min_k,i2wavefront->lo);
96 }
97 wavefront_t* const d2wavefront = wf_components->d2wavefronts[s];
98 if (d2wavefront != NULL) {
99 max_k = MAX(max_k,d2wavefront->hi);
100 min_k = MIN(min_k,d2wavefront->lo);
101 }
102 }
103 // Return
104 *out_max_k = max_k;
105 *out_min_k = min_k;
106}

110void wavefront_display_print_element(
111 FILE* const stream,
112 wavefront_aligner_t* const wf_aligner,
113 wavefront_t* const wavefront,
114 const int k,
115 const int bt_length) {
116 // Check limits
117 if (wavefront!=NULL && wavefront->lo <= k && k <= wavefront->hi) {
118 const wf_offset_t offset = wavefront->offsets[k];
119 if (offset >= 0) {
120 fprintf(stream,"[%2d]",(int)offset);
121 // Print pcigar
122 if (bt_length > 0) {
123 char cigar_buffer[32];
124 const int pcigar_length = pcigar_unpack(wavefront->bt_pcigar[k],cigar_buffer);
125 const int effective_pcigar_length = MIN(bt_length,pcigar_length);
126 fprintf(stream,"%.*s",effective_pcigar_length,cigar_buffer);
127 PRINT_CHAR_REP(stream,' ',bt_length-effective_pcigar_length);
128 }
129 } else {
130 fprintf(stream,"[ *]");
131 PRINT_CHAR_REP(stream,' ',bt_length);
132 }
133 } else {
134 PRINT_CHAR_REP(stream,' ',bt_length+4);
135 }
136}
137void wavefront_display_print_frame(
138 FILE* const stream,
139 const int score_begin,
140 const int score_end,
141 const int row_width,
142 const int bt_length) {
143 // Align Y-label [k=000]
144 PRINT_CHAR_REP(stream,' ',WF_DISPLAY_YLABEL_LENGTH);
145 // Frame
146 fprintf(stream,"+");
147 const int num_blocks = score_end-score_begin+1;
148 const int frame_length = (num_blocks*row_width) + (num_blocks-1);
149 PRINT_CHAR_REP(stream,'-',frame_length);
150 fprintf(stream,"+\n");
151}
152void wavefront_display_print_score(
153 FILE* const stream,
154 const int score_begin,
155 const int score_end,
156 const int row_width,
157 const int bt_length) {
158 // Score label
159 PRINT_CHAR_REP(stream,' ',WF_DISPLAY_YLABEL_LENGTH);
160 int s;
161 for (s=score_begin;s<=score_end;++s) {
162 fprintf(stream,"|");
163 if (row_width >= 10) {
164 PRINT_CHAR_REP(stream,' ',row_width-10);
165 fprintf(stream,"%4d-score",s);
166 } else {
167 fprintf(stream,"s=%2d",s);
168 }
169 }
170 fprintf(stream,"|\n");
171}

187void wavefront_display_print_header(
188 FILE* const stream,
189 wavefront_aligner_t* const wf_aligner,
190 const int score_begin,
191 const int score_end,
192 const int bt_length) {
193 // Parameters
194 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
195 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
196 const int row_width = wavefront_display_compute_row_width(distance_metric,bt_length);
197 // Score header
198 fprintf(stream,"\n>[SCORE %d-%d]\n",score_begin,score_end);
199 // Frame
200 wavefront_display_print_frame(stream,score_begin,score_end,row_width,bt_length);
201 // Score header
202 wavefront_display_print_score(stream,score_begin,score_end,row_width,bt_length);
203 // Frame
204 wavefront_display_print_frame(stream,score_begin,score_end,row_width,bt_length);
205 // Wavefront labels
206 PRINT_CHAR_REP(stream,' ',WF_DISPLAY_YLABEL_LENGTH); // Align [k= ]
207 int score;
208 for (score=score_begin;score<=score_end;++score) {
209 const int s = (wf_components->memory_modular) ? score%wf_components->max_score_scope : score;
210 fprintf(stream,"|");
211 wavefront_display_print_header_component(stream,wf_components->mwavefronts[s]," M",bt_length);
212 if (distance_metric <= gap_linear) continue;
213 wavefront_display_print_header_component(stream,wf_components->i1wavefronts[s],"I1",bt_length);
214 wavefront_display_print_header_component(stream,wf_components->d1wavefronts[s],"D1",bt_length);
215 if (distance_metric == gap_affine) continue;
216 wavefront_display_print_header_component(stream,wf_components->i2wavefronts[s],"I2",bt_length);
217 wavefront_display_print_header_component(stream,wf_components->d2wavefronts[s],"D2",bt_length);
218 }
219 fprintf(stream,"|\n");
220 // Frame
221 wavefront_display_print_frame(stream,score_begin,score_end,row_width,bt_length);
222}

226void wavefront_aligner_print_block(
227 FILE* const stream,
228 wavefront_aligner_t* const wf_aligner,
229 const int score_begin,
230 const int score_end,
231 int bt_length) {
232 // Parameters
233 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
234 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
235 if (!wf_components->bt_piggyback) bt_length = 0; // Check BT
236 // Compute dinmensions
237 int max_k, min_k;
238 wavefront_display_compute_limits(wf_aligner,score_begin,score_end,&max_k,&min_k);
239 // Header
240 wavefront_display_print_header(stream,wf_aligner,score_begin,score_end,bt_length);
241 // Traverse all diagonals
242 int k;
243 for (k=max_k;k>=min_k;k--) {
244 fprintf(stream,"[k=%3d] ",k);
245 // Traverse all scores
246 int i;
247 for (i=score_begin;i<=score_end;++i) {
248 const int s = (wf_components->memory_modular) ? i%wf_components->max_score_scope : i;
249 fprintf(stream,"|");
250 // Fetch wavefront
251 wavefront_t* const mwavefront = wf_components->mwavefronts[s];
252 wavefront_display_print_element(stream,wf_aligner,mwavefront,k,bt_length);
253 if (distance_metric <= gap_linear) continue;
254 wavefront_t* const i1wavefront = wf_components->i1wavefronts[s];
255 wavefront_t* const d1wavefront = wf_components->d1wavefronts[s];
256 wavefront_display_print_element(stream,wf_aligner,i1wavefront,k,bt_length);
257 wavefront_display_print_element(stream,wf_aligner,d1wavefront,k,bt_length);
258 if (distance_metric == gap_affine) continue;
259 wavefront_t* const i2wavefront = wf_components->i2wavefronts[s];
260 wavefront_t* const d2wavefront = wf_components->d2wavefronts[s];
261 wavefront_display_print_element(stream,wf_aligner,i2wavefront,k,bt_length);
262 wavefront_display_print_element(stream,wf_aligner,d2wavefront,k,bt_length);
263 }
264 fprintf(stream,"|\n");
265 }
266 // Footer
267 const int row_width = wavefront_display_compute_row_width(distance_metric,bt_length);
268 wavefront_display_print_frame(stream,score_begin,score_end,row_width,bt_length);
269}
270void wavefront_aligner_print(
271 FILE* const stream,
272 wavefront_aligner_t* const wf_aligner,
273 const int score_begin,
274 const int score_end,
275 const int num_wfs_per_row,
276 const int backtrace_length) {
277 // Print wavefronts by chunks
278 int s;
279 for (s=MAX(score_begin,0);s<=score_end;s+=num_wfs_per_row-1) {
280 const int block_score_end = MIN(s+num_wfs_per_row-1,score_end);
281 wavefront_aligner_print_block(stream,wf_aligner,s,block_score_end,backtrace_length);
282 if (block_score_end == score_end) break;
283 }
