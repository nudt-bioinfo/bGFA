41/*
42 * Compute Kernels
43 */
44void wavefront_compute_linear_idm(
45 wavefront_aligner_t* const wf_aligner,
46 const wavefront_set_t* const wavefront_set,
47 const int lo,
48 const int hi) {
49 // Parameters
50 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
51 const int pattern_length = sequences->pattern_length;
52 const int text_length = sequences->text_length;
53 // In Offsets
54 const wf_offset_t* const m_misms = wavefront_set->in_mwavefront_misms->offsets;
55 const wf_offset_t* const m_open1 = wavefront_set->in_mwavefront_open1->offsets;
56 // Out Offsets
57 wf_offset_t* const out_m = wavefront_set->out_mwavefront->offsets;
58 // Compute-Next kernel loop
59 int k;
60 PRAGMA_LOOP_VECTORIZE
61 for (k=lo;k<=hi;++k) {
62 // Compute maximum Offset
63 const wf_offset_t ins1 = m_open1[k-1];
64 const wf_offset_t del1 = m_open1[k+1];
65 const wf_offset_t misms = m_misms[k];
66 wf_offset_t max = MAX(del1,MAX(misms,ins1)+1);
67 // Adjust offset out of boundaries !(h>tlen,v>plen) (here to allow vectorization)
68 const wf_unsigned_offset_t h = WAVEFRONT_H(k,max); // Make unsigned to avoid checking negative
69 const wf_unsigned_offset_t v = WAVEFRONT_V(k,max); // Make unsigned to avoid checking negative
70 if (h > text_length) max = WAVEFRONT_OFFSET_NULL;
71 if (v > pattern_length) max = WAVEFRONT_OFFSET_NULL;
72 out_m[k] = max;
73 }
74}
75/*
76 * Compute Kernel (Piggyback)
77 */
78void wavefront_compute_linear_idm_piggyback(
79 wavefront_aligner_t* const wf_aligner,
80 const wavefront_set_t* const wavefront_set,
81 const int lo,
82 const int hi) {
83 // Parameters
84 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
85 const int pattern_length = sequences->pattern_length;
86 const int text_length = sequences->text_length;
87 // In M
88 const wf_offset_t* const m_misms = wavefront_set->in_mwavefront_misms->offsets;
89 const pcigar_t* const m_misms_bt_pcigar = wavefront_set->in_mwavefront_misms->bt_pcigar;
90 const bt_block_idx_t* const m_misms_bt_prev = wavefront_set->in_mwavefront_misms->bt_prev;
91 // In I/D
92 const wf_offset_t* const m_open1 = wavefront_set->in_mwavefront_open1->offsets;
93 const pcigar_t* const m_open1_bt_pcigar = wavefront_set->in_mwavefront_open1->bt_pcigar;
94 const bt_block_idx_t* const m_open1_bt_prev = wavefront_set->in_mwavefront_open1->bt_prev;
95 // Out
96 wf_offset_t* const out_m = wavefront_set->out_mwavefront->offsets;
97 pcigar_t* const out_m_bt_pcigar = wavefront_set->out_mwavefront->bt_pcigar;
98 bt_block_idx_t* const out_m_bt_prev = wavefront_set->out_mwavefront->bt_prev;
99 // Compute-Next kernel loop
100 int k;
101 PRAGMA_LOOP_VECTORIZE // Ifs predicated by the compiler
102 for (k=lo;k<=hi;++k) {
103 // Compute maximum Offset
104 const wf_offset_t ins1 = m_open1[k-1] + 1;
105 const wf_offset_t del1 = m_open1[k+1];
106 const wf_offset_t misms = m_misms[k] + 1;
107 wf_offset_t max = MAX(del1,MAX(misms,ins1));
108 // Update pcigar & bt-block
109 if (max == ins1) {
110 out_m_bt_pcigar[k] = PCIGAR_PUSH_BACK_INS(m_open1_bt_pcigar[k-1]);
111 out_m_bt_prev[k] = m_open1_bt_prev[k-1];
112 }
113 if (max == del1) {
114 out_m_bt_pcigar[k] = PCIGAR_PUSH_BACK_DEL(m_open1_bt_pcigar[k+1]);
115 out_m_bt_prev[k] = m_open1_bt_prev[k+1];
116 }
117 if (max == misms) {
118 out_m_bt_pcigar[k] = PCIGAR_PUSH_BACK_MISMS(m_misms_bt_pcigar[k]);
119 out_m_bt_prev[k] = m_misms_bt_prev[k];
120 }
121 // Adjust offset out of boundaries !(h>tlen,v>plen) (here to allow vectorization)
122 const wf_unsigned_offset_t h = WAVEFRONT_H(k,max); // Make unsigned to avoid checking negative
123 const wf_unsigned_offset_t v = WAVEFRONT_V(k,max); // Make unsigned to avoid checking negative
124 if (h > text_length) max = WAVEFRONT_OFFSET_NULL;
125 if (v > pattern_length) max = WAVEFRONT_OFFSET_NULL;
126 out_m[k] = max;
127 }
128}
129/*
130 * Compute Wavefronts (gap-linear)
131 */
132void wavefront_compute_linear_dispatcher(
133 wavefront_aligner_t* const wf_aligner,
134 wavefront_set_t* const wavefront_set,
135 const int lo,
136 const int hi) {
137 // Parameters
138 const bool bt_piggyback = wf_aligner->wf_components.bt_piggyback;
139 const int num_threads = wavefront_compute_num_threads(wf_aligner,lo,hi);
140 // Multithreading dispatcher
141 if (num_threads == 1) {
142 // Compute next wavefront
143 if (bt_piggyback) {
144 wavefront_compute_linear_idm_piggyback(wf_aligner,wavefront_set,lo,hi);
145 } else {
146 wavefront_compute_linear_idm(wf_aligner,wavefront_set,lo,hi);
147 }
148 } else {
149#ifdef WFA_PARALLEL
150 // Compute next wavefront in parallel
151 #pragma omp parallel num_threads(num_threads)
152 {
153 int t_lo, t_hi;
154 const int thread_id = omp_get_thread_num();
155 const int thread_num = omp_get_num_threads();
156 wavefront_compute_thread_limits(thread_id,thread_num,lo,hi,&t_lo,&t_hi);
157 if (bt_piggyback) {
158 wavefront_compute_linear_idm_piggyback(wf_aligner,wavefront_set,t_lo,t_hi);
159 } else {
160 wavefront_compute_linear_idm(wf_aligner,wavefront_set,t_lo,t_hi);
161 }
162 }
163#endif
164 }

166void wavefront_compute_linear(
167 wavefront_aligner_t* const wf_aligner,
168 const int score) {
169 // Select wavefronts
170 wavefront_set_t wavefront_set;
171 wavefront_compute_fetch_input(wf_aligner,&wavefront_set,score);
172 // Check null wavefronts
173 if (wavefront_set.in_mwavefront_misms->null &&
174 wavefront_set.in_mwavefront_open1->null) {
175 wf_aligner->align_status.num_null_steps++; // Increment null-steps
176 wavefront_compute_allocate_output_null(wf_aligner,score); // Null s-wavefront
177 return;
178 }
179 wf_aligner->align_status.num_null_steps = 0;
180 // Set limits
181 int hi, lo;
182 wavefront_compute_limits_input(wf_aligner,&wavefront_set,&lo,&hi);
183 // Allocate wavefronts
184 wavefront_compute_allocate_output(wf_aligner,&wavefront_set,score,lo,hi);
185 // Init wavefront ends
186 wavefront_compute_init_ends(wf_aligner,&wavefront_set,lo,hi);
187 // Compute Wavefronts
188 wavefront_compute_linear_dispatcher(wf_aligner,&wavefront_set,lo,hi);
189 // Offload backtrace (if necessary)
190 if (wf_aligner->wf_components.bt_piggyback) {
191 wavefront_backtrace_offload_linear(wf_aligner,&wavefront_set,lo,hi);
192 }
193 // Process wavefront ends
194 wavefront_compute_process_ends(wf_aligner,&wavefront_set,score);
195}
