64void wavefront_sequences_init_allocate(
65 wavefront_sequences_t* const wf_sequences,
66 const int pattern_length,
67 const int text_length) {
68 // Compute dimensions
69 const int buffer_size = pattern_length + text_length + 3*WF_SEQUENCES_PADDING;
70 // Check internal buffer allocated
71 if (wf_sequences->seq_buffer_allocated < buffer_size) {
72 // Free
73 if (wf_sequences->seq_buffer != NULL) free(wf_sequences->seq_buffer);
74 // Allocate
75 const int proposed_size = buffer_size + buffer_size/2;
76 wf_sequences->seq_buffer = calloc(proposed_size,1);
77 wf_sequences->seq_buffer_allocated = proposed_size;
78 }
79 // Assign memory
80 wf_sequences->pattern_buffer = wf_sequences->seq_buffer + WF_SEQUENCES_PADDING;
81 wf_sequences->text_buffer = wf_sequences->seq_buffer + WF_SEQUENCES_PADDING + pattern_length + WF_SEQUENCES_PADDING;
82}
83void wavefront_sequences_init_copy(
84 char* const buffer_dst,
85 const char* const sequence,
86 const int sequence_length,
87 const int padding_length,
88 const char padding_value,
89 const bool reverse) {
90 // Copy sequence
91 if (reverse) {
92 int i;
93 for (i=0;i<sequence_length;i++) {
94 buffer_dst[i] = sequence[sequence_length-1-i];
95 }
96 } else {
97 memcpy(buffer_dst,sequence,sequence_length);
98 }
99 // Add end padding
100 buffer_dst[sequence_length] = padding_value;
101}
102void wavefront_sequences_init_decode2bits(
103 char* const buffer_dst,
104 const uint8_t* const sequence,
105 const int sequence_length,
106 const int padding_length,
107 const char padding_value,
108 const bool reverse) {
109 // Parameters
110 const char dna_packed2bits_decode[4] = {'A','C','G','T'};
111 // Compute dimensions
112 const int num_words = DIV_CEIL(sequence_length,8);
113 int buffer_pos = (reverse) ? sequence_length-1 : 0;
114 // Decode and copy sequence
115 int word_num;
116 for (word_num=0;word_num<num_words;++word_num) {
117 // Fetch next word
118 const uint8_t word = sequence[word_num];
119 // Decode 4 letters packed
120 const char letter0 = dna_packed2bits_decode[(word & 3)];
121 const char letter1 = dna_packed2bits_decode[(word>>2 & 3)];
122 const char letter2 = dna_packed2bits_decode[(word>>4 & 3)];
123 const char letter3 = dna_packed2bits_decode[(word>>6 & 3)];
124 if (reverse) {
125 buffer_dst[buffer_pos ] = letter0;
126 buffer_dst[buffer_pos-1] = letter1;
127 buffer_dst[buffer_pos-2] = letter2;
128 buffer_dst[buffer_pos-3] = letter3;
129 buffer_pos -= 4;
130 } else {
131 buffer_dst[buffer_pos ] = letter0;
132 buffer_dst[buffer_pos+1] = letter1;
133 buffer_dst[buffer_pos+2] = letter2;
134 buffer_dst[buffer_pos+3] = letter3;
135 buffer_pos += 4;
136 }
137 }
138 // Add end padding
139 buffer_dst[sequence_length] = padding_value;
140}
141void wavefront_sequences_init_ascii(
142 wavefront_sequences_t* const wf_sequences,
143 const char* const pattern,
144 const int pattern_length,
145 const char* const text,
146 const int text_length,
147 const bool reverse) {
148 // Mode
149 wf_sequences->mode = wf_sequences_ascii;
150 wf_sequences->reverse = reverse;
151 // Allocate buffers
152 wavefront_sequences_init_allocate(wf_sequences,pattern_length,text_length);
153 // Copy internal sequences
154 wavefront_sequences_init_copy(wf_sequences->pattern_buffer,
155 pattern,pattern_length,WF_SEQUENCES_PADDING,WF_SEQUENCES_PATTERN_EOS,reverse);
156 wf_sequences->pattern_buffer_length = pattern_length;
157 wavefront_sequences_init_copy(wf_sequences->text_buffer,
158 text,text_length,WF_SEQUENCES_PADDING,WF_SEQUENCES_TEXT_EOS,reverse);
159 wf_sequences->text_buffer_length = text_length;
160 // Set pattern
161 wf_sequences->pattern = wf_sequences->pattern_buffer;
162 wf_sequences->pattern_begin = 0;
163 wf_sequences->pattern_length = pattern_length;
164 wf_sequences->pattern_eos = wf_sequences->pattern[pattern_length];
165 // Set text
166 wf_sequences->text = wf_sequences->text_buffer;
167 wf_sequences->text_begin = 0;
168 wf_sequences->text_length = text_length;
169 wf_sequences->text_eos = wf_sequences->text[text_length];
170}

225bool wavefront_sequences_cmp(
226 wavefront_sequences_t* const wf_sequences,
227 const int pattern_pos,
228 const int text_pos) {
229 // Select mode
230 if (wf_sequences->mode == wf_sequences_lambda) {
231 // Custom function to compare sequences
232 alignment_match_funct_t match_funct = wf_sequences->match_funct;
233 void* match_funct_arguments = wf_sequences->match_funct_arguments;
234 // Check coordinates (EOS)
235 const int pattern_length = wf_sequences->pattern_length;
236 const int text_length = wf_sequences->text_length;
237 if (pattern_pos >= pattern_length || text_pos >= text_length) return false;
238 // Compare using lambda (given coordinates)
239 const int pattern_begin = wf_sequences->pattern_begin;
240 const int text_begin = wf_sequences->text_begin;
241 if (wf_sequences->reverse) {
242 const int pattern_end = pattern_begin + pattern_length - 1;
243 const int text_end = text_begin + text_length - 1;
244 return match_funct(pattern_end-pattern_pos,text_end-text_pos,match_funct_arguments);
245 } else {
246 return match_funct(pattern_begin+pattern_pos,text_begin+text_pos,match_funct_arguments);
247 }
248 } else {
249 // Compare regular strings
250 return wf_sequences->pattern[pattern_pos] == wf_sequences->text[text_pos];
251 }
252}

274void wavefront_sequences_set_bounds(
275 wavefront_sequences_t* const wf_sequences,
276 const int pattern_begin,
277 const int pattern_end,
278 const int text_begin,
279 const int text_end) {
280 // Select mode
281 if (wf_sequences->mode != wf_sequences_lambda) {
282 // Restore previous EOS char
283 const int pattern_length_old = wf_sequences->pattern_length;
284 const int text_length_old = wf_sequences->text_length;
285 wf_sequences->pattern[pattern_length_old] = wf_sequences->pattern_eos;
286 wf_sequences->text[text_length_old] = wf_sequences->text_eos;
287 // Focus on the new section of the sequences
288 if (wf_sequences->reverse) {
289 // Compare given coordinates
290 wf_sequences->pattern = wf_sequences->pattern_buffer + (wf_sequences->pattern_buffer_length - pattern_end);
291 wf_sequences->text = wf_sequences->text_buffer + (wf_sequences->text_buffer_length - text_end);
292 } else {
293 wf_sequences->pattern = wf_sequences->pattern_buffer + pattern_begin;
294 wf_sequences->text = wf_sequences->text_buffer + text_begin;
295 }
296 // Save EOS char and truncate sequence
297 const int pattern_length_new = pattern_end - pattern_begin;
298 const int text_length_new = text_end - text_begin;
299 wf_sequences->pattern_eos = wf_sequences->pattern[pattern_length_new];
300 wf_sequences->text_eos = wf_sequences->text[text_length_new];
301 wf_sequences->pattern[pattern_length_new] = WF_SEQUENCES_PATTERN_EOS;
302 wf_sequences->text[text_length_new] = WF_SEQUENCES_TEXT_EOS;
303 }
304 // Set bounds
305 wf_sequences->pattern_begin = pattern_begin;
306 wf_sequences->pattern_length = pattern_end - pattern_begin;
307 wf_sequences->text_begin = text_begin;
308 wf_sequences->text_length = text_end - text_begin;
309}
