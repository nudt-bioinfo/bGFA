42 * CIGAR
43 */
44typedef struct {
45 // Alignment operations
46 char* operations; // Raw alignment operations
47 int max_operations; // Maximum buffer size
48 int begin_offset; // Begin offset
49 int end_offset; // End offset
50 // Score and end position (useful for partial alignments like Z-dropped)
51 int score; // Computed scored
52 int end_v; // Alignment-end vertical coordinate (pattern characters aligned)
53 int end_h; // Alignment-end horizontal coordinate (text characters aligned)
54 // CIGAR (SAM compliant)
55 bool has_misms; // Show 'X' and '=', instead of just 'M'
56 uint32_t* cigar_buffer; // CIGAR-operations (max_operations length)
57 int cigar_length; // Total CIGAR-operations
58} cigar_t;
