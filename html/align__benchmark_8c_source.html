54/*
55 * WFA lambda (custom match function)
56 */
57typedef struct {
58 char* pattern;
59 int pattern_length;
60 char* text;
61 int text_length;
62} match_function_params_t;
63match_function_params_t lambda_params;
64// Simplest Extend-matching function (for testing purposes)
65int lambda_function(int v,int h,void* arguments) {
66 // Extract parameters
67 match_function_params_t* const match_arguments = (match_function_params_t*)arguments;
68 // Check match
69 if (v >= match_arguments->pattern_length || h >= match_arguments->text_length) return 0;
70 return (match_arguments->pattern[v] == match_arguments->text[h]);
71}

86void align_pairwise_test() {
87 // Patters & Texts
88 char * pattern = "GATTACA";
89 char * text = "GATCACTA";
90
91 // Penalties
92 linear_penalties_t linear_penalties = {
93 .match = 0,
94 .mismatch = 4,
95 .indel = 2,
96 };
97 affine_penalties_t affine_penalties = {
98 .match = 0,
99 .mismatch = 4, //9,
100 .gap_opening = 6, //13,
101 .gap_extension = 2,
102 };
103 // Ends
104 const int pattern_begin_free = 0;
105 const int pattern_end_free = 0;
106 const int text_begin_free = 0;
107 const int text_end_free = 0;
108 const bool endsfree =
109 pattern_begin_free>0 || pattern_end_free>0 ||
110 text_begin_free>0 || text_end_free>0;
111 /*
112 * Gap-Affine
113 */
114 // Allocate
115 wavefront_aligner_attr_t attributes = wavefront_aligner_attr_default;
116 attributes.distance_metric = gap_affine;
117 attributes.linear_penalties = linear_penalties;
118 attributes.affine_penalties = affine_penalties;
119 attributes.heuristic.strategy = wf_heuristic_none;
120 attributes.heuristic.min_wavefront_length = 256;
121 attributes.heuristic.max_distance_threshold = 4096;
122 attributes.heuristic.steps_between_cutoffs = 10;
123 attributes.alignment_scope = compute_alignment; // compute_score
124 attributes.memory_mode = wavefront_memory_med;
125 attributes.alignment_form.span = (endsfree) ? alignment_endsfree : alignment_end2end;
126 attributes.alignment_form.pattern_begin_free = pattern_begin_free;
127 attributes.alignment_form.pattern_end_free = pattern_end_free;
128 attributes.alignment_form.text_begin_free = text_begin_free;
129 attributes.alignment_form.text_end_free = text_end_free;
130 attributes.plot.enabled = false;
131 wavefront_aligner_t* const wf_aligner = wavefront_aligner_new(&attributes);
132 // Align
133 wavefront_align(wf_aligner,
134 pattern,strlen(pattern),text,strlen(text));
135 // CIGAR
136 fprintf(stderr,">> WFA2");
137 cigar_print_pretty(stderr,wf_aligner->cigar,pattern,strlen(pattern),text,strlen(text));
138 fprintf(stderr,"SCORE: %d \n",cigar_score_gap_affine(
139 wf_aligner->cigar,&affine_penalties));
140 // Plot
141 if (attributes.plot.enabled) {
142 FILE* const wf_plot = fopen("test.wfa","w");
143 wavefront_plot_print(wf_plot,wf_aligner);
144 fclose(wf_plot);
145 }
146 // Free
147 wavefront_aligner_delete(wf_aligner);
148}

152wavefront_aligner_t* align_input_configure_wavefront(
153 align_input_t* const align_input) {
154 // Set attributes
155 wavefront_aligner_attr_t attributes = wavefront_aligner_attr_default;
156 attributes.memory_mode = parameters.wfa_memory_mode;
157 if (parameters.wfa_score_only) {
158 attributes.alignment_scope = compute_score;
159 }
160 // WF-Heuristic
161 switch (parameters.wfa_heuristic) {
162 case wf_heuristic_none:
163 attributes.heuristic.strategy = wf_heuristic_none;
164 break;
165 case wf_heuristic_banded_static:
166 attributes.heuristic.strategy = wf_heuristic_banded_static;
167 attributes.heuristic.min_k = parameters.wfa_heuristic_p1;
168 attributes.heuristic.max_k = parameters.wfa_heuristic_p2;
169 break;
170 case wf_heuristic_banded_adaptive:
171 attributes.heuristic.strategy = wf_heuristic_banded_adaptive;
172 attributes.heuristic.min_k = parameters.wfa_heuristic_p1;
173 attributes.heuristic.max_k = parameters.wfa_heuristic_p2;
174 attributes.heuristic.steps_between_cutoffs = parameters.wfa_heuristic_p3;
175 break;
176 case wf_heuristic_wfadaptive:
177 attributes.heuristic.strategy = wf_heuristic_wfadaptive;
178 attributes.heuristic.min_wavefront_length = parameters.wfa_heuristic_p1;
179 attributes.heuristic.max_distance_threshold = parameters.wfa_heuristic_p2;
180 attributes.heuristic.steps_between_cutoffs = parameters.wfa_heuristic_p3;
181 break;
182 case wf_heuristic_xdrop:
183 attributes.heuristic.strategy = wf_heuristic_xdrop;
184 attributes.heuristic.xdrop = parameters.wfa_heuristic_p1;
185 attributes.heuristic.steps_between_cutoffs = parameters.wfa_heuristic_p2;
186 break;
187 case wf_heuristic_zdrop:
188 attributes.heuristic.strategy = wf_heuristic_zdrop;
189 attributes.heuristic.zdrop = parameters.wfa_heuristic_p1;
190 attributes.heuristic.steps_between_cutoffs = parameters.wfa_heuristic_p2;
191 break;
192 default:
193 break;
194 }
195 // Select flavor
196 switch (parameters.algorithm) {
197 case alignment_indel_wavefront:
198 attributes.distance_metric = indel;
199 break;
200 case alignment_edit_wavefront:
201 attributes.distance_metric = edit;
202 break;
203 case alignment_gap_linear_wavefront:
204 attributes.distance_metric = gap_linear;
205 attributes.linear_penalties = parameters.linear_penalties;
206 break;
207 case alignment_gap_affine_wavefront:
208 attributes.distance_metric = gap_affine;
209 attributes.affine_penalties = parameters.affine_penalties;
210 break;
211 case alignment_gap_affine2p_wavefront:
212 attributes.distance_metric = gap_affine_2p;
213 attributes.affine2p_penalties = parameters.affine2p_penalties;
214 break;
215 default:
216 return NULL; // No WF selected
217 break;
218 }
219 // Select alignment form
220 if (parameters.align_span_extension) {
221 attributes.alignment_form.span = alignment_endsfree;
222 attributes.alignment_form.extension = true;
223 } else if (parameters.align_span_endsfree) {
224 attributes.alignment_form.span = alignment_endsfree;
225 attributes.alignment_form.extension = false;
226 } else { // Global
227 attributes.alignment_form.span = alignment_end2end;
228 attributes.alignment_form.extension = false;
229 }
230 // Misc
231 attributes.plot.enabled = (parameters.plot != 0);
232 attributes.plot.align_level = (parameters.plot < 0) ? -1 : parameters.plot - 1;
233 attributes.system.verbose = parameters.verbose;
234 attributes.system.max_memory_abort = parameters.wfa_max_memory;
235 attributes.system.max_alignment_steps = parameters.wfa_max_steps;
236 attributes.system.max_num_threads = parameters.wfa_max_threads;
237 // Allocate
238 return wavefront_aligner_new(&attributes);
239}

278void align_input_configure_local(
279 align_input_t* const align_input) {
280 // Ends-free configuration
281 if (parameters.align_span_endsfree) {
282 const int plen = align_input->pattern_length;
283 const int tlen = align_input->text_length;
284 align_input->pattern_begin_free = nominal_prop_u32(plen,parameters.pattern_begin_free);

287 align_input->text_end_free = nominal_prop_u32(tlen,parameters.text_end_free);

290 align_input->pattern_begin_free,align_input->pattern_end_free,
291 align_input->text_begin_free,align_input->text_end_free);
292 }
293 }
294 // Custom extend-match function
295 if (parameters.wfa_lambda) {
296 lambda_params.pattern = align_input->pattern;
297 lambda_params.pattern_length = align_input->pattern_length;
298 lambda_params.text = align_input->text;
299 lambda_params.text_length = align_input->text_length;

310bool align_benchmark_read_input(
311 FILE* input_file,
312 char** line1,
313 char** line2,
314 size_t* line1_allocated,
315 size_t* line2_allocated,
316 const int seqs_processed,
317 align_input_t* const align_input) {
318 // Parameters
319 int line1_length=0, line2_length=0;
320 // Read queries
321 line1_length = getline(line1,line1_allocated,input_file);
322 if (line1_length==-1) return false;
323 line2_length = getline(line2,line2_allocated,input_file);
324 if (line1_length==-1) return false;
325 // Configure input
326 align_input->sequence_id = seqs_processed;
327 align_input->pattern = *line1 + 1;
328 align_input->pattern_length = line1_length - 2;
329 align_input->pattern[align_input->pattern_length] = '\0';
330 align_input->text = *line2 + 1;
331 align_input->text_length = line2_length - 2;
332 if (align_input->text[align_input->text_length] == '\n') {
333 align_input->text[align_input->text_length] = '\0';
334 }
335 return true;
336}

449void align_benchmark_sequential() {
450 // PROFILE
451 timer_reset(&parameters.timer_global);
452 // I/O files
453 parameters.input_file = fopen(parameters.input_filename, "r");
454 if (parameters.input_file == NULL) {
455 fprintf(stderr,"Input file '%s' couldn't be opened\n",parameters.input_filename);
456 exit(1);
457 }
458 if (parameters.output_filename != NULL) {
459 parameters.output_file = fopen(parameters.output_filename, "w");
460 }
461 // Global configuration
462 align_input_t align_input;
463 align_input_configure_global(&align_input);
464 // Read-align loop
465 int seqs_processed = 0, progress = 0;
466 while (true) {
467 // Read input sequence-pair
468 const bool input_read = align_benchmark_read_input(
469 parameters.input_file,&parameters.line1,&parameters.line2,
470 &parameters.line1_allocated,&parameters.line2_allocated,
471 seqs_processed,&align_input);
472 if (!input_read) break;
473 // Execute the selected algorithm
474 timer_start(&parameters.timer_global); // PROFILE
475 align_benchmark_run_algorithm(&align_input);
476 timer_stop(&parameters.timer_global); // PROFILE
477 // Update progress
478 ++seqs_processed;
479 if (++progress == parameters.progress) {
480 progress = 0;
481 if (parameters.verbose >= 0) align_benchmark_print_progress(seqs_processed);
482 }
483 // DEBUG
484 // mm_allocator_print(stderr,align_input.wf_aligner->mm_allocator,false);
485 // mm_allocator_print(stderr,align_input.wf_aligner->bialigner->alg_forward->mm_allocator,false);
486 // mm_allocator_print(stderr,align_input.wf_aligner->bialigner->alg_reverse->mm_allocator,false);
487 // mm_allocator_print(stderr,align_input.wf_aligner->bialigner->wf_forward_base->mm_allocator,false);
488 // mm_allocator_print(stderr,align_input.wf_aligner->bialigner->wf_reverse_base->mm_allocator,false);
489 // Plot
490 if (parameters.plot != 0) align_benchmark_plot_wf(&align_input,seqs_processed);
491 }
492 // Print benchmark results
493 if (parameters.verbose >= 0) align_benchmark_print_results(&align_input,seqs_processed,true);
494 // Free
495 align_benchmark_free(&align_input);
496 fclose(parameters.input_file);
497 if (parameters.output_file) fclose(parameters.output_file);
498 free(parameters.line1);
499 free(parameters.line2);
500}
501void align_benchmark_parallel() {
502 // PROFILE
503 timer_reset(&parameters.timer_global);
504 // Open input file
505 parameters.input_file = fopen(parameters.input_filename, "r");
506 if (parameters.input_file == NULL) {
507 fprintf(stderr,"Input file '%s' couldn't be opened\n",parameters.input_filename);
508 exit(1);
509 }
510 if (parameters.output_filename != NULL) {
511 parameters.output_file = fopen(parameters.output_filename, "w");
512 }
513 // Global configuration
514 align_input_t align_input[parameters.num_threads];
515 for (int tid=0;tid<parameters.num_threads;++tid) {
516 align_input_configure_global(align_input+tid);
517 }
518 // Read-align loop
519 sequence_buffer_t* const sequence_buffer = sequence_buffer_new(2*parameters.batch_size,100);
520 int seqs_processed = 0, progress = 0, seqs_batch = 0;
521 while (true) {
522 // Read batch-input sequence-pair
523 sequence_buffer_clear(sequence_buffer);
524 for (seqs_batch=0;seqs_batch<parameters.batch_size;++seqs_batch) {
525 const bool seqs_pending = align_benchmark_read_input(
526 parameters.input_file,&parameters.line1,&parameters.line2,
527 &parameters.line1_allocated,&parameters.line2_allocated,
528 seqs_processed,align_input);
529 if (!seqs_pending) break;
530 // Add pair pattern-text
531 sequence_buffer_add_pair(sequence_buffer,
532 align_input->pattern,align_input->pattern_length,
533 align_input->text,align_input->text_length);
534 }
535 if (seqs_batch == 0) break;
536 // Parallel processing of the sequences batch
537 timer_start(&parameters.timer_global); // PROFILE
538#ifdef WFA_PARALLEL
539 #pragma omp parallel num_threads(parameters.num_threads)
540 {
541 int tid = omp_get_thread_num();
542
543 #pragma omp for
544 for (int seq_idx=0;seq_idx<seqs_batch;++seq_idx) {
545 // Configure sequence
546 sequence_offset_t* const offset = sequence_buffer->offsets + seq_idx;
547 align_input[tid].sequence_id = seqs_processed;
548 align_input[tid].pattern = sequence_buffer->buffer + offset->pattern_offset;
549 align_input[tid].pattern_length = offset->pattern_length;
550 align_input[tid].text = sequence_buffer->buffer + offset->text_offset;
551 align_input[tid].text_length = offset->text_length;
552 // Execute the selected algorithm
553 align_benchmark_run_algorithm(align_input+tid);
554 }
555 }
556#endif
557 timer_stop(&parameters.timer_global); // PROFILE
558 // Update progress
559 seqs_processed += seqs_batch;
560 progress += seqs_batch;
561 if (progress >= parameters.progress) {
562 progress -= parameters.progress;
563 if (parameters.verbose >= 0) align_benchmark_print_progress(seqs_processed);
564 }
565 // DEBUG
566 //mm_allocator_print(stderr,align_input->wf_aligner->mm_allocator,false);
567 }
568 // Print benchmark results
569 if (parameters.verbose >= 0) align_benchmark_print_results(align_input,seqs_processed,true);
570 // Free
571 for (int tid=0;tid<parameters.num_threads;++tid) {
572 align_benchmark_free(align_input+tid);
573 }
574 sequence_buffer_delete(sequence_buffer);
575 fclose(parameters.input_file);
576 if (parameters.output_file) fclose(parameters.output_file);
577 free(parameters.line1);
578 free(parameters.line2);
579}
