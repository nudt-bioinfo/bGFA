48FORCE_INLINE wf_offset_t wavefront_extend_matches_kernel_charwise(
49 wavefront_aligner_t* const wf_aligner,
50 const int k,
51 wf_offset_t offset) {
52 // Fetch pattern/text
53 char* pattern_ptr = wf_aligner->sequences.pattern + WAVEFRONT_V(k,offset);
54 char* text_ptr = wf_aligner->sequences.text + WAVEFRONT_H(k,offset);
55 // Compare 64-bits blocks
56 while (*pattern_ptr == *text_ptr) {
57 // Increment offset
58 offset++;
59 // Next chars
60 ++pattern_ptr;
61 ++text_ptr;
62 }
63 // Return extended offset
64 return offset;
65}
66FORCE_INLINE wf_offset_t wavefront_extend_matches_kernel_blockwise(
67 wavefront_aligner_t* const wf_aligner,
68 const int k,
69 wf_offset_t offset) {
70 // Fetch pattern/text blocks
71 uint64_t* pattern_blocks = (uint64_t*)(wf_aligner->sequences.pattern+WAVEFRONT_V(k,offset));
72 uint64_t* text_blocks = (uint64_t*)(wf_aligner->sequences.text+WAVEFRONT_H(k,offset));
73 // Compare 64-bits blocks
74 uint64_t cmp = *pattern_blocks ^ *text_blocks;
75 while (__builtin_expect(cmp==0,0)) {
76 // Increment offset (full block)
77 offset += 8;
78 // Next blocks
79 ++pattern_blocks;
80 ++text_blocks;
81 // Compare
82 cmp = *pattern_blocks ^ *text_blocks;
83 }
84 // Count equal characters
85 const int equal_right_bits = __builtin_ctzl(cmp);
86 const int equal_chars = DIV_FLOOR(equal_right_bits,8);
87 offset += equal_chars;
88 // Return extended offset
89 return offset;
90}
91/*
92 * Wavefront-Extend Inner Kernels
93 * Wavefront offset extension comparing characters
94 * Remember:
95 * - No offset is out of boundaries !(h>tlen,v>plen)
96 * - if (h==tlen,v==plen) extension won't increment (sentinels)
97 */
98FORCE_NO_INLINE void wavefront_extend_matches_packed_end2end(
99 wavefront_aligner_t* const wf_aligner,
100 wavefront_t* const mwavefront,
101 const int lo,
102 const int hi) {
103 wf_offset_t* const offsets = mwavefront->offsets;
104 int k;
105 for (k=lo;k<=hi;++k) {
106 // Fetch offset
107 const wf_offset_t offset = offsets[k];
108 if (offset == WAVEFRONT_OFFSET_NULL) continue;
109 // Extend offset
110 offsets[k] = wavefront_extend_matches_kernel(wf_aligner,k,offset);
111 }
112}

122 // Fetch offset

124 if (offset == WAVEFRONT_OFFSET_NULL) continue;

166/*
167 * Wavefront-Extend Inner Kernel (Custom match function)
168 */
169bool wavefront_extend_matches_custom(
170 wavefront_aligner_t* const wf_aligner,
171 wavefront_t* const mwavefront,
172 const int score,
173 const int lo,
174 const int hi,
175 const bool endsfree,
176 wf_offset_t* const max_antidiag) {
177 // Parameters
178 wavefront_sequences_t* const seqs = &wf_aligner->sequences;
179 // Extend diagonally each wavefront point
180 wf_offset_t* const offsets = mwavefront->offsets;
181 *max_antidiag = 0;
182 int k;
183 for (k=lo;k<=hi;++k) {
184 // Check offset
185 wf_offset_t offset = offsets[k];
186 if (offset == WAVEFRONT_OFFSET_NULL) continue;
187 // Count equal characters
188 int v = WAVEFRONT_V(k,offset);
189 int h = WAVEFRONT_H(k,offset);
190 while (wavefront_sequences_cmp(seqs,v,h)) {
191 h++; v++; offset++;
192 }
193 // Update offset
194 offsets[k] = offset;
195 // Compute max
196 const wf_offset_t antidiag = WAVEFRONT_ANTIDIAGONAL(k,offset);
197 if (*max_antidiag < antidiag) *max_antidiag = antidiag;
198 // Check ends-free reaching boundaries
199 if (endsfree && wavefront_termination_endsfree(wf_aligner,mwavefront,score,k,offset)) {
200 return true; // Quit (we are done)
201 }
202 }
203 // Alignment not finished
204 return false;
205}
