42/*
43 * System timer
44 */
45void timer_get_system_time(struct timespec *ts) {
46#ifdef __MACH__ // OS X does not have clock_gettime, use clock_get_time
47 clock_serv_t cclock;
48 mach_timespec_t mts;
49 host_get_clock_service(mach_host_self(),CALENDAR_CLOCK,&cclock);
50 clock_get_time(cclock,&mts);
51 mach_port_deallocate(mach_task_self(),cclock);
52 ts->tv_sec = mts.tv_sec;
53 ts->tv_nsec = mts.tv_nsec;
54#else
55 clock_gettime(CLOCK_REALTIME,ts);
56#endif
57}

65void timer_stop(profiler_timer_t* const timer) {
66 timer_pause(timer);

70void timer_pause(profiler_timer_t* const timer) {
71 struct timespec end_timer;
72 timer_get_system_time(&end_timer);
73 timer->accumulated += TIME_DIFF_NS(timer->begin_timer,end_timer);
74}
75void timer_continue(profiler_timer_t* const timer) {
76 timer_get_system_time(&timer->begin_timer);
77}
78void timer_reset(profiler_timer_t* const timer) {
79 timer->accumulated = 0;
80 counter_reset(&timer->time_ns);

82uint64_t timer_get_current_lap_ns(profiler_timer_t* const timer) {
83 struct timespec end_timer;
84 timer_get_system_time(&end_timer);

111void timer_print_total(
112 FILE* const stream,
113 const profiler_timer_t* const timer) {
114 const uint64_t total_time_ns = timer_get_total_ns(timer);
115 // Print Total
116 if (total_time_ns >= 60000000000ull) {
117 fprintf(stream,"%7.2f m ",TIMER_CONVERT_NS_TO_M(total_time_ns));
118 } else if (total_time_ns >= 1000000000) {
119 fprintf(stream,"%7.2f s ",TIMER_CONVERT_NS_TO_S(total_time_ns));
120 } else if (total_time_ns >= 1000000) {
121 fprintf(stream,"%7.2f ms",TIMER_CONVERT_NS_TO_MS(total_time_ns));
122 } else if (total_time_ns >= 1000) {
123 fprintf(stream,"%7.2f us",TIMER_CONVERT_NS_TO_US(total_time_ns));
124 } else {
125 fprintf(stream,"%7" PRIu64 " ns",total_time_ns);
126 }
127}
128void timer_print(
129 FILE* const stream,
130 const profiler_timer_t* const timer,
131 const profiler_timer_t* const ref_timer) {
132 const uint64_t total_time_ns = timer_get_total_ns(timer);
133 // Print Total
134 timer_print_total(stream,timer);
135 // Print percentage wrt reference
136 if (ref_timer!=NULL) {
137 if (total_time_ns==0) {
138 fprintf(stream," ( 0.00 %%)");
139 } else {
140 const uint64_t total_ref_time_ns = timer_get_total_ns(ref_timer);
141 if (total_ref_time_ns==0) {
142 fprintf(stream," ( n/a %%)");
143 } else {
144 const double percentage = (double)(total_time_ns*100)/(double)total_ref_time_ns;
145 fprintf(stream," (%6.02f %%)",percentage);
146 }
147 }
148 }
149 // Print Calls
150 const uint64_t num_calls = timer_get_num_samples(timer);
151 if (num_calls >= 1000000000) {
152 fprintf(stream," (%5" PRIu64 " Gcalls",num_calls/1000000000);
153 } else if (num_calls >= 1000000) {
154 fprintf(stream," (%5" PRIu64 " Mcalls",num_calls/1000000);
155 } else if (num_calls >= 1000) {
156 fprintf(stream," (%5" PRIu64 " Kcalls",num_calls/1000);
157 } else if (num_calls > 1 || num_calls == 0) {
158 fprintf(stream," (%5" PRIu64 " calls",num_calls);
159 } else {
160 fprintf(stream," (%5" PRIu64 " call",num_calls);
161 }
162 // Print time/call
163 if (num_calls==0) {
164 fprintf(stream,", n/a s/call)\n");
165 return;
166 } else {
167 const uint64_t ns_per_call = total_time_ns / num_calls;
168 if (ns_per_call > 1000000000) {
169 fprintf(stream,",%7.2f s/call",TIMER_CONVERT_NS_TO_S(ns_per_call));
170 } else if (ns_per_call > 1000000) {
171 fprintf(stream,",%7.2f ms/call",TIMER_CONVERT_NS_TO_MS(ns_per_call));
172 } else if (ns_per_call > 1000) {
173 fprintf(stream,",%7.2f us/call",TIMER_CONVERT_NS_TO_US(ns_per_call));
174 } else {
175 fprintf(stream,",%7" PRIu64 " ns/call",ns_per_call);
176 }
177 }
178 // Print Max
179 const uint64_t min_ns = timer_get_min_ns(timer);
180 if (min_ns > 1000000000) {
181 fprintf(stream," {min%.2fs",TIMER_CONVERT_NS_TO_S(min_ns));
182 } else if (min_ns > 1000000) {
183 fprintf(stream," {min%.2fms",TIMER_CONVERT_NS_TO_MS(min_ns));
184 } else if (min_ns > 1000) {
185 fprintf(stream," {min%.2fus",TIMER_CONVERT_NS_TO_US(min_ns));
186 } else {
187 fprintf(stream," {min%" PRIu64 "ns",min_ns);
188 }
189 // Print Min
190 const uint64_t max_ns = timer_get_max_ns(timer);
191 if (max_ns > 1000000000) {
192 fprintf(stream,",Max%.2fs})\n",TIMER_CONVERT_NS_TO_S(max_ns));
193 } else if (max_ns > 1000000) {
194 fprintf(stream,",Max%.2fms})\n",TIMER_CONVERT_NS_TO_MS(max_ns));
195 } else if (max_ns > 1000) {
196 fprintf(stream,",Max%.2fus})\n",TIMER_CONVERT_NS_TO_US(max_ns));
197 } else {
198 fprintf(stream,",Max%" PRIu64 "ns})\n",max_ns);
199 }
