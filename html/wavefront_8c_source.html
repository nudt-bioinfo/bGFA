39void wavefront_allocate(
40 wavefront_t* const wavefront,
41 const int wf_elements_allocated,
42 const bool allocate_backtrace,
43 mm_allocator_t* const mm_allocator) {
44 // Allocate memory
45 wavefront->wf_elements_allocated = wf_elements_allocated;
46 wavefront->offsets_mem = mm_allocator_calloc(
47 mm_allocator,wf_elements_allocated,wf_offset_t,false);
48 if (allocate_backtrace) {
49 wavefront->bt_pcigar_mem = mm_allocator_calloc(
50 mm_allocator,wf_elements_allocated,pcigar_t,false);
51 wavefront->bt_prev_mem = mm_allocator_calloc(
52 mm_allocator,wf_elements_allocated,bt_block_idx_t,false);
53 } else {
54 wavefront->bt_pcigar_mem = NULL;
55 }
56}
57void wavefront_resize(
58 wavefront_t* const wavefront,
59 const int wf_elements_allocated,
60 mm_allocator_t* const mm_allocator) {
61 // Set new size
62 wavefront->wf_elements_allocated = wf_elements_allocated;
63 // Reallocate offsets (Content is lost)
64 mm_allocator_free(mm_allocator,wavefront->offsets_mem);
65 wavefront->offsets_mem = mm_allocator_calloc(
66 mm_allocator,wf_elements_allocated,wf_offset_t,false);
67 // Reallocate backtrace (Content is lost)
68 if (wavefront->bt_pcigar_mem) {
69 mm_allocator_free(mm_allocator,wavefront->bt_pcigar_mem);
70 mm_allocator_free(mm_allocator,wavefront->bt_prev_mem);
71 wavefront->bt_pcigar_mem = mm_allocator_calloc(
72 mm_allocator,wf_elements_allocated,pcigar_t,false);
73 wavefront->bt_prev_mem = mm_allocator_calloc(
74 mm_allocator,wf_elements_allocated,bt_block_idx_t,false);
75 }
76}
77void wavefront_free(
78 wavefront_t* const wavefront,
79 mm_allocator_t* const mm_allocator) {
80 mm_allocator_free(mm_allocator,wavefront->offsets_mem);
81 if (wavefront->bt_pcigar_mem) {
82 mm_allocator_free(mm_allocator,wavefront->bt_pcigar_mem);
83 mm_allocator_free(mm_allocator,wavefront->bt_prev_mem);
84 }
85}

87 * Initialization

89void wavefront_init(
90 wavefront_t* const wavefront,
91 const int min_lo,
92 const int max_hi) {
93 // Limits
94 wavefront->null = false;
95 wavefront->lo = 1;
96 wavefront->hi = -1;
97 // Elements
98 wavefront->offsets = wavefront->offsets_mem - min_lo; // Center at k=0
99 if (wavefront->bt_pcigar_mem) {
100 wavefront->bt_occupancy_max = 0;
101 wavefront->bt_pcigar = wavefront->bt_pcigar_mem - min_lo; // Center at k=0
102 wavefront->bt_prev = wavefront->bt_prev_mem - min_lo; // Center at k=0
103 }
104 // Internals
105 wavefront->wf_elements_allocated_min = min_lo;
106 wavefront->wf_elements_allocated_max = max_hi;
107 wavefront->wf_elements_init_min = 0;
108 wavefront->wf_elements_init_max = 0;
109}
110void wavefront_init_null(
111 wavefront_t* const wavefront,
112 const int min_lo,
113 const int max_hi) {
114 // Limits
115 wavefront->null = true;
116 wavefront->lo = 1;
117 wavefront->hi = -1;
118 // Elements
119 wavefront->offsets = wavefront->offsets_mem - min_lo; // Center at k=0
120 if (wavefront->bt_pcigar_mem) {
121 wavefront->bt_occupancy_max = 0;
122 wavefront->bt_pcigar = wavefront->bt_pcigar_mem - min_lo; // Center at k=0
123 wavefront->bt_prev = wavefront->bt_prev_mem - min_lo; // Center at k=0
124 }
125 // Initialize
126 const int wf_elements = WAVEFRONT_LENGTH(min_lo,max_hi);
127 int i;
128 for (i=0;i<wf_elements;++i) {
129 wavefront->offsets_mem[i] = WAVEFRONT_OFFSET_NULL;
130 }
131 if (wavefront->bt_pcigar_mem) { // TODO: Really needed?
132 memset(wavefront->bt_pcigar_mem,0,wf_elements*sizeof(pcigar_t));
133 memset(wavefront->bt_prev_mem,0,wf_elements*sizeof(bt_block_idx_t));
134 }
135 // Internals
136 wavefront->wf_elements_allocated_min = min_lo;
137 wavefront->wf_elements_allocated_max = max_hi;
138 wavefront->wf_elements_init_min = min_lo;
139 wavefront->wf_elements_init_max = max_hi;
140}
141void wavefront_init_victim(
142 wavefront_t* const wavefront,
143 const int min_lo,
144 const int max_hi) {
145 // Delegate init
146 wavefront_init(wavefront,min_lo,max_hi);
147 // Set Null
148 wavefront->null = true;
149}

167uint64_t wavefront_get_size(
168 wavefront_t* const wavefront) {
169 uint64_t total_size = wavefront->wf_elements_allocated*sizeof(wf_offset_t);
170 if (wavefront->bt_pcigar_mem) {
171 total_size += wavefront->wf_elements_allocated*(sizeof(pcigar_t)+sizeof(bt_block_idx_t));
172 }
173 return total_size;
