22double ConvertProcess(CommandParser parse)
23{
24 struct timeval start_time, end_time;
25
26 GFA gfa;
27 // Determine input suffix and validate
28 const std::string &in_path = parse.convert_args.gfa_path;
29 size_t last_dot = in_path.find_last_of('.');
30 if (last_dot == std::string::npos || last_dot == in_path.size() - 1)
31 {
32 std::cout << "[Convert] Input file has no valid extension: "
33 << in_path << std::endl;
34 }
35
36 std::string in_ext = in_path.substr(last_dot + 1);
37 for (auto &ch : in_ext)
38 ch = static_cast<char>(std::tolower(static_cast<unsigned char>(ch)));
39
40 if (in_ext != "gfa" && in_ext != "bgfa")
41 {
42 std::cout << "[Convert] Unsupported input format: ."
43 << in_ext << ", expected .gfa or .bgfa" << std::endl;
44 }
45
46 // Auto output naming: gfa -> bgfa, bgfa -> gfa
47 if (parse.convert_args.output_path == "auto")
48 {
49 const std::string base = in_path.substr(0, last_dot);
50 parse.convert_args.output_path = (in_ext == "gfa") ? (base + ".bgfa") : (base + ".gfa");
51 }
52 std::cout << "Converting " << in_path << " to " << parse.convert_args.output_path << std::endl;
53
54 if (parse.convert_args.path_mode.empty())
55 parse.convert_args.path_mode = "inc";
56 for (auto &ch : parse.convert_args.path_mode)
57 ch = static_cast<char>(std::tolower(static_cast<unsigned char>(ch)));
58
59 std::cout << "\t Path Mode: " << parse.convert_args.path_mode << std::endl;
60 std::cout << "\t Segment ID Storage: "
61 << (parse.convert_args.segment_no_id ? "disabled (order-assigned)" : "enabled")
62 << std::endl;
63
64 gettimeofday(&start_time, NULL);
65 gfa = GFA(in_path, parse.convert_args.path_mode);
66 if (in_ext == "gfa" && parse.convert_args.normalized)
67 {
68 uword_t merged = gfa.mergeUniqueSuccessorByReverseTable();
69 std::cout << "\t Normalized merge applied, merged pairs: " << merged << std::endl;
70 }
71
72 // Determine conversion mode: default "all"
73 std::string mode = parse.convert_args.output_mode;
74 for (auto &ch : mode)
75 ch = static_cast<char>(std::tolower(static_cast<unsigned char>(ch)));
76 if (mode.empty())
77 mode = "all";
78
79 if (in_ext == "gfa")
80 {
81 if (mode == "split")
82 {
83 // Determine base path without extension from output or input
84 std::string base_out;
85 if (parse.convert_args.output_path == "auto")
86 {
87 base_out = in_path.substr(0, last_dot);
88 }
89 else
90 {
91 const std::string &out = parse.convert_args.output_path;
92 size_t out_last_dot = out.find_last_of('.');
93 if (out_last_dot != std::string::npos)
94 {
95 std::string out_ext = out.substr(out_last_dot + 1);
96 for (auto &ch2 : out_ext)
97 ch2 = static_cast<char>(std::tolower(static_cast<unsigned char>(ch2)));
98 base_out = (out_ext == "bgfa") ? out.substr(0, out_last_dot) : out;
99 }
100 else
101 {
102 base_out = out;
103 }
104 }
105
106 // Write split files and main BGFA via GFA method
107 gfa.write2BgfaSplit(base_out, !parse.convert_args.segment_no_id);
108 // Ensure convert output_path reflects main bgfa
109 parse.convert_args.output_path = base_out + ".bgfa";
110 }
111 else
112 {
113 // default: full conversion
114 gfa.write2Bgfa(parse.convert_args.output_path, !parse.convert_args.segment_no_id);
115 }
116 }
117 else
118 {
119 // bgfa -> gfa
120 gfa.write2GFA(parse.convert_args.output_path);
121 }
122 gettimeofday(&end_time, NULL);
123
124 double total_time = (double)(end_time.tv_sec - start_time.tv_sec) +
125 (double)(end_time.tv_usec - start_time.tv_usec) / 1e6;
126
127 std::cout << "Data successfully written to " << parse.convert_args.output_path << std::endl;
128 return total_time;
129}

200double MergeProcess(CommandParser parse)
201{
202 struct timeval start_time, end_time;
203 if (WORD_BIT < 32)
204 {
205 std::cerr << "Error: merge bGFA requires WORD_BIT >= 32." << std::endl;
206 return 0;
207 }
208
209 std::vector<GFA> gfa_list;
210 GFA merged_gfa;
211
212 gettimeofday(&start_time, NULL);
213
214 for (const auto &graph_path : parse.merge_args.graph_path)
215 {
216 GFA gfa(graph_path);
217 if (parse.merge_args.merge_type == 1 &&
218 parse.merge_args.details_type == 1)
219 gfa.setSegmentsDis(); // set distance for distance-aided merge
220
221 gfa_list.push_back(gfa);
222 }
223
224 if (gfa_list.empty())
225 throw std::invalid_argument("No input graphs provided.");
226
227 GFA ref = gfa_list[0];
228 std::vector<GFA> others;
229 for (size_t i = 1; i < gfa_list.size(); ++i)
230 others.push_back(gfa_list[i]);
231
232 if (parse.merge_args.merge_type == 0) // strict merge
233 {
234 std::cout << "Merging " << parse.merge_args.graph_path.size()
235 << " files into " << parse.merge_args.output_path
236 << " with strict match." << std::endl;
237 // Use first graph as reference, merge others into it using exact merge
238
239 mergeGFAStrict(ref, others);
240 }
241 else if (parse.merge_args.merge_type == 1)
242 {
243 std::cout << "Merging " << parse.merge_args.graph_path.size()
244 << " files into " << parse.merge_args.output_path;
245
246 if (parse.merge_args.details_type == 0)
247 {
248 std::cout << " with exact match." << std::endl;
249
250 mergeGFAExact(ref, others, parse.merge_args.merge_limited, parse.merge_args.alignment_kernel);
251 }
252 else if (parse.merge_args.details_type == 1)
253 {
254 std::cout << " with distance-aided exact match." << std::endl;
255
256 mergeGFAExactByDistance(ref, others, parse.merge_args.merge_limited, parse.merge_args.alignment_kernel);
257 }
258 else if (parse.merge_args.details_type == 2)
259 {
260 std::cout << " with minimizer-aided exact match." << std::endl;
261
262 mergeGFAExactByMinimize(ref, others, parse.merge_args.merge_limited, parse.merge_args.alignment_kernel);
263 }
264 else
265 {
266 throw std::invalid_argument("Invalid details type: " + std::to_string(parse.merge_args.details_type));
267 }
268 }
269 else
270 {
271 throw std::invalid_argument("Invalid merge type: " + std::to_string(parse.merge_args.merge_type));
272 }
273
274 ref.write2Bgfa(parse.merge_args.output_path);
275
276 gettimeofday(&end_time, NULL);
277
278 double total_time = (double)(end_time.tv_sec - start_time.tv_sec) +
279 (double)(end_time.tv_usec - start_time.tv_usec) / 1e6;
280 return total_time;
281}
