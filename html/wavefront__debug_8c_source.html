40bool wavefront_check_alignment(
41 FILE* const stream,
42 wavefront_aligner_t* const wf_aligner) {
43 // Parameters
44 wavefront_sequences_t* const sequences = (wf_aligner->bialigner==NULL) ?
45 &wf_aligner->sequences : &wf_aligner->bialigner->wf_forward->sequences;
46 const char* const pattern = sequences->pattern_buffer;
47 const int pattern_length = sequences->pattern_buffer_length;
48 const char* const text = sequences->text_buffer;
49 const int text_length = sequences->text_buffer_length;
50 // CIGAR
51 cigar_t* const cigar = wf_aligner->cigar;
52 char* const operations = cigar->operations;
53 const int begin_offset = cigar->begin_offset;
54 const int end_offset = cigar->end_offset;
55 // Traverse CIGAR
56 bool alignment_correct = true;
57 int pattern_pos=0, text_pos=0, i;
58 for (i=begin_offset;i<end_offset;++i) {
59 switch (operations[i]) {
60 case 'M': {
61 // Check match
62 if (sequences->mode != wf_sequences_lambda) {
63 const bool is_match = (pattern[pattern_pos]==text[text_pos]);
64 if (!is_match) {
65 fprintf(stream,"[WFA::Check] Alignment not matching (pattern[%d]=%c != text[%d]=%c)\n",
66 pattern_pos,pattern[pattern_pos],
67 text_pos,text[text_pos]);
68 alignment_correct = false;
69 break;
70 }
71 }
72 ++pattern_pos;
73 ++text_pos;
74 break;
75 }
76 case 'X': {
77 // Check mismatch
78 if (sequences->mode != wf_sequences_lambda) {
79 const bool is_match = (pattern[pattern_pos]==text[text_pos]);
80 if (is_match) {
81 fprintf(stream,"[WFA::Check] Alignment not mismatching (pattern[%d]=%c == text[%d]=%c)\n",
82 pattern_pos,pattern[pattern_pos],
83 text_pos,text[text_pos]);
84 alignment_correct = false;
85 break;
86 }
87 }
88 ++pattern_pos;
89 ++text_pos;
90 break;
91 }
92 case 'I':
93 ++text_pos;
94 break;
95 case 'D':
96 ++pattern_pos;
97 break;
98 default:
99 fprintf(stream,"[WFA::Check] Unknown edit operation '%c'\n",operations[i]);
100 exit(1);
101 break;
102 }
103 }
104 // Check alignment length
105 if (pattern_pos != pattern_length) {
106 fprintf(stream,
107 "[WFA::Check] Alignment incorrect length (pattern-aligned=%d,pattern-length=%d)\n",
108 pattern_pos,pattern_length);
109 alignment_correct = false;
110 }
111 if (text_pos != text_length) {
112 fprintf(stream,
113 "[WFA::Check] Alignment incorrect length (text-aligned=%d,text-length=%d)\n",
114 text_pos,text_length);
115 alignment_correct = false;
116 }
117 // Return
118 return alignment_correct;
119}

123void wavefront_report_lite(
124 FILE* const stream,
125 wavefront_aligner_t* const wf_aligner,
126 const bool alignment_completed) {
127 // Parameters
128 wavefront_sequences_t* const sequences = (wf_aligner->bialigner==NULL) ?
129 &wf_aligner->sequences : &wf_aligner->bialigner->wf_forward->sequences;
130 const char* const pattern = sequences->pattern;
131 const int pattern_length = sequences->pattern_length;
132 const char* const text = sequences->text;
133 const int text_length = sequences->text_length;
134 const int status = wf_aligner->align_status.status;
135 const uint64_t memory_used = wf_aligner->align_status.memory_used;
136 const bool has_cigar = alignment_completed && !cigar_is_null(wf_aligner->cigar);
137 // BANNER (#0)
138 if (alignment_completed) {
139 fprintf(stream,"[WFA::Debug]");
140 } else {
141 fprintf(stream,"[WFA::Debug::BEGIN]");
142 }
143 // SCORE (#1)
144 // const int score = wavefront_compute_classic_score(
145 // wf_aligner,pattern_length,text_length,wf_aligner->cigar->score);
146 const int score = wf_aligner->cigar->score;
147 if (alignment_completed && score!=INT32_MIN) {
148 fprintf(stream,"\t%d",score);
149 if (has_cigar) {
150 const int edit_dist = cigar_score_edit(wf_aligner->cigar);
151 fprintf(stream,"/%1.2f",(float)edit_dist/(float)MAX(pattern_length,text_length));
152 }
153 } else {
154 fprintf(stream,"\t*");
155 }
156 // PATTERN_LENGTH (#2)
157 fprintf(stream,"\t%d",pattern_length);
158 // TEXT_LENGTH (#3)
159 fprintf(stream,"\t%d",text_length);
160 // STATUS (#4)
161 fprintf(stream,"\t%s",wavefront_align_strerror_short(status));
162 // TIME (#5)
163 if (alignment_completed) {
164 fprintf(stream,"\t%2.3f",TIMER_GET_TOTAL_MS(&wf_aligner->system.timer));
165 } else {
166 fprintf(stream,"\t-");
167 }
168 // MEMORY (#6)
169 if (alignment_completed) {
170 fprintf(stream,"\t%luMB\t",CONVERT_B_TO_MB(memory_used));
171 } else {
172 fprintf(stream,"\t-\t");
173 }
174 // ATTRIBUTES (#7)
175 fprintf(stream,"[");
176 wavefront_aligner_print_mode(stream,wf_aligner);
177 fprintf(stream,";");
178 wavefront_aligner_print_scope(stream,wf_aligner);
179 fprintf(stream,";");
180 wavefront_penalties_print(stream,&wf_aligner->penalties);
181 fprintf(stream,";");
182 wavefront_aligner_print_conf(stream,wf_aligner);
183 fprintf(stream,";");
184 wavefront_heuristic_print(stream,&wf_aligner->heuristic);
185 fprintf(stream,";");
186 fprintf(stream,"(%d,%d,%d)",
187 wf_aligner->wf_components.num_wavefronts,
188 wf_aligner->wf_components.historic_min_lo,
189 wf_aligner->wf_components.historic_max_hi);
190 fprintf(stream,"]\t");
191 // CIGAR (#8)
192 if (!has_cigar) {
193 fprintf(stream,"-");
194 } else {
195 cigar_print(stream,wf_aligner->cigar,true);
196 }
197 // SEQUENCES (#9 #10)
198 if (sequences->mode == wf_sequences_lambda) {
199 fprintf(stream,"\t-\t-");
200 } else {
201 fprintf(stream,"\t%.*s\t%.*s",pattern_length,pattern,text_length,text);
202 }
203 fprintf(stream,"\n");
204}
205/*
206 * Debug
207 */
208void wavefront_debug_begin(
209 wavefront_aligner_t* const wf_aligner) {
210 // Check verbose level
211 if (wf_aligner->system.verbose >= 1) {
212 timer_reset(&wf_aligner->system.timer);
213 timer_start(&wf_aligner->system.timer);
214 if (wf_aligner->system.verbose >= 4) {
215 wavefront_report_lite(stderr,wf_aligner,false);
216 }
217 }
218}
219void wavefront_debug_end(
220 wavefront_aligner_t* const wf_aligner) {
221 // Print Summary
222 if (wf_aligner->system.verbose >= 1) {
223 timer_stop(&wf_aligner->system.timer);
224 wavefront_report_lite(stderr,wf_aligner,true);
225 }
226}
227/*
228 * Check
229 */
230void wavefront_debug_check_correct(
231 wavefront_aligner_t* const wf_aligner) {
232 // Check correct
233 if (wf_aligner->system.check_alignment_correct &&
234 wf_aligner->align_status.status == WF_STATUS_ALG_COMPLETED &&
235 wf_aligner->alignment_scope == compute_alignment) {
236 if (!wavefront_check_alignment(stderr,wf_aligner)) {
237 fprintf(stderr,"[WFA::Check] Error: Alignment incorrect\n");
238 exit(1);
239 }
240 }
