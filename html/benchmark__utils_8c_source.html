103void benchmark_print_output_lite(
104 FILE* const stream,
105 align_input_t* const align_input,
106 const int score,

117 // Free

143void benchmark_print_output(
144 align_input_t* const align_input,
145 const distance_metric_t distance_metric,
146 const bool score_only,
147 cigar_t* const cigar) {
148 if (align_input->output_file) {
149 // Compute score
150 int score = -1;
151 if (score_only) {
152 score = cigar->score;
153 } else if (cigar->begin_offset < cigar->end_offset) {
154 switch (distance_metric) {
155 case indel:
156 case edit:
157 score = cigar_score_edit(cigar);
158 break;
159 case gap_linear:
160 score = cigar_score_gap_linear(cigar,&align_input->linear_penalties);
161 break;
162 case gap_affine:
163 score = cigar_score_gap_affine(cigar,&align_input->affine_penalties);
164 break;
165 case gap_affine_2p:
166 score = cigar_score_gap_affine2p(cigar,&align_input->affine2p_penalties);
167 break;
168 default:
169 break;
170 }
171 }
172 // Print summary
173 if (align_input->output_full) {
174 benchmark_print_output_full(align_input->output_file,align_input,score,cigar);
175 } else {
176 benchmark_print_output_lite(align_input->output_file,align_input,score,cigar);
177 }
178 }
