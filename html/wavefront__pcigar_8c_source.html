36/*
37 * Packed CIGAR operations
38 */
39typedef struct {
40 char operation;
41 int inc_v;
42 int inc_h;
43 affine_matrix_type matrix_type;
44} pcigar_op_t;
45pcigar_op_t pcigar_lut[4] = {
46 { .operation = '?', .inc_v = 0, .inc_h = 0, .matrix_type = affine_matrix_M }, // 00 - None
47 { .operation = 'D', .inc_v = 1, .inc_h = 0, .matrix_type = affine_matrix_D }, // 01 - DELETION
48 { .operation = 'X', .inc_v = 1, .inc_h = 1, .matrix_type = affine_matrix_M }, // 10 - MISMATCH
49 { .operation = 'I', .inc_v = 0, .inc_h = 1, .matrix_type = affine_matrix_I }, // 11 - INSERTION
50};
51// Precomputed string of Matches
52char matches_lut[8] = "MMMMMMMM";
53#define CIGAR_8MATCHES_UINT64 *((uint64_t*)matches_lut)

58int pcigar_get_length(
59 const pcigar_t pcigar) {
60 int cigar_length = PCIGAR_MAX_LENGTH;
61 if (!PCIGAR_IS_UTILISED(pcigar,PCIGAR_FULL_MASK)) {
62 const int free_slots = PCIGAR_FREE_SLOTS(pcigar);
63 cigar_length -= free_slots;
64 }
65 return cigar_length;
66}
67int pcigar_unpack(
68 pcigar_t pcigar,
69 char* cigar_buffer) {
70 // Compute pcigar length and shift to the end of the word
71 int pcigar_length = PCIGAR_MAX_LENGTH;
72 if (!PCIGAR_IS_UTILISED(pcigar,PCIGAR_FULL_MASK)) {
73 const int free_slots = PCIGAR_FREE_SLOTS(pcigar);
74 pcigar_length -= free_slots;
75 pcigar <<= free_slots*2;
76 }
77 // Unpack BT-blocks
78 int i;
79 for (i=0;i<pcigar_length;++i) {
80 // Extract next CIGAR operation
81 const int cigar_op = (int)PCIGAR_EXTRACT(pcigar); // Extract
82 PCIGAR_POP_FRONT(pcigar); // Shift
83 // Add operation using LUT
84 pcigar_op_t* const op = pcigar_lut + cigar_op;
85 *(cigar_buffer++) = op->operation;
86 }
87 return pcigar_length;
88}
89/*
90 * PCIGAR extend exact-matches
91 */
92int pcigar_unpack_extend(
93 const char* const pattern,
94 const int pattern_length,
95 const char* const text,
96 const int text_length,
97 int v,
98 int h,
99 char* cigar_buffer) {
100 int num_matches = 0;
101 // Fetch pattern/text blocks
102 uint64_t* pattern_blocks = (uint64_t*)(pattern+v);
103 uint64_t* text_blocks = (uint64_t*)(text+h);
104 uint64_t pattern_block = *pattern_blocks;
105 uint64_t text_block = *text_blocks;
106 // Compare 64-bits blocks
107 uint64_t cmp = pattern_block ^ text_block;
108 while (cmp==0 && (v+8) < pattern_length && (h+8) < text_length) {
109 // Increment offset
110 v += 8;
111 h += 8;
112 num_matches += 8;
113 // Dump matches in block
114 *((uint64_t*)cigar_buffer) = CIGAR_8MATCHES_UINT64;
115 cigar_buffer += 8;
116 // Next blocks
117 ++pattern_blocks;
118 ++text_blocks;
119 // Fetch & Compare
120 pattern_block = *pattern_blocks;
121 text_block = *text_blocks;
122 cmp = pattern_block ^ text_block;
123 }
124 // Count equal characters
125 num_matches += __builtin_ctzl(cmp)/8;
126 *((uint64_t*)cigar_buffer) = CIGAR_8MATCHES_UINT64;
127 // Return total matches
128 return num_matches;
129}
130int pcigar_unpack_extend_custom(
131 const int pattern_length,
132 const int text_length,
133 alignment_match_funct_t const match_funct,
134 void* const match_funct_arguments,
135 int v,
136 int h,
137 char* cigar_buffer) {
138 int num_matches = 0;
139 while (v < pattern_length && h < text_length) {
140 // Check match
141 if (!match_funct(v,h,match_funct_arguments)) break;
142 ++v; ++h;
143 // Increment matches
144 *cigar_buffer = 'M';
145 ++cigar_buffer;
146 ++num_matches;
147 }
148 return num_matches;
149}

153void pcigar_unpack_linear(
154 pcigar_t pcigar,
155 wavefront_sequences_t* const sequences,
156 int* const v_pos,
157 int* const h_pos,
158 char* cigar_buffer,
159 int* const cigar_length) {
160 // Parameters
161 char* const pattern = sequences->pattern;
162 const int pattern_length = sequences->pattern_length;
163 char* const text = sequences->text;
164 const int text_length = sequences->text_length;
165 char* const cigar_buffer_base = cigar_buffer;
166 // Compute pcigar length and shift to the end of the word
167 int pcigar_length = PCIGAR_MAX_LENGTH;
168 if (!PCIGAR_IS_UTILISED(pcigar,PCIGAR_FULL_MASK)) {
169 const int free_slots = PCIGAR_FREE_SLOTS(pcigar);
170 pcigar_length -= free_slots;
171 pcigar <<= free_slots*2;
172 }
173 // Unpack BT-blocks
174 int v = *v_pos, h = *h_pos, i;
175 for (i=0;i<pcigar_length;++i) {
176 // Extend exact-matches
177 int num_matches;
178 if (sequences->mode == wf_sequences_lambda) { // Custom extend-match function
179 num_matches = pcigar_unpack_extend_custom(
180 pattern_length,text_length,sequences->match_funct,
181 sequences->match_funct_arguments,v,h,cigar_buffer);
182 } else {
183 num_matches = pcigar_unpack_extend(
184 pattern,pattern_length,text,text_length,v,h,cigar_buffer);
185 }
186 // Update location
187 v += num_matches;
188 h += num_matches;
189 cigar_buffer += num_matches;
190 // Extract next CIGAR operation
191 const int cigar_op = (int)PCIGAR_EXTRACT(pcigar); // Extract
192 PCIGAR_POP_FRONT(pcigar); // Shift
193 // Add operation using LUT
194 pcigar_op_t* const op = pcigar_lut + cigar_op;
195 *(cigar_buffer++) = op->operation;
196 v += op->inc_v;
197 h += op->inc_h;
198 }
199 // Update length/positions
200 *cigar_length = cigar_buffer - cigar_buffer_base;
201 *v_pos = v;
202 *h_pos = h;
203}
204void pcigar_unpack_affine(
205 pcigar_t pcigar,
206 wavefront_sequences_t* const sequences,
207 int* const v_pos,
208 int* const h_pos,
209 char* cigar_buffer,
210 int* const cigar_length,
211 affine_matrix_type* const current_matrix_type) {
212 // Parameters
213 char* const pattern = sequences->pattern;
214 const int pattern_length = sequences->pattern_length;
215 char* const text = sequences->text;
216 const int text_length = sequences->text_length;
217 char* const cigar_buffer_base = cigar_buffer;
218 // Compute pcigar length and shift to the end of the word
219 int pcigar_length = PCIGAR_MAX_LENGTH;
220 if (!PCIGAR_IS_UTILISED(pcigar,PCIGAR_FULL_MASK)) {
221 const int free_slots = PCIGAR_FREE_SLOTS(pcigar);
222 pcigar_length -= free_slots;
223 pcigar <<= free_slots*2;
224 }
225 // Unpack BT-blocks
226 affine_matrix_type matrix_type = *current_matrix_type;
227 int v = *v_pos, h = *h_pos, i;
228 for (i=0;i<pcigar_length;++i) {
229 // Extend exact-matches
230 if (matrix_type == affine_matrix_M) { // Extend only on the M-wavefront
231 int num_matches;
232 if (sequences->mode == wf_sequences_lambda) { // Custom extend-match function
233 num_matches = pcigar_unpack_extend_custom(
234 pattern_length,text_length,sequences->match_funct,
235 sequences->match_funct_arguments,v,h,cigar_buffer);
236 } else {
237 num_matches = pcigar_unpack_extend(
238 pattern,pattern_length,text,text_length,v,h,cigar_buffer);
239 }
240 // Update location
241 v += num_matches;
242 h += num_matches;
243 cigar_buffer += num_matches;
244 }
245 // Extract next CIGAR operation
246 const int cigar_op = (int)PCIGAR_EXTRACT(pcigar); // Extract
247 PCIGAR_POP_FRONT(pcigar); // Shift
248 // Add operation using LUT
249 pcigar_op_t* const op = pcigar_lut + cigar_op;
250 // Avoid X after I/D (used to encode gap-close)
251 if (matrix_type != affine_matrix_M && op->operation=='X') {
252 matrix_type = affine_matrix_M;
253 continue;
254 }
255 // Add operation
256 *(cigar_buffer++) = op->operation;
257 v += op->inc_v;
258 h += op->inc_h;
259 matrix_type = op->matrix_type;
260 }
261 // Update length/positions
262 *cigar_length = cigar_buffer - cigar_buffer_base;
263 *v_pos = v;
264 *h_pos = h;
265 *current_matrix_type = matrix_type;
266}
