33#define BPM_ALPHABET_LENGTH 4
34#define BPM_W64_LENGTH UINT64_LENGTH
35#define BPM_W64_SIZE UINT64_SIZE
36#define BPM_W64_ONES UINT64_MAX
37#define BPM_W64_MASK (1ull<<63)

39/*
40 * Pattern Accessors
41 */
42#define BPM_PATTERN_PEQ_IDX(word_pos,encoded_character) ((word_pos*BPM_ALPHABET_LENGTH)+(encoded_character))
43#define BPM_PATTERN_BDP_IDX(position,num_words,word_pos) ((position)*(num_words)+(word_pos))
44/*
45 * Advance block functions (Improved)
46 * const @vector Eq,mask;
47 * return (Pv,Mv,PHout,MHout);
48 */
49#define BPM_ADVANCE_BLOCK(Eq,mask,Pv,Mv,PHin,MHin,PHout,MHout) \
50 /* Computes modulator vector {Xv,Xh} ( cases A&C ) */ \
51 const uint64_t Xv = Eq | Mv; \
52 const uint64_t _Eq = Eq | MHin; \
53 const uint64_t Xh = (((_Eq & Pv) + Pv) ^ Pv) | _Eq; \
54 /* Calculate Hout */ \
55 uint64_t Ph = Mv | ~(Xh | Pv); \
56 uint64_t Mh = Pv & Xh; \
57 /* Account Hout that propagates for the next block */ \
58 PHout = (Ph & mask)!=0; \
59 MHout = (Mh & mask)!=0; \
60 /* Hout become the Hin of the next cell */ \
61 Ph <<= 1; \
62 Mh <<= 1; \
63 /* Account Hin coming from the previous block */ \
64 Ph |= PHin; \
65 Mh |= MHin; \
66 /* Finally, generate the Vout */ \
67 Pv = Mh | ~(Xv | Ph); \
68 Mv = Ph & Xv

72void edit_bpm_pattern_compile(
73 bpm_pattern_t* const bpm_pattern,
74 char* const pattern,
75 const int pattern_length,
76 mm_allocator_t* const mm_allocator) {
77 // Calculate dimensions
78 const uint64_t pattern_num_words64 = DIV_CEIL(pattern_length,BPM_W64_LENGTH);
79 const uint64_t PEQ_length = pattern_num_words64*BPM_W64_LENGTH;
80 const uint64_t pattern_mod = pattern_length%BPM_W64_LENGTH;
81 // Init fields
82 bpm_pattern->pattern = pattern;
83 bpm_pattern->pattern_length = pattern_length;
84 bpm_pattern->pattern_num_words64 = pattern_num_words64;
85 bpm_pattern->pattern_mod = pattern_mod;
86 // Allocate memory
87 const uint64_t aux_vector_size = pattern_num_words64*BPM_W64_SIZE;
88 const uint64_t PEQ_size = BPM_ALPHABET_LENGTH*aux_vector_size;
89 const uint64_t score_size = pattern_num_words64*UINT64_SIZE;
90 const uint64_t total_memory = PEQ_size + 3*aux_vector_size + 2*score_size + (pattern_num_words64+1)*UINT64_SIZE;
91 void* memory = mm_allocator_malloc(mm_allocator,total_memory);
92 bpm_pattern->PEQ = memory; memory += PEQ_size;
93 bpm_pattern->P = memory; memory += aux_vector_size;
94 bpm_pattern->M = memory; memory += aux_vector_size;
95 bpm_pattern->level_mask = memory; memory += aux_vector_size;
96 bpm_pattern->score = memory; memory += score_size;
97 bpm_pattern->init_score = memory; memory += score_size;
98 bpm_pattern->pattern_left = memory;
99 // Init PEQ
100 memset(bpm_pattern->PEQ,0,PEQ_size);
101 uint64_t i;
102 for (i=0;i<pattern_length;++i) {
103 const uint8_t enc_char = dna_encode(pattern[i]);
104 const uint64_t block = i/BPM_W64_LENGTH;
105 const uint64_t mask = 1ull<<(i%BPM_W64_LENGTH);
106 bpm_pattern->PEQ[BPM_PATTERN_PEQ_IDX(block,enc_char)] |= mask;
107 }
108 for (;i<PEQ_length;++i) { // Padding
109 const uint64_t block = i/BPM_W64_LENGTH;
110 const uint64_t mask = 1ull<<(i%BPM_W64_LENGTH);
111 uint64_t j;
112 for (j=0;j<BPM_ALPHABET_LENGTH;++j) {
113 bpm_pattern->PEQ[BPM_PATTERN_PEQ_IDX(block,j)] |= mask;
114 }
115 }
116 // Init auxiliary data
117 uint64_t pattern_left = pattern_length;
118 const uint64_t top = pattern_num_words64-1;
119 memset(bpm_pattern->level_mask,0,aux_vector_size);
120 for (i=0;i<top;++i) {
121 bpm_pattern->level_mask[i] = BPM_W64_MASK;
122 bpm_pattern->init_score[i] = BPM_W64_LENGTH;
123 bpm_pattern->pattern_left[i] = pattern_left;
124 pattern_left = (pattern_left > BPM_W64_LENGTH) ? pattern_left-BPM_W64_LENGTH : 0;
125 }
126 for (;i<=pattern_num_words64;++i) {
127 bpm_pattern->pattern_left[i] = pattern_left;
128 pattern_left = (pattern_left > BPM_W64_LENGTH) ? pattern_left-BPM_W64_LENGTH : 0;
129 }
130 if (pattern_mod > 0) {
131 const uint64_t mask_shift = pattern_mod-1;
132 bpm_pattern->level_mask[top] = 1ull<<(mask_shift);
133 bpm_pattern->init_score[top] = pattern_mod;
134 } else {
135 bpm_pattern->level_mask[top] = BPM_W64_MASK;
136 bpm_pattern->init_score[top] = BPM_W64_LENGTH;
137 }
138}

156 bpm_matrix->Pv = Pv;
157 // CIGAR
158 bpm_matrix->cigar = cigar_new(pattern_length+text_length);
159}

171void edit_bpm_reset_search_cutoff(
172 uint8_t* const top_level,
173 uint64_t* const P,
174 uint64_t* const M,
175 int64_t* const score,
176 const int64_t* const init_score,
177 const uint64_t max_distance) {
178 // Calculate the top level (maximum bit-word for cut-off purposes)
179 const uint8_t y = (max_distance>0) ? (max_distance+(BPM_W64_LENGTH-1))/BPM_W64_LENGTH : 1;
180 *top_level = y;
181 // Reset score,P,M
182 uint64_t i;
183 P[0]=BPM_W64_ONES;
184 M[0]=0;
185 score[0] = init_score[0];
186 for (i=1;i<y;++i) {
187 P[i]=BPM_W64_ONES;
188 M[i]=0;
189 score[i] = score[i-1] + init_score[i];
190 }
191}
192void edit_bpm_compute_matrix(
193 bpm_matrix_t* const bpm_matrix,
194 bpm_pattern_t* const bpm_pattern,
195 char* const text,
196 const uint64_t text_length,
197 uint64_t max_distance) {
198 // Pattern variables
199 const uint64_t* PEQ = bpm_pattern->PEQ;
200 const uint64_t num_words64 = bpm_pattern->pattern_num_words64;
201 const uint64_t* const level_mask = bpm_pattern->level_mask;
202 int64_t* const score = bpm_pattern->score;
203 const int64_t* const init_score = bpm_pattern->init_score;
204 uint64_t* const Pv = bpm_matrix->Pv;
205 uint64_t* const Mv = bpm_matrix->Mv;
206 const uint64_t max_distance__1 = max_distance+1;
207 const uint8_t top = num_words64-1;
208 uint8_t top_level;
209 edit_bpm_reset_search_cutoff(&top_level,Pv,Mv,score,init_score,max_distance);
210 // Advance in DP-bit_encoded matrix
211 uint64_t text_position;
212 for (text_position=0;text_position<text_length;++text_position) {
213 // Fetch next character
214 const uint8_t enc_char = dna_encode(text[text_position]);
215 // Advance all blocks
216 uint64_t i,PHin=1,MHin=0,PHout,MHout;
217 for (i=0;i<top_level;++i) {
218 /* Calculate Step Data */
219 const uint64_t bdp_idx = BPM_PATTERN_BDP_IDX(text_position,num_words64,i);
220 const uint64_t next_bdp_idx = bdp_idx+num_words64;
221 uint64_t Pv_in = Pv[bdp_idx];
222 uint64_t Mv_in = Mv[bdp_idx];
223 const uint64_t mask = level_mask[i];
224 const uint64_t Eq = PEQ[BPM_PATTERN_PEQ_IDX(i,enc_char)];
225 /* Compute Block */
226 BPM_ADVANCE_BLOCK(Eq,mask,Pv_in,Mv_in,PHin,MHin,PHout,MHout);
227 /* Adjust score and swap propagate Hv */
228 score[i] += PHout-MHout;
229 Pv[next_bdp_idx] = Pv_in;
230 Mv[next_bdp_idx] = Mv_in;
231 PHin=PHout;
232 MHin=MHout;
233 }
234 // Cut-off
235 const uint8_t last = top_level-1;
236 if (score[last]<=max_distance__1 && last<top) {
237 const uint64_t last_score = score[last]+(MHin-PHin);
238 const uint64_t Peq = PEQ[BPM_PATTERN_PEQ_IDX(top_level,enc_char)];
239 if (last_score<=max_distance && (MHin || (Peq & 1))) {
240 // Init block V
241 const uint64_t bdp_idx = BPM_PATTERN_BDP_IDX(text_position,num_words64,top_level);
242 const uint64_t next_bdp_idx = bdp_idx+num_words64;
243 uint64_t Pv_in = BPM_W64_ONES;
244 uint64_t Mv_in = 0;
245 Pv[bdp_idx] = BPM_W64_ONES;
246 Mv[bdp_idx] = 0;
247 const uint64_t mask = level_mask[top_level];
248 /* Compute Block */
249 BPM_ADVANCE_BLOCK(Peq,mask,Pv_in,Mv_in,PHin,MHin,PHout,MHout);
250 /* Save Block Pv,Mv */
251 Pv[next_bdp_idx]=Pv_in;
252 Mv[next_bdp_idx]=Mv_in;
253 /* Set score & increment the top level block */
254 score[top_level] = last_score + init_score[top_level] + (PHout-MHout);
255 ++top_level;
256 } else {
257 while (score[top_level-1] > (max_distance+init_score[top_level-1])) {
258 --top_level;
259 }
260 }
261 } else {
262 while (score[top_level-1] > (max_distance+init_score[top_level-1])) {
263 --top_level;
264 }
265 }
266 }
267 // Return optimal column/distance
268 // Check match
269 const int64_t current_score = score[top_level-1];
270 if (top_level==num_words64 && current_score<=max_distance) {
271 bpm_matrix->min_score = score[top_level-1];
272 bpm_matrix->min_score_column = text_length-1;
273 } else {
274 bpm_matrix->min_score = UINT64_MAX;
275 bpm_matrix->min_score_column = UINT64_MAX;
276 }
277}
278void edit_bpm_backtrace_matrix(
279 bpm_matrix_t* const bpm_matrix,
280 const bpm_pattern_t* const bpm_pattern,
281 char* const text) {
282 // Parameters
283 char* const pattern = bpm_pattern->pattern;
284 const uint64_t pattern_length = bpm_pattern->pattern_length;
285 const uint64_t* const Pv = bpm_matrix->Pv;
286 const uint64_t* const Mv = bpm_matrix->Mv;
287 char* const operations = bpm_matrix->cigar->operations;
288 int op_sentinel = bpm_matrix->cigar->end_offset-1;
289 // Retrieve the alignment. Store the match
290 const uint64_t num_words64 = bpm_pattern->pattern_num_words64;
291 int64_t h = bpm_matrix->min_score_column;
292 int64_t v = pattern_length - 1;
293 while (v >= 0 && h >= 0) {
294 const uint8_t block = v / UINT64_LENGTH;
295 const uint64_t bdp_idx = BPM_PATTERN_BDP_IDX(h+1,num_words64,block);
296 const uint64_t mask = 1L << (v % UINT64_LENGTH);
297 // CIGAR operation Test
298 if (Pv[bdp_idx] & mask) {
299 operations[op_sentinel--] = 'D';
300 --v;
301 } else if (Mv[(bdp_idx-num_words64)] & mask) {
302 operations[op_sentinel--] = 'I';
303 --h;
304 } else if ((text[h]==pattern[v])) {
305 operations[op_sentinel--] = 'M';
306 --h;
307 --v;
308 } else {
309 operations[op_sentinel--] = 'X';
310 --h;
311 --v;
312 }
313 }
314 while (h>=0) {operations[op_sentinel--] = 'I'; --h;}
315 while (v>=0) {operations[op_sentinel--] = 'D'; --v;}
316 bpm_matrix->cigar->begin_offset = op_sentinel+1;
317}
318void edit_bpm_compute(
319 bpm_matrix_t* const bpm_matrix,
320 bpm_pattern_t* const bpm_pattern,
321 char* const text,
322 const int text_length,
323 const int max_distance) {
324 // Fill Matrix (Pv,Mv)
325 edit_bpm_compute_matrix(
326 bpm_matrix,bpm_pattern,
327 text,text_length,max_distance);
328 // Check distance
329 if (bpm_matrix->min_score == UINT64_MAX) return;
330 // Backtrace and generate CIGAR
331 edit_bpm_backtrace_matrix(bpm_matrix,bpm_pattern,text);
332}
