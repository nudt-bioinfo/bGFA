370 void toBgfaBin(std::ofstream &output_file, bool store_id = true) const
371 {
372 if (!output_file.is_open())
373 {
374 throw std::runtime_error(
375 "[Write to bGFA File] Output file is not open");
376 }
377
378 uint8_t type_flag = 1; // Type flag for segment
379
380 // output_file.write(reinterpret_cast<const char *>(&type_flag), sizeof(uword_t));
381 if (store_id)
382 {
383 output_file.write(reinterpret_cast<const char *>(&id_),
384 sizeof(uword_t));
385 }
386 output_file.write(reinterpret_cast<const char *>(&sosr_),
387 sizeof(uword_t));
388
389 if (store_id)
390 {
391 if (length_ > (WORD_BIT - 6) / 2)
392 {
393 output_file.write(reinterpret_cast<const char *>(&length_),
394 sizeof(uword_t));
395
396 for (const auto &value : sequence_)
397 {
398 output_file.write(reinterpret_cast<const char *>(&value),
399 sizeof(uword_t));
400 }
401 }
402 else
403 {
404 uword_t val = (((uword_t(1) << 5) | length_) << (WORD_BIT - 6)) |
405 (sequence_[0] >> 6);
406 output_file.write(reinterpret_cast<const char *>(&val),
407 sizeof(uword_t));
408 }
409 }
410 else
411 {
412 const uword_t compact1_cap = (WORD_BIT - 6) / 2;
413 const uword_t compact1_len_max = std::min<uword_t>(compact1_cap, 0x0F);
414 const uword_t compact2_cap = (2 * WORD_BIT - 8) / 2;
415 const uword_t compact2_len_max = std::min<uword_t>(compact2_cap, 0x3F);
416
417 if (length_ > compact2_len_max)
418 {
419 // 00: plain length + sequence words
420 output_file.write(reinterpret_cast<const char *>(&length_),
421 sizeof(uword_t));
422 for (const auto &value : sequence_)
423 {
424 output_file.write(reinterpret_cast<const char *>(&value),
425 sizeof(uword_t));
426 }
427 }
428 else if (length_ <= compact1_len_max)
429 {
430 // 10: [2-bit tag=10][4-bit length][WORD_BIT-6 bits sequence]
431 uword_t payload = 0;
432 for (uword_t i = 0; i < length_; ++i)
433 {
434 const uint8_t enc = GlobalVariant::char2bin[static_cast<uint8_t>(getBaseAt(i))] & 0x3;
435 const uword_t shift = (WORD_BIT - 8) - 2 * i;
436 payload |= (static_cast<uword_t>(enc) << shift);
437 }
438 const uword_t val = (uword_t(2) << (WORD_BIT - 2)) |
439 ((length_ & 0x0F) << (WORD_BIT - 6)) |
440 payload;
441 output_file.write(reinterpret_cast<const char *>(&val),
442 sizeof(uword_t));
443 }
444 else
445 {
446 // 11: [2-bit tag=11][6-bit length][WORD_BIT-8 bits sequence] + [next uword sequence]
447 uword_t payload_hi = 0;
448 uword_t payload_lo = 0;
449 const uword_t hi_bits = WORD_BIT - 8;
450
451 for (uword_t i = 0; i < length_; ++i)
452 {
453 const uint8_t enc = GlobalVariant::char2bin[static_cast<uint8_t>(getBaseAt(i))] & 0x3;
454 const uword_t bit_pos = 2 * i;
455
456 const uint8_t b0 = (enc >> 1) & 0x1;
457 const uint8_t b1 = enc & 0x1;
458
459 const uword_t p0 = bit_pos;
460 const uword_t p1 = bit_pos + 1;
461
462 if (p0 < hi_bits)
463 {
464 if (b0)
465 payload_hi |= (uword_t(1) << (hi_bits - 1 - p0));
466 }
467 else
468 {
469 const uword_t q = p0 - hi_bits;
470 if (b0)
471 payload_lo |= (uword_t(1) << (WORD_BIT - 1 - q));
472 }
473
474 if (p1 < hi_bits)
475 {
476 if (b1)
477 payload_hi |= (uword_t(1) << (hi_bits - 1 - p1));
478 }
479 else
480 {
481 const uword_t q = p1 - hi_bits;
482 if (b1)
483 payload_lo |= (uword_t(1) << (WORD_BIT - 1 - q));
484 }
485 }
486
487 const uword_t val = (uword_t(3) << (WORD_BIT - 2)) |
488 ((length_ & 0x3F) << (WORD_BIT - 8)) |
489 payload_hi;
490 output_file.write(reinterpret_cast<const char *>(&val),
491 sizeof(uword_t));
492 output_file.write(reinterpret_cast<const char *>(&payload_lo),
493 sizeof(uword_t));
494 }
495 }

1417 void parsePathIncrement(std::vector<std::pair<uword_t, char>> tokens)
1418 {
1419 uword_t idx = 0;
1420 uword_t node;
1421
1422 node = packIdDirFlag(tokens[idx].first, tokens[idx].second, 0);
1423 nodes_.push_back(node); // nodes[0], seg[0]
1424 node = packIdDirFlag(tokens[tokens.size() - 1].first, tokens[tokens.size() - 1].second, 0);
1425 nodes_.push_back(node); // nodes[1], seg[-1] (end placeholder)
1426 idx++; // idx = 1
1427
1428 // If only one node, finalize
1429 if (tokens.size() == 1)
1430 {
1431 return;
1432 }
1433
1434 // Save second node at nodes_[2]
1435 node = packIdDirFlag(tokens[idx].first, tokens[idx].second, 0);
1436 nodes_.push_back(node); // nodes[2], seg1
1437 idx++; // idx = 2
1438
1439 uword_t inc_head_pos = idx; // position of "head" before a potential increment group
1440 uword_t inc_pre_id = tokens[1].first; // previous id to compute deltas
1441
1442 // State for an ongoing increment block
1443 bool in_block = false;
1444 uword_t inc_word = 0; // packs 4 x 8-bit increments
1445 uword_t inc_count = 0; // number of increments recorded in current block (max 16)
1446 uword_t dir_word_index = 0; // nodes_ index where direction word is stored when in_block
1447
1448 while (idx + 1 < tokens.size())
1449 {
1450 // Consider starting from tokens[idx]
1451 uword_t cur_id = tokens[idx].first;
1452 char cur_dir = tokens[idx].second;
1453 uword_t delta = cur_id < inc_pre_id ? PATH_MAX_INC + 1 : (cur_id - inc_pre_id);
1454
1455 if (!in_block)
1456 {
1457 // 1.完整模式, 保存当前段，continue
1458 if (delta > PATH_MAX_INC)
1459 {
1460 uword_t node = packIdDirFlag(cur_id, cur_dir, 0);
1461 nodes_.push_back(node);
1462 inc_head_pos = nodes_.size() - 1;
1463 inc_pre_id = cur_id;
1464 idx++;
1465 continue;
1466 }
1467
1468 // 2.可以采用增量，但是不具有4个可用段（含当前）, 保存所有段，break
1469 if (idx + PATH_INC_SIZE - 1 >= tokens.size())
1470 {
1471 for (; idx < tokens.size(); ++idx)
1472 {
1473 uword_t node = packIdDirFlag(tokens[idx].first, tokens[idx].second, 0);
1474 nodes_.push_back(node);
1475 inc_head_pos = nodes_.size() - 1;
1476 inc_pre_id = tokens[idx].first;
1477 }
1478 break;
1479 }
1480
1481 // 3.可以采用增量，具有4个可用段（含当前）
1482 uword_t prev = inc_pre_id;
1483 bool ok = true;
1484 inc_word = 0;
1485 uword_t dir_word = 0;
1486 for (int k = 0; k < PATH_INC_SIZE; ++k)
1487 {
1488 uword_t nid = tokens[idx + k].first;
1489 if (nid > PATH_MAX_INC + prev)
1490 {
1491 ok = false;
1492 inc_word = 0;
1493 break;
1494 }
1495 dir_word = setDirWord(dir_word, tokens[idx + k].second, k);
1496 inc_word = setIncWord(inc_word, nid - prev, k);
1497 prev = nid;
1498 }
1499
1500 // 3.1.四个段增量超出范围，保存当前段，continue
1501 if (!ok)
1502 {
1503 uword_t node = packIdDirFlag(cur_id, cur_dir, 0);
1504 nodes_.push_back(node);
1505 inc_head_pos = nodes_.size() - 1;
1506 inc_pre_id = cur_id;
1507 idx++;
1508 continue;
1509 }
1510
1511 // 3.2.四个段增量未超出范围
1512 // 为增量开始头结点进行标记
1513 nodes_[inc_head_pos] |= 1u;
1514
1515 // 打包四个dir到dir_word
1516 nodes_.push_back(dir_word);
1517 dir_word_index = nodes_.size() - 1;
1518
1519 // 打包四个inc到inc_word
1520 nodes_.push_back(inc_word);
1521 inc_word = 0;
1522
1523 inc_pre_id = prev;
1524 inc_count = PATH_INC_SIZE;
1525 in_block = true; // reset for subsequent chunks
1526 idx += PATH_INC_SIZE;
1527 continue;
1528 }
1529 else
1530 {
1531 // Extend current block while delta within PATH_MAX_INC and count <= 16
1532 if (delta <= PATH_MAX_INC && inc_count < 16)
1533 {
1534 // Append direction (2 bits) into existing direction word
1535 nodes_[dir_word_index] =
1536 setDirWord(nodes_[dir_word_index], cur_dir, inc_count);
1537
1538 // Append increment into the current 4-slot word
1539 uword_t slot = inc_count % 4;
1540 inc_word = setIncWord(inc_word, delta, slot);
1541 inc_count++;
1542 inc_pre_id = cur_id;
1543
1544 // Every 4 increments, flush inc_word
1545 if ((inc_count % 4) == 0)
1546 {
1547 nodes_.push_back(inc_word);
1548 inc_word = 0;
1549 }
1550 idx++;
1551 continue;
1552 }
1553
1554 // 不满足距离需求，需要将现有块存储（若inc_count为4的倍数已在之前flush，无需再存储）
1555 if (inc_count % 4)
1556 {
1557 nodes_.push_back(inc_word);
1558 inc_word = 0;
1559 }
1560
1561 inc_count = 0;
1562 in_block = false;
1563
1564 // 存储下一个节点为完整模式
1565 uword_t node = packIdDirFlag(cur_id, cur_dir, 0);
1566 nodes_.push_back(node);
1567 inc_head_pos = nodes_.size() - 1;
1568 inc_pre_id = cur_id;
1569 idx++;
1570 continue;
1571 }
1572 }
1573
1574 // Finalize: flush any pending increments if in a block
1575 if (in_block && (inc_count % 4))
1576 {
1577 nodes_.push_back(inc_word);
1578 }
1579 }

1600 {
1601 std::ostringstream oss;
1602
1603 oss << "P\t" << id_ << "\t";
1604
1605 if (nodes_.empty())
1606 return std::string();
1607
1608 // Print start node (nodes_[0])
1609 uword_t start_id = (nodes_[0] >> 2);
1610 char start_dir = GlobalVariant::bin2dir[(nodes_[0] >> 1) & 0x01];
1611 oss << start_id << start_dir;
1612 uword_t prev_id = start_id;
1613
1614 // Iterate over encoded sequence, skipping nodes_[1] (end placeholder)
1615 size_t i = 2;
1616 while (i < nodes_.size())
1617 {
1618 uword_t val = nodes_[i];
1619 if ((val & 0x01u) == 1u)
1620 {
1621 // Head of an increment block
1622 uword_t head_id = (val >> 2);
1623 char head_dir = GlobalVariant::bin2dir[(val >> 1) & 0x01];
1624 oss << "," << head_id << head_dir;
1625 prev_id = head_id;
1626
1627 if (i + 1 >= nodes_.size())
1628 break; // malformed; no direction word
1629 uword_t dir_word = nodes_[i + 1];
1630
1631 // Count directions encoded: 2 bits per node; 00 indicates termination
1632 int dir_count = 0;
1633 for (int p = 0; p < 16; ++p)
1634 {
1635 uword_t code = (dir_word >> (2 * p)) & 0x03u;
1636 if (code == 0u)
1637 break;
1638 ++dir_count;
1639 }
1640 int inc_words = (dir_count + PATH_INC_SIZE - 1) / PATH_INC_SIZE;
1641
1642 // Collect increment words following direction word
1643 std::vector<uword_t> incs;
1644 incs.reserve(inc_words);
1645 size_t base = i + 2;
1646 for (int w = 0; w < inc_words && (base + w) < nodes_.size(); ++w)
1647 {
1648 incs.push_back(nodes_[base + w]);
1649 }
1650
1651 // Reconstruct nodes from increments and directions
1652 for (int m = 0; m < dir_count; ++m)
1653 {
1654 uword_t inc_word = incs[m / PATH_INC_SIZE];
1655 uword_t delta = (inc_word >> (PATH_INC_SHIFT * (m % PATH_INC_SIZE))) & 0xFFu;
1656 prev_id += delta;
1657
1658 uword_t code = (dir_word >> (2 * m)) & 0x03u; // 2:'-', 3:'+'
1659 char dir_char = (code == 3u) ? '+' : '-';
1660 oss << "," << prev_id << dir_char;
1661 }
1662
1663 // Advance past head, dir_word and inc_words
1664 i = base + inc_words;
1665 continue;
1666 }
1667 else
1668 {
1669 // Normal node stored as full id+dir
1670 uword_t id = (val >> 2);
1671 char dir_c = GlobalVariant::bin2dir[(val >> 1) & 0x01];
1672 oss << "," << id << dir_c;
1673 prev_id = id;
1674 ++i;
1675 }
1676 }
1677
1678 return oss.str();
1679 }

2743 }

2844

2846 new_walks.reserve(walks_.size());
