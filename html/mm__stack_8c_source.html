98 // Allocate handler

100 // Memory segments
101 mm_stack->segments = vector_new(MM_STACK_INITIAL_SEGMENTS,mm_stack_segment_t*);
102 mm_stack->segment_size = segment_size;
103#ifndef MM_STACK_FORCE_MALLOC
104 mm_stack_segment_new(mm_stack);
105#endif
106 mm_stack->current_segment_idx = 0;
107 // Malloc memory
108 mm_stack->malloc_requests = vector_new(MM_STACK_INITIAL_MALLOC_REQUESTS,void*);
109 // Stack states
110 mm_stack->states = vector_new(MM_STACK_INITIAL_STATES,mm_stack_state_t);
111 // Return
112 return mm_stack;

114void mm_stack_clear(
115 mm_stack_t* const mm_stack) {
116 // Clear first memory segment and discard the rest
117 mm_stack_segment_t* const segment = *vector_get_elm(mm_stack->segments,0,mm_stack_segment_t*);
118 mm_stack_segment_clear(segment);
119 mm_stack->current_segment_idx = 0;
120 // Free malloc memory
121 VECTOR_ITERATE(mm_stack->malloc_requests,mem_ptr,m,void*) {
122 free(*mem_ptr);
123 }
124 vector_clear(mm_stack->malloc_requests);
125 // Clear states
126 vector_clear(mm_stack->states);
127}
128void mm_stack_delete(
129 mm_stack_t* const mm_stack) {
130 // Delete memory segments
131 VECTOR_ITERATE(mm_stack->segments,segment_ptr,p,mm_stack_segment_t*) {
132 mm_stack_segment_delete(*segment_ptr);
133 }
134 vector_delete(mm_stack->segments);
135 // Free malloc memory
136 VECTOR_ITERATE(mm_stack->malloc_requests,mem_ptr,m,void*) {
137 free(*mem_ptr);
138 }
139 vector_delete(mm_stack->malloc_requests);
140 // Clear states
141 vector_delete(mm_stack->states);
142 // Free handler
143 free(mm_stack);
144}

148mm_stack_segment_t* mm_stack_fetch_segment(
149 mm_stack_t* const mm_stack,
150 const uint64_t num_bytes) {
151 // Fetch current segment
152 mm_stack_segment_t* const curr_segment =
153 *vector_get_elm(mm_stack->segments,mm_stack->current_segment_idx,mm_stack_segment_t*);
154// // Check overall segment size
155// if (num_bytes > curr_segment->size/2) { // Never buy anything you cannot afford twice
156// return NULL; // Memory request over max-request size
157// }
158 // Check available segment size
159 if (curr_segment->used + num_bytes <= curr_segment->size) {
160 return curr_segment;
161 }
162 // Check overall segment size
163 if (num_bytes > curr_segment->size) {
164 return NULL; // Memory request over segment size
165 }
166 // Get free segment
167 const uint64_t num_segments = vector_get_used(mm_stack->segments);
168 ++(mm_stack->current_segment_idx);
169 if (mm_stack->current_segment_idx < num_segments) {
170 // Get next segment
171 mm_stack_segment_t* const segment =
172 *vector_get_elm(mm_stack->segments,mm_stack->current_segment_idx,mm_stack_segment_t*);
173 // Clear
174 mm_stack_segment_clear(segment);
175 // Return
176 return segment;
177 }
178 // Add new segment
179 return mm_stack_segment_new(mm_stack);
180}
181void* mm_stack_allocate(
182 mm_stack_t* const mm_stack,
183 const uint64_t num_bytes,
184 const bool zero_mem,
185 const uint64_t align_bytes) {
186 // Zero check
187 if (num_bytes == 0) {
188 fprintf(stderr,"MMStack error. Zero bytes requested\n");
189 exit(1);
190 }
191 // Add payload
192 const uint64_t num_bytes_allocated = num_bytes + align_bytes;
193 // Fetch segment
194#ifdef MM_STACK_FORCE_MALLOC
195 mm_stack_segment_t* const segment = NULL; // Force malloc memory
196#else
197 mm_stack_segment_t* const segment = mm_stack_fetch_segment(mm_stack,num_bytes_allocated);
198#endif
199 // Allocate memory
200 void* memory_base ;
201 if (segment != NULL) {
202 // Segment-memory
203 memory_base = segment->memory + segment->used;
204 if (zero_mem) memset(memory_base,0,num_bytes_allocated); // Set zero
205 segment->used += num_bytes_allocated; // Update segment
206 } else {
207 // Malloc-memory
208 memory_base = malloc(num_bytes_allocated);
209 if (zero_mem) memset(memory_base,0,num_bytes_allocated); // Set zero
210 // Add malloc-request
211 vector_insert(mm_stack->malloc_requests,memory_base,void*);
212 }
213 // Check alignment
214 if (align_bytes == 0) return memory_base;
215 // Align memory request
216 void* memory_aligned = memory_base + align_bytes;
217 memory_aligned = memory_aligned - ((uintptr_t)memory_aligned % align_bytes);
218 return memory_aligned;
219}

223void mm_stack_push(
224 mm_stack_t* const mm_stack) {
225 // Get new stack-state
226 mm_stack_state_t* stack_state;
227 vector_alloc_new(mm_stack->states,mm_stack_state_t,stack_state);
228 // Store current state
229 mm_stack_segment_t* const current_segment =
230 *vector_get_elm(mm_stack->segments,mm_stack->current_segment_idx,mm_stack_segment_t*);
231 stack_state->segment_idx = mm_stack->current_segment_idx;
232 stack_state->segment_used = current_segment->used;
233 stack_state->num_malloc_requests = vector_get_used(mm_stack->malloc_requests);
234}
235void mm_stack_pop(
236 mm_stack_t* const mm_stack) {
237 // Get last stack-state
238 mm_stack_state_t* const stack_state = vector_get_last_elm(mm_stack->states,mm_stack_state_t);
239 vector_dec_used(mm_stack->states);
240 // Restore segment-memory state
241 mm_stack->current_segment_idx = stack_state->segment_idx;
242 mm_stack_segment_t* const current_segment =
243 *(vector_get_elm(mm_stack->segments,stack_state->segment_idx,mm_stack_segment_t*));
244 current_segment->used = stack_state->segment_used;
245 // Restore malloc-memory state (free requests)
246 const uint64_t total_malloc_requests = vector_get_used(mm_stack->malloc_requests);
247 void** const malloc_requests = vector_get_mem(mm_stack->malloc_requests,void*);
248 uint64_t i;
249 for (i=stack_state->num_malloc_requests;i<total_malloc_requests;++i) {
250 free(*(malloc_requests+i)); // Free
251 }
252 vector_set_used(mm_stack->malloc_requests,stack_state->num_malloc_requests);
253}
