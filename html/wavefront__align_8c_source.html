46 * Checks
47 */
48void wavefront_align_presets__checks(
49 wavefront_aligner_t* const wf_aligner,
50 const int pattern_length,
51 const int text_length) {
52 // Parameters
53 alignment_form_t* const form = &wf_aligner->alignment_form;
54 /*
55 * Configuration presets
56 */
57 if (form->span == alignment_endsfree && wf_aligner->alignment_form.extension) {
58 // Configure WF-extend mode
59 form->pattern_begin_free = 0;
60 form->pattern_end_free = pattern_length;
61 form->text_begin_free = 0;
62 form->text_end_free = text_length;
63 }
64 /*
65 * Checks
66 */
67 if (wf_aligner->bialigner != NULL) {
68 const bool ends_free =
69 form->pattern_begin_free > 0 ||
70 form->pattern_end_free > 0 ||
71 form->text_begin_free > 0 ||
72 form->text_end_free > 0;
73 if (ends_free) {
74 fprintf(stderr,"[WFA] BiWFA ends-free has not been tested properly yet (let me know and I'll do it)\n");
75 exit(1);
76 }
77 if (wf_aligner->alignment_form.extension) {
78 fprintf(stderr,"[WFA] BiWFA extension is not implemented yet (let me know and I'll add it)\n");
79 exit(1);
80 }
81 }
82 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
83 const bool is_heuristic_drop =
84 (wf_aligner->heuristic.strategy & wf_heuristic_xdrop) ||
85 (wf_aligner->heuristic.strategy & wf_heuristic_zdrop);
86 if (is_heuristic_drop && (distance_metric==edit || distance_metric==indel)) {
87 fprintf(stderr,"[WFA] Heuristics drops are not compatible with 'edit'/'indel' distance metrics\n");
88 exit(1);
89 }
90 if (form->span == alignment_endsfree) {
91 if (form->pattern_begin_free > pattern_length ||
92 form->pattern_end_free > pattern_length ||
93 form->text_begin_free > text_length ||
94 form->text_end_free > text_length) {
95 fprintf(stderr,"[WFA] Ends-free parameters must be not larger than the sequences "
96 "(P0=%d,Pf=%d,T0=%d,Tf=%d). Must be (P0<=|P|,Pf<=|P|,T0<=|T|,Tf<=|T|) where (|P|,|T|)=(%d,%d)\n",
97 form->pattern_begin_free,form->pattern_end_free,
98 form->text_begin_free,form->text_end_free,
99 pattern_length,text_length);
100 exit(1);
101 }
102 }
103}
104/*
105 * Wavefront Alignment Unidirectional
106 */
107void wavefront_align_unidirectional_cleanup(
108 wavefront_aligner_t* const wf_aligner) {
109 // Compute memory used
110 uint64_t memory_used = wavefront_aligner_get_size(wf_aligner);
111 wf_aligner->align_status.memory_used = memory_used;
112 // Reap memory (controlled reaping)
113 if (memory_used > wf_aligner->system.max_memory_resident) {
114 // Wavefront components
115 wavefront_components_reap(&wf_aligner->wf_components);
116 // Check memory
117 memory_used = wavefront_aligner_get_size(wf_aligner);
118 wf_aligner->align_status.memory_used = memory_used;
119 // Slab
120 if (memory_used > wf_aligner->system.max_memory_resident) {
121 wavefront_slab_reap(wf_aligner->wavefront_slab);
122 if (wf_aligner->bialigner != NULL) {
123 wavefront_bialigner_reap(wf_aligner->bialigner);
124 }
125 }
126 }
127}
128void wavefront_align_unidirectional(
129 wavefront_aligner_t* const wf_aligner) {
130 // Wavefront align sequences
131 wavefront_unialign_init(wf_aligner,affine2p_matrix_M,affine2p_matrix_M); // Init
132 wavefront_unialign(wf_aligner); // Align
133 // Finish
134 if (wf_aligner->align_status.status == WF_STATUS_MAX_STEPS_REACHED) return; // Alignment paused
135 wavefront_align_unidirectional_cleanup(wf_aligner);
136}

147/*
148 * Wavefront Alignment Dispatcher
149 */
150int wavefront_align_lambda(
151 wavefront_aligner_t* const wf_aligner,
152 alignment_match_funct_t match_funct,
153 void* match_funct_arguments,
154 const int pattern_length,
155 const int text_length) {
156 // Checks
157 wavefront_align_presets__checks(wf_aligner,pattern_length,text_length);
158 wavefront_debug_begin(wf_aligner);
159 // Plot
160 if (wf_aligner->plot != NULL) wavefront_plot_resize(wf_aligner->plot,pattern_length,text_length);
161 // Dispatcher
162 if (wf_aligner->bialigner == NULL) {
163 // Prepare Sequences
164 wavefront_sequences_init_lambda(&wf_aligner->sequences,
165 match_funct,match_funct_arguments,
166 pattern_length,text_length,false);
167 wavefront_align_unidirectional(wf_aligner);
168 } else {
169 // Prepare Sequences
170 wavefront_bialigner_set_sequences_lambda(wf_aligner->bialigner,
171 match_funct,match_funct_arguments,
172 pattern_length,text_length);
173 // Align
174 wavefront_align_bidirectional(wf_aligner);
175 }
176 // DEBUG
177 wavefront_debug_end(wf_aligner);
178 wavefront_debug_check_correct(wf_aligner);
179 // Return
180 return wf_aligner->align_status.status;
181}

242/*
243 * Wavefront Alignment Resume (Experimental)
244 */
245int wavefront_align_resume(
246 wavefront_aligner_t* const wf_aligner) {
247 // Parameters
248 wavefront_align_status_t* const align_status = &wf_aligner->align_status;
249 // Check current alignment status
250 if (align_status->status != WF_STATUS_MAX_STEPS_REACHED ||
251 wf_aligner->bialigner != NULL) {
252 fprintf(stderr,"[WFA] Alignment cannot be resumed\n");
253 exit(1);
254 }
255 // Resume aligning sequences
256 wavefront_unialign(wf_aligner);
257 // Finish alignment
258 if (align_status->status == WF_STATUS_MAX_STEPS_REACHED) {
259 return WF_STATUS_MAX_STEPS_REACHED; // Alignment paused
260 }
261 wavefront_align_unidirectional_cleanup(wf_aligner);
262 // DEBUG
263 wavefront_debug_check_correct(wf_aligner);
264 // Return
265 return align_status->status;
266}
