47#define PCIGAR_NULL 0ul
48#define PCIGAR_DELETION 1ul
49#define PCIGAR_MISMATCH 2ul
50#define PCIGAR_INSERTION 3ul
51
52#define PCIGAR_POP_FRONT(pcigar) pcigar = pcigar << 2
53#define PCIGAR_PUSH_BACK(pcigar,operation) ((pcigar<<2) | operation)
54
55#define PCIGAR_PUSH_BACK_INS(pcigar) ((pcigar<<2) | PCIGAR_INSERTION)
56#define PCIGAR_PUSH_BACK_DEL(pcigar) ((pcigar<<2) | PCIGAR_DELETION)
57#define PCIGAR_PUSH_BACK_MISMS(pcigar) ((pcigar<<2) | PCIGAR_MISMATCH)

59#ifdef PCIGAR_32BITS
60 typedef uint32_t pcigar_t;
61 #define PCIGAR_MAX_LENGTH 16
62 #define PCIGAR_FULL_MASK 0x40000000u /* Completely full */
63 #define PCIGAR_ALMOST_FULL_MASK 0x10000000u /* 15-slots busy or more */
64 #define PCIGAR_HALF_FULL_MASK 0x00010000u /* 9-slots busy or more */
65 #define PCIGAR_IS_UTILISED(pcigar,mask) ((pcigar) >= mask)
66 #define PCIGAR_EXTRACT(pcigar) ((pcigar) >> 30)
67 #define PCIGAR_FREE_SLOTS(pcigar) ((pcigar)!=0) ? __builtin_clz(pcigar)/2 : PCIGAR_MAX_LENGTH;
68#else
69 typedef uint64_t pcigar_t;
70 #define PCIGAR_MAX_LENGTH 32
71 #define PCIGAR_FULL_MASK 0x4000000000000000ul /* Completely full */
72 #define PCIGAR_ALMOST_FULL_MASK 0x1000000000000000ul /* 31-slots busy or more */
73 #define PCIGAR_HALF_FULL_MASK 0x0000000100000000ul /* 17-slots busy or more */
74 #define PCIGAR_IS_UTILISED(pcigar,mask) ((pcigar) >= mask)
75 #define PCIGAR_EXTRACT(pcigar) ((pcigar) >> 62)
76 #define PCIGAR_FREE_SLOTS(pcigar) ((pcigar)!=0) ? __builtin_clzl(pcigar)/2 : PCIGAR_MAX_LENGTH;
77#endif
