42/*
43 * Error codes & messages
44 */
45// [OK]
46#define WF_STATUS_ALG_COMPLETED 0 // Success (Complete alignment found)
47#define WF_STATUS_ALG_PARTIAL 1 // Success (Partial alignment found)
48// [FAIL]
49#define WF_STATUS_MAX_STEPS_REACHED -100 // Maximum number of WFA-steps reached
50#define WF_STATUS_OOM -200 // Maximum memory limit reached
51#define WF_STATUS_UNATTAINABLE -300 // Alignment unattainable under configured heuristics
52// [INTERNAL]
53#define WF_STATUS_OK -1 // Computing alignment (in progress)
54#define WF_STATUS_END_REACHED -2 // Alignment end reached
55#define WF_STATUS_END_UNREACHABLE -3 // Alignment end unreachable under current configuration (eg Z-drop)
56// Error messages
57char *wavefront_align_strerror(const int error_code);
