44vector_t* vector_new_(
45 const uint64_t num_initial_elements,
46 const uint64_t element_size) {
47 vector_t* const vector_buffer = (vector_t*) malloc(sizeof(vector_t));
48 vector_buffer->element_size = element_size;
49 vector_buffer->elements_allocated = num_initial_elements;
50 vector_buffer->memory = malloc(num_initial_elements*element_size);
51 if (!vector_buffer->memory) {
52 fprintf(stderr,"Could not create new vector (%" PRIu64 " bytes requested)",
53 num_initial_elements*element_size);
54 exit(1);
55 }
56 vector_buffer->used = 0;
57 return vector_buffer;
58}
59void vector_delete(
60 vector_t* const vector) {
61 free(vector->memory);
62 free(vector);

64void vector_cast(
65 vector_t* const vector,
66 const uint64_t element_size) {
67 vector->elements_allocated = (vector->elements_allocated*vector->element_size)/element_size;
68 vector->element_size = element_size;
69 vector->used = 0;
70}
71void vector_reserve(
72 vector_t* const vector,
73 const uint64_t num_elements,
74 const bool zero_mem) {
75 if (vector->elements_allocated < num_elements) {
76 const uint64_t proposed = (float)vector->elements_allocated*VECTOR_EXPAND_FACTOR;
77 vector->elements_allocated = num_elements>proposed?num_elements:proposed;
78 vector->memory = realloc(vector->memory,vector->elements_allocated*vector->element_size);
79 if (!vector->memory) {
80 fprintf(stderr,"Could not reserve vector (%" PRIu64 " bytes requested)",
81 vector->elements_allocated*vector->element_size);
82 exit(1);
83 }
84 }
85 if (zero_mem) {
86 memset(vector->memory+vector->used*vector->element_size,0,
87 (vector->elements_allocated-vector->used)*vector->element_size);
88 }
89}

108void vector_copy(
109 vector_t* const vector_to,
110 vector_t* const vector_from) {
111 // Prepare
112 vector_cast(vector_to,vector_from->element_size);
113 vector_reserve(vector_to,vector_from->used,false);
114 // Copy
115 vector_set_used(vector_to,vector_from->used);
116 memcpy(vector_to->memory,vector_from->memory,vector_from->used*vector_from->element_size);
117}
118vector_t* vector_dup(
119 vector_t* const vector_src) {
120 vector_t* const vector_cpy = vector_new_(vector_src->used,vector_src->element_size);
121 // Copy
122 vector_set_used(vector_cpy,vector_src->used);
123 memcpy(vector_cpy->memory,vector_src->memory,vector_src->used*vector_src->element_size);
124 return vector_cpy;
125}
