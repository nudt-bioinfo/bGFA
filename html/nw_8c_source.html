38void nw_traceback(
39 score_matrix_t* const score_matrix,
40 linear_penalties_t* const penalties,
41 cigar_t* const cigar) {
42 // Parameters
43 int** const dp = score_matrix->columns;
44 char* const operations = cigar->operations;
45 cigar->end_offset = cigar->max_operations;
46 int op_sentinel = cigar->end_offset - 1;
47 // Compute traceback
48 int h = score_matrix->num_columns-1;
49 int v = score_matrix->num_rows-1;
50 while (h>0 && v>0) {
51 if (dp[h][v] == dp[h][v-1]+penalties->indel) {
52 operations[op_sentinel--] = 'D';
53 --v;
54 } else if (dp[h][v] == dp[h-1][v]+penalties->indel) {
55 operations[op_sentinel--] = 'I';
56 --h;
57 } else {
58 operations[op_sentinel--] =
59 (dp[h][v] == dp[h-1][v-1] + penalties->mismatch) ? 'X' : 'M';
60 --h;
61 --v;
62 }
63 }
64 while (h>0) {operations[op_sentinel--] = 'I'; --h;}
65 while (v>0) {operations[op_sentinel--] = 'D'; --v;}
66 cigar->begin_offset = op_sentinel + 1;
67}
68void nw_align(
69 score_matrix_t* const score_matrix,
70 linear_penalties_t* const penalties,
71 const char* const pattern,
72 const int pattern_length,
73 const char* const text,
74 const int text_length,
75 cigar_t* const cigar) {
76 // Parameters
77 int** dp = score_matrix->columns;
78 int h, v;
79 // Init DP (No ends-free)
80 dp[0][0] = 0;
81 for (v=1;v<=pattern_length;++v) {
82 dp[0][v] = dp[0][v-1] + penalties->indel;
83 }
84 for (h=1;h<=text_length;++h) {
85 dp[h][0] = dp[h-1][0] + penalties->indel;
86 }
87 // Compute DP
88 for (h=1;h<=text_length;++h) {
89 for (v=1;v<=pattern_length;++v) {
90 int min = dp[h-1][v-1] + ((pattern[v-1]==text[h-1]) ? penalties->match : penalties->mismatch); // Misms
91 min = MIN(min,dp[h-1][v]+penalties->indel); // Ins
92 min = MIN(min,dp[h][v-1]+penalties->indel); // Del
93 dp[h][v] = min;
94 }
95 }
96 // Compute traceback
97 nw_traceback(score_matrix,penalties,cigar);
98}
