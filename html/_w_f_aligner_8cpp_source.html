46WFAligner::WFAligner(
47 const AlignmentScope alignmentScope,
48 const MemoryModel memoryModel) {
49 this->attributes = wavefront_aligner_attr_default;
50 switch (memoryModel) {
51 case MemoryHigh: this->attributes.memory_mode = wavefront_memory_high; break;
52 case MemoryMed: this->attributes.memory_mode = wavefront_memory_med; break;
53 case MemoryLow: this->attributes.memory_mode = wavefront_memory_low; break;
54 case MemoryUltralow: this->attributes.memory_mode = wavefront_memory_ultralow; break;
55 default: this->attributes.memory_mode = wavefront_memory_high; break;
56 }
57 this->attributes.alignment_scope = (alignmentScope==Score) ? compute_score : compute_alignment;
58 // this->attributes.system.verbose = 2; // DEBUG
59 // this->attributes.plot.enabled = true; // DEBUG
60 this->wfAligner = nullptr;
61}
62WFAligner::~WFAligner() {
63 wavefront_aligner_delete(wfAligner);
64}

288void WFAligner::getAlignment(
289 char** const cigarOperations,
290 int* cigarLength) {
291 *cigarOperations = wfAligner->cigar->operations + wfAligner->cigar->begin_offset;
292 *cigarLength = wfAligner->cigar->end_offset - wfAligner->cigar->begin_offset;
293}
294std::string WFAligner::getAlignment() {
295 // Fetch Alignment
296 char* const buffer = wfAligner->cigar->operations + wfAligner->cigar->begin_offset;
297 const int length = wfAligner->cigar->end_offset - wfAligner->cigar->begin_offset;
298 // Create string and return
299 return std::string(buffer,length);
300}
301void WFAligner::getCIGAR(
302 const bool showMismatches,
303 uint32_t** const cigarOperations,
304 int* const numCigarOperations) {
305 cigar_get_CIGAR(wfAligner->cigar,showMismatches,cigarOperations,numCigarOperations);
306}
307std::string WFAligner::getCIGAR(
308 const bool showMismatches) {
309 // Check length
310 const int alignmentLength = wfAligner->cigar->end_offset - wfAligner->cigar->begin_offset;
311 if (alignmentLength <= 0) return std::string();
312 // Allocate
313 char* const buffer = new char[2*alignmentLength];
314 const int bufferLength = cigar_sprint_SAM_CIGAR(buffer,wfAligner->cigar,showMismatches);
315 // Create string
316 std::string cigarString = std::string(buffer,bufferLength);
317 // Free & return
318 delete[] buffer;
319 return cigarString;
320}
