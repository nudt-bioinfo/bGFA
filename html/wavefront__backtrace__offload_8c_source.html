36/*
37 * Backtrace-blocks offloading
38 */
39void wavefront_backtrace_offload_blocks_selective(
40 wf_offset_t* const out_offsets,
41 pcigar_t* const out_bt_pcigar,
42 bt_block_idx_t* const out_bt_prev,
43 const int lo,
44 const int hi,
45 const pcigar_t occupation_mask,
46 wf_backtrace_buffer_t* const bt_buffer) {
47 // Fetch BT-buffer free memory
48 int bt_blocks_available;
49 bt_block_t* bt_block_mem;
50 bt_block_idx_t global_pos = wf_backtrace_buffer_get_mem(bt_buffer,&bt_block_mem,&bt_blocks_available);
51 bt_block_idx_t current_pos = global_pos;
52 const int max_pos = current_pos + bt_blocks_available;
53 // Check PCIGAR buffers full and off-load if needed
54 int k;
55 for (k=lo;k<=hi;++k) {
56 if (out_offsets[k]>=0 && PCIGAR_IS_UTILISED(out_bt_pcigar[k],occupation_mask)) {
57 // Store
58 bt_block_mem->pcigar = out_bt_pcigar[k];
59 bt_block_mem->prev_idx = out_bt_prev[k];
60 bt_block_mem++;
61 // Reset
62 out_bt_pcigar[k] = 0;
63 out_bt_prev[k] = current_pos;
64 current_pos++;
65 // Update pos
66 if (current_pos >= max_pos) {
67 wf_backtrace_buffer_add_used(bt_buffer,current_pos-global_pos);
68 global_pos = wf_backtrace_buffer_get_mem(bt_buffer,&bt_block_mem,&bt_blocks_available);
69 }
70 }
71 }
72 wf_backtrace_buffer_add_used(bt_buffer,current_pos-global_pos);
73}
74/*
75 * Backtrace offloading (linear)
76 */
77int wavefront_backtrace_offload_blocks_linear(
78 wavefront_aligner_t* const wf_aligner,
79 wf_offset_t* const out_offsets,
80 pcigar_t* const out_bt_pcigar,
81 bt_block_idx_t* const out_bt_prev,
82 const int lo,
83 const int hi) {
84 // Parameters
85 const wavefront_memory_t wavefront_memory = wf_aligner->memory_mode;
86 wf_backtrace_buffer_t* const bt_buffer = wf_aligner->wf_components.bt_buffer;
87 // Select memory-mode
88 switch (wavefront_memory) {
89 case wavefront_memory_med:
90 wavefront_backtrace_offload_blocks_selective(
91 out_offsets,out_bt_pcigar,out_bt_prev,
92 lo,hi,PCIGAR_HALF_FULL_MASK,bt_buffer);
93 return PCIGAR_MAX_LENGTH/2; // Half occupancy
94 break;
95 case wavefront_memory_low:
96 wavefront_backtrace_offload_blocks_selective(
97 out_offsets,out_bt_pcigar,out_bt_prev,
98 lo,hi,PCIGAR_FULL_MASK,bt_buffer);
99 return PCIGAR_MAX_LENGTH-1; // At least 1-slots free
100 break;
101 default:
102 fprintf(stderr,"[WFA::compute] Wrong memory-mode\n");
103 exit(1);
104 }
105}
106void wavefront_backtrace_offload_linear(
107 wavefront_aligner_t* const wf_aligner,
108 const wavefront_set_t* const wavefront_set,
109 const int lo,
110 const int hi) {
111 // Paramters
112 wavefront_t* const wf_m = wavefront_set->out_mwavefront;
113 const wavefront_t* const m_misms = wavefront_set->in_mwavefront_misms;
114 const wavefront_t* const m_open1 = wavefront_set->in_mwavefront_open1;
115 // Compute BT occupancy maximum
116 int occ_max_m = 0, occ_max_indel = 0;
117 if (!m_open1->null) occ_max_indel = m_open1->bt_occupancy_max;
118 if (!m_misms->null) occ_max_m = m_misms->bt_occupancy_max;
119 const int occ_max = MAX(occ_max_indel,occ_max_m) + 1;
120 // Set new occupancy
121 wf_m->bt_occupancy_max = occ_max;
122 // Offload if necessary (Gap-Linear)
123 if (!wf_m->null && occ_max >= PCIGAR_MAX_LENGTH) {
124 wf_offset_t* const out_m = wavefront_set->out_mwavefront->offsets;
125 pcigar_t* const out_m_bt_pcigar = wavefront_set->out_mwavefront->bt_pcigar;
126 bt_block_idx_t* const out_m_bt_prev = wavefront_set->out_mwavefront->bt_prev;
127 wavefront_set->out_mwavefront->bt_occupancy_max =
128 wavefront_backtrace_offload_blocks_linear(
129 wf_aligner,out_m,out_m_bt_pcigar,out_m_bt_prev,lo,hi);
130 }

163void wavefront_backtrace_offload_occupation_affine(
164 wavefront_aligner_t* const wf_aligner,
165 const wavefront_set_t* const wavefront_set) {
166 // Parameters
167 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
168 // Select distance metric
169 int occ_max_m = 0;
170 int occ_max_i1 = 0, occ_max_i2 = 0;
171 int occ_max_d1 = 0, occ_max_d2 = 0;
172 if (distance_metric == gap_affine) {
173 // Parameters
174 const wavefront_t* const m_misms = wavefront_set->in_mwavefront_misms;
175 const wavefront_t* const m_open1 = wavefront_set->in_mwavefront_open1;
176 const wavefront_t* const i1_ext = wavefront_set->in_i1wavefront_ext;
177 const wavefront_t* const d1_ext = wavefront_set->in_d1wavefront_ext;
178 // Compute BT occupancy maximum
179 if (!m_open1->null) {
180 occ_max_i1 = m_open1->bt_occupancy_max + 1;
181 occ_max_d1 = m_open1->bt_occupancy_max + 1;
182 }
183 if (!i1_ext->null) occ_max_i1 = MAX(occ_max_i1,i1_ext->bt_occupancy_max+1);
184 if (!d1_ext->null) occ_max_d1 = MAX(occ_max_d1,d1_ext->bt_occupancy_max+1);
185 if (!m_misms->null) occ_max_m = m_misms->bt_occupancy_max;
186 if (occ_max_m < occ_max_i1) occ_max_m = occ_max_i1;
187 if (occ_max_m < occ_max_d1) occ_max_m = occ_max_d1;
188 ++occ_max_m;
189 // Set new occupancy
190 wavefront_set->out_i1wavefront->bt_occupancy_max = occ_max_i1;
191 wavefront_set->out_d1wavefront->bt_occupancy_max = occ_max_d1;
192 wavefront_set->out_mwavefront->bt_occupancy_max = occ_max_m;
193 } else { // distance_metric == gap_affine_2p
194 // Parameters
195 const wavefront_t* const m_misms = wavefront_set->in_mwavefront_misms;
196 const wavefront_t* const m_open1 = wavefront_set->in_mwavefront_open1;
197 const wavefront_t* const m_open2 = wavefront_set->in_mwavefront_open2;
198 const wavefront_t* const i1_ext = wavefront_set->in_i1wavefront_ext;
199 const wavefront_t* const i2_ext = wavefront_set->in_i2wavefront_ext;
200 const wavefront_t* const d1_ext = wavefront_set->in_d1wavefront_ext;
201 const wavefront_t* const d2_ext = wavefront_set->in_d2wavefront_ext;
202 // Compute BT occupancy maximum (I)
203 if (!m_open1->null) {
204 occ_max_i1 = m_open1->bt_occupancy_max + 1;
205 occ_max_d1 = m_open1->bt_occupancy_max + 1;
206 }
207 // Compute BT occupancy maximum (D)
208 if (!i1_ext->null) occ_max_i1 = MAX(occ_max_i1,i1_ext->bt_occupancy_max+1);
209 if (!d1_ext->null) occ_max_d1 = MAX(occ_max_d1,d1_ext->bt_occupancy_max+1);
210 if (!m_open2->null) {
211 occ_max_i2 = m_open2->bt_occupancy_max + 1;
212 occ_max_d2 = m_open2->bt_occupancy_max + 1;
213 }
214 if (!i2_ext->null) occ_max_i2 = MAX(occ_max_i2,i2_ext->bt_occupancy_max+1);
215 if (!d2_ext->null) occ_max_d2 = MAX(occ_max_d2,d2_ext->bt_occupancy_max+1);
216 // Compute BT occupancy maximum (M)
217 if (!m_misms->null) occ_max_m = m_misms->bt_occupancy_max;
218 if (occ_max_m < occ_max_i1) occ_max_m = occ_max_i1;
219 if (occ_max_m < occ_max_i2) occ_max_m = occ_max_i2;
220 if (occ_max_m < occ_max_d1) occ_max_m = occ_max_d1;
221 if (occ_max_m < occ_max_d2) occ_max_m = occ_max_d2;
222 ++occ_max_m;
223 // Set new occupancy
224 wavefront_set->out_i1wavefront->bt_occupancy_max = occ_max_i1;
225 wavefront_set->out_i2wavefront->bt_occupancy_max = occ_max_i2;
226 wavefront_set->out_d1wavefront->bt_occupancy_max = occ_max_d1;
227 wavefront_set->out_d2wavefront->bt_occupancy_max = occ_max_d2;
228 wavefront_set->out_mwavefront->bt_occupancy_max = occ_max_m;
229 }

231void wavefront_backtrace_offload_affine(
232 wavefront_aligner_t* const wf_aligner,
233 const wavefront_set_t* const wavefront_set,
234 const int lo,
235 const int hi) {
236 // Parameters
237 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
238 // Compute maximum occupancy
239 wavefront_backtrace_offload_occupation_affine(wf_aligner,wavefront_set);
240 // Offload if necessary (Gap-Affine)
241 const wavefront_t* const wf_m = wavefront_set->out_mwavefront;
242 if (!wf_m->null && wf_m->bt_occupancy_max >= PCIGAR_MAX_LENGTH-1) {
243 wf_offset_t* const out_m = wavefront_set->out_mwavefront->offsets;
244 pcigar_t* const out_m_bt_pcigar = wavefront_set->out_mwavefront->bt_pcigar;
245 bt_block_idx_t* const out_m_bt_prev = wavefront_set->out_mwavefront->bt_prev;
246 wavefront_set->out_mwavefront->bt_occupancy_max =
247 wavefront_backtrace_offload_blocks_affine(
248 wf_aligner,out_m,out_m_bt_pcigar,out_m_bt_prev,lo,hi);
249 }
250 const wavefront_t* const wf_i1 = wavefront_set->out_i1wavefront;
251 if (!wf_i1->null && wf_i1->bt_occupancy_max >= PCIGAR_MAX_LENGTH-1) {
252 wf_offset_t* const out_i1 = wavefront_set->out_i1wavefront->offsets;

262 pcigar_t* const out_d1_bt_pcigar = wavefront_set->out_d1wavefront->bt_pcigar;
263 bt_block_idx_t* const out_d1_bt_prev = wavefront_set->out_d1wavefront->bt_prev;

267 }
268 if (distance_metric == gap_affine) return;
269 // Offload if necessary (Gap-Affine-2p)
270 const wavefront_t* const wf_i2 = wavefront_set->out_i2wavefront;
271 if (!wf_i2->null && wf_i2->bt_occupancy_max >= PCIGAR_MAX_LENGTH-1) {
272 wf_offset_t* const out_i2 = wavefront_set->out_i2wavefront->offsets;

274 bt_block_idx_t* const out_i2_bt_prev = wavefront_set->out_i2wavefront->bt_prev;
275 wavefront_set->out_i2wavefront->bt_occupancy_max =
276 wavefront_backtrace_offload_blocks_affine(
277 wf_aligner,out_i2,out_i2_bt_pcigar,out_i2_bt_prev,lo,hi);
278 }
279 const wavefront_t* const wf_d2 = wavefront_set->out_d2wavefront;
280 if (!wf_d2->null && wf_d2->bt_occupancy_max >= PCIGAR_MAX_LENGTH-1) {
281 wf_offset_t* const out_d2 = wavefront_set->out_d2wavefront->offsets;
282 pcigar_t* const out_d2_bt_pcigar = wavefront_set->out_d2wavefront->bt_pcigar;
283 bt_block_idx_t* const out_d2_bt_prev = wavefront_set->out_d2wavefront->bt_prev;
284 wavefront_set->out_d2wavefront->bt_occupancy_max =
