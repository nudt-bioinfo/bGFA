39void benchmark_indel_wavefront(
40 align_input_t* const align_input) {
41 // Parameters
42 wavefront_aligner_t* const wf_aligner = align_input->wf_aligner;
43 // Align
44 timer_start(&align_input->timer);
45 if (align_input->wfa_match_funct == NULL) {
46 wavefront_align(wf_aligner,
47 align_input->pattern,align_input->pattern_length,
48 align_input->text,align_input->text_length);
49 } else {
50 wavefront_align_lambda(wf_aligner,
51 align_input->wfa_match_funct,align_input->wfa_match_funct_arguments,
52 align_input->pattern_length,align_input->text_length);
53 }
54 timer_stop(&align_input->timer);
55 // DEBUG
56 if (align_input->debug_flags) {
57 benchmark_check_alignment(align_input,wf_aligner->cigar);
58 }
59 // Output
60 if (align_input->output_file) {
61 const int score_only = (wf_aligner->alignment_scope == compute_score);
62 benchmark_print_output(align_input,indel,score_only,wf_aligner->cigar);
63 }
64}
