113static inline uint32_t *ksw_push_cigar(void *km, int *n_cigar, int *m_cigar, uint32_t *cigar, uint32_t op, int len)
114{
115 if (*n_cigar == 0 || op != (cigar[(*n_cigar) - 1]&0xf)) {
116 if (*n_cigar == *m_cigar) {
117 *m_cigar = *m_cigar? (*m_cigar)<<1 : 4;
118 cigar = (uint32_t*)krealloc(km, cigar, (*m_cigar) << 2);
119 }
120 cigar[(*n_cigar)++] = len<<4 | op;
121 } else cigar[(*n_cigar)-1] += len<<4;
122 return cigar;
123}

125// In the backtrack matrix, value p[] has the following structure:
126// bit 0-2: which type gets the max - 0 for H, 1 for E, 2 for F, 3 for \tilde{E} and 4 for \tilde{F}
127// bit 3/0x08: 1 if a continuation on the E state (bit 5/0x20 for a continuation on \tilde{E})
128// bit 4/0x10: 1 if a continuation on the F state (bit 6/0x40 for a continuation on \tilde{F})
129static inline void ksw_backtrack(void *km, int is_rot, int is_rev, int min_intron_len, const uint8_t *p, const int *off, const int *off_end, int n_col, int i0, int j0,
130 int *m_cigar_, int *n_cigar_, uint32_t **cigar_)
131{ // p[] - lower 3 bits: which type gets the max; bit
132 int n_cigar = 0, m_cigar = *m_cigar_, i = i0, j = j0, r, state = 0;
133 uint32_t *cigar = *cigar_, tmp;
134 while (i >= 0 && j >= 0) { // at the beginning of the loop, _state_ tells us which state to check
135 int force_state = -1;
136 if (is_rot) {
137 r = i + j;
138 if (i < off[r]) force_state = 2;
139 if (off_end && i > off_end[r]) force_state = 1;
140 tmp = force_state < 0? p[(size_t)r * n_col + i - off[r]] : 0;
141 } else {
142 if (j < off[i]) force_state = 2;
143 if (off_end && j > off_end[i]) force_state = 1;
144 tmp = force_state < 0? p[(size_t)i * n_col + j - off[i]] : 0;
145 }
146 if (state == 0) state = tmp & 7; // if requesting the H state, find state one maximizes it.
147 else if (!(tmp >> (state + 2) & 1)) state = 0; // if requesting other states, _state_ stays the same if it is a continuation; otherwise, set to H
148 if (state == 0) state = tmp & 7; // TODO: probably this line can be merged into the "else if" line right above; not 100% sure
149 if (force_state >= 0) state = force_state;
150 if (state == 0) cigar = ksw_push_cigar(km, &n_cigar, &m_cigar, cigar, KSW_CIGAR_MATCH, 1), --i, --j;
151 else if (state == 1 || (state == 3 && min_intron_len <= 0)) cigar = ksw_push_cigar(km, &n_cigar, &m_cigar, cigar, KSW_CIGAR_DEL, 1), --i;
152 else if (state == 3 && min_intron_len > 0) cigar = ksw_push_cigar(km, &n_cigar, &m_cigar, cigar, KSW_CIGAR_N_SKIP, 1), --i;
153 else cigar = ksw_push_cigar(km, &n_cigar, &m_cigar, cigar, KSW_CIGAR_INS, 1), --j;
154 }
155 if (i >= 0) cigar = ksw_push_cigar(km, &n_cigar, &m_cigar, cigar, min_intron_len > 0 && i >= min_intron_len? KSW_CIGAR_N_SKIP : KSW_CIGAR_DEL, i + 1); // first deletion
156 if (j >= 0) cigar = ksw_push_cigar(km, &n_cigar, &m_cigar, cigar, KSW_CIGAR_INS, j + 1); // first insertion
157 if (!is_rev)
158 for (i = 0; i < n_cigar>>1; ++i) // reverse CIGAR
159 tmp = cigar[i], cigar[i] = cigar[n_cigar-1-i], cigar[n_cigar-1-i] = tmp;
160 *m_cigar_ = m_cigar, *n_cigar_ = n_cigar, *cigar_ = cigar;
161}

163static inline void ksw_cigar2eqx(void *km, const uint8_t *query, const uint8_t *target, int nc0, const uint32_t *ci0, int *mc1, int *nc1, uint32_t **ci1)
164{
165 int i, k, x = 0, y = 0;
166 *nc1 = 0;
167 for (k = 0; k < nc0; ++k) {
168 int op = ci0[k]&0xf, len = ci0[k]>>4;
169 if (op == KSW_CIGAR_MATCH) {
170 for (i = 0; i < len; ++i) {
171 if (target[x + i] == query[y + i]) ksw_push_cigar(km, nc1, mc1, *ci1, KSW_CIGAR_EQ, 1);
172 else ksw_push_cigar(km, nc1, mc1, *ci1, KSW_CIGAR_X, 1);
173 }
174 x += len, y += len;
175 } else {
176 ksw_push_cigar(km, nc1, mc1, *ci1, op, len);
177 if (op == KSW_CIGAR_DEL || op == KSW_CIGAR_N_SKIP) x += len;
178 else if (op == KSW_CIGAR_INS) y += len;
179 else if (op == KSW_CIGAR_EQ || op == KSW_CIGAR_X) x += len, y += len;
180 }
181 }

191static inline int ksw_apply_zdrop(ksw_extz_t *ez, int is_rot, int32_t H, int a, int b, int zdrop, int8_t e)
192{
193 int r, t;
194 if (is_rot) r = a, t = b;
195 else r = a + b, t = a;
196 if (H > (int32_t)ez->max) {
197 ez->max = H, ez->max_t = t, ez->max_q = r - t;
198 } else if (t >= ez->max_t && r - t >= ez->max_q) {
199 int tl = t - ez->max_t, ql = (r - t) - ez->max_q, l;
200 l = tl > ql? tl - ql : ql - tl;
201 if (zdrop >= 0 && ez->max - H > zdrop + l * e) {
202 ez->zdropped = 1;
203 return 1;
204 }
205 }
206 return 0;
207}
