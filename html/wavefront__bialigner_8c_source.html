40wavefront_bialigner_t* wavefront_bialigner_new(
41 wavefront_aligner_attr_t* const attributes,
42 wavefront_plot_t* const plot) {
43 // Allocate
44 wavefront_bialigner_t* const wf_bialigner = malloc(sizeof(wavefront_bialigner_t));
45 // Configure subsidiary aligners
46 wavefront_aligner_attr_t subsidiary_attr = wavefront_aligner_attr_default;
47 // Inherit attributes from master aligner
48 subsidiary_attr.distance_metric = attributes->distance_metric;
49 subsidiary_attr.linear_penalties = attributes->linear_penalties;
50 subsidiary_attr.affine_penalties = attributes->affine_penalties;
51 subsidiary_attr.affine2p_penalties = attributes->affine2p_penalties;
52 // Set specifics for subsidiary aligners
53 subsidiary_attr.heuristic = attributes->heuristic; // Inherit same heuristic
54 subsidiary_attr.memory_mode = wavefront_memory_high; // Classic WFA
55 subsidiary_attr.alignment_scope = compute_score; // BiWFAs are score-only
56 subsidiary_attr.alignment_form.extension = false; // Deactivate extension mode
57 // Set other parameter for subsidiary aligners
58 subsidiary_attr.system = attributes->system; // Inherit system configuration
59 // Allocate forward/reverse aligners
60 wf_bialigner->wf_forward = wavefront_aligner_new(&subsidiary_attr);
61 wf_bialigner->wf_forward->align_mode = wf_align_biwfa_breakpoint_forward;
62 wf_bialigner->wf_forward->plot = plot;
63 wf_bialigner->wf_reverse = wavefront_aligner_new(&subsidiary_attr);
64 wf_bialigner->wf_reverse->align_mode = wf_align_biwfa_breakpoint_reverse;
65 wf_bialigner->wf_reverse->plot = plot;
66 // Allocate subsidiary aligner
67 subsidiary_attr.alignment_scope = compute_alignment;
68 subsidiary_attr.heuristic.strategy = wf_heuristic_none; // Not inherited
69 wf_bialigner->wf_base = wavefront_aligner_new(&subsidiary_attr);
70 wf_bialigner->wf_base->align_mode = wf_align_biwfa_subsidiary;
71 wf_bialigner->wf_base->plot = plot;
72 // Return
73 return wf_bialigner;
74}
