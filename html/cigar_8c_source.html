35/*
36 * SAM CIGAR Operations
37 */
38#define SAM_CIGAR_MATCH 0
39#define SAM_CIGAR_INS 1
40#define SAM_CIGAR_DEL 2
41#define SAM_CIGAR_N_SKIP 3
42#define SAM_CIGAR_EQ 7
43#define SAM_CIGAR_X 8
44/* ... */
45#define SAM_CIGAR_NA 15
46
47const uint8_t sam_cigar_lut[256] =
48{
49 [0 ... 255] = SAM_CIGAR_NA,
50 ['M'] = SAM_CIGAR_MATCH,
51 ['I'] = SAM_CIGAR_INS,
52 ['D'] = SAM_CIGAR_DEL,
53 ['N'] = SAM_CIGAR_N_SKIP,
54 ['='] = SAM_CIGAR_EQ,
55 ['X'] = SAM_CIGAR_X,
56};

61cigar_t* cigar_new(
62 const int max_operations) {
63 // Allocate
64 cigar_t* const cigar = malloc(sizeof(cigar_t));
65 // Allocate alignment-operations buffer
66 cigar->max_operations = max_operations;
67 cigar->operations = malloc(cigar->max_operations);
68 cigar->begin_offset = 0;
69 cigar->end_offset = 0;
70 cigar->score = INT32_MIN;
71 cigar->end_v = -1;
72 cigar->end_h = -1;
73 // CIGAR
74 cigar->cigar_length = 0;
75 cigar->cigar_buffer = calloc(max_operations,sizeof(uint32_t));
76 // Return
77 return cigar;
78}
79void cigar_clear(
80 cigar_t* const cigar) {
81 // Alignment operations
82 cigar->begin_offset = 0;
83 cigar->end_offset = 0;
84 // Score and end position
85 cigar->score = INT32_MIN;
86 cigar->end_v = -1;
87 cigar->end_h = -1;
88 // CIGAR
89 cigar->cigar_length = 0;
90}
91void cigar_resize(
92 cigar_t* const cigar,
93 const int max_operations) {
94 // Check maximum operations
95 if (max_operations > cigar->max_operations) {
96 cigar->max_operations = max_operations;
97 free(cigar->operations); // Free
98 free(cigar->cigar_buffer); // Free
99 cigar->operations = malloc(max_operations); // Allocate
100 cigar->cigar_buffer = calloc(max_operations,sizeof(uint32_t)); // Allocate
101 }
102 cigar_clear(cigar);
103}

106 free(cigar->operations);

125void cigar_append_forward(
126 cigar_t* const cigar_dst,
127 cigar_t* const cigar_src) {
128 // Parameters
129 const int cigar_length = cigar_src->end_offset - cigar_src->begin_offset;
130 char* const operations_src = cigar_src->operations + cigar_src->begin_offset;
131 char* const operations_dst = cigar_dst->operations + cigar_dst->end_offset;
132 // Append forward
133 memcpy(operations_dst,operations_src,cigar_length);
134 // Update offset
135 cigar_dst->end_offset += cigar_length;
136}
137void cigar_append_reverse(
138 cigar_t* const cigar_dst,
139 cigar_t* const cigar_src) {
140 // Parameters
141 const int begin_offset = cigar_src->begin_offset;
142 const int end_offset = cigar_src->end_offset;
143 const int cigar_length = end_offset - begin_offset;
144 char* const operations_src = cigar_src->operations + begin_offset;
145 char* const operations_dst = cigar_dst->operations + cigar_dst->end_offset;
146 // Append reverse
147 int i;
148 for (i=0;i<cigar_length;++i) {
149 operations_dst[i] = operations_src[cigar_length-1-i];
150 }
151 // Update offset
152 cigar_dst->end_offset += cigar_length;
153}

178/*
179 * SAM-compliant CIGAR
180 */
181void cigar_compute_CIGAR(
182 cigar_t* const cigar,
183 const bool show_mismatches) {
184 // Prepare CIGAR (SAM compliant)
185 if (cigar->cigar_length==0 || cigar->has_misms!=show_mismatches) {
186 const char* const operations = cigar->operations;
187 const int begin_offset = cigar->begin_offset;
188 const int end_offset = cigar->end_offset;
189 // Check null CIGAR
190 if (begin_offset >= end_offset) {
191 cigar->cigar_length = 0;
192 return;
193 }
194 // Generate CIGAR
195 uint32_t* const cigar_buffer = cigar->cigar_buffer;
196 int cigar_length = 0;
197 char last_op = operations[begin_offset];
198 uint32_t last_op_len = 1;
199 int i;
200 for (i=begin_offset+1;i<end_offset;++i) {
201 // Fetch operation
202 char op = operations[i];
203 if (!show_mismatches && op=='X') op = 'M';
204 // Check previous operations
205 if (op == last_op) {
206 ++last_op_len;
207 } else {
208 // Dump operation
209 if (show_mismatches && last_op=='M') {
210 cigar_buffer[cigar_length++] = (last_op_len << 4) | ((uint32_t)SAM_CIGAR_EQ);
211 } else {
212 cigar_buffer[cigar_length++] = (last_op_len << 4) | ((uint32_t)sam_cigar_lut[(int)last_op]);
213 }
214 // Save new operation
215 last_op = op;
216 last_op_len = 1;
217 }
218 }
219 // Dump last operation
220 if (show_mismatches && last_op=='M') {
221 cigar_buffer[cigar_length++] = (last_op_len << 4) | ((uint32_t)SAM_CIGAR_EQ);
222 } else {
223 cigar_buffer[cigar_length++] = (last_op_len << 4) | ((uint32_t)sam_cigar_lut[(int)last_op]);
224 }
225 // Set as ready
226 cigar->has_misms = show_mismatches;
227 cigar->cigar_length = cigar_length;
228 }
229}
230void cigar_get_CIGAR(
231 cigar_t* const cigar,
232 const bool show_mismatches,
233 uint32_t** const cigar_buffer,
234 int* const cigar_length) {
235 // Compute CIGAR
236 cigar_compute_CIGAR(cigar,show_mismatches);
237 // Return
238 *cigar_buffer = cigar->cigar_buffer;
239 *cigar_length = cigar->cigar_length;
240}

244int cigar_score_edit(
245 cigar_t* const cigar) {
246 int score = 0, i;
247 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
248 switch (cigar->operations[i]) {
249 case 'M': break;
250 case 'X':
251 case 'D':
252 case 'I': ++score; break;
253 default:
254 fprintf(stderr,"[CIGAR] Computing CIGAR score: Unknown operation\n");
255 exit(1);
256 }
257 }
258 return score;
259}
260int cigar_score_gap_linear(
261 cigar_t* const cigar,
262 linear_penalties_t* const penalties) {
263 int score = 0, i;
264 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
265 switch (cigar->operations[i]) {
266 case 'M': score -= penalties->match; break;
267 case 'X': score -= penalties->mismatch; break;
268 case 'I': score -= penalties->indel; break;
269 case 'D': score -= penalties->indel; break;
270 default:
271 fprintf(stderr,"[CIGAR] Computing CIGAR score: Unknown operation\n");
272 exit(1);
273 }
274 }
275 return score;
276}
277int cigar_score_gap_affine(
278 cigar_t* const cigar,
279 affine_penalties_t* const penalties) {
280 char last_op = '\0';
281 int score = 0, i;
282 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
283 switch (cigar->operations[i]) {
284 case 'M':
285 score -= penalties->match;
286 break;
287 case 'X':
288 score -= penalties->mismatch;
289 break;
290 case 'D':
291 score -= penalties->gap_extension + ((last_op=='D') ? 0 : penalties->gap_opening);
292 break;
293 case 'I':
294 score -= penalties->gap_extension + ((last_op=='I') ? 0 : penalties->gap_opening);
295 break;
296 default:
297 fprintf(stderr,"[CIGAR] Computing CIGAR score: Unknown operation\n");
298 exit(1);
299 }
300 last_op = cigar->operations[i];
301 }
302 return score;
303}
304int cigar_score_gap_affine2p_score_op(
305 const char operation,
306 const int length,
307 affine2p_penalties_t* const penalties) {
308 switch (operation) {
309 case 'M':
310 return penalties->match*length;
311 case 'X':
312 return penalties->mismatch*length;
313 case 'D':
314 case 'I': {
315 const int score1 = penalties->gap_opening1 + penalties->gap_extension1*length;
316 const int score2 = penalties->gap_opening2 + penalties->gap_extension2*length;
317 return MIN(score1,score2);
318 }
319 default:
320 fprintf(stderr,"[CIGAR] Computing CIGAR score: Unknown operation\n");
321 exit(1);
322 }
323}
324int cigar_score_gap_affine2p(
325 cigar_t* const cigar,
326 affine2p_penalties_t* const penalties) {
327 char last_op = '\0';
328 int score = 0, op_length = 0;
329 int i;
330 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
331 // Account for operation
332 if (cigar->operations[i] != last_op && last_op != '\0') {
333 score -= cigar_score_gap_affine2p_score_op(last_op,op_length,penalties);
334 op_length = 0;
335 }
336 last_op = cigar->operations[i];
337 ++op_length;
338 }
339 // Account for last operation
340 score -= cigar_score_gap_affine2p_score_op(last_op,op_length,penalties);
341 return score;
342}

346int cigar_cmp(
347 cigar_t* const cigar_a,
348 cigar_t* const cigar_b) {
349 // Compare lengths
350 const int length_cigar_a = cigar_a->end_offset - cigar_a->begin_offset;
351 const int length_cigar_b = cigar_b->end_offset - cigar_b->begin_offset;
352 if (length_cigar_a != length_cigar_b) return length_cigar_a - length_cigar_b;
353 // Compare operations
354 char* const operations_a = cigar_a->operations + cigar_a->begin_offset;
355 char* const operations_b = cigar_b->operations + cigar_b->begin_offset;
356 int i;
357 for (i=0;i<length_cigar_a;++i) {
358 if (operations_a[i] != operations_b[i]) {
359 return operations_a[i] - operations_b[i];
360 }
361 }
362 // Equal
363 return 0;
364}

376void cigar_discover_mismatches(
377 const char* const pattern,
378 const int pattern_length,
379 const char* const text,
380 const int text_length,
381 cigar_t* const cigar) {
382 // Refine adding mismatches
383 int i, p=0, t=0;
384 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
385 // Check limits
386 if (p >= pattern_length || t >= text_length) break;
387 switch (cigar->operations[i]) {
388 case 'M':
389 cigar->operations[i] = (pattern[p]==text[t]) ? 'M' : 'X';
390 ++p; ++t;
391 break;
392 case 'I':
393 ++t;
394 break;
395 case 'D':
396 ++p;
397 break;
398 default:
399 fprintf(stderr,"[CIGAR] Wrong edit operation\n");
400 exit(1);
401 break;
402 }
403 }
404 while (p < pattern_length) { cigar->operations[i++] = 'D'; ++p; };
405 while (t < text_length) { cigar->operations[i++] = 'I'; ++t; };
406 cigar->end_offset = i;
407 cigar->operations[cigar->end_offset] = '\0';
408 // // DEBUG
409 // printf("Score=%ld\nPath-length=%" PRIu64 "\nCIGAR=%s\n",
410 // gaba_alignment->score,gaba_alignment->plen,
411 // cigar->operations);
412}
413/*
414 * Maxtrim
415 * Reduce the CIGAR to the maximal scoring sequence, starting from
416 * the beginning, under a given distance function
417 *
418 */
419bool cigar_maxtrim_gap_linear(
420 cigar_t* const cigar,
421 linear_penalties_t* const penalties) {
422 // Parameters
423 const char* const operations = cigar->operations;
424 const int begin_offset = cigar->begin_offset;
425 const int end_offset = cigar->end_offset;
426 const int match_score = (penalties->match!=0) ? penalties->match : -1;
427 // Max-score
428 int max_score = 0, max_score_offset = begin_offset, max_end_v = 0, max_end_h = 0;
429 // Traverse all cigar
430 int score = 0, end_v = 0, end_h = 0, i;
431 for (i=begin_offset;i<end_offset;++i) {
432 // Update score
433 switch (operations[i]) {
434 case 'M':
435 score -= match_score;
436 ++end_v; ++end_h;
437 break;
438 case 'X':
439 score -= penalties->mismatch;
440 ++end_v; ++end_h;
441 break;
442 case 'I':
443 score -= penalties->indel;
444 ++end_h;
445 break;
446 case 'D':
447 score -= penalties->indel;
448 ++end_v;
449 break;
450 }
451 // Compare max
452 if (max_score < score) {
453 max_score = score;
454 max_score_offset = i;
455 max_end_v = end_v;
456 max_end_h = end_h;
457 }
458 }
459 // Keep the max-scoring part of the cigar
460 const bool cigar_trimmed = (max_score_offset != end_offset-1);
461 if (max_score == 0) {
462 cigar_clear(cigar);
463 } else {
464 cigar->operations[max_score_offset+1] = '\0';
465 cigar->end_offset = max_score_offset + 1;
466 cigar->score = max_score;
467 cigar->end_v = max_end_v;
468 cigar->end_h = max_end_h;
469 }
470 // Return
471 return cigar_trimmed;
472}

529int cigar_maxtrim_gap_affine2p_score_op(
530 const char operation,
531 const int length,
532 affine2p_penalties_t* const penalties,
533 int* const end_v,
534 int* const end_h) {
535 switch (operation) {
536 case 'M': {
537 *end_v += length; *end_h += length;
538 const int match_score = (penalties->match!=0) ? penalties->match : -1;
539 return match_score*length;
540 }
541 case 'X':
542 *end_v += length; *end_h += length;
543 return penalties->mismatch*length;
544 case 'D': {
545 *end_v += length;
546 const int score1 = penalties->gap_opening1 + penalties->gap_extension1*length;
547 const int score2 = penalties->gap_opening2 + penalties->gap_extension2*length;
548 return MIN(score1,score2);
549 }
550 case 'I': {
551 *end_h += length;
552 const int score1 = penalties->gap_opening1 + penalties->gap_extension1*length;
553 const int score2 = penalties->gap_opening2 + penalties->gap_extension2*length;
554 return MIN(score1,score2);
555 }
556 default:
557 fprintf(stderr,"[CIGAR] Computing CIGAR score: Unknown operation\n");
558 exit(1);
559 }
560}
561bool cigar_maxtrim_gap_affine2p(
562 cigar_t* const cigar,
563 affine2p_penalties_t* const penalties) {
564 // Parameters
565 const char* const operations = cigar->operations;
566 const int begin_offset = cigar->begin_offset;
567 const int end_offset = cigar->end_offset;
568 if (begin_offset >= end_offset) return false;
569 // Max-score
570 int max_score = 0, max_score_offset = begin_offset, max_end_v = 0, max_end_h = 0;
571 // Traverse all cigar
572 char last_op = '\0';
573 int score = 0, end_v = 0, end_h = 0, op_length = 0;
574 int i;
575 for (i=begin_offset;i<end_offset;++i) {
576 // Account for operation
577 const char operation = operations[i];
578 if (operation != last_op && last_op != '\0') {
579 score -= cigar_maxtrim_gap_affine2p_score_op(last_op,op_length,penalties,&end_v,&end_h);
580 op_length = 0;
581 // Compare max
582 if (max_score < score) {
583 max_score = score;
584 max_score_offset = i - 1;
585 max_end_v = end_v;
586 max_end_h = end_h;
587 }
588 }
589 last_op = operation;
590 ++op_length;
591 }
592 // Account for last operation
593 score -= cigar_maxtrim_gap_affine2p_score_op(last_op,op_length,penalties,&end_v,&end_h);
594 if (max_score < score) {
595 max_score = score;
596 max_score_offset = end_offset - 1;
597 max_end_v = end_v;
598 max_end_h = end_h;
599 }
600 // Keep the max-scoring part of the cigar
601 const bool cigar_trimmed = (max_score_offset != end_offset-1);
602 if (max_score == 0) {
603 cigar_clear(cigar);
604 } else {
605 cigar->operations[max_score_offset+1] = '\0';
606 cigar->end_offset = max_score_offset + 1;
607 cigar->score = max_score;
608 cigar->end_v = max_end_v;
609 cigar->end_h = max_end_h;
610 }
611 // Return
612 return cigar_trimmed;
613}
614/*
615 * Check
616 */
617bool cigar_check_alignment(
618 FILE* const stream,
619 const char* const pattern,
620 const int pattern_length,
621 const char* const text,
622 const int text_length,
623 cigar_t* const cigar,
624 const bool verbose) {
625 // Parameters
626 char* const operations = cigar->operations;
627 // Traverse CIGAR
628 int pattern_pos=0, text_pos=0, i;
629 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
630 switch (operations[i]) {
631 case 'M':
632 // Check match
633 if (pattern[pattern_pos] != text[text_pos]) {
634 if (verbose) {
635 fprintf(stream,
636 "[CIGAR] Alignment not matching (pattern[%d]=%c != text[%d]=%c)\n",
637 pattern_pos,pattern[pattern_pos],text_pos,text[text_pos]);
638 }
639 return false;
640 }
641 ++pattern_pos;
642 ++text_pos;
643 break;
644 case 'X':
645 // Check mismatch
646 if (pattern[pattern_pos] == text[text_pos]) {
647 if (verbose) {
648 fprintf(stream,
649 "[CIGAR] Alignment not mismatching (pattern[%d]=%c == text[%d]=%c)\n",
650 pattern_pos,pattern[pattern_pos],text_pos,text[text_pos]);
651 }
652 return false;
653 }
654 ++pattern_pos;
655 ++text_pos;
656 break;
657 case 'I':
658 ++text_pos;
659 break;
660 case 'D':
661 ++pattern_pos;
662 break;
663 default:
664 fprintf(stream,"[CIGAR] Unknown edit operation '%c'\n",operations[i]);
665 exit(1);
666 break;
667 }
668 }
669 // Check alignment length
670 if (pattern_pos != pattern_length) {
671 if (verbose) {
672 fprintf(stream,
673 "[CIGAR] Alignment incorrect length (pattern-aligned=%d,pattern-length=%d)\n",
674 pattern_pos,pattern_length);
675 }
676 return false;
677 }
678 if (text_pos != text_length) {
679 if (verbose) {
680 fprintf(stream,
681 "[CIGAR] Alignment incorrect length (text-aligned=%d,text-length=%d)\n",
682 text_pos,text_length);
683 }
684 return false;
685 }
686 // OK
687 return true;
688}

706int cigar_sprint(
707 char* const buffer,
708 cigar_t* const cigar,
709 const bool print_matches) {
710 // Check null
711 if (cigar_is_null(cigar)) {
712 buffer[0] = '\0';
713 return 0;
714 }
715 // Parameters
716 const char* const operations = cigar->operations;

780void cigar_print_pretty(
781 FILE* const stream,
782 cigar_t* const cigar,
783 const char* const pattern,
784 const int pattern_length,
785 const char* const text,
786 const int text_length) {
787 // Parameters
788 char* const operations = cigar->operations;
789 const int begin_offset = cigar->begin_offset;
790 const int end_offset = cigar->end_offset;
791 // Allocate alignment buffers
792 const int max_buffer_length = text_length + pattern_length + 1;
793 char* const mem = calloc(3*max_buffer_length,1);
794 char* const pattern_alg = mem;
795 char* const ops_alg = pattern_alg + max_buffer_length;
796 char* const text_alg = ops_alg + max_buffer_length;
797 // Compute alignment buffers
798 int i, alg_pos = 0, pattern_pos = 0, text_pos = 0;
799 for (i=begin_offset;i<end_offset;++i) {
800 switch (operations[i]) {
801 case 'M':
802 if (pattern[pattern_pos] != text[text_pos]) {
803 pattern_alg[alg_pos] = pattern[pattern_pos];
804 ops_alg[alg_pos] = 'X';
805 text_alg[alg_pos++] = text[text_pos];
806 } else {
807 pattern_alg[alg_pos] = pattern[pattern_pos];
808 ops_alg[alg_pos] = '|';
809 text_alg[alg_pos++] = text[text_pos];
810 }
811 pattern_pos++; text_pos++;
812 break;
813 case 'X':
814 if (pattern[pattern_pos] != text[text_pos]) {
815 pattern_alg[alg_pos] = pattern[pattern_pos++];
816 ops_alg[alg_pos] = ' ';
817 text_alg[alg_pos++] = text[text_pos++];
818 } else {
819 pattern_alg[alg_pos] = pattern[pattern_pos++];
820 ops_alg[alg_pos] = 'X';
821 text_alg[alg_pos++] = text[text_pos++];
822 }
823 break;
824 case 'I':
825 pattern_alg[alg_pos] = '-';
826 ops_alg[alg_pos] = ' ';
827 text_alg[alg_pos++] = text[text_pos++];
828 break;
829 case 'D':
830 pattern_alg[alg_pos] = pattern[pattern_pos++];
831 ops_alg[alg_pos] = ' ';
832 text_alg[alg_pos++] = '-';
833 break;
834 default:
835 break;
836 }
837 }
838 i=0;
839 while (pattern_pos < pattern_length) {
840 pattern_alg[alg_pos+i] = pattern[pattern_pos++];
841 ops_alg[alg_pos+i] = '?';
842 ++i;
843 }
844 i=0;
845 while (text_pos < text_length) {
846 text_alg[alg_pos+i] = text[text_pos++];
847 ops_alg[alg_pos+i] = '?';
848 ++i;
849 }
850 // Print string
851 fprintf(stream," ALIGNMENT ");
852 cigar_print(stream,cigar,true);
853 fprintf(stream,"\n");
854 fprintf(stream," ETRACE ");
855 cigar_print(stream,cigar,false);
856 fprintf(stream,"\n");
857 fprintf(stream," CIGAR ");
858 cigar_print_SAM_CIGAR(stream,cigar,false);
859 fprintf(stream,"\n");
860 fprintf(stream," PATTERN %s\n",pattern_alg);
861 fprintf(stream," %s\n",ops_alg);
862 fprintf(stream," TEXT %s\n",text_alg);
863 // Free
864 free(mem);
865}
