40void wavefront_plot_heatmaps_allocate(
41 wavefront_plot_t* const wf_plot,
42 const int pattern_length,
43 const int text_length) {
44 wavefront_plot_attr_t* const attributes = &wf_plot->attributes;
45 // Compute dimensions
46 const int resolution_points = attributes->resolution_points;
47 const int min_v = (wf_plot->min_v == -1) ? 0 : wf_plot->min_v;
48 const int max_v = (wf_plot->max_v == -1) ? pattern_length-1 : wf_plot->max_v;
49 const int min_h = (wf_plot->min_h == -1) ? 0 : wf_plot->min_h;
50 const int max_h = (wf_plot->max_h == -1) ? text_length-1 : wf_plot->max_h;
51 // Behavior
52 wf_plot->behavior_heatmap = heatmap_new(heatmap_value,
53 min_v,max_v,min_h,max_h,resolution_points);
54 // Wavefront Components
55 wf_plot->m_heatmap = heatmap_new(heatmap_min,
56 min_v,max_v,min_h,max_h,resolution_points);
57 wf_plot->i1_heatmap = NULL;
58 wf_plot->d1_heatmap = NULL;
59 wf_plot->i2_heatmap = NULL;
60 wf_plot->d2_heatmap = NULL;
61 if (wf_plot->distance_metric < gap_affine) return;
62 // Gap-affine
63 wf_plot->i1_heatmap = heatmap_new(heatmap_min,
64 min_v,max_v,min_h,max_h,resolution_points);
65 wf_plot->d1_heatmap = heatmap_new(heatmap_min,
66 min_v,max_v,min_h,max_h,resolution_points);
67 if (wf_plot->distance_metric == gap_affine) return;
68 // Gap-affine-2p
69 wf_plot->i2_heatmap = heatmap_new(heatmap_min,
70 min_v,max_v,min_h,max_h,resolution_points);
71 wf_plot->d2_heatmap = heatmap_new(heatmap_min,
72 min_v,max_v,min_h,max_h,resolution_points);
73}
74void wavefront_plot_heatmaps_free(

77 heatmap_delete(wf_plot->m_heatmap);
78 if (wf_plot->i1_heatmap) heatmap_delete(wf_plot->i1_heatmap);
79 if (wf_plot->d1_heatmap) heatmap_delete(wf_plot->d1_heatmap);
80 if (wf_plot->i2_heatmap) heatmap_delete(wf_plot->i2_heatmap);
81 if (wf_plot->d2_heatmap) heatmap_delete(wf_plot->d2_heatmap);
82}

124void wavefront_plot_component(
125 wavefront_aligner_t* const wf_aligner,
126 wavefront_t* const wavefront,
127 const int score,
128 heatmap_t* const wf_heatmap,
129 const bool extend) {
130 // Check wavefront
131 if (wavefront == NULL) return;
132 // Parameters
133 wavefront_sequences_t* const sequences = &wf_aligner->sequences;
134 const int pattern_begin = sequences->pattern_begin;
135 const int pattern_length = sequences->pattern_length;
136 const int text_begin = sequences->text_begin;
137 const int text_length = sequences->text_length;
138 const char* const pattern = sequences->pattern;
139 const char* const text = sequences->text;
140 const bool reverse = (wf_aligner->align_mode == wf_align_biwfa_breakpoint_reverse);
141 // Traverse all offsets
142 int k;
143 for (k=wavefront->lo;k<=wavefront->hi;++k) {
144 const wf_offset_t offset = wavefront->offsets[k];
145 if (offset < 0) continue;
146 // Compute local coordinates
147 int v_local = WAVEFRONT_V(k,offset);
148 int h_local = WAVEFRONT_H(k,offset);
149 if (v_local < 0 || v_local >= pattern_length) continue;
150 if (h_local < 0 || h_local >= text_length) continue;
151 // Compute global coordinates
152 int v_global, h_global;
153 if (reverse) {
154 v_global = pattern_begin + (pattern_length - 1 - v_local);
155 h_global = text_begin + (text_length - 1 - h_local);
156 } else {
157 v_global = pattern_begin + v_local;
158 h_global = text_begin + h_local;
159 }
160 // Plot
161 if (reverse) {
162 if (h_local>0 && v_local>0) heatmap_set(wf_heatmap,v_global+1,h_global+1,score);
163 } else {
164 if (h_local>0 && v_local>0) heatmap_set(wf_heatmap,v_global-1,h_global-1,score);
165 }
166 // Simulate extension
167 if (extend) {
168 while (v_local < pattern_length &&
169 h_local < text_length &&
170 pattern[v_local] == text[h_local]) {
171 if (reverse) {
172 v_global--; h_global--;
173 } else {
174 v_global++; h_global++;
175 }
176 v_local++; h_local++;
177 if (reverse) {
178 heatmap_set(wf_heatmap,v_global+1,h_global+1,score);
179 } else {
180 heatmap_set(wf_heatmap,v_global-1,h_global-1,score);
181 }
182 }
183 }
184 }

186void wavefront_plot(
187 wavefront_aligner_t* const wf_aligner,
188 const int score,
189 const int align_level) {
190 // Check plotting enabled wrt align-level
191 if (wf_aligner->align_mode == wf_align_biwfa_breakpoint_forward ||
192 wf_aligner->align_mode == wf_align_biwfa_breakpoint_reverse) {
193 if (align_level != wf_aligner->plot->attributes.align_level) return;
194 }
195 if (wf_aligner->align_mode == wf_align_biwfa_subsidiary &&
196 wf_aligner->plot->attributes.align_level != -1) return;
197 // Parameters
198 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
199 wavefront_components_t* const wf_components = &wf_aligner->wf_components;
200 const int score_mod = (wf_components->memory_modular) ? score%wf_components->max_score_scope : score;
201 // Plot wavefront components
202 wavefront_plot_component(wf_aligner,
203 wf_components->mwavefronts[score_mod],
204 score,wf_aligner->plot->m_heatmap,true);
205 if (distance_metric < gap_affine) return;
206 // Gap-affine
207 wavefront_plot_component(wf_aligner,
208 wf_components->i1wavefronts[score_mod],
209 score,wf_aligner->plot->i1_heatmap,false);
210 wavefront_plot_component(wf_aligner,
211 wf_components->d1wavefronts[score_mod],
212 score,wf_aligner->plot->d1_heatmap,false);
213 if (distance_metric == gap_affine) return;
214 // Gap-affine-2p
215 wavefront_plot_component(wf_aligner,
216 wf_components->i2wavefronts[score_mod],
217 score,wf_aligner->plot->i2_heatmap,false);
218 wavefront_plot_component(wf_aligner,
219 wf_components->d2wavefronts[score_mod],
220 score,wf_aligner->plot->d2_heatmap,false);
221}

225void wavefront_plot_print_cigar(
226 FILE* const stream,
227 cigar_t* const cigar,
228 const char target_operation) {
229 int i, h=0, v=0, count=0;
230 for (i=cigar->begin_offset;i<cigar->end_offset;++i) {
231 // Check operation
232 const char operation = cigar->operations[i];
233 switch (operation) {
234 case 'M': case 'X': ++h; ++v; break;
235 case 'I': ++h; break;
236 case 'D': ++v; break;
237 default: break;
238 }
239 // Print point
240 if (operation == target_operation && h>0 && v>0) {
241 if (count++ > 0) fprintf(stream,";");
242 fprintf(stream,"%d,%d",h-1,v-1);
243 }
244 }

246void wavefront_plot_print(
247 FILE* const stream,
248 wavefront_aligner_t* const wf_aligner) {
249 // Parameters
250 const distance_metric_t distance_metric = wf_aligner->penalties.distance_metric;
251 wavefront_plot_t* const wf_plot = wf_aligner->plot;
252 wavefront_sequences_t* sequences = NULL;
253 if (wf_aligner->bialigner == NULL) {
254 sequences = &wf_aligner->sequences;
255 } else {
256 sequences = &wf_aligner->bialigner->wf_forward->sequences;
257 wavefront_sequences_set_bounds(sequences,
258 0,sequences->pattern_buffer_length,
259 0,sequences->text_buffer_length);
260 }
261 const int pattern_length = sequences->pattern_buffer_length;
262 const int text_length = sequences->text_buffer_length;
263 // Check plot
264 if (wf_aligner->plot == NULL) {
265 fprintf(stream,"# WFA-plot not enabled\n");
266 return;
267 }
268 // Metadata
269 if (sequences->mode == wf_sequences_lambda) {
270 fprintf(stream,"# PatternLength %d\n",pattern_length);
271 fprintf(stream,"# TextLength %d\n",text_length);
272 fprintf(stream,"# Pattern -\n");
273 fprintf(stream,"# Text -\n");
274 } else {
275 fprintf(stream,"# PatternLength %d\n",pattern_length);
276 fprintf(stream,"# Pattern %.*s\n",pattern_length,sequences->pattern);
277 fprintf(stream,"# TextLength %d\n",text_length);
278 fprintf(stream,"# Text %.*s\n",text_length,sequences->text);
279 }
280 fprintf(stream,"# Penalties ");
281 wavefront_penalties_print(stream,&wf_aligner->penalties);
282 fprintf(stream,"\n");
283 // Alignment mode
284 fprintf(stream,"# WFAMode ");
285 wavefront_heuristic_t* const wf_heuristic = &wf_aligner->heuristic;
286 wavefront_heuristic_print(stream,wf_heuristic);
287 fprintf(stream,"\n");
288 // Wavefront components
289 fprintf(stream,"# Heatmap M\n"); heatmap_print(stream,wf_plot->m_heatmap);
290 if (distance_metric == gap_affine) {
291 fprintf(stream,"# Heatmap I1\n"); heatmap_print(stream,wf_plot->i1_heatmap);
292 fprintf(stream,"# Heatmap D1\n"); heatmap_print(stream,wf_plot->d1_heatmap);
293 }
294 if (distance_metric == gap_affine_2p) {
295 fprintf(stream,"# Heatmap I2\n"); heatmap_print(stream,wf_plot->i2_heatmap);
296 fprintf(stream,"# Heatmap D2\n"); heatmap_print(stream,wf_plot->d2_heatmap);
297 }
298 // Extend
299 fprintf(stream,"# Heatmap Extend\n"); heatmap_print(stream,wf_plot->behavior_heatmap);
300 // CIGAR
301 if (wf_aligner->alignment_scope == compute_alignment) {
302 fprintf(stream,"# List CIGAR-M ");
303 wavefront_plot_print_cigar(stream,wf_aligner->cigar,'M');
304 fprintf(stream,"\n");
305 fprintf(stream,"# List CIGAR-X ");
306 wavefront_plot_print_cigar(stream,wf_aligner->cigar,'X');
307 fprintf(stream,"\n");
308 fprintf(stream,"# List CIGAR-I ");
309 wavefront_plot_print_cigar(stream,wf_aligner->cigar,'I');
310 fprintf(stream,"\n");
311 fprintf(stream,"# List CIGAR-D ");
312 wavefront_plot_print_cigar(stream,wf_aligner->cigar,'D');
313 fprintf(stream,"\n");
314 }
